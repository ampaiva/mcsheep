 166.        private static void safeClose(InitialContext context) {
 167.            if (context == null) {
 168.                return;
 169.            }
 170.            try {
 171.                context.close();
 172.            } catch (Throwable t) {
 173.                // just log
 174.                log.info("Ignoring a problem which occurred while closing: " + context, t);
 175.            }
 176.            context = null;
 177.        }
