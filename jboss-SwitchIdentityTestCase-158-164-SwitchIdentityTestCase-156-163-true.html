 369.         * A {@link ServerSetupTask} instance which creates remoting mappings for this test case.
 370.         *
 371.         * @author Josef Cacek
 372.         */
 373.        static class RemotingSetup implements ServerSetupTask {
 374.    
 375.            public void setup(ManagementClient managementClient, String containerId) throws Exception {
 376.                final List updates = new LinkedList();
 377.                LOGGER.info("Adding socket binding");
 378.                // /socket-binding-group=standard-sockets/remote-destination-outbound-socket-binding=ejb-outbound:add(host=localhost,port=8080) {allow-resource-service-restart=true}
 379.                ModelNode socketBindingModelNode = Util.createAddOperation(ADDR_SOCKET_BINDING);
 380.                socketBindingModelNode.get(HOST).set(Utils.getHost(managementClient));
 381.                socketBindingModelNode.get(PORT).set(8080);
 382.                socketBindingModelNode.get(OPERATION_HEADERS, ALLOW_RESOURCE_SERVICE_RESTART).set(true);
 383.                updates.add(socketBindingModelNode);
 384.    
 385.                final ModelNode compositeOp = new ModelNode();
 386.                compositeOp.get(OP).set(COMPOSITE);
 387.                compositeOp.get(OP_ADDR).setEmptyList();
 388.                ModelNode steps = compositeOp.get(STEPS);
 389.    
 390.                // /subsystem=remoting/remote-outbound-connection=ejb-outbound-connection:add(outbound-socket-binding-ref=ejb-outbound,username=ConnectionUser,security-realm=ejb-outbound-realm) {allow-resource-service-restart=true}
 391.                final ModelNode remotingConnectorModelNode = Util.createAddOperation(ADDR_REMOTING_CONNECTOR);
 392.                remotingConnectorModelNode.get("outbound-socket-binding-ref").set(EJB_OUTBOUND_SOCKET_BINDING);
 393.                remotingConnectorModelNode.get("username").set(EJBUtil.CONNECTION_PASSWORD);
 394.                remotingConnectorModelNode.get("security-realm").set(EJB_OUTBOUND_REALM);
 395.                remotingConnectorModelNode.get("protocol").set("http-remoting");
 396.                remotingConnectorModelNode.get(OPERATION_HEADERS, ALLOW_RESOURCE_SERVICE_RESTART).set(true);
 397.                steps.add(remotingConnectorModelNode);
 398.    
 399.                // /subsystem=remoting/remote-outbound-connection=ejb-outbound-connection/property=SSL_ENABLED:add(value=false)
 400.                final ModelNode sslPropertyModelNode = Util.createAddOperation(ADDR_REMOTING_CONNECTOR.append("property",
 401.                        "SSL_ENABLED"));
 402.                sslPropertyModelNode.get(VALUE).set(false);
 403.                sslPropertyModelNode.get(OPERATION_HEADERS, ALLOW_RESOURCE_SERVICE_RESTART).set(true);
 404.                steps.add(sslPropertyModelNode);
 405.    
 406.                updates.add(compositeOp);
 407.                Utils.applyUpdates(updates, managementClient.getControllerClient());
 408.    
 409.            }
 410.    
 411.            public void tearDown(ManagementClient managementClient, String containerId) throws Exception {
 412.                final List updates = new ArrayList();
 413.    
 414.                // /subsystem=remoting/remote-outbound-connection=ejb-outbound-connection:remove()
 415.                ModelNode op = Util.createRemoveOperation(ADDR_REMOTING_CONNECTOR);
 416.                op.get(OPERATION_HEADERS, ROLLBACK_ON_RUNTIME_FAILURE).set(false);
 417.                op.get(OPERATION_HEADERS, ALLOW_RESOURCE_SERVICE_RESTART).set(true);
 418.                updates.add(op);
 419.    
 420.                // /socket-binding-group=standard-sockets/remote-destination-outbound-socket-binding=ejb-outbound:remove()
 421.                op = Util.createRemoveOperation(ADDR_SOCKET_BINDING);
 422.                op.get(OPERATION_HEADERS, ROLLBACK_ON_RUNTIME_FAILURE).set(false);
 423.                op.get(OPERATION_HEADERS, ALLOW_RESOURCE_SERVICE_RESTART).set(true);
 424.                updates.add(op);
 425.    
 426.                Utils.applyUpdates(updates, managementClient.getControllerClient());
 427.            }
