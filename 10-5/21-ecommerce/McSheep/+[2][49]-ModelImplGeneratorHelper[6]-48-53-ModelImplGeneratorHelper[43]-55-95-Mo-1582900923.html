 125.                                    flagMap.put("Date", "");
 126.                                } else if (c.getType().equals("Time")

 129.                                    flagMap.put("Time", "");
 130.                                }
 131.    
 132.                                // private fields
 133.                                privateFieldsSb.append("\r\tprivate ")
 134.                                        .append(c.getType()).append(" ")
 135.                                        .append(c.getName()).append(";").append("\r\n");
 136.    
 137.                                // getter
 138.                                setterAndGetterSb.append("\r\tpublic ")
 139.                                        .append(c.getType()).append(" get")
 140.                                        .append(nameCapitalition).append("(")
 141.                                        .append(") {").append("\r\t\treturn ")
 142.                                        .append(c.getName()).append(";")
 143.                                        .append("\r\t}").append("\r\n");
 144.    
 145.                                // setter

 147.                                        .append(" set").append(nameCapitalition)
 148.                                        .append("(").append(c.getType()).append(" ")
 149.                                        .append(nameUnCapitalition).append(") {")
 150.                                        .append("\r\t\tthis.")
 151.                                        .append(nameUnCapitalition).append(" = ")
 152.                                        .append(nameUnCapitalition).append(";")
 153.                                        .append("\r\t}").append("\r\n");
 154.                                
 155.                                // mock
 156.                                String valueObj = TypeTransformEnum.getMockValueByTypeInXml(c.getType(), nameCapitalition);
 157.                                mockSb.append("\r\t\t").append("${modelVariable}.set")
 158.                                        .append(nameCapitalition)
 159.                                        .append("(")
 160.                                        .append(valueObj)
 161.                                        .append(");");
 162.                            }
 163.    
 164.                        } // end: column
 165.                    } // end: if:column
 166.    
 167.                    String modelName = entity.getPackageName() + ".intf."
 168.                            + entity.getName();
 169.                    importsSb.append("import " + modelName + ";\n");
 170.                    String outputTemplate = template
 171.                            .replace("${packageName}",
 172.                                    entity.getPackageName() + ".impl")
 173.                            .replace("${imports}", importsSb.toString())
 174.                            .replace(
 175.                                    "${modelImpl}",
 176.                                    StringUtils.capitalize(entity.getName())
 177.                                            + "Impl")
 178.                            .replace("${model}",
 179.                                    StringUtils.capitalize(entity.getName()))
 180.                            .replace("${privateFields}", privateFieldsSb.toString())
 181.                            .replace("${setterAndGetter}",
 182.                                    setterAndGetterSb.toString())
 183.                            .replace("${mockSetValue}", mockSb.toString())
 184.                            .replace("${modelVariable}",
 185.                                    entity.getName().toLowerCase())
 186.                                    ;
 187.    
 188.                    String packageFileName = entity.getPackageName().replace(".",
 189.                            "/");
 190.                    StringBuilder fileNameSb = new StringBuilder(
 191.                            DMConstants.sourceDirectory);
 192.                    fileNameSb.append(packageFileName).append("/").append("impl/")
 193.                            .append(entity.getName()).append("Impl")
 194.                            .append(".java");
 195.    
 196.                    // write to source file.
 197.                    GeneratorHelper
 198.                            .writeToDestFile(outputTemplate, fileNameSb.toString());
 199.                } // end: entity
 200.            } // end: if:entity

 202.        }
