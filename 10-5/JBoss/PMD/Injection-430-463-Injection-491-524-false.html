 370.                int to = input.indexOf("}");
 371.                int dv = input.indexOf(":", from + 2);
 372.    
 373.                if (dv != -1) {
 374.                    if (dv > to)
 375.                        dv = -1;
 376.                }
 377.    
 378.                String systemProperty = "";
 379.                String defaultValue = "";
 380.                if (dv == -1) {

 383.                        systemProperty = File.separator;
 384.                    } else if (":".equals(s)) {

 388.                    }
 389.                } else {
 390.                    systemProperty = WildFlySecurityManager.getSystemPropertiesPrivileged().getProperty(input.substring(from + 2, dv));
 391.                    defaultValue = input.substring(dv + 1, to);
 392.                }
 393.                String prefix = "";
 394.                String postfix = "";
 395.    
 396.                if (from != 0) {
 397.                    prefix = input.substring(0, from);
 398.                }
 399.    
 400.                if (to + 1 < input.length() - 1) {
 401.                    postfix = input.substring(to + 1);
 402.                }
 403.    
 404.                if (systemProperty != null && !systemProperty.trim().equals("")) {
 405.                    input = prefix + systemProperty + postfix;
 406.                } else if (!defaultValue.trim().equals("")) {
 407.                    input = prefix + defaultValue + postfix;
 408.                } else {
 409.                    input = prefix + postfix;
 410.                }
 411.            }
 412.            return input;
 413.        }
