 367.            } else if (propagate && previous != null && previous.getOutgoingRunAs() != null) {

 373.        private boolean authenticate(SecurityContext context, Subject subject) {
 374.            SecurityContextUtil util = context.getUtil();
 375.            SubjectInfo subjectInfo = getSubjectInfo(context);
 376.            if (subject == null) {
 377.                subject = new Subject();
 378.            }
 379.            Principal principal = util.getUserPrincipal();
 380.            Principal auditPrincipal = principal;
 381.            Object credential = util.getCredential();
 382.            Identity unauthenticatedIdentity = null;
 383.    
 384.            boolean authenticated = false;
 385.            if (principal == null) {
 386.                unauthenticatedIdentity = getUnauthenticatedIdentity();
 387.                subjectInfo.addIdentity(unauthenticatedIdentity);
 388.                auditPrincipal = unauthenticatedIdentity.asPrincipal();
 389.                subject.getPrincipals().add(auditPrincipal);
 390.                authenticated = true;
 391.            } else {
 392.                subject.getPrincipals().add(principal);
 393.            }
 394.    
 395.            if (authenticated == false) {
 396.                AuthenticationManager authenticationManager = context.getAuthenticationManager();
 397.                authenticated = authenticationManager.isValid(principal, credential, subject);
 398.            }
 399.            if (authenticated == true) {
 400.                subjectInfo.setAuthenticatedSubject(subject);
 401.            }
 402.    
 403.            AuditManager auditManager = context.getAuditManager();
 404.            if (auditManager != null) {
 405.                audit(authenticated ? AuditLevel.SUCCESS : AuditLevel.FAILURE, auditManager, auditPrinci...
 406.            }
 407.    
 408.            return authenticated;
 409.        }
