  50.    /**
  51.     * <p/>
  52.     * This implementation of {@code org.omg.PortableInterceptor.ClientRequestInterceptor} inserts the s...
  53.     * service (SAS) context into outgoing IIOP requests and handles the SAS messages received from the ...
  54.     * service in the SAS context of incoming IIOP replies.
  55.     *
  56.     * @author <a href="mailto:reverbel@ime.usp.br">Francisco Reverbel</a>

  58.     */
  59.    public class SASClientInterceptor extends LocalObject implements ClientRequestInterceptor {
  60.

  63.         */

  67.

  69.

 100.        @Override
 101.        public void send_request(ClientRequestInfo ri) {
 102.            try {
 103.                CompoundSecMech secMech = CSIv2Util.getMatchingSecurityMech(ri, codec,
 104.                        EstablishTrustInClient.value,  /* client supports */
 105.                        (short) 0                       /* client requires */);
 106.                if (secMech == null) {
 107.                    return;
 108.                }
 109.    
 110.    
 111.                if ((secMech.as_context_mech.target_supports & EstablishTrustInClient.value) != 0) {
 112.                    Principal p = SecurityActions.getPrincipal();
 113.                    if (p != null) {
 114.                        byte[] encodedTargetName = secMech.as_context_mech.target_name;
 115.    
 116.                        // The name scope needs to be externalized.
 117.                        String name = p.getName();
 118.                        if (name.indexOf('@') < 0) {
 119.                            byte[] decodedTargetName = CSIv2Util.decodeGssExportedName(encodedTargetName...
 120.                            String targetName = new String(decodedTargetName, StandardCharsets.UTF_8);
 121.                            name += "@" + targetName; // "@default"
 122.                        }
 123.                        byte[] username = name.getBytes(StandardCharsets.UTF_8);
 124.                        // I don't know why there is not a better way to go from char[] -> byte[].
 125.    
 126.                        Object credential = SecurityActions.getCredential();
 127.                        byte[] password = {};
 128.                        if (credential instanceof char[]) {
 129.                            String tmp = new String((char[]) credential);
 130.                            password = tmp.getBytes(StandardCharsets.UTF_8);
 131.                        } else if (credential instanceof byte[])
 132.                            password = (byte[]) credential;
 133.                        else if (credential != null) {
 134.                            String tmp = credential.toString();
 135.                            password = tmp.getBytes(StandardCharsets.UTF_8);
 136.                        }
 137.    
 138.    
 139.                        // create authentication token.
 140.                        InitialContextToken authenticationToken = new InitialContextToken(username, pass...
 141.                                encodedTargetName);
 142.                        // ASN.1-encode it, as defined in RFC 2743.
 143.                        byte[] encodedAuthenticationToken = CSIv2Util.encodeInitialContextToken(authenti...
 144.                                codec);
 145.    
 146.                        // create EstablishContext message with the encoded token.
 147.                        EstablishContext message = new EstablishContext(0, // stateless ctx id
 148.                                noAuthorizationToken, absentIdentityToken, encodedAuthenticationToken);
 149.    
 150.    
 151.                        // create SAS context with the EstablishContext message.
 152.                        SASContextBody contextBody = new SASContextBody();
 153.                        contextBody.establish_msg(message);
 154.    
 155.                        // stuff the SAS context into the outgoing request.
 156.                        Any any = ORB.init().create_any();
 157.                        SASContextBodyHelper.insert(any, contextBody);
 158.                        ServiceContext sc = new ServiceContext(sasContextId, codec.encode_value(any));
 159.                        ri.add_request_service_context(sc, true /*replace existing context*/);
 160.                    }
 161.                }
 162.            } catch (Exception e) {
 163.                throw IIOPLogger.ROOT_LOGGER.unexpectedException(e);
 164.            }
 165.        }

 176.                SASContextBody contextBody = SASContextBodyHelper.extract(msg);
 177.    
 178.                // At this point contextBody should contain a CompleteEstablishContext message, which do...

 183.                    // should not happen.
 184.                    throw IIOPLogger.ROOT_LOGGER.unexpectedContextErrorInSASReply(0, CompletionStatus.CO...
 185.                }
 186.            } catch (BAD_PARAM e) {
 187.                // no service context with sasContextId: do nothing
 188.            } catch (FormatMismatch e) {

 190.            } catch (TypeMismatch e) {
 191.                throw IIOPLogger.ROOT_LOGGER.errorParsingSASReply(e, 0, CompletionStatus.COMPLETED_YES);...
 192.            }
 193.        }
