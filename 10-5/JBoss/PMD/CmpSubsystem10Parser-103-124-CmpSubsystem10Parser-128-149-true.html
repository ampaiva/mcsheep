 145.                throw missingRequired(reader, Collections.singleton(Attribute.NAME));
 146.            }
 147.    
 148.            final ModelNode op = Util.createAddOperation(parentAddress.append(HILO_KEY_GENERATOR, name));
 149.            final EnumSet required = EnumSet.of(Element.DATA_SOURCE, Element.TABLE_NAME, Element.ID_COLUMN, Element.SEQUENCE_COLUMN, Element.SEQUENCE_NAME);
 150.    
 151.            while (reader.hasNext() && reader.nextTag() != END_ELEMENT) {
 152.                final String value = reader.getElementText();
 153.                final String tag = reader.getLocalName();
 154.                final Element element = Element.forName(tag);
 155.                required.remove(element);
 156.    
 157.                SimpleAttributeDefinition attribute = HiLoKeyGeneratorResourceDefinition.ATTRIBUTE_MAP.get(tag);
 158.                if(attribute == null) {
 159.                    throw unexpectedElement(reader);
 160.                }
 161.                attribute.parseAndSetParameter(value, op, reader);
 162.            }
 163.            if(!required.isEmpty()) {
 164.                throw missingRequiredElement(reader, required);
 165.            }
 166.            return op;
 167.        }

 169.        public void writeContent(final XMLExtendedStreamWriter writer, final SubsystemMarshallingContext context) throws XMLStreamException {
 170.            context.startSubsystemElement(Namespace.CURRENT.getUriString(), false);
 171.    
 172.            final ModelNode model = context.getModelNode();
 173.            if (model.hasDefined(UUID_KEY_GENERATOR) || model.hasDefined(HILO_KEY_GENERATOR)) {
 174.                writer.writeStartElement(Element.KEY_GENERATORS.getLocalName());
 175.    
 176.                if (model.hasDefined(UUID_KEY_GENERATOR)) {
 177.                    for (Property keyGen : model.get(UUID_KEY_GENERATOR).asPropertyList()) {
 178.                        final String name = keyGen.getName();
 179.                        final ModelNode keyGenModel = keyGen.getValue();
 180.                        writeUuid(writer, name, keyGenModel);
 181.                    }
 182.                }
 183.                if (model.hasDefined(HILO_KEY_GENERATOR)) {
 184.                    for (Property keyGen : model.get(HILO_KEY_GENERATOR).asPropertyList()) {
 185.                        final String name = keyGen.getName();
 186.                        final ModelNode keyGenModel = keyGen.getValue();
 187.                        writeHilo(writer, name, keyGenModel);
 188.                    }
 189.                }
 190.                writer.writeEndElement();
 191.            }
 192.            writer.writeEndElement();
 193.        }
 194.    
 195.        private void writeHilo(final XMLExtendedStreamWriter writer, final String name, final ModelNode model) throws XMLStreamException {
 196.            writer.writeStartElement(Element.HILO.getLocalName());
 197.            writer.writeAttribute(Attribute.NAME.getLocalName(), name);
 198.            HiLoKeyGeneratorResourceDefinition.JNDI_NAME.marshallAsAttribute(model, writer);
 199.    
 200.            for(SimpleAttributeDefinition attribute : HiLoKeyGeneratorResourceDefinition.ATTRIBUTES) {
 201.                if (!attribute.equals(HiLoKeyGeneratorResourceDefinition.JNDI_NAME))
 202.                    attribute.marshallAsElement(model, writer);
 203.            }
 204.            writer.writeEndElement();
 205.        }

 207.        private void writeUuid(final XMLExtendedStreamWriter writer, final String name, final ModelNode model) throws XMLStreamException {
 208.            writer.writeStartElement(Element.UUID.getLocalName());
 209.            writer.writeAttribute(Attribute.NAME.getLocalName(), name);
 210.            UUIDKeyGeneratorResourceDefinition.JNDI_NAME.marshallAsAttribute(model, writer);
 211.            writer.writeEndElement();
 212.        }
