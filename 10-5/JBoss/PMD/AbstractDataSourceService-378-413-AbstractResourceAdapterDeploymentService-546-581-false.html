 126.        public void unregisterAll(String deploymentName) {
 127.            if (value != null) {
 128.                DEPLOYMENT_CONNECTOR_LOGGER.debugf("Unregistering: %s", deploymentName);
 129.    
 130.                if (registry != null && registry.getValue() != null) {
 131.                    registry.getValue().unregisterResourceAdapterDeployment(value);
 132.                }
 133.    
 134.                if (managementRepository != null && managementRepository.getValue() != null &&
 135.                    value.getDeployment() != null && value.getDeployment().getConnector() != null) {
 136.                    managementRepository.getValue().getConnectors().remove(value.getDeployment().getConnector());
 137.                }
 138.    
 139.                if (mdr != null && mdr.getValue() != null && value.getDeployment() != null
 140.                        && value.getDeployment().getCfs() != null && value.getDeployment().getCfJndiNames() != null) {
 141.                    for (int i = 0; i < value.getDeployment().getCfs().length; i++) {
 142.                        try {
 143.                            String cf = value.getDeployment().getCfs()[i].getClass().getName();
 144.                            String jndi = value.getDeployment().getCfJndiNames()[i];
 145.    
 146.                            mdr.getValue().unregisterJndiMapping(value.getDeployment().getURL().toExternalForm(), cf, jndi);
 147.    
 148.                        } catch (Throwable nfe) {
 149.                            DEPLOYMENT_CONNECTOR_LOGGER.debug("Exception during JNDI unbinding", nfe);
 150.                        }
 151.                    }
 152.                }
 153.    
 154.                if (mdr != null && mdr.getValue() != null && value.getDeployment().getAos() != null
 155.                        && value.getDeployment().getAoJndiNames() != null) {
 156.                    for (int i = 0; i < value.getDeployment().getAos().length; i++) {
 157.                        try {
 158.                            String ao = value.getDeployment().getAos()[i].getClass().getName();
 159.                            String jndi = value.getDeployment().getAoJndiNames()[i];
 160.    
 161.                            mdr.getValue().unregisterJndiMapping(value.getDeployment().getURL().toExternalForm(), ao, jndi);
 162.                        } catch (Throwable nfe) {
 163.                            DEPLOYMENT_CONNECTOR_LOGGER.debug("Exception during JNDI unbinding", nfe);
 164.                        }
 165.                    }
 166.                }
 167.    
 168.                if (value!= null && value.getDeployment() != null && value.getDeployment().getRecovery() != null && txInt !=null && txInt.getValue() != null) {
 169.                    XAResourceRecoveryRegistry rr = txInt.getValue().getRecoveryRegistry();
 170.    
 171.                    if (rr != null) {
 172.                        for (XAResourceRecovery recovery : value.getDeployment().getRecovery()) {
 173.                            if (recovery!= null) {
 174.                                try {
 175.                                    recovery.shutdown();
 176.                                } catch (Exception e) {
 177.                                    DEPLOYMENT_CONNECTOR_LOGGER.error("Error during recovery shutdown", e);
 178.                                } finally {
 179.                                    rr.removeXAResourceRecovery(recovery);
 180.                                }
 181.                            }
 182.                        }
 183.                    }
 184.                }
 185.    
 186.                if (value.getDeployment() != null && value.getDeployment().getConnectionManagers() != null) {
 187.                    for (ConnectionManager cm : value.getDeployment().getConnectionManagers()) {
 188.                        cm.shutdown();
 189.                    }
 190.                }
 191.    
 192.                if (value.getDeployment() != null && value.getDeployment().getResourceAdapter() != null) {
 193.                    ClassLoader old = WildFlySecurityManager.getCurrentContextClassLoaderPrivileged();
 194.                    try {
 195.                        WildFlySecurityManager.setCurrentContextClassLoaderPrivileged(value.getDeployment().getResourceAdapter().getClass().getClassLoader());
 196.                        value.getDeployment().getResourceAdapter().stop();
 197.                    } catch (Throwable nfe) {
 198.                        DEPLOYMENT_CONNECTOR_LOGGER.warn("Exception during stopping resource adapter", nfe);
 199.                    } finally {
 200.                        WildFlySecurityManager.setCurrentContextClassLoaderPrivileged(old);
 201.                    }
 202.                }
 203.            }
 204.            if (raRepositoryRegistrationId != null  && raRepository != null && raRepository.getValue() != null) {
 205.                try {
 206.                    raRepository.getValue().unregisterResourceAdapter(raRepositoryRegistrationId);
 207.                } catch (Throwable e) {
 208.                    DEPLOYMENT_CONNECTOR_LOGGER.debug("Failed to unregister RA from RA Repository", e);
 209.                }
 210.            }
 211.            if (connectorServicesRegistrationName != null) {
 212.                try {
 213.                    ConnectorServices.unregisterResourceAdapterIdentifier(connectorServicesRegistrationName);
 214.                } catch (Throwable e) {
 215.                    DEPLOYMENT_CONNECTOR_LOGGER.debug("Failed to unregister RA from ConnectorServices", e);
 216.                }
 217.            }
 218.            if (mdrRegistrationName != null && mdr != null && mdr.getValue() != null) {
 219.                try {
 220.                    mdr.getValue().unregisterResourceAdapter(mdrRegistrationName);
 221.                } catch (Throwable e) {
 222.                    DEPLOYMENT_CONNECTOR_LOGGER.debug("Failed to unregister RA from MDR", e);
 223.                }
 224.            }
 225.    
 226.        }

 297.        protected final void cleanupStartAsync(final StartContext context, final String deploymentName,
 298.                    final ServiceName serviceName, final Throwable cause) {
 299.            ExecutorService executorService = getLifecycleExecutorService();
 300.            Runnable r = new Runnable() {
 301.                @Override
 302.                public void run() {
 303.                    try {
 304.                        // TODO -- one of the 3 previous synchronous calls to this method don't had the TCCL set,
 305.                        // but the other two don't. I (BES 2013/10/21) intepret from that that setting the TCCL
 306.                        // was not necessary and in caller that had it set it was an artifact of
 307.                        WritableServiceBasedNamingStore.pushOwner(serviceName);
 308.                        unregisterAll(deploymentName);
 309.                    } finally {
 310.                        WritableServiceBasedNamingStore.popOwner();
 311.                        context.failed(ConnectorLogger.ROOT_LOGGER.failedToStartRaDeployment(cause, deploymentName));
 312.                    }
 313.                }
 314.            };
 315.            try {
 316.                executorService.execute(r);
 317.            } catch (RejectedExecutionException e) {
 318.                r.run();
 319.            } finally {
 320.                context.asynchronous();
 321.            }
 322.        }

 324.        protected void stopAsync(final StopContext context, final String deploymentName, final ServiceName serviceName) {
 325.            ExecutorService executorService = getLifecycleExecutorService();
 326.            Runnable r = new Runnable() {
 327.                @Override
 328.                public void run() {
 329.                    try {
 330.                        DEPLOYMENT_CONNECTOR_LOGGER.debugf("Stopping service %s", serviceName);
 331.                        WritableServiceBasedNamingStore.pushOwner(serviceName);
 332.                        unregisterAll(deploymentName);
 333.                    } finally {

 336.                    }
 337.                }
 338.            };
 339.            try {
 340.                executorService.execute(r);
 341.            } catch (RejectedExecutionException e) {
 342.                r.run();
 343.            } finally {
 344.                context.asynchronous();
 345.            }
