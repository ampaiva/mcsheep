 150.                            UndertowLogger.ROOT_LOGGER.couldNotLoadWebSocketConfig(clazz.name().toString(), e);
 151.                        }
 152.                    }
 153.                }
 154.    
 155.                WebSocketDeploymentInfo webSocketDeploymentInfo = new WebSocketDeploymentInfo();
 156.    
 157.                doDeployment(webSocketDeploymentInfo, annotatedEndpoints, config, endpoints);
 158.    
 159.                installWebsockets(phaseContext, webSocketDeploymentInfo);
 160.    
 161.            } finally {
 162.                Thread.currentThread().setContextClassLoader(oldCl);
 163.            }
 164.    
 165.        }
 166.    
 167.        private void installWebsockets(final DeploymentPhaseContext phaseContext, final WebSocketDeploymentInfo webSocketDeploymentInfo) {
 168.            DeploymentUnit deploymentUnit = phaseContext.getDeploymentUnit();
 169.            deploymentUnit.putAttachment(UndertowAttachments.WEB_SOCKET_DEPLOYMENT_INFO, webSocketDeploymentInfo);
 170.        }
 171.    
 172.        private void doDeployment(final WebSocketDeploymentInfo container, final Set> annotatedEndpoints, final Set> serverApplicationConfigClasses, final Set> endpoints) throws DeploymentUnitProcessingException {
 173.    
 174.            Set> allScannedEndpointImplementations = new HashSet<>(endpoints);
 175.            Set> allScannedAnnotatedEndpoints = new HashSet<>(annotatedEndpoints);
 176.            Set> newAnnotatatedEndpoints = new HashSet<>();
 177.            Set serverEndpointConfigurations = new HashSet<>();
 178.    
 179.            final Set configInstances = new HashSet<>();
 180.            for (Class clazz : serverApplicationConfigClasses) {
 181.                try {
 182.                    configInstances.add(clazz.newInstance());
 183.                } catch (InstantiationException | IllegalAccessException e) {
 184.                    JsrWebSocketLogger.ROOT_LOGGER.couldNotInitializeConfiguration(clazz, e);
 185.                }
 186.            }
 187.    
 188.    
 189.            if (!configInstances.isEmpty()) {
 190.                for (ServerApplicationConfig config : configInstances) {
 191.                    Set> returnedEndpoints = config.getAnnotatedEndpointClasses(allScannedAnnotatedEndpoints);
 192.                    if (returnedEndpoints != null) {
 193.                        newAnnotatatedEndpoints.addAll(returnedEndpoints);
 194.                    }
 195.                    Set endpointConfigs = config.getEndpointConfigs(allScannedEndpointImplementations);
 196.                    if (endpointConfigs != null) {
 197.                        serverEndpointConfigurations.addAll(endpointConfigs);
 198.                    }
 199.                }
 200.            } else {
 201.                newAnnotatatedEndpoints.addAll(allScannedAnnotatedEndpoints);
 202.            }
 203.    
 204.            //annotated endpoints first
 205.            for (Class endpoint : newAnnotatatedEndpoints) {
 206.                container.addEndpoint(endpoint);
 207.            }
 208.    
 209.            for (final ServerEndpointConfig endpoint : serverEndpointConfigurations) {
 210.                container.addEndpoint(endpoint);
 211.            }
 212.        }

 216.
