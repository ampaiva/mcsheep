 249.                }
 250.

 254.                for (PersistenceUnitMetadataHolder holder : puList) {

 263.                            final PersistenceProviderDeploymentHolder persistenceProviderDeploymentHolder = getPersistenceProviderDeploymentHolder(deploymentUnit);

 266.                            final boolean twoPhaseBootStrapCapable = (adaptor instanceof TwoPhaseBootstrapCapable) && Configuration.allowTwoPhaseBootstrap(pu);
 267.    
 268.                            if (startEarly) {
 269.                                if (twoPhaseBootStrapCapable) {
 270.                                    deployPersistenceUnitPhaseOne(phaseContext, deploymentUnit, eeModuleDescription, components, serviceTarget, classLoader, pu, adaptor);
 271.                                }
 272.                                else if (false == Configuration.needClassFileTransformer(pu)) {
 273.                                    // will start later when startEarly == false
 274.                                    JPA_LOGGER.tracef("persistence unit %s in deployment %s is configured to not need class transformer to be set, no class rewriting will be allowed",
 275.                                        pu.getPersistenceUnitName(), deploymentUnit.getName());
 276.                                }
 277.                                else {
 278.                                    // we need class file transformer to work, don't allow cdi bean manager to be access since that
 279.                                    // could cause application classes to be loaded (workaround by setting jboss.as.jpa.classtransformer to false).  WFLY-1463
 280.                                    final boolean allowCdiBeanManagerAccess = false;
 281.                                    deployPersistenceUnit(phaseContext, deploymentUnit, eeModuleDescription, components, serviceTarget, classLoader, pu, startEarly, provider, adaptor, allowCdiBeanManagerAccess);
 282.                                }
 283.                            }
 284.                            else { // !startEarly
 285.                                if (twoPhaseBootStrapCapable) {
 286.                                    deployPersistenceUnitPhaseTwo(phaseContext, deploymentUnit, eeModuleDescription, components, serviceTarget, classLoader, pu, provider, adaptor);
 287.                                } else if (false == Configuration.needClassFileTransformer(pu)) {
 288.                                    final boolean allowCdiBeanManagerAccess = true;
 289.                                    // PUs that have Configuration.JPA_CONTAINER_CLASS_TRANSFORMER = false will start during INSTALL phase
 290.                                    deployPersistenceUnit(phaseContext, deploymentUnit, eeModuleDescription, components, serviceTarget, classLoader, pu, startEarly, provider, adaptor, allowCdiBeanManagerAccess);
 291.                                }
 292.                            }
 293.    
 294.                        }
 295.                        else {
 296.                            JPA_LOGGER.tracef("persistence unit %s in deployment %s is not container managed (%s is set to false)",
 297.                                    pu.getPersistenceUnitName(), deploymentUnit.getName(), Configuration.JPA_CONTAINER_MANAGED);
 298.                        }
 299.                    }
 300.                }
 301.            }

 320.        private static void deployPersistenceUnit(
 321.                final DeploymentPhaseContext phaseContext,
 322.                final DeploymentUnit deploymentUnit,
 323.                final EEModuleDescription eeModuleDescription,
 324.                final Collection components,

 333.            try {
 334.                ValidatorFactory validatorFactory = null;
 335.                final HashMap properties = new HashMap();

 337.                    // Get the CDI-enabled ValidatorFactory
 338.                    validatorFactory = deploymentUnit.getAttachment(BeanValidationAttachments.VALIDATOR_FACTORY);
 339.                }
 340.    
 341.                final PersistenceUnitServiceImpl service = new PersistenceUnitServiceImpl(classLoader, pu, adaptor, provider, PersistenceUnitRegistryImpl.INSTANCE, deploymentUnit.getServiceName(), validatorFactory);

 344.

 346.                adaptor.addProviderProperties(properties, pu);

 349.                deploymentUnit.putAttachment(JpaAttachments.PERSISTENCE_UNIT_SERVICE_KEY, puServiceName);
 350.    
 351.                deploymentUnit.addToAttachmentList(Attachments.DEPLOYMENT_COMPLETE_SERVICES, puServiceName);
 352.

 361.                    if (jtaDataSource.startsWith("java:")) {

 364.                        }

 372.                    }
 373.                }

 375.                    if (nonJtaDataSource.startsWith("java:")) {
 376.                        builder.addDependency(ContextNames.bindInfoForEnvEntry(eeModuleDescription.getApplicationName(), eeModuleDescription.getModuleName(), eeModuleDescription.getModuleName(), false, nonJtaDataSource).getBinderServiceName(), ManagedReferenceFactory.class, new ManagedReferenceFactoryInjector(service.getNonJtaDataSourceInjector()));

 379.                        builder.addDependency(AbstractDataSourceService.SERVICE_NAME_BASE.append(nonJtaDataSource), new CastingInjector<>(service.getNonJtaDataSourceInjector(), DataSource.class));
 380.                        useDefaultDataSource = false;
 381.                    }
 382.                }
 383.                // JPA 2.0 8.2.1.5, container provides default JTA datasource
 384.                if (useDefaultDataSource) {
 385.                    // try the default datasource defined in the ee subsystem
 386.                    String defaultJtaDataSource = null;
 387.                    if (eeModuleDescription != null) {
 388.                        defaultJtaDataSource = eeModuleDescription.getDefaultResourceJndiNames().getDataSource();
 389.                    }
 390.    
 391.                    if (defaultJtaDataSource == null ||
 392.                            defaultJtaDataSource.isEmpty()) {
 393.                        // try the datasource defined in the jpa subsystem
 394.                        defaultJtaDataSource = adjustJndi(JPAService.getDefaultDataSourceName());
 395.                    }
 396.                    if (defaultJtaDataSource != null &&
 397.                        !defaultJtaDataSource.isEmpty()) {
 398.                        builder.addDependency(AbstractDataSourceService.SERVICE_NAME_BASE.append(defaultJtaDataSource), new CastingInjector<>(service.getJtaDataSourceInjector(), DataSource.class));

 400.                    }
 401.                }
 402.    
 403.                // JPA 2.1 sections 3.5.1 + 9.1 require the CDI bean manager to be passed to the peristence provider
 404.                // if the persistence unit is contained in a deployment that is a CDI bean archive (has beans.xml).
 405.                if (allowCdiBeanManagerAccess && WeldDeploymentMarker.isPartOfWeldDeployment(deploymentUnit)) {
 406.                    builder.addDependency(beanManagerServiceName(deploymentUnit),  new CastingInjector(service.getBeanManagerInjector(), BeanManager.class));
 407.                }
 408.    
 409.                try {
 410.                    // save a thread local reference to the builder for setting up the second level cache dependencies
 411.                    CacheDeploymentListener.setInternalDeploymentServiceBuilder(builder);
 412.                    adaptor.addProviderDependencies(pu);
 413.                }
 414.                finally {
 415.                    CacheDeploymentListener.clearInternalDeploymentServiceBuilder();
 416.                }
 417.    
 418.                /**
 419.                 * handle extension that binds a transaction scoped entity manager to specified JNDI location
 420.                 */
 421.                entityManagerBind(eeModuleDescription, serviceTarget, pu, puServiceName);
 422.    
 423.                /**
 424.                 * handle extension that binds an entity manager factory to specified JNDI location
 425.                 */
 426.                entityManagerFactoryBind(eeModuleDescription, serviceTarget, pu, puServiceName);
 427.    
 428.                builder.setInitialMode(ServiceController.Mode.ACTIVE)
 429.                    .addInjection(service.getPropertiesInjector(), properties);
 430.    
 431.                // get async executor from Services.addServerExecutorDependency
 432.                addServerExecutorDependency(builder, service.getExecutorInjector(), false);
 433.    
 434.                builder.install();
 435.    
 436.                JPA_LOGGER.tracef("added PersistenceUnitService for '%s'.  PU is ready for injector action.", puServiceName);
 437.                addManagementConsole(deploymentUnit, pu, adaptor);
 438.    
 439.            } catch (ServiceRegistryException e) {
 440.                throw JpaLogger.ROOT_LOGGER.failedToAddPersistenceUnit(e, pu.getPersistenceUnitName());
 441.            }
 442.        }

 616.    
 617.                // add dependency on first phase
 618.                builder.addDependency(puServiceName.append(FIRST_PHASE), new CastingInjector<>(service.getPhaseOnePersistenceUnitServiceImplInjector(), PhaseOnePersistenceUnitServiceImpl.class));
 619.    
 620.                boolean useDefaultDataSource = Configuration.allowDefaultDataSourceUse(pu);
 621.                final String jtaDataSource = adjustJndi(pu.getJtaDataSourceName());
 622.                final String nonJtaDataSource = adjustJndi(pu.getNonJtaDataSourceName());
 623.    
 624.                if (jtaDataSource != null && jtaDataSource.length() > 0) {
 625.                    if (jtaDataSource.startsWith("java:")) {
 626.                        if (jtaDataSource.equals(EE_DEFAULT_DATASOURCE)) { // explicit use of default datasource
 627.                            useDefaultDataSource = true;
 628.                        }
 629.                        else {
 630.                            builder.addDependency(ContextNames.bindInfoForEnvEntry(eeModuleDescription.getApplicationName(), eeModuleDescription.getModuleName(), eeModuleDescription.getModuleName(), false, jtaDataSource).getBinderServiceName(), ManagedReferenceFactory.class, new ManagedReferenceFactoryInjector(service.getJtaDataSourceInjector()));
 631.                            useDefaultDataSource = false;
 632.                        }
 633.                    } else {
 634.                        builder.addDependency(AbstractDataSourceService.SERVICE_NAME_BASE.append(jtaDataSource), new CastingInjector<>(service.getJtaDataSourceInjector(), DataSource.class));
 635.                        useDefaultDataSource = false;
 636.                    }
 637.                }
 638.                if (nonJtaDataSource != null && nonJtaDataSource.length() > 0) {
 639.                    if (nonJtaDataSource.startsWith("java:")) {
 640.                        builder.addDependency(ContextNames.bindInfoForEnvEntry(eeModuleDescription.getApplicationName(), eeModuleDescription.getModuleName(), eeModuleDescription.getModuleName(), false, nonJtaDataSource).getBinderServiceName(), ManagedReferenceFactory.class, new ManagedReferenceFactoryInjector(service.getNonJtaDataSourceInjector()));
 641.                        useDefaultDataSource = false;
 642.                    } else {
 643.                        builder.addDependency(AbstractDataSourceService.SERVICE_NAME_BASE.append(nonJtaDataSource), new CastingInjector<>(service.getNonJtaDataSourceInjector(), DataSource.class));
 644.                        useDefaultDataSource = false;
 645.                    }
 646.                }
 647.                // JPA 2.0 8.2.1.5, container provides default JTA datasource
 648.                if (useDefaultDataSource) {
 649.                    // try the default datasource defined in the ee subsystem
 650.                    String defaultJtaDataSource = null;
 651.                    if (eeModuleDescription != null) {
 652.                        defaultJtaDataSource = eeModuleDescription.getDefaultResourceJndiNames().getDataSource();
 653.                    }
 654.    
 655.                    if (defaultJtaDataSource == null ||
 656.                            defaultJtaDataSource.isEmpty()) {
 657.                        // try the datasource defined in the jpa subsystem
 658.                        defaultJtaDataSource = adjustJndi(JPAService.getDefaultDataSourceName());
 659.                    }
 660.                    if (defaultJtaDataSource != null &&
 661.                        !defaultJtaDataSource.isEmpty()) {
 662.                        builder.addDependency(AbstractDataSourceService.SERVICE_NAME_BASE.append(defaultJtaDataSource), new CastingInjector<>(service.getJtaDataSourceInjector(), DataSource.class));
 663.                        JPA_LOGGER.tracef("%s is using the default data source '%s'", puServiceName, defaultJtaDataSource);
 664.                    }
 665.                }
 666.    
 667.                // JPA 2.1 sections 3.5.1 + 9.1 require the CDI bean manager to be passed to the peristence provider
 668.                // if the persistence unit is contained in a deployment that is a CDI bean archive (has beans.xml).
 669.                if (WeldDeploymentMarker.isPartOfWeldDeployment(deploymentUnit)) {
 670.                    builder.addDependency(beanManagerServiceName(deploymentUnit),  new CastingInjector(service.getBeanManagerInjector(), BeanManager.class));
 671.                }
 672.    
 673.                try {
 674.                    // save a thread local reference to the builder for setting up the second level cache dependencies
 675.                    CacheDeploymentListener.setInternalDeploymentServiceBuilder(builder);
 676.                    adaptor.addProviderDependencies(pu);
 677.                }
 678.                finally {
 679.                    CacheDeploymentListener.clearInternalDeploymentServiceBuilder();
 680.                }
 681.    
 682.    
 683.                /**
 684.                 * handle extension that binds a transaction scoped entity manager to specified JNDI location
 685.                 */
 686.                entityManagerBind(eeModuleDescription, serviceTarget, pu, puServiceName);
 687.    
 688.                /**
 689.                 * handle extension that binds an entity manager factory to specified JNDI location
 690.                 */
 691.                entityManagerFactoryBind(eeModuleDescription, serviceTarget, pu, puServiceName);
 692.    
 693.                builder.setInitialMode(ServiceController.Mode.ACTIVE)
 694.                    .addInjection(service.getPropertiesInjector(), properties);
 695.    
 696.                // get async executor from Services.addServerExecutorDependency
 697.                addServerExecutorDependency(builder, service.getExecutorInjector(), false);
 698.    
 699.                builder.install();
 700.    
 701.                JPA_LOGGER.tracef("added PersistenceUnitService (phase 2 of 2) for '%s'.  PU is ready for injector action.", puServiceName);
 702.                addManagementConsole(deploymentUnit, pu, adaptor);
 703.    
 704.            } catch (ServiceRegistryException e) {
 705.                throw JpaLogger.ROOT_LOGGER.failedToAddPersistenceUnit(e, pu.getPersistenceUnitName());
 706.            }
 707.        }
