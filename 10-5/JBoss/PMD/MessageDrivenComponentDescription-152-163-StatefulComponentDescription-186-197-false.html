  94.            private final MethodIdentifier methodIdentifier;
  95.            private final boolean retainIfException;

  97.            StatefulRemoveMethod(final MethodIdentifier method, final boolean retainIfException) {
  98.                if (method == null) {
  99.                    throw EjbLogger.ROOT_LOGGER.removeMethodIsNull();
 100.                }
 101.                this.methodIdentifier = method;
 102.                this.retainIfException = retainIfException;
 103.            }

 113.            @Override
 114.            public boolean equals(Object o) {
 115.                if (this == o) return true;
 116.                if (o == null || getClass() != o.getClass()) return false;
 117.    
 118.                StatefulRemoveMethod that = (StatefulRemoveMethod) o;
 119.    
 120.                if (!methodIdentifier.equals(that.methodIdentifier)) return false;
 121.    
 122.                return true;
 123.            }
 124.    
 125.            @Override
 126.            public int hashCode() {

 306.        private void addStatefulInstanceAssociatingInterceptor(final EJBViewDescription view) {
 307.            view.getConfigurators().add(new ViewConfigurator() {
 308.                @Override
 309.                public void configure(final DeploymentPhaseContext context, final ComponentConfiguration componentConfiguration, ViewDescription description, ViewConfiguration viewConfiguration) throws DeploymentUnitProcessingException {
 310.                    EJBViewDescription ejbViewDescription = (EJBViewDescription) view;
 311.                    //if this is a home interface we add a different interceptor
 312.                    if (ejbViewDescription.getMethodIntf() == MethodIntf.HOME || ejbViewDescription.getMethodIntf() == MethodIntf.LOCAL_HOME) {
 313.                        for (Method method : viewConfiguration.getProxyFactory().getCachedMethods()) {
 314.                            if ((method.getName().equals("hashCode") && method.getParameterTypes().length == 0) ||
 315.                                    method.getName().equals("equals") && method.getParameterTypes().length == 1 &&
 316.                                            method.getParameterTypes()[0] == Object.class) {
 317.                                viewConfiguration.addClientInterceptor(method, ComponentTypeIdentityInterceptorFactory.INSTANCE, InterceptorOrder.Client.EJB_EQUALS_HASHCODE);
 318.                            }
 319.                        }
 320.                    } else {
 321.                        // interceptor factory return an interceptor which sets up the session id on component view instance creation
 322.                        final InterceptorFactory sessionIdGeneratingInterceptorFactory = StatefulComponentSessionIdGeneratingInterceptor.FACTORY;
 323.    
 324.                        // add the session id generating interceptor to the start of the *post-construct interceptor chain of the ComponentViewInstance*
 325.                        viewConfiguration.addClientPostConstructInterceptor(sessionIdGeneratingInterceptorFactory, InterceptorOrder.ClientPostConstruct.INSTANCE_CREATE);
 326.    
 327.                        for (Method method : viewConfiguration.getProxyFactory().getCachedMethods()) {
 328.                            if ((method.getName().equals("hashCode") && method.getParameterTypes().length == 0) ||
 329.                                    method.getName().equals("equals") && method.getParameterTypes().length == 1 &&
 330.                                            method.getParameterTypes()[0] == Object.class) {
 331.                                viewConfiguration.addClientInterceptor(method, StatefulIdentityInterceptor.FACTORY, InterceptorOrder.Client.EJB_EQUALS_HASHCODE);
 332.                            }
 333.                        }
 334.                    }
 335.                }
 336.            });
 337.            if (view.getMethodIntf() != MethodIntf.LOCAL_HOME && view.getMethodIntf() != MethodIntf.HOME) {
 338.                view.getConfigurators().add(new ViewConfigurator() {
 339.                    @Override
 340.                    public void configure(DeploymentPhaseContext context, ComponentConfiguration componentConfiguration, ViewDescription description, ViewConfiguration configuration) throws DeploymentUnitProcessingException {
 341.                        // add the instance associating interceptor to the *start of the invocation interceptor chain*
 342.                        configuration.addViewInterceptor(StatefulComponentInstanceInterceptor.FACTORY, InterceptorOrder.View.ASSOCIATING_INTERCEPTOR);
 343.                    }
 344.                });
 345.            }
 346.    
 347.        }

 349.        private void addRemoveMethodInterceptor(final ViewDescription view) {
 350.            view.getConfigurators().add(new ViewConfigurator() {
 351.                @Override
 352.                public void configure(DeploymentPhaseContext context, ComponentConfiguration componentConfiguration, ViewDescription description, ViewConfiguration configuration) throws DeploymentUnitProcessingException {
 353.                    final StatefulComponentDescription statefulComponentDescription = (StatefulComponentDescription) componentConfiguration.getComponentDescription();
 354.                    final Collection removeMethods = statefulComponentDescription.getRemoveMethods();
 355.                    if (removeMethods.isEmpty()) {
 356.                        return;
 357.                    }
 358.                    for (final Method viewMethod : configuration.getProxyFactory().getCachedMethods()) {
 359.                        final MethodIdentifier viewMethodIdentifier = MethodIdentifier.getIdentifierForMethod(viewMethod);
 360.                        for (final StatefulRemoveMethod removeMethod : removeMethods) {
 361.                            if (removeMethod.methodIdentifier.equals(viewMethodIdentifier)) {
 362.    
 363.                                //we do not want to add this if it is the Ejb(Local)Object.remove() method, as that is handed elsewhere
 364.                                final boolean object = EJBObject.class.isAssignableFrom(configuration.getViewClass()) || EJBLocalObject.class.isAssignableFrom(configuration.getViewClass());
 365.                                if (!object || !viewMethodIdentifier.getName().equals("remove") || viewMethodIdentifier.getParameterTypes().length != 0) {
 366.                                    configuration.addViewInterceptor(viewMethod, new ImmediateInterceptorFactory(new StatefulRemoveInterceptor(removeMethod.retainIfException)), InterceptorOrder.View.SESSION_REMOVE_INTERCEPTOR);
 367.                                }
 368.                                break;
 369.                            }
 370.                        }
 371.                    }
 372.                }
 373.            });
 374.        }
