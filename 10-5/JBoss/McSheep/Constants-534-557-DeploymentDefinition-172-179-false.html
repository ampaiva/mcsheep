  84.                return INSTANCE;

  87.            @Override
  88.            protected void executeRuntimeStep(OperationContext context, ModelNode operation) throws OperationFailedException {
  89.    
  90.                final PathAddress address = PathAddress.pathAddress(operation.get(ModelDescriptionConstants.OP_ADDR));
  91.    
  92.                final Resource web = context.readResourceFromRoot(address.subAddress(0, address.size()), false);
  93.                final ModelNode subModel = web.getModel();
  94.    
  95.                final String host = VIRTUAL_HOST.resolveModelAttribute(context, subModel).asString();
  96.                final String path = CONTEXT_ROOT.resolveModelAttribute(context, subModel).asString();
  97.                final String server = SERVER.resolveModelAttribute(context, subModel).asString();
  98.    
  99.                final ServiceController controller = context.getServiceRegistry(false).getService(UndertowService.deploymentServiceName(server, host, path));
 100.                final UndertowDeploymentService deploymentService = (UndertowDeploymentService) controller.getService();
 101.                Deployment deployment = deploymentService.getDeployment();
 102.                SessionManager sessionManager = deployment.getSessionManager();
 103.    
 104.                SessionStat stat = SessionStat.getStat(operation.require(ModelDescriptionConstants.NAME).asString());
 105.                SessionManagerStatistics sms = sessionManager instanceof SessionManagerStatistics ? (SessionManagerStatistics) sessionManager : null;
 106.    
 107.                if (stat == null) {
 108.                    context.getFailureDescription().set(UndertowLogger.ROOT_LOGGER.unknownMetric(operation.require(ModelDescriptionConstants.NAME).asString()));
 109.                } else {
 110.                    ModelNode result = new ModelNode();
 111.                    switch (stat) {
 112.                        case ACTIVE_SESSIONS:
 113.                            result.set(sessionManager.getActiveSessions().size());
 114.                            break;
 115.                        case EXPIRED_SESSIONS:
 116.                            if(sms == null) {
 117.                                result.set(0);
 118.                            } else {
 119.                                result.set((int)sms.getExpiredSessionCount());
 120.                            }
 121.                            break;
 122.                        case MAX_ACTIVE_SESSIONS:
 123.                            if(sms == null) {
 124.                                result.set(0);
 125.                            } else {
 126.                                result.set((int)sms.getMaxActiveSessions());
 127.                            }
 128.                            break;
 129.                        case SESSIONS_CREATED:
 130.                            if(sms == null) {
 131.                                result.set(0);
 132.                            } else {
 133.                                result.set((int)sms.getCreatedSessionCount());
 134.                            }
 135.                            break;
 136.                        //case DUPLICATED_SESSION_IDS:
 137.                        //    result.set(sm.getDuplicates());
 138.                        //    break;
 139.                        case SESSION_AVG_ALIVE_TIME:
 140.                            if(sms == null) {
 141.                                result.set(0);
 142.                            } else {
 143.                                result.set((int)sms.getAverageSessionAliveTime());
 144.                            }
 145.                            break;
 146.                        case SESSION_MAX_ALIVE_TIME:
 147.                            if(sms == null) {
 148.                                result.set(0);
 149.                            } else {
 150.                                result.set((int)sms.getMaxSessionAliveTime());
 151.                            }
 152.                            break;
 153.                        case REJECTED_SESSIONS:
 154.                            if(sms == null) {
 155.                                result.set(0);
 156.                            } else {
 157.                                result.set((int)sms.getRejectedSessions());
 158.                            }
 159.                            break;
 160.                        default:
 161.                            throw new IllegalStateException(UndertowLogger.ROOT_LOGGER.unknownMetric(stat));
 162.                    }
 163.                    context.getResult().set(result);
 164.                }
 165.    
 166.                context.stepCompleted();
 167.            }
 168.    
 169.        }
