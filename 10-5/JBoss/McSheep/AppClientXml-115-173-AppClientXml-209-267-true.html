 345.            // Handle elements
 346.            while (reader.nextTag() != END_ELEMENT) {
 347.                requireNamespace(reader, expectedNs);
 348.                final Element element = Element.forName(reader.getLocalName());
 349.                switch (element) {
 350.                    case SOCKET_BINDING: {
 351.                        // FIXME JBAS-8825
 352.                        final String bindingName = parseSocketBinding(reader, interfaces, groupAddress, updates);
 353.                        if (!uniqueBindingNames.add(bindingName)) {
 354.                            throw ControllerLogger.ROOT_LOGGER.alreadyDeclared(Element.SOCKET_BINDING.getLocalName(), Element.OUTBOUND_SOCKET_BINDING.getLocalName(), bindingName, Element.SOCKET_BINDING_GROUP.getLocalName(), socketBindingGroupName, reader.getLocation());
 355.                        }
 356.                        break;
 357.                    }
 358.                    case OUTBOUND_SOCKET_BINDING: {
 359.                        final String bindingName = parseOutboundSocketBinding(reader, interfaces, groupAddress, updates);
 360.                        if (!uniqueBindingNames.add(bindingName)) {
 361.                            throw ControllerLogger.ROOT_LOGGER.alreadyDeclared(Element.SOCKET_BINDING.getLocalName(), Element.OUTBOUND_SOCKET_BINDING.getLocalName(), bindingName, Element.SOCKET_BINDING_GROUP.getLocalName(), socketBindingGroupName, reader.getLocation());
 362.                        }
 363.                        break;
 364.                    }
 365.                    default:
 366.                        throw unexpectedElement(reader);
 367.                }
 368.            }
 369.        }

 371.        private void parseServerProfile(final XMLExtendedStreamReader reader, final ModelNode address, final List list)
 372.                throws XMLStreamException {
 373.            // Attributes
 374.            requireNoAttributes(reader);
 375.    
 376.            // Content
 377.            final Set configuredSubsystemTypes = new HashSet();
 378.            while (reader.hasNext() && reader.nextTag() != END_ELEMENT) {
 379.                if (Element.forName(reader.getLocalName()) != Element.SUBSYSTEM) {
 380.                    throw unexpectedElement(reader);
 381.                }
 382.                if (!configuredSubsystemTypes.add(reader.getNamespaceURI())) {
 383.                    throw AppClientLogger.ROOT_LOGGER.duplicateSubsystemDeclaration(reader.getLocation());
 384.                }
 385.                // parse subsystem
 386.                final List subsystems = new ArrayList();
 387.                reader.handleAny(subsystems);
 388.    
 389.                // Process subsystems
 390.                for (final ModelNode update : subsystems) {
 391.                    // Process relative subsystem path address
 392.                    final ModelNode subsystemAddress = address.clone();
 393.                    for (final Property path : update.get(OP_ADDR).asPropertyList()) {
 394.                        subsystemAddress.add(path.getName(), path.getValue().asString());
 395.                    }
 396.                    update.get(OP_ADDR).set(subsystemAddress);
 397.                    list.add(update);
 398.                }
 399.            }
 400.        }
