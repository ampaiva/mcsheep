 398.            Set<String> names = attributes.getAttributeNames();
 399.            List<HttpSessionActivationListener> listeners = new ArrayList<>(names.size());
 400.            for (String name: names) {
 401.                Object attribute = attributes.getAttribute(name);
 402.                if (attribute instanceof HttpSessionActivationListener) {
 403.                    listeners.add((HttpSessionActivationListener) attribute);
 404.                }
 405.            }
 406.            return listeners;
 407.        }
