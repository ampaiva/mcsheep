  51.    
  52.        static final RTSSubsystemAdd INSTANCE = new RTSSubsystemAdd();
  53.    
  54.        private RTSSubsystemAdd() {
  55.        }
  56.    
  57.        @Override
  58.        protected void populateModel(ModelNode operation, ModelNode model) throws OperationFailedExcepti...
  59.            if (RTSLogger.ROOT_LOGGER.isTraceEnabled()) {
  60.                RTSLogger.ROOT_LOGGER.trace("RTSSubsystemAdd.populateModel");
  61.            }
  62.    
  63.            RTSSubsystemDefinition.SERVER.validateAndSet(operation, model);
  64.            RTSSubsystemDefinition.HOST.validateAndSet(operation, model);
  65.            RTSSubsystemDefinition.SOCKET_BINDING.validateAndSet(operation, model);
  66.        }
  67.    
  68.        @Override
  69.        public void performBoottime(OperationContext context, ModelNode operation, ModelNode model)
  70.                throws OperationFailedException {
  71.    
  72.            if (RTSLogger.ROOT_LOGGER.isTraceEnabled()) {
  73.                RTSLogger.ROOT_LOGGER.trace("RTSSubsystemAdd.performBoottime");
  74.            }
  75.    
  76.            registerCoordinatorService(context, model);
  77.            registerParticipantService(context, model);
  78.            registerVolatileParticipantService(context, model);
  79.            registerInboundBridgeService(context);
  80.    
  81.            registerDeploymentProcessors(context);
  82.        }
  83.    
  84.        private void registerInboundBridgeService(final OperationContext context) {
  85.    
  86.            final InboundBridgeService inboundBridgeService = new InboundBridgeService();
  87.            final ServiceBuilder<InboundBridgeService> inboundBridgeServiceBuilder = context
  88.                    .getServiceTarget()
  89.                    .addService(RTSSubsystemExtension.INBOUND_BRIDGE, inboundBridgeService)
  90.                    .addDependency(TxnServices.JBOSS_TXN_ARJUNA_RECOVERY_MANAGER);
  91.    
  92.            inboundBridgeServiceBuilder
  93.                    .setInitialMode(ServiceController.Mode.ACTIVE)
  94.                    .install();
  95.        }
  96.    
  97.        private void registerCoordinatorService(final OperationContext context, final ModelNode model) {...
  98.    
  99.            final String socketBindingName = model.get(Attribute.SOCKET_BINDING.getLocalName()).asString...
 100.            final String serverName = model.get(Attribute.SERVER.getLocalName()).asString();
 101.            final String hostName = model.get(Attribute.HOST.getLocalName()).asString();
 102.            final CoordinatorService coordinatorService = new CoordinatorService();
 103.            final ServiceBuilder<CoordinatorService> coordinatorServiceBuilder = context
 104.                    .getServiceTarget()
 105.                    .addService(RTSSubsystemExtension.COORDINATOR, coordinatorService)
 106.                    .addDependency(SocketBinding.JBOSS_BINDING_NAME.append(socketBindingName), SocketBin...
 107.                            coordinatorService.getInjectedSocketBinding())
 108.                    .addDependency(UndertowService.virtualHostName(serverName, hostName), Host.class,
 109.                            coordinatorService.getInjectedHost());
 110.    
 111.            coordinatorServiceBuilder
 112.                    .setInitialMode(ServiceController.Mode.ACTIVE)
 113.                    .install();
 114.        }

 156.        private void registerDeploymentProcessors(final OperationContext context) {
 157.            context.addStep(new AbstractDeploymentChainStep() {
 158.                public void execute(final DeploymentProcessorTarget processorTarget) {
 159.                    processorTarget.addDeploymentProcessor(RTSSubsystemExtension.SUBSYSTEM_NAME, Phase.P...
 160.                            Phase.POST_MODULE_RTS_PROVIDERS, new InboundBridgeDeploymentProcessor());
 161.                }
 162.            }, OperationContext.Stage.RUNTIME);
 163.        }
