 256.        /**

 258.         * and create the timer table if necessary.

 260.        private void checkDatabase() {

 275.                preparedStatement.setString(3, "NON-EXISTENT");

 277.            } catch (SQLException e) {

 279.                if (connection != null) {
 280.                    try {

 282.                        String[] statements = createTable.split(";");
 283.                        for (final String sql : statements) {
 284.                            try {
 285.                                statement = connection.createStatement();
 286.                                statement.executeUpdate(sql);
 287.                            } finally {
 288.                                safeClose(statement);
 289.                            }
 290.                        }
 291.                    } catch (SQLException e1) {
 292.                        EjbLogger.ROOT_LOGGER.couldNotCreateTable(e1);
 293.                    }
 294.                } else {
 295.                    EjbLogger.ROOT_LOGGER.couldNotCreateTable(e);
 296.                }
 297.            } finally {
 298.                safeClose(resultSet);
 299.                safeClose(preparedStatement);
 300.                safeClose(statement);
 301.                safeClose(connection);
 302.            }
 303.        }

 358.                        knownTimerIds.get(timerEntity.getTimedObjectId()).add(timerEntity.getId());
 359.                    }

 363.                    statement.setTimestamp(2, timestamp(timerEntity.getPreviousRun()));

 375.            } finally {
