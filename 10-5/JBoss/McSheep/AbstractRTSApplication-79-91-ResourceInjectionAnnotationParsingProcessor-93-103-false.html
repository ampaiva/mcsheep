 178.        protected void processMethodResource(final DeploymentPhaseContext phaseContext, final MethodInfo methodInfo, final String name, final String type, final EEModuleClassDescription classDescription, final AnnotationInstance annotation, final EEModuleDescription eeModuleDescription, final Module module, final EEApplicationClasses applicationClasses, final PropertyReplacer replacer) throws DeploymentUnitProcessingException {
 179.            final String methodName = methodInfo.name();
 180.            if (!methodName.startsWith("set") || methodInfo.args().length != 1) {
 181.                throw EeLogger.ROOT_LOGGER.setterMethodOnly("@Resource", methodInfo);
 182.            }
 183.    
 184.            final String contextNameSuffix = methodName.substring(3, 4).toLowerCase(Locale.ENGLISH) + methodName.substring(4);

 186.    
 187.            final String injectionType = isEmpty(type) || type.equals(Object.class.getName()) ? methodInfo.args()[0].name().toString() : type;
 188.            final InjectionTarget targetDescription = new MethodInjectionTarget(methodInfo.declaringClass().name().toString(), methodName, methodInfo.args()[0].name().toString());
 189.            process(phaseContext, classDescription, annotation, injectionType, localContextName, targetDescription, eeModuleDescription, module, applicationClasses, replacer);
 190.        }
 191.    
 192.        protected void processClassResource(final DeploymentPhaseContext phaseContext, final String name, final String type, final EEModuleClassDescription classDescription, final AnnotationInstance annotation, final EEModuleDescription eeModuleDescription, final Module module, final EEApplicationClasses applicationClasses, final PropertyReplacer replacer) throws DeploymentUnitProcessingException {
 193.            if (isEmpty(name)) {
 194.                throw EeLogger.ROOT_LOGGER.annotationAttributeMissing("@Resource", "name");
 195.            }
 196.            final String realType;
 197.            if (isEmpty(type)) {
 198.                realType = Object.class.getName();
 199.            } else {
 200.                realType = type;
 201.            }
 202.            process(phaseContext, classDescription, annotation, realType, name, null, eeModuleDescription, module, applicationClasses, replacer);
 203.        }
 204.    
 205.        protected void process(final DeploymentPhaseContext phaseContext, final EEModuleClassDescription classDescription, final AnnotationInstance annotation, final String injectionType, final String localContextName, final InjectionTarget targetDescription, final EEModuleDescription eeModuleDescription, final Module module, final EEApplicationClasses applicationClasses, final PropertyReplacer replacer) throws DeploymentUnitProcessingException {
 206.            final EEResourceReferenceProcessorRegistry registry = phaseContext.getDeploymentUnit().getAttachment(Attachments.RESOURCE_REFERENCE_PROCESSOR_REGISTRY);
 207.            final AnnotationValue lookupAnnotation = annotation.value("lookup");
 208.            String lookup = (lookupAnnotation == null) ? null : replacer.replaceProperties(lookupAnnotation.asString());
 209.            // if "lookup" hasn't been specified then fallback on "mappedName" which we treat the same as "lookup"
 210.            if (isEmpty(lookup)) {
 211.                final AnnotationValue mappedNameAnnotationValue = annotation.value("mappedName");
 212.                lookup = (mappedNameAnnotationValue == null) ? null : replacer.replaceProperties(mappedNameAnnotationValue.asString());
 213.            }
 214.    
 215.            if (isEmpty(lookup) && FIXED_LOCATIONS.containsKey(injectionType)) {
 216.                lookup = FIXED_LOCATIONS.get(injectionType);
 217.            }
 218.            InjectionSource valueSource = null;
 219.            final boolean isEnvEntryType = this.isEnvEntryType(injectionType, module);
 220.            if (!isEmpty(lookup)) {
 221.                valueSource = new LookupInjectionSource(lookup, JAVAX_NAMING_CONTEXT.equals(injectionType));
 222.            } else if (isEnvEntryType) {
 223.                // if it's an env-entry type then we do *not* create a BindingConfiguration to bind to the ENC
 224.                // since the binding (value) for env-entry is always driven from a deployment descriptor.
 225.                // The deployment descriptor processing and subsequent binding in the ENC is taken care off by a
 226.                // different Deployment unit processor. If the value isn't specified in the deployment descriptor,
 227.                // then there will be no binding the ENC and that's what is expected by the Java EE 6 spec. Furthermore,
 228.                // if the @Resource is an env-entry binding then the injection target will be optional since in the absence of
 229.                // an env-entry-value, there won't be a binding and effectively no injection. This again is as expected by spec.
 230.            } else {
 231.                //otherwise we just try and handle it
 232.                //if we don't have a value source we will try and inject from a lookup
 233.                //and the user has to configure the value in a deployment descriptor
 234.                final EEResourceReferenceProcessor resourceReferenceProcessor = registry.getResourceReferenceProcessor(injectionType);
 235.                if (resourceReferenceProcessor != null) {
 236.                    valueSource = resourceReferenceProcessor.getResourceReferenceBindingSource();
 237.                }
 238.            }
 239.    
 240.            // EE.5.2.4
 241.            // Each injection of an object corresponds to a JNDI lookup. Whether a new
 242.            // instance of the requested object is injected, or whether a shared instance is
 243.            // injected, is determined by the rules described above.
 244.    
 245.            // Because of performance we allow any type of InjectionSource.
 246.    
 247.            if (valueSource == null) {
 248.                // the ResourceInjectionConfiguration is created by LazyResourceInjection
 249.                if (targetDescription != null) {
 250.                    final LookupInjectionSource optionalInjection = new LookupInjectionSource(localContextName, true);
 251.                    final ResourceInjectionConfiguration injectionConfiguration = new ResourceInjectionConfiguration(targetDescription, optionalInjection, true);
 252.                    classDescription.addResourceInjection(injectionConfiguration);
 253.                }
 254.            } else {
 255.                // our injection comes from the local lookup, no matter what.
 256.                final InjectionSource injectionSource = new LookupInjectionSource(localContextName);
 257.                final ResourceInjectionConfiguration injectionConfiguration = targetDescription != null ?
 258.                        new ResourceInjectionConfiguration(targetDescription, injectionSource) : null;
 259.    
 260.                final BindingConfiguration bindingConfiguration = new BindingConfiguration(localContextName, valueSource);
 261.                classDescription.getBindingConfigurations().add(bindingConfiguration);
 262.    
 263.                if (injectionConfiguration != null) {
 264.                    classDescription.addResourceInjection(injectionConfiguration);
 265.                }
 266.            }
 267.        }
 268.    
 269.        private boolean isEmpty(final String string) {
 270.            return string == null || string.isEmpty();
 271.        }
 272.    
 273.        private boolean isEnvEntryType(final String type, final Module module) {
 274.            if (SIMPLE_ENTRIES.contains(type)) {
 275.                return true;
 276.            }
 277.            try {
 278.                return module.getClassLoader().loadClass(type).isEnum();
 279.            } catch (ClassNotFoundException e) {
 280.                return false;
 281.            }
 282.        }
