  99.        private void parseMailSession(final XMLExtendedStreamReader reader, List<ModelNode> list, final ...
 100.            String jndiName = null;
 101.            final ModelNode operation = new ModelNode();
 102.            for (int i = 0; i < reader.getAttributeCount(); i++) {
 103.                Attribute attr = Attribute.forName(reader.getAttributeLocalName(i));
 104.                String value = reader.getAttributeValue(i);
 105.                if (attr == Attribute.JNDI_NAME) {
 106.                    jndiName = value;
 107.                    JNDI_NAME.parseAndSetParameter(value, operation, reader);
 108.                } else if (attr == Attribute.DEBUG) {
 109.                    DEBUG.parseAndSetParameter(value, operation, reader);
 110.                } else if (attr == Attribute.FROM) {
 111.                    FROM.parseAndSetParameter(value, operation, reader);
 112.                }
 113.            }
 114.            if (jndiName == null) {
 115.                throw ParseUtils.missingRequired(reader, EnumSet.of(Attribute.JNDI_NAME));
 116.            }
 117.            final PathAddress address = parent.append(MAIL_SESSION, jndiName);
 118.            operation.get(OP_ADDR).set(address.toModelNode());
 119.            operation.get(OP).set(ADD);
 120.            list.add(operation);
 121.            while (reader.hasNext() && reader.nextTag() != END_ELEMENT) {
 122.                switch (Namespace.forUri(reader.getNamespaceURI())) {
 123.                    case MAIL_1_0:
 124.                    case MAIL_1_1: {
 125.                        final Element element = Element.forName(reader.getLocalName());
 126.                        switch (element) {
 127.                            case SMTP_SERVER: {
 128.                                parseServerConfig(reader, SMTP, address, list);
 129.                                break;
 130.                            }
 131.                            case POP3_SERVER: {
 132.                                parseServerConfig(reader, POP3, address, list);
 133.                                break;
 134.                            }
 135.                            case IMAP_SERVER: {
 136.                                parseServerConfig(reader, IMAP, address, list);
 137.                                break;
 138.                            }
 139.                            case CUSTOM_SERVER: {
 140.                                parseCustomServerConfig(reader, address, list);
 141.                                break;
 142.                            }
 143.                            default: {
 144.                                reader.handleAny(list);
 145.                                break;
 146.                            }
 147.                        }
 148.                        break;
 149.                    }
 150.                    default: {
 151.                        throw unexpectedElement(reader);
 152.                    }
 153.                }
 154.            }
 155.        }
