  75.            // no attributes
  76.            requireNoAttributes(reader);
  77.    
  78.            final ModelNode subsystem = new ModelNode();
  79.            subsystem.get(OP).set(ADD);
  80.            subsystem.get(OP_ADDR).add(SUBSYSTEM, WSExtension.SUBSYSTEM_NAME);
  81.    
  82.            final List<ModelNode> endpointConfigs = new ArrayList<ModelNode>();
  83.    
  84.            // elements
  85.            final EnumSet<Element> encountered = EnumSet.noneOf(Element.class);
  86.            while (reader.hasNext() && reader.nextTag() != END_ELEMENT) {
  87.                switch (Namespace.forUri(reader.getNamespaceURI())) {
  88.                    case WEBSERVICES_1_0: {
  89.                        final Element element = Element.forName(reader.getLocalName());
  90.                        if (element != Element.ENDPOINT_CONFIG && !encountered.add(element)) {
  91.                            throw unexpectedElement(reader);
  92.                        }
  93.                        switch (element) {
  94.                            case MODIFY_WSDL_ADDRESS: {
  95.                                boolean b = Boolean.parseBoolean(parseElementNoAttributes(reader));
  96.                                subsystem.get(MODIFY_WSDL_ADDRESS).set(b);
  97.                                break;
  98.                            }
  99.                            case WSDL_HOST: {
 100.                                subsystem.get(WSDL_HOST).set(parseElementNoAttributes(reader));
 101.                                break;
 102.                            }
 103.                            case WSDL_PORT: {
 104.                                int port = Integer.valueOf(parseElementNoAttributes(reader));
 105.                                subsystem.get(WSDL_PORT).set(port);
 106.                                break;
 107.                            }
 108.                            case WSDL_SECURE_PORT: {
 109.                                int port = Integer.valueOf(parseElementNoAttributes(reader));
 110.                                subsystem.get(WSDL_SECURE_PORT).set(port);
 111.                                break;
 112.                            }
 113.                            case ENDPOINT_CONFIG: {
 114.                                readEndpointConfig(reader, subsystem.get(OP_ADDR), endpointConfigs);
 115.                                break;
 116.                            }
 117.                            default: {
 118.                                throw unexpectedElement(reader);
 119.                            }
 120.                        }
 121.                        break;
 122.                    }
 123.                }
 124.            }
 125.    
 126.            list.add(subsystem);
 127.            list.addAll(endpointConfigs);
 128.        }

 137.        private void readEndpointConfig(final XMLExtendedStreamReader reader, final ModelNode address, f...
 138.            String configName = null;
 139.    
 140.            final EnumSet<Element> encountered = EnumSet.noneOf(Element.class);
 141.            while (reader.nextTag() != END_ELEMENT) {
 142.                final Element element = Element.forName(reader.getLocalName());
 143.                if (element != Element.PROPERTY
 144.                        && !encountered.add(element)) {
 145.                    throw unexpectedElement(reader);
 146.                }
 147.                switch (element) {
 148.                    case CONFIG_NAME: {
 149.                        configName = parseElementNoAttributes(reader);
 150.                        final ModelNode node = new ModelNode();
 151.                        node.get(OP).set(ADD);
 152.                        node.get(OP_ADDR).set(address).add(ENDPOINT_CONFIG, configName);
 153.                        operationList.add(node);
 154.                        break;
 155.                    }
 156.                    case PRE_HANDLER_CHAINS: {
 157.                        parseHandlerChains(reader, configName, operationList, true);
 158.                        break;
 159.                    }
 160.                    case POST_HANDLER_CHAINS: {
 161.                        parseHandlerChains(reader, configName, operationList, false);
 162.                        break;
 163.                    }
 164.                    case PROPERTY : {
 165.                        final ModelNode operation = parseProperty(reader, configName);
 166.                        operationList.add(operation);
 167.                        break;
 168.                    }
 169.                    default: {
 170.                        throw unexpectedElement(reader);
 171.                    }
 172.                }
 173.            }

 232.        private void parseHandlerChain(final XMLExtendedStreamReader reader, final String configName, fi...
 233.            String handlerChainId = null;
 234.            final EnumSet<Element> encountered = EnumSet.noneOf(Element.class);
 235.            final int count = reader.getAttributeCount();
 236.            for (int i = 0; i < count; i++) {
 237.                requireNoNamespaceAttribute(reader, i);
 238.                final String value = reader.getAttributeValue(i);
 239.                final Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
 240.                switch (attribute) {
 241.                case ID:
 242.                    handlerChainId = value;
 243.                    break;
 244.                default:
 245.                    throw unexpectedAttribute(reader, i);
 246.                }
 247.            }
 248.            if (handlerChainId == null) {
 249.                handlerChainId = "auto-generated-" + System.currentTimeMillis();
 250.            }
 251.            String protocolBindings = null;
 252.            final List<ModelNode> addHandlerOperations = new LinkedList<ModelNode>();
 253.            while (reader.nextTag() != END_ELEMENT) {
 254.                final Element element = Element.forName(reader.getLocalName());
 255.                if (element != Element.HANDLER && !encountered.add(element)) {
 256.                    throw unexpectedElement(reader);
 257.                }
 258.                switch (element) {
 259.                    case PROTOCOL_BINDINGS: {
 260.                        protocolBindings = parseElementNoAttributes(reader);
 261.                        break;
 262.                    }
 263.                    case HANDLER: {
 264.                        parseHandler(reader, configName, handlerChainId, isPreHandlerChain, addHandlerOp...
 265.                        break;
 266.                    }
 267.                    default: {
 268.                        throw unexpectedElement(reader);
 269.                    }
 270.                }
 271.            }
 272.            final ModelNode operation = new ModelNode();
 273.            final String handlerChainType = isPreHandlerChain ? PRE_HANDLER_CHAIN : POST_HANDLER_CHAIN;
 274.            operation.get(OP).set(ADD);
 275.            operation.get(OP_ADDR).add(SUBSYSTEM, WSExtension.SUBSYSTEM_NAME).add(ENDPOINT_CONFIG, confi...
 276.            if (protocolBindings != null) {
 277.                operation.get(Constants.PROTOCOL_BINDINGS).set(protocolBindings);
 278.            }
 279.            operationList.add(operation);
 280.            operationList.addAll(addHandlerOperations);
 281.        }
