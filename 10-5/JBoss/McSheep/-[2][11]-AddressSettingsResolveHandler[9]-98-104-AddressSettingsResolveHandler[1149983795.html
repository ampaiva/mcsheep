 355.            final String name = operation.require(ModelDescriptionConstants.NAME).asString();

 359.                context.getResult().set(started);
 360.            } else if (VERSION.getName().equals(name)) {
 361.                String version = server.getVersion().getFullVersion();

 363.            } else if (ACTIVE.getName().equals(name)) {
 364.                boolean active = server.isActive();
 365.                context.getResult().set(active);
 366.            } else {
 367.                // Bug
 368.                throw MessagingLogger.ROOT_LOGGER.unsupportedAttribute(name);
 369.            }
