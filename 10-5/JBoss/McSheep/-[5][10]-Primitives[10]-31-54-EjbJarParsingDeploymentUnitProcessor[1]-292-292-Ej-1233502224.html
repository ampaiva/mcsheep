  47.

  50.            MONTH_ALIAS.put("feb", 2);
  51.            MONTH_ALIAS.put("mar", 3);
  52.            MONTH_ALIAS.put("apr", 4);

  57.            MONTH_ALIAS.put("sep", 9);

  59.            MONTH_ALIAS.put("nov", 11);

  61.    
  62.        }

  64.        private static final int OFFSET = MONTH_ALIAS.get("jan") - Calendar.JANUARY;
  65.    
  66.        private SortedSet<Integer> offsetAdjustedMonths = new TreeSet<Integer>();
  67.    
  68.        public Month(String value) {
  69.            super(value);
  70.            if (OFFSET != 0) {
  71.                for (Integer month : this.absoluteValues) {
  72.                    this.offsetAdjustedMonths.add(month - OFFSET);
  73.                }
  74.            } else {
  75.                this.offsetAdjustedMonths = this.absoluteValues;
  76.            }
  77.        }

  91.            // month doesn't support relative values, so always return false
  92.            return false;
  93.        }

  96.        protected boolean accepts(ScheduleExpressionType scheduleExprType) {
  97.            switch (scheduleExprType) {
  98.                case RANGE:
  99.                case LIST:
 100.                case SINGLE_VALUE:
 101.                case WILDCARD:
 102.                    return true;
 103.                // month doesn't support increment
 104.                case INCREMENT:
 105.                default:
 106.                    return false;
 107.            }
 108.        }

 110.        @Override
 111.        protected Integer parseInt(String alias) {
 112.            try {
 113.                return super.parseInt(alias);
 114.            } catch (NumberFormatException nfe) {
 115.                if (MONTH_ALIAS != null) {
 116.                    String lowerCaseAlias = alias.toLowerCase(Locale.ENGLISH);
 117.                    return MONTH_ALIAS.get(lowerCaseAlias);
 118.                }
 119.            }
 120.            return null;
 121.        }

 123.        public Integer getNextMatch(Calendar currentCal) {
 124.            if (this.scheduleExpressionType == ScheduleExpressionType.WILDCARD) {
 125.                return currentCal.get(Calendar.MONTH);
 126.            }
 127.            if (this.offsetAdjustedMonths.isEmpty()) {
 128.                return null;
 129.            }
 130.            int currentMonth = currentCal.get(Calendar.MONTH);
 131.            for (Integer month : this.offsetAdjustedMonths) {
 132.                if (currentMonth == month) {
 133.                    return currentMonth;
 134.                }
 135.                if (month > currentMonth) {
 136.                    return month;
 137.                }
 138.            }
 139.            return this.offsetAdjustedMonths.first();
 140.        }
 141.    
 142.        public Integer getFirstMatch() {
 143.            if (this.scheduleExpressionType == ScheduleExpressionType.WILDCARD) {
 144.                return Calendar.JANUARY;
 145.            }
 146.            if (this.offsetAdjustedMonths.isEmpty()) {
 147.                throw EjbLogger.ROOT_LOGGER.invalidExpressionSeconds(this.origValue);
 148.            }
 149.            return this.offsetAdjustedMonths.first();
 150.        }
