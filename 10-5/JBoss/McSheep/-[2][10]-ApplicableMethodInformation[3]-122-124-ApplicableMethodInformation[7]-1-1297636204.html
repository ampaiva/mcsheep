 359.        private abstract static class PopulatingMap<K, V> extends HashMap<K, V> {
 360.            V pick(K key) {
 361.                V value = get(key);
 362.                if (value == null) {
 363.                    value = populate();
 364.                    put(key, value);
 365.                }
 366.                return value;
 367.            }
 368.    
 369.            abstract V populate();
 370.        }

 185.        private Method resolveRealMethod(final Method method) {
 186.            if (method.isBridge() || method.isSynthetic()) {
 187.                Method[] declaredMethods =  WildFlySecurityManager.doUnchecked(new PrivilegedAction<Meth...
 188.                    @Override
 189.                    public Method[] run() {
 190.                        return method.getDeclaringClass().getDeclaredMethods();
 191.                    }
 192.                });
 193.                for (Method m : declaredMethods) {
 194.                    if (m.getName().equals(method.getName())
 195.                            && m.getParameterTypes().length == method.getParameterTypes().length
 196.                            && !m.isBridge()
 197.                            && !m.isSynthetic()) {
 198.                        if(!method.getReturnType().isAssignableFrom(m.getReturnType())) {
 199.                            continue;
 200.                        }
 201.                        for(int i = 0; i < method.getParameterTypes().length; ++i) {
 202.                            if(!method.getParameterTypes()[i].isAssignableFrom(m.getParameterTypes()[i])...
 203.                                continue;
 204.                            }
 205.                        }
 206.                        return m;
 207.                    }
 208.                }
 209.            }
 210.            return method;
 211.        }

 321.        /**
 322.         * Makes an array usable as a key.
 323.         *
 324.         * @author <a href="mailto:cdewolf@redhat.com">Carlo de Wolf</a>
 325.         */
 326.        private static class ArrayKey {
 327.            private final Object[] a;
 328.            private final int hashCode;
 329.            private transient String s;
 330.    
 331.            ArrayKey(Object... a) {
 332.                this.a = a;

 334.            }
 335.    
 336.            @Override
 337.            public boolean equals(Object obj) {
 338.                if (!(obj instanceof ArrayKey))
 339.                    return false;

 341.            }
 342.    
 343.            @Override
 344.            public int hashCode() {
 345.                return hashCode;
 346.            }
 347.    
 348.            @Override
 349.            public String toString() {
 350.                if (s == null)
 351.                    this.s = Arrays.toString(a);
 352.                return s;
 353.            }
 354.        }

 356.        /**
 357.         * @author <a href="mailto:cdewolf@redhat.com">Carlo de Wolf</a>
 358.         */
 359.        private abstract static class PopulatingMap<K, V> extends HashMap<K, V> {
 360.            V pick(K key) {
 361.                V value = get(key);
 362.                if (value == null) {
 363.                    value = populate();
 364.                    put(key, value);
 365.                }
 366.                return value;
 367.            }
 368.    
 369.            abstract V populate();
 370.        }
 371.    
 372.        private static <K, V> V get(Map<K, V> map, K key) {
 373.            if (map == null)
 374.                return null;
 375.            return map.get(key);
 376.        }
