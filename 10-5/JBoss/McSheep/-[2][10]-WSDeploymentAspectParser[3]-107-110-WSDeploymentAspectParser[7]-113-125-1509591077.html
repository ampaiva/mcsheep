  79.        public static List<DeploymentAspect> parse(XMLStreamReader reader, ClassLoader loader) throws XM...
  80.            int iterate;
  81.            try {
  82.                iterate = reader.nextTag();
  83.            } catch (XMLStreamException e) {
  84.                // skip non-tag elements
  85.                iterate = reader.nextTag();
  86.            }
  87.            List<DeploymentAspect> deploymentAspects = null;
  88.            switch (iterate) {
  89.                case END_ELEMENT: {
  90.                    // we're done
  91.                    break;
  92.                }
  93.                case START_ELEMENT: {

  95.                    if (match(reader, NS, DEPLOYMENT_ASPECTS)) {
  96.                        deploymentAspects = parseDeploymentAspects(reader, loader);
  97.                    } else {
  98.                        throw WSLogger.ROOT_LOGGER.unexpectedElement(reader.getLocalName());
  99.                    }
 100.                }
 101.            }
 102.            return deploymentAspects;
 103.        }

 129.            String deploymentAspectClass = reader.getAttributeValue(null, CLASS);

 139.                    WildFlySecurityManager.setCurrentContextClassLoaderPrivileged(loader);

 142.                    WildFlySecurityManager.setCurrentContextClassLoaderPrivileged(orig);

 153.                    case XMLStreamConstants.END_ELEMENT: {

 172.        @SuppressWarnings("rawtypes")

 209.                            } catch (Exception e) {

 211.                            }

 224.                }

 248.        private static Object parseSimpleValue(XMLStreamReader reader, String propClass) throws XMLStrea...
 249.            if (String.class.getName().equals(propClass)) {

 322.        @SuppressWarnings({ "rawtypes", "unchecked" })

 324.                throws XMLStreamException {

 345.                        } else {
