 109.            this.minPoolSize = minPoolSize;
 110.            this.properties = properties;
 111.            if (transactionSupport == null)
 112.                transactionSupport = TransactionSupport.TransactionSupportLevel.NoTransaction;
 113.            this.transactionSupport = transactionSupport;
 114.            this.module = module;
 115.            this.bindInfo = bindInfo;
 116.        }

 120.            return bindInfo;

 123.        @Override
 124.        public void start(StartContext context) throws StartException {
 125.            ROOT_LOGGER.debugf("started DirectConnectionFactoryActivatorService %s", context.getController().getName());
 126.            String cfInterface = null;
 127.    
 128.    
 129.            try {
 130.    
 131.                Connector cmd = mdr.getValue().getResourceAdapter(raId);
 132.    
 133.                ResourceAdapter ra = cmd.getResourceadapter();
 134.                if (ra.getOutboundResourceadapter() != null) {
 135.                    for (ConnectionDefinition cd : ra.getOutboundResourceadapter().getConnectionDefinitions()) {
 136.                        if (cd.getConnectionFactoryInterface().getValue().equals(interfaceName))
 137.                            cfInterface = cd.getConnectionFactoryInterface().getValue();
 138.                    }
 139.                }
 140.    
 141.    
 142.                if (cfInterface == null || !cfInterface.equals(interfaceName)) {
 143.                    throw ConnectorLogger.ROOT_LOGGER.invalidConnectionFactory(cfInterface, resourceAdapter, jndiName);
 144.                }
 145.    
 146.                Map raConfigProperties = new HashMap();
 147.                Map mcfConfigProperties = new HashMap();
 148.                String securitySetting = null;
 149.                String securitySettingDomain = null;
 150.    
 151.                if (properties != null) {
 152.                    for (Map.Entry prop : properties.entrySet()) {
 153.                        String key = prop.getKey();
 154.                        String value = prop.getValue();
 155.                        if (key.startsWith("ironjacamar.security")) {
 156.                            securitySetting = value;
 157.                        } else if (key.startsWith("ironjacamar.security.domain")) {
 158.                            securitySettingDomain = value;
 159.                        } else {
 160.                            if (key.startsWith("ra.")) {
 161.                                raConfigProperties.put(key.substring(3), value);
 162.                            } else if (key.startsWith("mcf.")) {
 163.                                mcfConfigProperties.put(key.substring(4), value);
 164.                            } else {
 165.                                mcfConfigProperties.put(key, value);
 166.                            }
 167.                        }
 168.                    }
 169.                }
 170.    
 171.                String mcfClass = null;
 172.    
 173.                if (ra.getOutboundResourceadapter() != null) {
 174.                    for (ConnectionDefinition cd :
 175.                            ra.getOutboundResourceadapter().getConnectionDefinitions()) {
 176.                        if (cd.getConnectionFactoryInterface().getValue().equals(cfInterface))
 177.                            mcfClass = cd.getManagedConnectionFactoryClass().getValue();
 178.                    }
 179.                }
 180.    
 181.    
 182.                Security security = null;
 183.                if (securitySetting != null) {
 184.                    if ("".equals(securitySetting)) {
 185.                        security = new SecurityImpl(null, null, false);
 186.                    } else if ("application".equals(securitySetting)) {
 187.                        security = new SecurityImpl(null, null, true);
 188.                    } else if ("domain".equals(securitySetting) && securitySettingDomain != null) {
 189.                        security = new SecurityImpl(securitySettingDomain, null, false);
 190.                    } else if ("domain-and-application".equals(securitySetting) && securitySettingDomain != null) {
 191.                        security = new SecurityImpl(null, securitySettingDomain, false);
 192.                    }
 193.                }
 194.    
 195.                if (security == null) {
 196.                    SUBSYSTEM_RA_LOGGER.noSecurityDefined(jndiName);
 197.                }
 198.    
 199.    
 200.                Pool pool = null;
 201.                Boolean isXA = Boolean.FALSE;
 202.                if (transactionSupport == TransactionSupport.TransactionSupportLevel.XATransaction) {
 203.                    pool = new XaPoolImpl(minPoolSize < 0 ? Defaults.MIN_POOL_SIZE : minPoolSize, Defaults.INITIAL_POOL_SIZE, maxPoolSize < 0 ? Defaults.MAX_POOL_SIZE : maxPoolSize,
 204.                            Defaults.PREFILL, Defaults.USE_STRICT_MIN, Defaults.FLUSH_STRATEGY,
 205.                            null, Defaults.IS_SAME_RM_OVERRIDE, Defaults.INTERLEAVING, Defaults.PAD_XID, Defaults.WRAP_XA_RESOURCE, Defaults.NO_TX_SEPARATE_POOL);
 206.                    isXA = Boolean.TRUE;
 207.                } else {
 208.                    pool = new PoolImpl(minPoolSize < 0 ? Defaults.MIN_POOL_SIZE : minPoolSize, Defaults.INITIAL_POOL_SIZE, maxPoolSize < 0 ? Defaults.MAX_POOL_SIZE : maxPoolSize,
 209.                            Defaults.PREFILL, Defaults.USE_STRICT_MIN, Defaults.FLUSH_STRATEGY, null);
 210.                }
 211.    
 212.                TransactionSupportEnum transactionSupportValue = TransactionSupportEnum.NoTransaction;
 213.                if (transactionSupport == TransactionSupport.TransactionSupportLevel.XATransaction) {
 214.                    transactionSupportValue = TransactionSupportEnum.XATransaction;
 215.                } else if (transactionSupport == TransactionSupport.TransactionSupportLevel.LocalTransaction) {
 216.                    transactionSupportValue = TransactionSupportEnum.LocalTransaction;
 217.                }
 218.    
 219.                org.jboss.jca.common.api.metadata.resourceadapter.ConnectionDefinition cd = new org.jboss.jca.common.metadata.resourceadapter.ConnectionDefinitionImpl(mcfConfigProperties, mcfClass, jndiName, poolName(cfInterface),
 220.                        Boolean.TRUE, Boolean.TRUE, Boolean.TRUE, Boolean.TRUE, Boolean.TRUE, Defaults.CONNECTABLE, Defaults.TRACKING,
 221.                        pool, null, null, security, null, isXA);
 222.    
 223.                Activation activation = new ActivationImpl(null, null, transactionSupportValue, Collections.singletonList(cd), Collections.emptyList(), raConfigProperties, Collections.emptyList(), null, null);
 224.    
 225.                String serviceName = jndiName;
 226.                serviceName = serviceName.replace(':', '_');
 227.                serviceName = serviceName.replace('/', '_');
 228.    
 229.                ResourceAdapterActivatorService activator = new ResourceAdapterActivatorService(cmd, activation, module.getClassLoader(), serviceName);
 230.                activator.setCreateBinderService(false);
 231.                activator.setBindInfo(bindInfo);
 232.                ServiceTarget serviceTarget = context.getChildTarget();
 233.                ServiceName activatorServiceName = ConnectorServices.RESOURCE_ADAPTER_ACTIVATOR_SERVICE.append(serviceName);
 234.                ServiceBuilder connectionFactoryServiceBuilder = serviceTarget
 235.                        .addService(activatorServiceName, activator)
 236.                        .addDependency(ConnectorServices.IRONJACAMAR_MDR, AS7MetadataRepository.class,
 237.                                activator.getMdrInjector())
 238.                        .addDependency(ConnectorServices.RA_REPOSITORY_SERVICE, ResourceAdapterRepository.class,
 239.                                activator.getRaRepositoryInjector())
 240.                        .addDependency(ConnectorServices.MANAGEMENT_REPOSITORY_SERVICE, ManagementRepository.class,
 241.                                activator.getManagementRepositoryInjector())
 242.                        .addDependency(ConnectorServices.RESOURCE_ADAPTER_REGISTRY_SERVICE,
 243.                                ResourceAdapterDeploymentRegistry.class, activator.getRegistryInjector())
 244.                        .addDependency(ConnectorServices.CONNECTOR_CONFIG_SERVICE,
 245.                                JcaSubsystemConfiguration.class, activator.getConfigInjector())
 246.                        .addDependency(SubjectFactoryService.SERVICE_NAME, SubjectFactory.class,
 247.                                activator.getSubjectFactoryInjector())
 248.                        .addDependency(ConnectorServices.CCM_SERVICE, CachedConnectionManager.class,
 249.                                activator.getCcmInjector())
 250.                        .addDependency(NamingService.SERVICE_NAME)
 251.                        .addDependency(ConnectorServices.TRANSACTION_INTEGRATION_SERVICE, TransactionIntegration.class,
 252.                                activator.getTxIntegrationInjector())
 253.                        .addDependency(TxnServices.JBOSS_TXN_TRANSACTION_MANAGER)
 254.                        .addDependency(ConnectorServices.BOOTSTRAP_CONTEXT_SERVICE.append("default"));
 255.    
 256.    
 257.                connectionFactoryServiceBuilder.setInitialMode(ServiceController.Mode.ACTIVE).install();
 258.    
 259.    
 260.            } catch (Exception e) {
 261.                throw new StartException(e);
 262.            }
 263.        }

 275.        private String poolName(final String cfInterface) {
 276.            if (cfInterface.indexOf(".") != -1) {
 277.                return cfInterface.substring(cfInterface.lastIndexOf(".") + 1);
 278.            } else {
 279.                return cfInterface;
 280.            }
 281.        }
