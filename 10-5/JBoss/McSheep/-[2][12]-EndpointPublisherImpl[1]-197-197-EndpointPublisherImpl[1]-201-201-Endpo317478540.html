  35.     * WebAppController allows for automatically starting/stopping a webapp (servlet) depending on the a...

  38.     * @author alessio.soldano@jboss.com

  41.    public class WebAppController {

  53.                String serverTempDir) {

  62.        public synchronized int incrementUsers() throws StartException {
  63.            if (count == 0) {
  64.                try {
  65.                    ctx = startWebApp(host);
  66.                } catch (Exception e) {
  67.                    throw new StartException(e);
  68.                }
  69.            }
  70.            return count++;
  71.        }
  72.    
  73.        public synchronized int decrementUsers() {
  74.            if (count == 0) {
  75.                throw new IllegalStateException();
  76.            }
  77.            count--;
  78.            if (count == 0) {
  79.                try {
  80.                    stopWebApp(ctx);
  81.                } catch (Exception e) {
  82.                    throw new RuntimeException(e);
  83.                }
  84.            }
  85.            return count;
  86.        }

  88.        private WebDeploymentController startWebApp(WebHost host) throws Exception {
  89.            WebDeploymentBuilder builder = new WebDeploymentBuilder();
  90.            WebDeploymentController deployment;
  91.            try {
  92.                builder.setContextRoot(contextRoot);
  93.                File docBase = new File(serverTempDir, contextRoot);
  94.                if (!docBase.exists()) {
  95.                    docBase.mkdirs();
  96.                }
  97.                builder.setDocumentRoot(docBase);
  98.                builder.setClassLoader(classloader);
  99.    
 100.                final int j = servletClass.indexOf(".");
 101.                final String servletName = j < 0 ? servletClass : servletClass.substring(j + 1);
 102.                final Class<?> clazz = classloader.loadClass(servletClass);
 103.                ServletBuilder servlet = new ServletBuilder();
 104.                servlet.setServletName(servletName);
 105.                servlet.setServlet((Servlet) clazz.newInstance());
 106.                servlet.setServletClass(clazz);
 107.                servlet.addUrlMapping(urlPattern);
 108.                builder.addServlet(servlet);
 109.    
 110.                deployment = host.addWebDeployment(builder);
 111.                deployment.create();
 112.    
 113.            } catch (Exception e) {
 114.                throw WSLogger.ROOT_LOGGER.createContextPhaseFailed(e);
 115.            }
 116.            try {
 117.                deployment.start();
 118.            } catch (Exception e) {
 119.                throw WSLogger.ROOT_LOGGER.startContextPhaseFailed(e);
 120.            }
 121.            return deployment;
 122.        }

 124.        private void stopWebApp(WebDeploymentController context) throws Exception {
 125.            try {
 126.                context.stop();
 127.            } catch (Exception e) {
 128.                throw WSLogger.ROOT_LOGGER.stopContextPhaseFailed(e);
 129.            }
 130.            try {
 131.                context.destroy();
 132.            } catch (Exception e) {
 133.                throw WSLogger.ROOT_LOGGER.destroyContextPhaseFailed(e);
 134.            }
 135.        }
