 531.            if(firstComponent.startsWith("java:")) {

 539.                return name.getSuffix(1);
 540.            } else {
 541.                return composeName(name, prefix);

 573.        private void check(Name name, Action... actions) throws NamingException {
 574.            final SecurityManager sm = System.getSecurityManager();
 575.            if (sm != null && WildFlySecurityManager.isChecking()) {
 576.                // build absolute name (including store's base name)
 577.                Name absoluteName = (Name) namingStore.getBaseName().clone();
 578.                if (name.isEmpty()) {
 579.                    absoluteName.addAll(prefix);
 580.                } else {
 581.                    final String firstComponent = name.get(0);
 582.                    if (firstComponent.startsWith("java:")) {
 583.                        absoluteName = name;
 584.                    } else if (firstComponent.isEmpty()) {
 585.                        absoluteName.addAll(name.getSuffix(1));
 586.                    } else {
 587.                        absoluteName.addAll(prefix);
 588.                        absoluteName.addAll(name);
 589.                    }
 590.                }
 591.                sm.checkPermission(new JndiPermission(absoluteName, actions));
 592.            }
