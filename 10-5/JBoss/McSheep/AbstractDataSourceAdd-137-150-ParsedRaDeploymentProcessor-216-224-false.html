 131.                String bootstrapCtx = null;
 132.    
 133.                if (ironJacamarXmlDescriptor != null && ironJacamarXmlDescriptor.getIronJacamar() != null && ironJacamarXmlDescriptor.getIronJacamar().getBootstrapContext() != null)
 134.                    bootstrapCtx = ironJacamarXmlDescriptor.getIronJacamar().getBootstrapContext();
 135.    
 136.                if (bootstrapCtx == null)
 137.                    bootstrapCtx = "default";
 138.                //Register an empty override model regardless of we're enabled or not - the statistics listener will add the relevant childresources
 139.                if (registration.isAllowsOverride() && registration.getOverrideModel(deploymentUnit.getName()) == null) {
 140.                    registration.registerOverrideModel(deploymentUnit.getName(), new OverrideDescriptionProvider() {
 141.                        @Override
 142.                        public Map getAttributeOverrideDescriptions(Locale locale) {
 143.                            return Collections.emptyMap();
 144.                        }
 145.    
 146.                        @Override
 147.                        public Map getChildTypeOverrideDescriptions(Locale locale) {
 148.                            return Collections.emptyMap();
 149.                        }
 150.                    });
 151.    
 152.                }
 153.                builder.addListener(new AbstractResourceAdapterDeploymentServiceListener(registration, deploymentUnit.getName(), deploymentResource, bootstrapCtx, deploymentUnit.getName(), false) {
 154.    
 155.                    @Override
 156.                    protected void registerIronjacamar(final ServiceController controller, final ManagementResourceRegistration subRegistration, final Resource subsystemResource) {
 157.                        //register ironJacamar
 158.                        try {
 159.                            subRegistration.registerSubModel(new IronJacamarResourceDefinition());
 160.                        } catch (IllegalArgumentException iae) {
 161.                            //ignore it: submodel already registered
 162.                        }
 163.                        AS7MetadataRepository mdr = ((ResourceAdapterDeploymentService) controller.getService()).getMdr();
 164.                        IronJacamarResourceCreator.INSTANCE.execute(subsystemResource, mdr);
 165.                    }
 166.    
 167.                    @Override
 168.                    protected CommonDeployment getDeploymentMetadata(final ServiceController controller) {
 169.                        return ((ResourceAdapterDeploymentService) controller.getService()).getRaDeployment();
 170.                    }
 171.                });
 172.    
 173.    
 174.                builder.setInitialMode(Mode.ACTIVE).install();
 175.            }
 176.        }

 178.    
 179.        public static ServiceBuilder process(final ConnectorXmlDescriptor connectorXmlDescriptor, final IronJacamarXmlDescriptor ironJacamarXmlDescriptor, final ClassLoader classLoader, final ServiceTarget serviceTarget, final Map annotationIndexes, final ServiceName duServiceName) throws DeploymentUnitProcessingException {
 180.    
 181.            Connector cmd = connectorXmlDescriptor != null ? connectorXmlDescriptor.getConnector() : null;
 182.            final Activation activation = ironJacamarXmlDescriptor != null ? ironJacamarXmlDescriptor.getIronJacamar() : null;
 183.    
 184.            try {
 185.                // Annotation merging
 186.                Annotations annotator = new Annotations();
 187.    
 188.                if (annotationIndexes != null && annotationIndexes.size() > 0) {
 189.                    DEPLOYMENT_CONNECTOR_LOGGER.debugf("ParsedRaDeploymentProcessor: Found %d annotationIndexes", annotationIndexes.size());
 190.                    for (Index index : annotationIndexes.values()) {
 191.                        // Don't apply any empty indexes, as IronJacamar doesn't like that atm.
 192.                        if (index.getKnownClasses() != null && index.getKnownClasses().size() > 0) {
 193.                            AnnotationRepository repository = new JandexAnnotationRepositoryImpl(index, classLoader);
 194.                            cmd = annotator.merge(cmd, repository, classLoader);
 195.                            DEPLOYMENT_CONNECTOR_LOGGER.debugf("ParsedRaDeploymentProcessor: CMD=%s", cmd);
 196.                        }
 197.                    }
 198.                }
 199.                if (annotationIndexes == null || annotationIndexes.size() == 0)
 200.                    DEPLOYMENT_CONNECTOR_LOGGER.debugf("ParsedRaDeploymentProcessor: Found 0 annotationIndexes");
 201.    
 202.                // FIXME: when the connector is null the Iron Jacamar data is ignored
 203.                if (cmd != null) {
 204.                    // Validate metadata
 205.                    cmd.validate();
 206.    
 207.                    // Merge metadata
 208.                    cmd = (new Merger()).mergeConnectorWithCommonIronJacamar(activation, cmd);
 209.                }
 210.    
 211.                final ServiceName deployerServiceName = ConnectorServices.RESOURCE_ADAPTER_DEPLOYER_SERVICE_PREFIX.append(connectorXmlDescriptor.getDeploymentName());
 212.                final ResourceAdapterDeploymentService raDeploymentService = new ResourceAdapterDeploymentService(connectorXmlDescriptor, cmd, activation, classLoader, deployerServiceName, duServiceName);
 213.    
 214.    
 215.                // Create the service
 216.                return Services.addServerExecutorDependency(
 217.                            serviceTarget.addService(deployerServiceName, raDeploymentService),
 218.                            raDeploymentService.getExecutorServiceInjector(), false)
 219.                        .addDependency(ConnectorServices.IRONJACAMAR_MDR, AS7MetadataRepository.class, raDeploymentService.getMdrInjector())
 220.                        .addDependency(ConnectorServices.RA_REPOSITORY_SERVICE, ResourceAdapterRepository.class, raDeploymentService.getRaRepositoryInjector())
 221.                        .addDependency(ConnectorServices.MANAGEMENT_REPOSITORY_SERVICE, ManagementRepository.class, raDeploymentService.getManagementRepositoryInjector())
 222.                        .addDependency(ConnectorServices.RESOURCE_ADAPTER_REGISTRY_SERVICE, ResourceAdapterDeploymentRegistry.class, raDeploymentService.getRegistryInjector())
 223.                        .addDependency(ConnectorServices.TRANSACTION_INTEGRATION_SERVICE, TransactionIntegration.class, raDeploymentService.getTxIntegrationInjector())
 224.                        .addDependency(ConnectorServices.CONNECTOR_CONFIG_SERVICE, JcaSubsystemConfiguration.class, raDeploymentService.getConfigInjector())
 225.                        .addDependency(SubjectFactoryService.SERVICE_NAME, SubjectFactory.class, raDeploymentService.getSubjectFactoryInjector())
 226.                        .addDependency(ConnectorServices.CCM_SERVICE, CachedConnectionManager.class, raDeploymentService.getCcmInjector())
 227.                        .addDependency(ConnectorServices.IDLE_REMOVER_SERVICE)
 228.                        .addDependency(ConnectorServices.CONNECTION_VALIDATOR_SERVICE)
 229.                        .addDependency(NamingService.SERVICE_NAME);
 230.            } catch (Throwable t) {
 231.                throw new DeploymentUnitProcessingException(t);
 232.            }
 233.        }
