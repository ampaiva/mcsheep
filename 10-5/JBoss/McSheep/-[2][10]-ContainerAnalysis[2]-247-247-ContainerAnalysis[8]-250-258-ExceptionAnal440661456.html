 292.        protected boolean isAccessor(Method m) {
 293.            Class returnType = m.getReturnType();
 294.            // JBAS-4473, look for get<name>()
 295.            String name = m.getName();
 296.            if (!(name.startsWith("get") && name.length() > "get".length()))
 297.                if (!(name.startsWith("is") && name.length() > "is".length())
 298.                        || !(returnType == Boolean.TYPE))
 299.                    return false;
 300.            if (returnType == Void.TYPE)
 301.                return false;
 302.            if (m.getParameterTypes().length != 0)
 303.                return false;
 304.    
 305.            return hasNonAppExceptions(m);
 306.        }

 396.        protected void analyzeMethods() {
 397.    
 398.            // The dynamic stub and skeleton strategy generation mechanism
 399.            // requires the inclusion of inherited methods in the analysis of
 400.            // remote interfaces. To speed things up, inherited methods are
 401.            // not considered in the analysis of a class or non-remote interface.
 402.            if (cls.isInterface() && java.rmi.Remote.class.isAssignableFrom(cls))
 403.                methods = cls.getMethods();
 404.            else
 405.                methods = cls.getDeclaredMethods();
 406.            m_flags = new byte[methods.length];
 407.            mutators = new int[methods.length];
 408.    
 409.            // Find read-write properties
 410.            for (int i = 0; i < methods.length; ++i)

 412.            for (int i = 0; i < methods.length; ++i) {
 413.    
 414.                if (isAccessor(methods[i]) && (m_flags[i] & M_READ) == 0) {
 415.                    String attrName = attributeReadName(methods[i].getName());
 416.                    Class iReturn = methods[i].getReturnType();
 417.                    for (int j = i + 1; j < methods.length; ++j) {
 418.                        if (isMutator(methods[j]) && (m_flags[j] & M_WRITE) == 0 &&
 419.                                attrName.equals(attributeWriteName(methods[j].getName()))) {
 420.                            Class[] jParams = methods[j].getParameterTypes();
 421.                            if (jParams.length == 1 && jParams[0] == iReturn) {
 422.                                m_flags[i] |= M_READ;
 423.                                m_flags[j] |= M_WRITE;
 424.                                mutators[i] = j;
 425.                                break;
 426.                            }
 427.                        }
 428.                    }
 429.                } else if (isMutator(methods[i]) && (m_flags[i] & M_WRITE) == 0) {
 430.                    String attrName = attributeWriteName(methods[i].getName());
 431.                    Class[] iParams = methods[i].getParameterTypes();
 432.                    for (int j = i + 1; j < methods.length; ++j) {
 433.                        if (isAccessor(methods[j]) && (m_flags[j] & M_READ) == 0 &&
 434.                                attrName.equals(attributeReadName(methods[j].getName()))) {
 435.                            Class jReturn = methods[j].getReturnType();
 436.                            if (iParams.length == 1 && iParams[0] == jReturn) {
 437.                                m_flags[i] |= M_WRITE;
 438.                                m_flags[j] |= M_READ;
 439.                                mutators[j] = i;
 440.                                break;
 441.                            }
 442.                        }
 443.                    }
 444.                }
 445.            }
 446.    
 447.            // Find read-only properties
 448.            for (int i = 0; i < methods.length; ++i)
 449.                if ((m_flags[i] & (M_READ | M_WRITE)) == 0 && isAccessor(methods[i]))
 450.                    m_flags[i] |= M_READONLY;
 451.    
 452.            // Check for overloaded and inherited methods
 453.            for (int i = 0; i < methods.length; ++i) {
 454.                if ((m_flags[i] & (M_READ | M_WRITE | M_READONLY)) == 0) {
 455.                    String iName = methods[i].getName();
 456.    
 457.                    for (int j = i + 1; j < methods.length; ++j) {
 458.                        if (iName.equals(methods[j].getName())) {
 459.                            m_flags[i] |= M_OVERLOADED;
 460.                            m_flags[j] |= M_OVERLOADED;
 461.                        }
 462.                    }
 463.                }
 464.    
 465.                if (methods[i].getDeclaringClass() != cls)
 466.                    m_flags[i] |= M_INHERITED;
 467.            }
 468.    
 469.        }
 470.    
 471.        /**
 472.         * Convert an attribute read method name in Java format to
 473.         * an attribute name in Java format.
 474.         */
 475.        protected String attributeReadName(String name) {
 476.            if (name.startsWith("get"))
 477.                name = name.substring(3);
 478.            else if (name.startsWith("is"))
 479.                name = name.substring(2);
 480.            else
 481.                throw IIOPLogger.ROOT_LOGGER.notAnAccessor(name);
 482.    
 483.            return name;
 484.        }
 485.    
 486.        /**
 487.         * Convert an attribute write method name in Java format to
 488.         * an attribute name in Java format.
 489.         */
 490.        protected String attributeWriteName(String name) {
 491.            if (name.startsWith("set"))
 492.                name = name.substring(3);
 493.            else
 494.                throw IIOPLogger.ROOT_LOGGER.notAnAccessor(name);
 495.    
 496.            return name;
 497.        }
 498.    
 499.        /**
 500.         * Analyse constants.
 501.         * This will fill in the <code>constants</code> array.
 502.         */
 503.        protected void analyzeConstants()
 504.                throws RMIIIOPViolationException {
 505.    
 506.            ArrayList a = new ArrayList();
 507.    
 508.            for (int i = 0; i < fields.length; ++i) {
 509.                if ((f_flags[i] & F_CONSTANT) == 0)
 510.                    continue;
 511.    
 512.                Class type = fields[i].getType();
 513.    
 514.                // Only map primitives and java.lang.String
 515.                if (!type.isPrimitive() && type != java.lang.String.class) {
 516.                    // It is an RMI/IIOP violation for interfaces.
 517.                    if (cls.isInterface())
 518.                        throw IIOPLogger.ROOT_LOGGER.badRMIIIOPConstantType(fields[i].getName(), cls.get...
 519.                    continue;
 520.                }
 521.    
 522.                String name = fields[i].getName();
 523.    
 524.                Object value;
 525.                try {
 526.                    value = fields[i].get(null);
 527.                } catch (Exception ex) {
 528.                    throw new RuntimeException(ex.toString());
 529.                }
 530.    
 531.                a.add(new ConstantAnalysis(name, type, value));
 532.            }
 533.    
 534.            constants = new ConstantAnalysis[a.size()];
 535.            constants = (ConstantAnalysis[]) a.toArray(constants);
 536.        }
 537.    
 538.        /**
 539.         * Analyse attributes.
 540.         * This will fill in the <code>attributes</code> array.
 541.         */
 542.        protected void analyzeAttributes() throws RMIIIOPViolationException {
 543.            ArrayList a = new ArrayList();
 544.    
 545.            for (int i = 0; i < methods.length; ++i) {
 546.                //if ((m_flags[i]&M_INHERITED) != 0)
 547.                //  continue;
 548.    
 549.                if ((m_flags[i] & (M_READ | M_READONLY)) != 0) {
 550.                    // Read method of an attribute.
 551.                    String name = attributeReadName(methods[i].getName());
 552.    
 553.                    if ((m_flags[i] & M_READONLY) != 0)
 554.                        a.add(new AttributeAnalysis(name, methods[i]));
 555.                    else
 556.                        a.add(new AttributeAnalysis(name, methods[i],
 557.                                methods[mutators[i]]));
 558.                }
 559.            }
 560.    
 561.            attributes = new AttributeAnalysis[a.size()];
 562.            attributes = (AttributeAnalysis[]) a.toArray(attributes);
 563.    
 564.        }
 565.    
 566.        /**
 567.         * Analyse operations.
 568.         * This will fill in the <code>operations</code> array.
 569.         * This implementation just creates an empty array; override
 570.         * in subclasses for a real analysis.
 571.         */
 572.        protected void analyzeOperations()
 573.                throws RMIIIOPViolationException {
 574.            operations = new OperationAnalysis[0];
 575.        }
 576.    
 577.        /**
 578.         * Fixup overloaded operation names.
 579.         * As specified in section 1.3.2.6.
 580.         */
 581.        protected void fixupOverloadedOperationNames()
 582.                throws RMIIIOPViolationException {
 583.            for (int i = 0; i < methods.length; ++i) {
 584.                if ((m_flags[i] & M_OVERLOADED) == 0)
 585.                    continue;
 586.    
 587.                // Find the operation
 588.                OperationAnalysis oa = null;
 589.                String javaName = methods[i].getName();
 590.                for (int opIdx = 0; oa == null && opIdx < operations.length; ++opIdx)
 591.                    if (operations[opIdx].getMethod().equals(methods[i]))
 592.                        oa = operations[opIdx];
 593.    
 594.                if (oa == null)
 595.                    continue; // This method is not mapped.
 596.    
 597.                // Calculate new IDL name
 598.                ParameterAnalysis[] params = oa.getParameters();
 599.                StringBuffer b = new StringBuffer(oa.getIDLName());
 600.                if (params.length == 0)
 601.                    b.append("__");
 602.                for (int j = 0; j < params.length; ++j) {
 603.                    String s = params[j].getTypeIDLName();
 604.    
 605.                    if (s.startsWith("::"))
 606.                        s = s.substring(2);
 607.    
 608.                    if (s.startsWith("_")) {
 609.                        // remove leading underscore in IDL escaped identifier
 610.                        s = s.substring(1);
 611.                    }
 612.    
 613.                    b.append('_');
 614.    
 615.                    while (!"".equals(s)) {
 616.                        int idx = s.indexOf("::");
 617.    
 618.                        b.append('_');
 619.    
 620.                        if (idx == -1) {
 621.                            b.append(s);
 622.                            s = "";
 623.                        } else {
 624.                            b.append(s.substring(0, idx));
 625.                            if (s.length() > idx + 2 && s.charAt(idx + 2) == '_') {
 626.                                // remove leading underscore in IDL escaped identifier
 627.                                s = s.substring(idx + 3);
 628.                            } else {
 629.                                s = s.substring(idx + 2);
 630.                            }
 631.                        }
 632.                    }
 633.                }
 634.    
 635.                // Set new IDL name
 636.                oa.setIDLName(b.toString());
 637.            }
 638.        }
 639.    
 640.        /**
 641.         * Fixup names differing only in case.
 642.         * As specified in section 1.3.2.7.
 643.         */
 644.        protected void fixupCaseNames()
 645.                throws RMIIIOPViolationException {
 646.            ArrayList entries = getContainedEntries();
 647.            boolean[] clash = new boolean[entries.size()];
 648.            String[] upperNames = new String[entries.size()];
 649.    
 650.            for (int i = 0; i < entries.size(); ++i) {
 651.                AbstractAnalysis aa = (AbstractAnalysis) entries.get(i);
 652.    
 653.                clash[i] = false;
 654.                upperNames[i] = aa.getIDLName().toUpperCase(Locale.ENGLISH);
 655.    
 656.                for (int j = 0; j < i; ++j) {
 657.                    if (upperNames[i].equals(upperNames[j])) {
 658.                        clash[i] = true;
 659.                        clash[j] = true;
 660.                    }
 661.                }
 662.            }
 663.    
 664.            for (int i = 0; i < entries.size(); ++i) {
 665.                if (!clash[i])
 666.                    continue;
 667.    
 668.                AbstractAnalysis aa = (AbstractAnalysis) entries.get(i);
 669.                boolean noUpper = true;
 670.                String name = aa.getIDLName();
 671.                StringBuffer b = new StringBuffer(name);
 672.                b.append('_');
 673.                for (int j = 0; j < name.length(); ++j) {
 674.                    if (!Character.isUpperCase(name.charAt(j)))
 675.                        continue;
 676.                    if (noUpper)
 677.                        noUpper = false;
 678.                    else
 679.                        b.append('_');
 680.                    b.append(j);
 681.                }
 682.    
 683.                aa.setIDLName(b.toString());
 684.            }
 685.        }

 693.        /**
 694.         * Return the class hash code, as specified in "The Common Object
 695.         * Request Broker: Architecture and Specification" (01-02-33),
 696.         * section 10.6.2.
 697.         */
 698.        protected void calculateClassHashCode() {
 699.            // The simple cases
 700.            if (cls.isInterface())
 701.                classHashCode = 0;
 702.            else if (!Serializable.class.isAssignableFrom(cls))
 703.                classHashCode = 0;
 704.            else if (Externalizable.class.isAssignableFrom(cls))
 705.                classHashCode = 1;
 706.            else // Go ask Util class for the hash code
 707.                classHashCode = Util.getClassHashCode(cls);
 708.        }

 710.        /**
 711.         * Escape non-ISO characters for an IR name.
 712.         */
 713.        protected String escapeIRName(String name) {
 714.            StringBuffer b = new StringBuffer();
 715.    
 716.            for (int i = 0; i < name.length(); ++i) {
 717.                char c = name.charAt(i);
 718.    
 719.                if (c < 256)
 720.                    b.append(c);
 721.                else
 722.                    b.append("\\U").append(toHexString((int) c));
 723.            }
 724.            return b.toString();
 725.        }

 727.        /**
 728.         * Return the IR global ID of the given class or interface.
 729.         * This is described in section 1.3.5.7.
 730.         * The returned string is in the RMI hashed format, like
 731.         * "RMI:java.util.Hashtable:C03324C0EA357270:13BB0F25214AE4B8".
 732.         */
 733.        protected void calculateRepositoryId() {
 734.            if (cls.isArray() || cls.isPrimitive())
 735.                throw IIOPLogger.ROOT_LOGGER.notAnClassOrInterface(cls.getName());
 736.    
 737.            if (cls.isInterface() &&
 738.                    org.omg.CORBA.Object.class.isAssignableFrom(cls) &&
 739.                    org.omg.CORBA.portable.IDLEntity.class.isAssignableFrom(cls)) {
 740.    
 741.                StringBuffer b = new StringBuffer("IDL:");
 742.                b.append(cls.getPackage().getName().replace('.', '/'));
 743.                b.append('/');
 744.                String base = cls.getName();
 745.                base = base.substring(base.lastIndexOf('.') + 1);
 746.                b.append(base).append(":1.0");
 747.                repositoryId = b.toString();
 748.            } else {
 749.                StringBuffer b = new StringBuffer("RMI:");
 750.                b.append(escapeIRName(cls.getName()));
 751.                memberPrefix = b.toString() + ".";
 752.                String hashStr = toHexString(classHashCode);
 753.                b.append(':').append(hashStr);
 754.                ObjectStreamClass osClass = ObjectStreamClass.lookup(cls);
 755.                if (osClass != null) {
 756.                    long serialVersionUID = osClass.getSerialVersionUID();
 757.                    String SVUID = toHexString(serialVersionUID);
 758.    
 759.                    if (classHashCode != serialVersionUID)
 760.                        b.append(':').append(SVUID);
 761.                    memberPostfix = ":" + hashStr + ":" + SVUID;
 762.                } else
 763.                    memberPostfix = ":" + hashStr;
 764.    
 765.                repositoryId = b.toString();
 766.            }
 767.        }

  26.    /**
  27.     * Exception analysis.
  28.     * <p/>
  29.     * Routines here are conforming to the "Java(TM) Language to IDL Mapping
  30.     * Specification", version 1.1 (01-06-07).
  31.     *
  32.     * @author <a href="mailto:osh@sparre.dk">Ole Husgaard</a>
  33.     */
  34.    public class ExceptionAnalysis extends ValueAnalysis {
  35.    
  36.        private static WorkCacheManager cache
  37.                = new WorkCacheManager(ExceptionAnalysis.class);
  38.    
  39.        private String exceptionRepositoryId;
  40.    
  41.        public static ExceptionAnalysis getExceptionAnalysis(Class cls)
  42.                throws RMIIIOPViolationException {
  43.            return (ExceptionAnalysis) cache.getAnalysis(cls);

  50.        protected ExceptionAnalysis(Class cls) {
  51.            super(cls);
  52.        }
  53.    
  54.        protected void doAnalyze() throws RMIIIOPViolationException {
  55.            super.doAnalyze();
  56.    
  57.            if (!Exception.class.isAssignableFrom(cls) ||
  58.                    RuntimeException.class.isAssignableFrom(cls))
  59.                throw IIOPLogger.ROOT_LOGGER.badRMIIIOPExceptionType(cls.getName(), "1.2.6");
  60.    
  61.            // calculate exceptionRepositoryId
  62.            StringBuffer b = new StringBuffer("IDL:");
  63.            String pkgName = cls.getPackage().getName();
  64.    
  65.            while (!"".equals(pkgName)) {
  66.                int idx = pkgName.indexOf('.');
  67.                String n = (idx == -1) ? pkgName : pkgName.substring(0, idx);
  68.                b.append(Util.javaToIDLName(n)).append('/');
  69.                pkgName = (idx == -1) ? "" : pkgName.substring(idx + 1);
  70.            }
  71.    
  72.            String base = cls.getName();
  73.            base = base.substring(base.lastIndexOf('.') + 1);
  74.            if (base.endsWith("Exception"))
  75.                base = base.substring(0, base.length() - 9);
  76.            base = Util.javaToIDLName(base + "Ex");
  77.    
  78.            b.append(base).append(":1.0");
  79.            exceptionRepositoryId = b.toString();
  80.        }
