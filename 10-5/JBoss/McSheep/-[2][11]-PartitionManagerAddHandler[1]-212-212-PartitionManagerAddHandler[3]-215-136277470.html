 500.        @SuppressWarnings("unchecked")
 501.        private <T> Class<T> loadClass(ModelNode moduleNode, String typeName) {
 502.            try {
 503.                Module module = getModule(moduleNode);
 504.    
 505.                if (module != null) {
 506.                    return (Class<T>) module.getClassLoader().loadClass(typeName);
 507.                } else {
 508.                    return (Class<T>) getClass().getClassLoader().loadClass(typeName);
 509.                }
 510.            } catch (ClassNotFoundException cnfe) {
 511.                throw ROOT_LOGGER.couldNotLoadClass(typeName, cnfe);
 512.            }
 513.        }

 482.        private Module getModule(ModelNode moduleNode) {
 483.            Module module;
 484.    
 485.            if (moduleNode.isDefined()) {
 486.                ModuleLoader moduleLoader = Module.getBootModuleLoader();
 487.                try {
 488.                    module = moduleLoader.loadModule(ModuleIdentifier.create(moduleNode.asString()));
 489.                } catch (ModuleLoadException e) {
 490.                    throw ROOT_LOGGER.moduleCouldNotLoad(moduleNode.asString(), e);
 491.                }
 492.            } else {
 493.                // fallback to caller module.
 494.                module = Module.getCallerModule();
 495.            }
 496.    
 497.            return module;
 498.        }

 500.        @SuppressWarnings("unchecked")
 501.        private <T> Class<T> loadClass(ModelNode moduleNode, String typeName) {
 502.            try {
 503.                Module module = getModule(moduleNode);
 504.    
 505.                if (module != null) {
 506.                    return (Class<T>) module.getClassLoader().loadClass(typeName);
 507.                } else {
 508.                    return (Class<T>) getClass().getClassLoader().loadClass(typeName);
 509.                }
 510.            } catch (ClassNotFoundException cnfe) {
 511.                throw ROOT_LOGGER.couldNotLoadClass(typeName, cnfe);
 512.            }
 513.        }
