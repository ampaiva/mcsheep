  78.        public Object lookup(final Name name, boolean dereference) throws NamingException {
  79.            if (name.isEmpty()) {
  80.                return new NamingContext(EMPTY_NAME, this, null);
  81.            }
  82.            final ServiceName lookupName = buildServiceName(name);
  83.            Object obj = lookup(name.toString(), lookupName, dereference);
  84.            if (obj == null) {
  85.                final ServiceName lower = boundServices.lower(lookupName);
  86.                if (lower != null && lower.isParentOf(lookupName)) {
  87.                    // Parent might be a reference or a link
  88.                    obj = lookup(name.toString(), lower, dereference);
  89.                    //if the lower is a context that has been explicitly bound then
  90.                    //we do not return a resolve result, as this will result in an
  91.                    //infinite loop
  92.                    if (!(obj instanceof NamingContext)) {
  93.                        checkReferenceForContinuation(name, obj);
  94.                        return new ResolveResult(obj, suffix(lower, lookupName));
  95.                    }
  96.                }
  97.    
  98.                final ServiceName ceiling = boundServices.ceiling(lookupName);
  99.                if (ceiling != null && lookupName.isParentOf(ceiling)) {
 100.                    if (lookupName.equals(ceiling)) {
 101.                        //the binder service returned null
 102.                        return null;
 103.                    }
 104.                    return new NamingContext((Name) name.clone(), this, null);
 105.                }
 106.                throw new NameNotFoundException(name.toString() + " -- " + lookupName);
 107.            }
 108.    
 109.            return obj;
 110.        }

 281.            final Enumeration<String> parts = name.getAll();
 282.            ServiceName current = serviceNameBase;

 285.                if (!currentPart.isEmpty()) {
 286.                    current = current.append(currentPart);

 294.            CompositeName name = new CompositeName();
 295.            int baseIndex = serviceNameBase.toArray().length;
 296.            for (int i = baseIndex; i < c.length; i++) {
 297.                try {
 298.                    name.add(c[i]);
 299.                } catch (InvalidNameException e) {
 300.                    throw new IllegalStateException(e);
 301.                }
 302.            }
 303.
