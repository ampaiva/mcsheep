 447.         *
 448.         * @param c         The container that the remainder of modules should be defined in.
 449.         * @param previous  The IDL module name, from root to <code>c</code>.
 450.         * @param remainder The java package name, relative to <code>c</code>.
 451.         * @return A reference to the IR module that represents the package.
 452.         */
 453.        private ModuleDefImpl ensurePackageExists(LocalContainer c,
 454.                                                  String previous,
 455.                                                  String remainder)
 456.                throws IRConstructionException {
 457.            if ("".equals(remainder))
 458.                return (ModuleDefImpl) c; // done
 459.    
 460.            int idx = remainder.indexOf('.');
 461.            String base;
 462.    
 463.            if (idx == -1)
 464.                base = remainder;
 465.            else
 466.                base = remainder.substring(0, idx);
 467.            base = Util.javaToIDLName(base);
 468.    
 469.            if (previous.equals(""))
 470.                previous = base;
 471.            else
 472.                previous = previous + "/" + base;
 473.            if (idx == -1)
 474.                remainder = "";
 475.            else
 476.                remainder = remainder.substring(idx + 1);
 477.    
 478.            LocalContainer next = null;
 479.            LocalContained contained = (LocalContained) c._lookup(base);
 480.    
 481.            if (contained instanceof LocalContainer)
 482.                next = (LocalContainer) contained;
 483.            else if (contained != null)
 484.                throw IIOPLogger.ROOT_LOGGER.collisionWhileCreatingPackage();
 485.    
 486.            if (next == null) {
 487.                String id = "IDL:" + previous + ":1.0";
 488.    
 489.                // Create module
 490.                ModuleDefImpl m = new ModuleDefImpl(id, base, "1.0", c, impl);
 491.    
 492.                c.add(base, m);
 493.    
 494.                if (idx == -1)
 495.                    return m; // done
 496.    
 497.                next = (LocalContainer) c._lookup(base); // Better be there now...
 498.            } else // Check that next _is_ a module
 499.                if (next.def_kind() != DefinitionKind.dk_Module)
 500.                    throw IIOPLogger.ROOT_LOGGER.collisionWhileCreatingPackage();
 501.    
 502.            return ensurePackageExists(next, previous, remainder);
 503.        }

 660.        /**
 661.         * Add an array.
 662.         */
 663.        private ValueBoxDefImpl addArray(Class cls)
 664.                throws RMIIIOPViolationException, IRConstructionException {
 665.            if (!cls.isArray())
 666.                throw IIOPLogger.ROOT_LOGGER.classIsNotArray(cls.getName());
 667.    
 668.            ValueBoxDefImpl vbDef;
 669.    
 670.            // Lookup: Has it already been added?
 671.            vbDef = (ValueBoxDefImpl) arrayMap.get(cls);
 672.            if (vbDef != null)
 673.                return vbDef; // Yes, just return it.
 674.    
 675.            int dimensions = 0;
 676.            Class compType = cls;
 677.    
 678.            do {
 679.                compType = compType.getComponentType();
 680.                ++dimensions;
 681.            } while (compType.isArray());
 682.    
 683.            String typeName;
 684.            String moduleName;
 685.            TypeCode typeCode;
 686.    
 687.            if (compType.isPrimitive()) {
 688.                if (compType == Boolean.TYPE) {
 689.                    typeName = "boolean";
 690.                    typeCode = orb.get_primitive_tc(TCKind.tk_boolean);
 691.                } else if (compType == Character.TYPE) {
 692.                    typeName = "wchar";
 693.                    typeCode = orb.get_primitive_tc(TCKind.tk_wchar);
 694.                } else if (compType == Byte.TYPE) {
 695.                    typeName = "octet";
 696.                    typeCode = orb.get_primitive_tc(TCKind.tk_octet);
 697.                } else if (compType == Short.TYPE) {
 698.                    typeName = "short";
 699.                    typeCode = orb.get_primitive_tc(TCKind.tk_short);
 700.                } else if (compType == Integer.TYPE) {
 701.                    typeName = "long";
 702.                    typeCode = orb.get_primitive_tc(TCKind.tk_long);
 703.                } else if (compType == Long.TYPE) {
 704.                    typeName = "long_long";
 705.                    typeCode = orb.get_primitive_tc(TCKind.tk_longlong);
 706.                } else if (compType == Float.TYPE) {
 707.                    typeName = "float";
 708.                    typeCode = orb.get_primitive_tc(TCKind.tk_float);
 709.                } else if (compType == Double.TYPE) {
 710.                    typeName = "double";
 711.                    typeCode = orb.get_primitive_tc(TCKind.tk_double);
 712.                } else {
 713.                    throw IIOPLogger.ROOT_LOGGER.unknownPrimitiveType(compType.getName());
 714.                }
 715.    
 716.                moduleName = "org.omg.boxedRMI";
 717.            } else {
 718.                typeCode = getTypeCode(compType); // map the component type.
 719.    
 720.                if (compType == java.lang.String.class)
 721.                    typeName = getJavaLangString().name();
 722.                else if (compType == java.lang.Object.class)
 723.                    typeName = getJavaLang_Object().name();
 724.                else if (compType == java.lang.Class.class)
 725.                    typeName = getJavaxRmiCORBAClassDesc().name();
 726.                else if (compType == java.io.Serializable.class)
 727.                    typeName = getJavaIoSerializable().name();
 728.                else if (compType == java.io.Externalizable.class)
 729.                    typeName = getJavaIoExternalizable().name();
 730.                else if (compType.isInterface() &&
 731.                        !RmiIdlUtil.isAbstractValueType(compType))
 732.                    typeName = ((InterfaceDefImpl) interfaceMap.get(compType)).name();
 733.                else if (Exception.class.isAssignableFrom(compType)) // exception type
 734.                    typeName = ((ExceptionDefImpl) exceptionMap.get(compType)).name();
 735.                else // must be value type
 736.                    typeName = ((ValueDefImpl) valueMap.get(compType)).name();
 737.    
 738.                moduleName = "org.omg.boxedRMI." + compType.getPackage().getName();
 739.            }
 740.    
 741.            // Get module to add array to.
 742.            ModuleDefImpl m = ensurePackageExists(moduleName);
 743.    
 744.            // Create an array of the types for the dimensions
 745.            Class[] types = new Class[dimensions];
 746.            types[dimensions - 1] = cls;
 747.            for (int i = dimensions - 2; i >= 0; --i)
 748.                types[i] = types[i + 1].getComponentType();
 749.    
 750.            // Create boxed sequences for all dimensions.
 751.            for (int i = 0; i < dimensions; ++i) {
 752.                Class type = types[i];
 753.    
 754.                typeCode = orb.create_sequence_tc(0, typeCode);
 755.                vbDef = (ValueBoxDefImpl) arrayMap.get(type);
 756.                if (vbDef == null) {
 757.                    String id = Util.getIRIdentifierOfClass(type);
 758.    
 759.                    SequenceDefImpl sdi = new SequenceDefImpl(typeCode, impl);
 760.    
 761.                    String name = "seq" + (i + 1) + "_" + typeName;
 762.    //            TypeCode boxTypeCode = new TypeCodeImpl(TCKind._tk_value_box,
 763.    //                                                    id, name, typeCode);
 764.                    TypeCode boxTypeCode = orb.create_value_box_tc(id, name, typeCode);
 765.                    vbDef = new ValueBoxDefImpl(id, name, "1.0", m, boxTypeCode, impl);
 766.    
 767.                    addTypeCode(type, vbDef.type());
 768.                    m.add(name, vbDef);
 769.                    impl.putSequenceImpl(id, typeCode, sdi, vbDef);
 770.    
 771.                    arrayMap.put(type, vbDef); // Remember we mapped this.
 772.    
 773.                    typeCode = boxTypeCode;
 774.                } else
 775.                    typeCode = vbDef.type();
 776.            }
 777.    
 778.            // Return the box of highest dimension.
 779.            return vbDef;
 780.        }

 783.         * Add an interface.
 784.         */
 785.        private InterfaceDefImpl addInterface(InterfaceAnalysis ia)
 786.                throws RMIIIOPViolationException, IRConstructionException {
 787.            InterfaceDefImpl iDef;
 788.            Class cls = ia.getCls();
 789.    
 790.            // Lookup: Has it already been added?
 791.            iDef = (InterfaceDefImpl) interfaceMap.get(cls);
 792.            if (iDef != null)
 793.                return iDef; // Yes, just return it.
 794.    
 795.            // Get module to add interface to.
 796.            ModuleDefImpl m = ensurePackageExists(cls.getPackage().getName());
 797.    
 798.            // Add superinterfaces
 799.            String[] base_interfaces = addInterfaces(ia);
 800.    
 801.            // Create the interface
 802.            String base = cls.getName();
 803.            base = base.substring(base.lastIndexOf('.') + 1);
 804.            base = Util.javaToIDLName(base);
 805.    
 806.            iDef = new InterfaceDefImpl(ia.getRepositoryId(),
 807.                    base, "1.0", m,
 808.                    base_interfaces, impl);
 809.            addTypeCode(cls, iDef.type());
 810.            m.add(base, iDef);
 811.            interfaceMap.put(cls, iDef); // Remember we mapped this.
 812.    
 813.            // Fill in constants
 814.            addConstants(iDef, ia);
 815.    
 816.            // Add attributes
 817.            addAttributes(iDef, ia);
 818.    
 819.            // Fill in operations
 820.            addOperations(iDef, ia);
 821.    
 822.            return iDef;
 823.        }

 831.            Class cls = va.getCls();

 836.                return vDef; // Yes, just return it.
 837.

 839.            ModuleDefImpl m = ensurePackageExists(cls.getPackage().getName());

 842.            String[] supported_interfaces = addInterfaces(va);
 843.    
 844.            // Add abstract base valuetypes
 845.            String[] abstract_base_valuetypes = addAbstractBaseValuetypes(va);

 847.            // Add superclass
 848.            ValueDefImpl superValue = null;
 849.            ValueAnalysis superAnalysis = va.getSuperAnalysis();

 851.                superValue = addValue(superAnalysis);

 854.            String base = cls.getName();
 855.            base = base.substring(base.lastIndexOf('.') + 1);
 856.            base = Util.javaToIDLName(base);
 857.    
 858.            TypeCode baseTypeCode;
 859.            if (superValue == null)
 860.                baseTypeCode = orb.get_primitive_tc(TCKind.tk_null);
 861.            else
 862.                baseTypeCode = superValue.type();

 864.            vDef = new ValueDefImpl(va.getRepositoryId(), base, "1.0",
 865.                    m,
 866.                    va.isAbstractValue(),
 867.                    va.isCustom(),
 868.                    supported_interfaces,
 869.                    abstract_base_valuetypes,
 870.                    baseTypeCode,
 871.                    impl);
 872.            addTypeCode(cls, vDef.type());
 873.            m.add(base, vDef);
 874.            valueMap.put(cls, vDef); // Remember we mapped this.
 875.    
 876.            // Fill in constants.
 877.            addConstants(vDef, va);
 878.    
 879.            // Add value members
 880.            ValueMemberAnalysis[] vmas = va.getMembers();
 881.            for (int i = 0; i < vmas.length; ++i) {
 882.                ValueMemberDefImpl vmDef;

 884.                String vmName = vmas[i].getIDLName();
 885.    
 886.                Class vmCls = vmas[i].getCls();
 887.                TypeCode typeCode = getTypeCode(vmCls);
 888.    
 889.                boolean vmPublic = vmas[i].isPublic();
 890.    
 891.                vmDef = new ValueMemberDefImpl(vmid, vmName, "1.0",
 892.                        typeCode, vmPublic, vDef, impl);

 894.            }
 895.    
 896.            // Add attributes
 897.            addAttributes(vDef, va);
 898.    
 899.            // TODO: Fill in operations.
 900.    
 901.            return vDef;
 902.        }

 217.                throws IRConstructionException {
 218.            if (cls == null)
 219.                throw IIOPLogger.ROOT_LOGGER.invalidNullClass();
 220.    
 221.            TypeCode ret = constantTypeCodeMap.get(cls);
 222.

 228.        /**
 229.         * Returns the TypeCode IDL TypeCodes for parameter, result, attribute
 230.         * and value member types.
 231.         * This may provoke a mapping of the class argument.
 232.         * <p/>
 233.         * Exception classes map to both values and exceptions. For these, this
 234.         * method returns the typecode for the value, and you can use the
 235.         * <code>getExceptionTypeCode</code> TODO method to get the typecode for the
 236.         * mapping to exception.
 237.         *
 238.         * @param cls The Java class denoting the java type.
 239.         */
 240.        private TypeCode getTypeCode(Class cls)
 241.                throws IRConstructionException, RMIIIOPViolationException {
 242.            if (cls == null)
 243.                throw IIOPLogger.ROOT_LOGGER.invalidNullClass();
 244.    
 245.            TypeCode ret = (TypeCode) typeCodeMap.get(cls);
 246.    
 247.            if (ret == null) {
 248.                if (cls == java.lang.String.class)
 249.                    ret = getJavaLangString().type();
 250.                else if (cls == java.lang.Object.class)
 251.                    ret = getJavaLang_Object().type();
 252.                else if (cls == java.lang.Class.class)
 253.                    ret = getJavaxRmiCORBAClassDesc().type();
 254.                else if (cls == java.io.Serializable.class)
 255.                    ret = getJavaIoSerializable().type();
 256.                else if (cls == java.io.Externalizable.class)
 257.                    ret = getJavaIoExternalizable().type();
 258.                else {
 259.                    // Try adding a mapping of the the class to the IR
 260.                    addClass(cls);
 261.    
 262.                    // Lookup again, it should be there now.
 263.                    ret = (TypeCode) typeCodeMap.get(cls);
 264.    
 265.                    if (ret == null)
 266.                        throw IIOPLogger.ROOT_LOGGER.unknownTypeCodeForClass(cls.getName());
 267.                    else
 268.                        return ret;
 269.                }
 270.    
 271.                typeCodeMap.put(cls, ret);
 272.            }
 273.    
 274.            return ret;
 275.        }

 452.         */
 453.        private ModuleDefImpl ensurePackageExists(LocalContainer c,
 454.                                                  String previous,
 455.                                                  String remainder)
 456.                throws IRConstructionException {
 457.            if ("".equals(remainder))
 458.                return (ModuleDefImpl) c; // done
 459.    
 460.            int idx = remainder.indexOf('.');
 461.            String base;
 462.    
 463.            if (idx == -1)
 464.                base = remainder;

 469.            if (previous.equals(""))
 470.                previous = base;
 471.            else
 472.                previous = previous + "/" + base;
 473.            if (idx == -1)
 474.                remainder = "";
 475.            else
 476.                remainder = remainder.substring(idx + 1);

 486.            if (next == null) {
 487.                String id = "IDL:" + previous + ":1.0";

 490.                ModuleDefImpl m = new ModuleDefImpl(id, base, "1.0", c, impl);
 491.    
 492.                c.add(base, m);
 493.    
 494.                if (idx == -1)
 495.                    return m; // done
 496.    
 497.                next = (LocalContainer) c._lookup(base); // Better be there now...
 498.            } else // Check that next _is_ a module
 499.                if (next.def_kind() != DefinitionKind.dk_Module)

 708.                    typeCode = orb.get_primitive_tc(TCKind.tk_float);
 709.                } else if (compType == Double.TYPE) {
 710.                    typeName = "double";
 711.                    typeCode = orb.get_primitive_tc(TCKind.tk_double);
 712.                } else {
 713.                    throw IIOPLogger.ROOT_LOGGER.unknownPrimitiveType(compType.getName());
 714.                }
 715.    
 716.                moduleName = "org.omg.boxedRMI";
 717.            } else {
 718.                typeCode = getTypeCode(compType); // map the component type.
 719.    
 720.                if (compType == java.lang.String.class)
 721.                    typeName = getJavaLangString().name();
 722.                else if (compType == java.lang.Object.class)
 723.                    typeName = getJavaLang_Object().name();
 724.                else if (compType == java.lang.Class.class)
 725.                    typeName = getJavaxRmiCORBAClassDesc().name();
 726.                else if (compType == java.io.Serializable.class)
 727.                    typeName = getJavaIoSerializable().name();
 728.                else if (compType == java.io.Externalizable.class)
 729.                    typeName = getJavaIoExternalizable().name();
 730.                else if (compType.isInterface() &&
 731.                        !RmiIdlUtil.isAbstractValueType(compType))
 732.                    typeName = ((InterfaceDefImpl) interfaceMap.get(compType)).name();
 733.                else if (Exception.class.isAssignableFrom(compType)) // exception type
 734.                    typeName = ((ExceptionDefImpl) exceptionMap.get(compType)).name();
 735.                else // must be value type
 736.                    typeName = ((ValueDefImpl) valueMap.get(compType)).name();
 737.    
 738.                moduleName = "org.omg.boxedRMI." + compType.getPackage().getName();
 739.            }
 740.    
 741.            // Get module to add array to.
 742.            ModuleDefImpl m = ensurePackageExists(moduleName);
 743.    
 744.            // Create an array of the types for the dimensions
 745.            Class[] types = new Class[dimensions];
 746.            types[dimensions - 1] = cls;
 747.            for (int i = dimensions - 2; i >= 0; --i)
 748.                types[i] = types[i + 1].getComponentType();
 749.    
 750.            // Create boxed sequences for all dimensions.
 751.            for (int i = 0; i < dimensions; ++i) {
 752.                Class type = types[i];
 753.    
 754.                typeCode = orb.create_sequence_tc(0, typeCode);
 755.                vbDef = (ValueBoxDefImpl) arrayMap.get(type);
 756.                if (vbDef == null) {
 757.                    String id = Util.getIRIdentifierOfClass(type);
 758.    
 759.                    SequenceDefImpl sdi = new SequenceDefImpl(typeCode, impl);
 760.    
 761.                    String name = "seq" + (i + 1) + "_" + typeName;
 762.    //            TypeCode boxTypeCode = new TypeCodeImpl(TCKind._tk_value_box,
 763.    //                                                    id, name, typeCode);
 764.                    TypeCode boxTypeCode = orb.create_value_box_tc(id, name, typeCode);
 765.                    vbDef = new ValueBoxDefImpl(id, name, "1.0", m, boxTypeCode, impl);
 766.    
 767.                    addTypeCode(type, vbDef.type());
 768.                    m.add(name, vbDef);
 769.                    impl.putSequenceImpl(id, typeCode, sdi, vbDef);
 770.    
 771.                    arrayMap.put(type, vbDef); // Remember we mapped this.
 772.    
 773.                    typeCode = boxTypeCode;
 774.                } else
 775.                    typeCode = vbDef.type();
 776.            }
 777.    
 778.            // Return the box of highest dimension.
 779.            return vbDef;
 780.        }

 782.        /**
 783.         * Add an interface.
 784.         */
 785.        private InterfaceDefImpl addInterface(InterfaceAnalysis ia)
 786.                throws RMIIIOPViolationException, IRConstructionException {
 787.            InterfaceDefImpl iDef;
 788.            Class cls = ia.getCls();
 789.    
 790.            // Lookup: Has it already been added?
 791.            iDef = (InterfaceDefImpl) interfaceMap.get(cls);
 792.            if (iDef != null)
 793.                return iDef; // Yes, just return it.
 794.    
 795.            // Get module to add interface to.
 796.            ModuleDefImpl m = ensurePackageExists(cls.getPackage().getName());
 797.    
 798.            // Add superinterfaces
 799.            String[] base_interfaces = addInterfaces(ia);
 800.    
 801.            // Create the interface
 802.            String base = cls.getName();
 803.            base = base.substring(base.lastIndexOf('.') + 1);
 804.            base = Util.javaToIDLName(base);
 805.    
 806.            iDef = new InterfaceDefImpl(ia.getRepositoryId(),
 807.                    base, "1.0", m,
 808.                    base_interfaces, impl);
 809.            addTypeCode(cls, iDef.type());
 810.            m.add(base, iDef);
 811.            interfaceMap.put(cls, iDef); // Remember we mapped this.
 812.    
 813.            // Fill in constants
 814.            addConstants(iDef, ia);
 815.    
 816.            // Add attributes
 817.            addAttributes(iDef, ia);
 818.    
 819.            // Fill in operations
 820.            addOperations(iDef, ia);
 821.    
 822.            return iDef;
 823.        }

 825.        /**
 826.         * Add a value type.
 827.         */
 828.        private ValueDefImpl addValue(ValueAnalysis va)
 829.                throws RMIIIOPViolationException, IRConstructionException {
 830.            ValueDefImpl vDef;
 831.            Class cls = va.getCls();
 832.    
 833.            // Lookup: Has it already been added?
 834.            vDef = (ValueDefImpl) valueMap.get(cls);
 835.            if (vDef != null)
 836.                return vDef; // Yes, just return it.
 837.    
 838.            // Get module to add value to.
 839.            ModuleDefImpl m = ensurePackageExists(cls.getPackage().getName());
 840.    
 841.            // Add implemented interfaces
 842.            String[] supported_interfaces = addInterfaces(va);
 843.    
 844.            // Add abstract base valuetypes
 845.            String[] abstract_base_valuetypes = addAbstractBaseValuetypes(va);
 846.    
 847.            // Add superclass
 848.            ValueDefImpl superValue = null;
 849.            ValueAnalysis superAnalysis = va.getSuperAnalysis();
 850.            if (superAnalysis != null)
 851.                superValue = addValue(superAnalysis);
 852.    
 853.            // Create the value
 854.            String base = cls.getName();
 855.            base = base.substring(base.lastIndexOf('.') + 1);
 856.            base = Util.javaToIDLName(base);
 857.    
 858.            TypeCode baseTypeCode;
 859.            if (superValue == null)
 860.                baseTypeCode = orb.get_primitive_tc(TCKind.tk_null);
 861.            else
 862.                baseTypeCode = superValue.type();
 863.    
 864.            vDef = new ValueDefImpl(va.getRepositoryId(), base, "1.0",
 865.                    m,
 866.                    va.isAbstractValue(),
 867.                    va.isCustom(),
 868.                    supported_interfaces,
 869.                    abstract_base_valuetypes,
 870.                    baseTypeCode,
 871.                    impl);
 872.            addTypeCode(cls, vDef.type());
 873.            m.add(base, vDef);
 874.            valueMap.put(cls, vDef); // Remember we mapped this.
 875.    
 876.            // Fill in constants.
 877.            addConstants(vDef, va);
 878.    
 879.            // Add value members
 880.            ValueMemberAnalysis[] vmas = va.getMembers();
 881.            for (int i = 0; i < vmas.length; ++i) {
 882.                ValueMemberDefImpl vmDef;
 883.                String vmid = va.getMemberRepositoryId(vmas[i].getJavaName());
 884.                String vmName = vmas[i].getIDLName();
 885.    
 886.                Class vmCls = vmas[i].getCls();
 887.                TypeCode typeCode = getTypeCode(vmCls);
 888.    
 889.                boolean vmPublic = vmas[i].isPublic();
 890.    
 891.                vmDef = new ValueMemberDefImpl(vmid, vmName, "1.0",
 892.                        typeCode, vmPublic, vDef, impl);
 893.                vDef.add(vmName, vmDef);
 894.            }
 895.    
 896.            // Add attributes
 897.            addAttributes(vDef, va);
 898.    
 899.            // TODO: Fill in operations.
 900.    
 901.            return vDef;
 902.        }
