 132.                        }
 133.                    }
 134.                }
 135.            }
 136.            //now allow the annotations to be overridden by the DD
 137.            final SessionBeanMetaData descriptorData = description.getDescriptorData();
 138.            if (descriptorData != null) {
 139.    
 140.                if (descriptorData.getHome() != null) {
 141.                    home = descriptorData.getHome();
 142.                }
 143.                if (descriptorData.getLocalHome() != null) {
 144.                    localHome = descriptorData.getLocalHome();
 145.                }
 146.            }
 147.            if (localHome != null) {
 148.                description.addLocalHome(localHome);
 149.            }
 150.            if (home != null) {
 151.                description.addRemoteHome(home);
 152.            }
 153.            // finally see if we have to infer the remote or local interface from the home/local home views, respectively
 154.            if (description.getEjbHomeView() != null && description.getEjbRemoteView() == null) {
 155.                final String remoteClassName = this.inferRemoteInterfaceFromHome(description.getEjbHomeView().getViewClassName(), module, deploymentReflectionIndex, description);
 156.                description.addEjbObjectView(remoteClassName);
 157.            }
 158.            if (description.getEjbLocalHomeView() != null && description.getEjbLocalView() == null) {
 159.                final String localClassName = this.inferLocalInterfaceFromLocalHome(description.getEjbLocalHomeView().getViewClassName(), module, deploymentReflectionIndex, description);
 160.                description.addEjbLocalObjectView(localClassName);
 161.            }
 162.        }

 165.        private String inferRemoteInterfaceFromHome(final String homeClassName, final Module module, final DeploymentReflectionIndex deploymentReflectionIndex, final SessionBeanComponentDescription description) throws ClassNotFoundException, DeploymentUnitProcessingException {
 166.            final Class homeClass = module.getClassLoader().loadClass(homeClassName);
 167.            final ClassReflectionIndex index = deploymentReflectionIndex.getClassIndex(homeClass);
 168.            Class remote = null;
 169.            for (final Method method : index.getMethods()) {
 170.                if (method.getName().startsWith("create")) {
 171.                    if (remote != null && remote != method.getReturnType()) {
 172.                        throw EjbLogger.ROOT_LOGGER.multipleCreateMethod(homeClass);
 173.                    }
 174.                    remote = method.getReturnType();
 175.                }
 176.            }
 177.            if(remote == null) {
 178.                throw EjbLogger.ROOT_LOGGER.couldNotDetermineRemoteInterfaceFromHome(homeClassName, description.getEJBName());
 179.            }
 180.            return remote.getName();
 181.        }

 183.        private String inferLocalInterfaceFromLocalHome(final String localHomeClassName, final Module module, final DeploymentReflectionIndex deploymentReflectionIndex, final SessionBeanComponentDescription description) throws ClassNotFoundException, DeploymentUnitProcessingException {
 184.            final Class localHomeClass = module.getClassLoader().loadClass(localHomeClassName);
 185.            final ClassReflectionIndex index = deploymentReflectionIndex.getClassIndex(localHomeClass);
 186.            Class localClass = null;
 187.            for (final Method method : index.getMethods()) {
 188.                if (method.getName().startsWith("create")) {
 189.                    if (localClass != null && localClass != method.getReturnType()) {
 190.                        throw EjbLogger.ROOT_LOGGER.multipleCreateMethod(localHomeClass);
 191.                    }
 192.                    localClass = method.getReturnType();
 193.                }
 194.            }
 195.            if (localClass == null) {
 196.                throw EjbLogger.ROOT_LOGGER.couldNotDetermineLocalInterfaceFromLocalHome(localHomeClassName, description.getEJBName());
 197.            }
 198.            return localClass.getName();
 199.        }
