  55.    /**
  56.     * The webservices extension.
  57.     *
  58.     * @author alessio.soldano@jboss.com

  63.    public final class WSExtension implements Extension {
  64.    
  65.        static final PathElement ENDPOINT_PATH = PathElement.pathElement(ENDPOINT);
  66.        static final PathElement CLIENT_CONFIG_PATH = PathElement.pathElement(CLIENT_CONFIG);
  67.        static final PathElement WSDL_URI_SCHEME_PATH = PathElement.pathElement(WSDL_URI_SCHEME);
  68.        static final PathElement ENDPOINT_CONFIG_PATH = PathElement.pathElement(ENDPOINT_CONFIG);
  69.        private static final PathElement PROPERTY_PATH = PathElement.pathElement(PROPERTY);
  70.        static final PathElement PRE_HANDLER_CHAIN_PATH = PathElement.pathElement(PRE_HANDLER_CHAIN);
  71.        static final PathElement POST_HANDLER_CHAIN_PATH = PathElement.pathElement(POST_HANDLER_CHAIN);
  72.        static final PathElement HANDLER_PATH = PathElement.pathElement(HANDLER);
  73.        public static final String SUBSYSTEM_NAME = "webservices";
  74.        static final PathElement SUBSYSTEM_PATH = PathElement.pathElement(SUBSYSTEM, SUBSYSTEM_NAME);
  75.        private static final String RESOURCE_NAME = WSExtension.class.getPackage().getName() + ".LocalDescriptions";
  76.    
  77.        private static final ModelVersion CURRENT_MODEL_VERSION = ModelVersion.create(2, 0, 0);
  78.    
  79.        // attributes on the endpoint element
  80.         static final AttributeDefinition ENDPOINT_WSDL = new SimpleAttributeDefinitionBuilder(
  81.                Constants.ENDPOINT_WSDL, ModelType.STRING, false)
  82.                .setStorageRuntime()
  83.                .build();
  84.        static final AttributeDefinition ENDPOINT_CLASS = new SimpleAttributeDefinitionBuilder(
  85.                Constants.ENDPOINT_CLASS, ModelType.STRING, false)
  86.                .setStorageRuntime()
  87.                .build();
  88.        static final AttributeDefinition ENDPOINT_CONTEXT = new SimpleAttributeDefinitionBuilder(
  89.                Constants.ENDPOINT_CONTEXT, ModelType.STRING, false)
  90.                .setStorageRuntime()
  91.                .build();
  92.        static final AttributeDefinition ENDPOINT_TYPE = new SimpleAttributeDefinitionBuilder(
  93.                Constants.ENDPOINT_TYPE, ModelType.STRING, true)
  94.                .setStorageRuntime()
  95.                .build();
  96.        static final AttributeDefinition ENDPOINT_NAME = new SimpleAttributeDefinitionBuilder(
  97.                Constants.ENDPOINT_NAME, ModelType.STRING, false)
  98.                .setStorageRuntime()
  99.                .build();
 100.    
 101.    
 102.        static StandardResourceDescriptionResolver getResourceDescriptionResolver(final String... keyPrefix) {
 103.            StringBuilder prefix = new StringBuilder(SUBSYSTEM_NAME);
 104.            for (String kp : keyPrefix) {
 105.                prefix.append('.').append(kp);
 106.            }
 107.            return new StandardResourceDescriptionResolver(prefix.toString(), RESOURCE_NAME, WSExtension.class.getClassLoader(), true, false);
 108.        }
 109.    
 110.        @Override
 111.        public void initialize(final ExtensionContext context) {
 112.            final boolean registerRuntimeOnly = context.isRuntimeOnlyRegistrationValid();
 113.            final SubsystemRegistration subsystem = context.registerSubsystem(SUBSYSTEM_NAME, CURRENT_MODEL_VERSION);
 114.            subsystem.registerXMLElementWriter(WSSubsystemWriter.getInstance());
 115.            // ws subsystem
 116.            ResourceBuilder propertyResource = ResourceBuilder.Factory.create(PROPERTY_PATH, getResourceDescriptionResolver(Constants.PROPERTY))
 117.                    .setAddOperation(PropertyAdd.INSTANCE)
 118.                    .setRemoveOperation(ReloadRequiredRemoveStepHandler.INSTANCE)
 119.                    .addReadWriteAttribute(Attributes.VALUE, null, new ReloadRequiredWriteAttributeHandler(Attributes.VALUE));
 120.    
 121.    
 122.            ResourceBuilder handlerBuilder = ResourceBuilder.Factory.create(HANDLER_PATH, getResourceDescriptionResolver(HANDLER))
 123.                    .setAddOperation(HandlerAdd.INSTANCE)
 124.                    .setRemoveOperation(ReloadRequiredRemoveStepHandler.INSTANCE)
 125.                    .addReadWriteAttribute(Attributes.CLASS, null, new ReloadRequiredWriteAttributeHandler(Attributes.CLASS));
 126.    
 127.            ResourceBuilder preHandler = ResourceBuilder.Factory.create(PRE_HANDLER_CHAIN_PATH, getResourceDescriptionResolver(Constants.PRE_HANDLER_CHAIN))
 128.                    .setAddOperation(HandlerChainAdd.INSTANCE)
 129.                    .setRemoveOperation(ReloadRequiredRemoveStepHandler.INSTANCE)
 130.                    .addReadWriteAttribute(Attributes.PROTOCOL_BINDINGS, null, new ReloadRequiredWriteAttributeHandler(Attributes.PROTOCOL_BINDINGS))
 131.                    .pushChild(handlerBuilder).pop();
 132.    
 133.            ResourceBuilder postHandler = ResourceBuilder.Factory.create(POST_HANDLER_CHAIN_PATH, getResourceDescriptionResolver(Constants.POST_HANDLER_CHAIN))
 134.                    .setAddOperation(HandlerChainAdd.INSTANCE)
 135.                    .setRemoveOperation(ReloadRequiredRemoveStepHandler.INSTANCE)
 136.                    .addReadWriteAttribute(Attributes.PROTOCOL_BINDINGS, null, new ReloadRequiredWriteAttributeHandler(Attributes.PROTOCOL_BINDINGS))
 137.                    .pushChild(handlerBuilder).pop();
 138.    
 139.            ResourceDefinition epConfigsDef = ResourceBuilder.Factory.create(ENDPOINT_CONFIG_PATH, getResourceDescriptionResolver(ENDPOINT_CONFIG))
 140.                    .setAddOperation(EndpointConfigAdd.INSTANCE)
 141.                    .setRemoveOperation(ReloadRequiredRemoveStepHandler.INSTANCE)
 142.                    .pushChild(propertyResource).pop()
 143.                    .pushChild(preHandler).pop()
 144.                    .pushChild(postHandler).pop()
 145.                    .build();
 146.    
 147.            ResourceDefinition clConfigsDef = ResourceBuilder.Factory.create(CLIENT_CONFIG_PATH, getResourceDescriptionResolver(CLIENT_CONFIG))
 148.                  .setAddOperation(ClientConfigAdd.INSTANCE)
 149.                  .setRemoveOperation(ReloadRequiredRemoveStepHandler.INSTANCE)
 150.                  .pushChild(propertyResource).pop()
 151.                  .pushChild(preHandler).pop()
 152.                  .pushChild(postHandler).pop()
 153.                  .build();
 154.    
 155.            ResourceDefinition subsystemResource = ResourceBuilder.Factory.createSubsystemRoot(SUBSYSTEM_PATH, getResourceDescriptionResolver(), WSSubsystemAdd.INSTANCE, WSSubsystemRemove.INSTANCE)
 156.                    .addReadWriteAttribute(Attributes.WSDL_HOST, null, new WSServerConfigAttributeHandler(Attributes.WSDL_HOST))
 157.                    .addReadWriteAttribute(Attributes.WSDL_PORT, null, new WSServerConfigAttributeHandler(Attributes.WSDL_PORT))
 158.                    .addReadWriteAttribute(Attributes.WSDL_SECURE_PORT, null, new WSServerConfigAttributeHandler(Attributes.WSDL_SECURE_PORT))
 159.                    .addReadWriteAttribute(Attributes.WSDL_URI_SCHEME, null, new WSServerConfigAttributeHandler(Attributes.WSDL_URI_SCHEME))
 160.                    .addReadWriteAttribute(Attributes.WSDL_PATH_REWRITE_RULE, null, new WSServerConfigAttributeHandler(Attributes.WSDL_PATH_REWRITE_RULE))
 161.                    .addReadWriteAttribute(Attributes.MODIFY_WSDL_ADDRESS, null, new WSServerConfigAttributeHandler(Attributes.MODIFY_WSDL_ADDRESS))
 162.                    .addReadWriteAttribute(Attributes.STATISTICS_ENABLED, null, new WSServerConfigAttributeHandler(Attributes.STATISTICS_ENABLED))
 163.                    .build();
 164.            ManagementResourceRegistration subsystemRegistration = subsystem.registerSubsystemModel(subsystemResource);
 165.            subsystemRegistration.registerSubModel(epConfigsDef);
 166.            subsystemRegistration.registerSubModel(clConfigsDef);
 167.    
 168.            if (registerRuntimeOnly) {
 169.                subsystem.registerDeploymentModel(ResourceBuilder.Factory.create(SUBSYSTEM_PATH, getResourceDescriptionResolver("deployment"))
 170.                        .pushChild(ENDPOINT_PATH)
 171.                        .addMetrics(WSEndpointMetrics.INSTANCE, WSEndpointMetrics.ATTRIBUTES)
 172.                        .addReadOnlyAttribute(ENDPOINT_CLASS)
 173.                        .addReadOnlyAttribute(ENDPOINT_CONTEXT)
 174.                        .addReadOnlyAttribute(ENDPOINT_NAME)
 175.                        .addReadOnlyAttribute(ENDPOINT_TYPE)
 176.                        .addReadOnlyAttribute(ENDPOINT_WSDL)
 177.                        .build());
 178.            }
 179.    
 180.            if (context.isRegisterTransformers()) {
 181.                registerTransformers1_1_0(subsystem);
 182.                registerTransformers1_2_0(subsystem);
 183.            }
 184.        }

 187.        @Override
 188.        public void initializeParsers(final ExtensionParsingContext context) {
 189.            context.setSubsystemXmlMapping(SUBSYSTEM_NAME, Namespace.WEBSERVICES_1_0.getUriString(), WSSubsystemLegacyReader.getInstance());
 190.            context.setSubsystemXmlMapping(SUBSYSTEM_NAME, Namespace.WEBSERVICES_1_1.getUriString(), WSSubsystem11Reader.getInstance());
 191.            context.setSubsystemXmlMapping(SUBSYSTEM_NAME, Namespace.WEBSERVICES_1_2.getUriString(), WSSubSystem12Reader.getInstance());
 192.            context.setSubsystemXmlMapping(SUBSYSTEM_NAME, Namespace.WEBSERVICES_2_0.getUriString(), WSSubSystem20Reader.getInstance());
 193.        }
 194.    
 195.        private void registerTransformers1_1_0(SubsystemRegistration registration) {
 196.            ModelVersion version = ModelVersion.create(1, 1, 0);
 197.    
 198.            ResourceTransformationDescriptionBuilder builder = TransformationDescriptionBuilder.Factory.createSubsystemInstance();
 199.            builder.getAttributeBuilder().setDiscard(DiscardAttributeChecker.ALWAYS, Attributes.STATISTICS_ENABLED);
 200.            builder.getAttributeBuilder().addRejectCheck(RejectAttributeChecker.SIMPLE_EXPRESSIONS, Attributes.SUBSYSTEM_ATTRIBUTES).end();
 201.            builder.getAttributeBuilder().setDiscard(DiscardAttributeChecker.ALWAYS, Attributes.WSDL_URI_SCHEME);
 202.            builder.rejectChildResource(CLIENT_CONFIG_PATH);
 203.    
 204.            ResourceTransformationDescriptionBuilder endpoint = builder.addChildResource(ENDPOINT_CONFIG_PATH);
 205.            endpoint.addChildResource(PRE_HANDLER_CHAIN_PATH).getAttributeBuilder().addRejectCheck(RejectAttributeChecker.SIMPLE_EXPRESSIONS, Attributes.PROTOCOL_BINDINGS).end();
 206.            endpoint.addChildResource(POST_HANDLER_CHAIN_PATH).getAttributeBuilder().addRejectCheck(RejectAttributeChecker.SIMPLE_EXPRESSIONS, Attributes.PROTOCOL_BINDINGS).end();
 207.    
 208.            TransformationDescription.Tools.register(builder.build(), registration, version);
 209.        }
 210.    
 211.        private void registerTransformers1_2_0(SubsystemRegistration registration) {
 212.            ModelVersion version = ModelVersion.create(1, 2, 0);
 213.            ResourceTransformationDescriptionBuilder builder = TransformationDescriptionBuilder.Factory.createSubsystemInstance();
 214.            builder.getAttributeBuilder().setDiscard(DiscardAttributeChecker.ALWAYS, Attributes.STATISTICS_ENABLED);
 215.            builder.getAttributeBuilder().setDiscard(DiscardAttributeChecker.ALWAYS, Attributes.WSDL_URI_SCHEME);
 216.            builder.getAttributeBuilder().setDiscard(DiscardAttributeChecker.ALWAYS, Attributes.WSDL_PATH_REWRITE_RULE);
 217.            TransformationDescription.Tools.register(builder.build(), registration, version);
 218.        }
