 321.            } else if (type.equals("long") || type.equals("java.lang.Long")) {

 323.            } else if (type.equals("float") || type.equals("java.lang.Float")) {

 325.            } else if (type.equals("double") || type.equals("java.lang.Double")) {
 326.                return Double.parseDouble(value);
 327.            } else if (type.equals("boolean") || type.equals("java.lang.Boolean")) {
 328.                return Boolean.parseBoolean(value);

 310.        private Object coerceToType(final String value, final String type) throws OperationFailedExcepti...
 311.            if (type == null || type.isEmpty() || type.equals(String.class.getName())) {
 312.                return value;
 313.            } else if (type.equals("char") || type.equals("java.lang.Character")) {
 314.                return value.charAt(0);
 315.            } else if (type.equals("byte") || type.equals("java.lang.Byte")) {
 316.                return Byte.parseByte(value);
 317.            } else if (type.equals("short") || type.equals("java.lang.Short")) {
 318.                return Short.parseShort(value);
 319.            } else if (type.equals("int") || type.equals("java.lang.Integer")) {
 320.                return Integer.parseInt(value);
 321.            } else if (type.equals("long") || type.equals("java.lang.Long")) {
 322.                return Long.parseLong(value);
 323.            } else if (type.equals("float") || type.equals("java.lang.Float")) {
 324.                return Float.parseFloat(value);
 325.            } else if (type.equals("double") || type.equals("java.lang.Double")) {
 326.                return Double.parseDouble(value);
 327.            } else if (type.equals("boolean") || type.equals("java.lang.Boolean")) {
 328.                return Boolean.parseBoolean(value);
 329.            } else if (type.equals(URL.class.getName())) {
 330.                try {
 331.                    return new URL(value);
 332.                } catch (MalformedURLException e) {
 333.                    throw NamingLogger.ROOT_LOGGER.unableToTransformURLBindingValue(value, e);
 334.                }
 335.            } else {
 336.                throw NamingLogger.ROOT_LOGGER.unsupportedSimpleBindingType(type);
 337.            }
 338.    
 339.        }
