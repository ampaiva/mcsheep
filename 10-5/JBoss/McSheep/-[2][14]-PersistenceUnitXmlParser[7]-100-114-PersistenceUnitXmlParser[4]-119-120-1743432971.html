  44.    /**
  45.     * Parse a persistence.xml into a list of persistence unit definitions.
  46.     *
  47.     * @author Scott Marlow
  48.     */
  49.    public class PersistenceUnitXmlParser extends MetaDataElementParser {
  50.        // cache the trace enabled flag
  51.        private static final boolean traceEnabled = JPA_LOGGER.isTraceEnabled();
  52.    
  53.        public static PersistenceUnitMetadataHolder parse(final XMLStreamReader reader, final PropertyRe...
  54.    
  55.            reader.require(START_DOCUMENT, null, null);  // check for a bogus document and throw error
  56.    
  57.            // Read until the first start element
  58.            Version version = null;
  59.            while (reader.hasNext() && reader.next() != START_ELEMENT) {
  60.                if (reader.getEventType() == DTD) {
  61.                    final String dtdLocation = readDTDLocation(reader);
  62.                    if (dtdLocation != null) {
  63.                        version = Version.forLocation(dtdLocation);
  64.                    }
  65.                }
  66.            }
  67.            final String schemaLocation = readSchemaLocation(reader);
  68.            if (schemaLocation != null) {
  69.                version = Version.forLocation(schemaLocation);
  70.            }
  71.            if (version == null || Version.UNKNOWN.equals(version)) {
  72.                // Look at the version attribute
  73.                String versionString = null;
  74.                final int count = reader.getAttributeCount();
  75.                for (int i = 0; i < count; i++) {
  76.                    final String attributeNamespace = reader.getAttributeNamespace(i);
  77.                    if (attributeNamespace != null && !attributeNamespace.isEmpty()) {
  78.                        continue;
  79.                    }
  80.                    final Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
  81.                    if (attribute == Attribute.VERSION) {
  82.                        versionString = reader.getAttributeValue(i);
  83.                    }
  84.                }
  85.                if ("1.0".equals(versionString)) {
  86.                    version = Version.JPA_1_0;
  87.                } else if ("1".equals(versionString)) {
  88.                    version = Version.JPA_1_0;
  89.                } else if ("2.0".equals(versionString)) {
  90.                    version = Version.JPA_2_0;
  91.                } else if ("2.1".equals(versionString)) {
  92.                    version = Version.JPA_2_1;
  93.                } else if ("2".equals(versionString)) {
  94.                    version = Version.JPA_2_0;
  95.                } else {
  96.                    version = Version.JPA_2_1;
  97.                }
  98.            }
  99.    
 100.            final int count = reader.getAttributeCount();
 101.            for (int i = 0; i < count; i++) {
 102.                final String value = reader.getAttributeValue(i);
 103.                final String attributeNamespace = reader.getAttributeNamespace(i);
 104.                if (attributeNamespace != null && !attributeNamespace.isEmpty()) {
 105.                    continue;
 106.                }
 107.                final Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
 108.                switch (attribute) {
 109.                    case VERSION:
 110.                        // log.info("version = " + value);
 111.                        // TODO:  handle version
 112.                        break;
 113.                    default:
 114.                        throw unexpectedAttribute(reader, i);
 115.                }
 116.            }
 117.            final List<PersistenceUnitMetadata> PUs = new ArrayList<PersistenceUnitMetadata>();
 118.            // until the ending PERSISTENCE tag
 119.            while (reader.hasNext() && reader.nextTag() != END_ELEMENT) {
 120.                final Element element = Element.forName(reader.getLocalName());
 121.                switch (element) {
 122.                    case PERSISTENCEUNIT:
 123.                        PersistenceUnitMetadata pu = parsePU(reader, version, propertyReplacer);
 124.                        PUs.add(pu);
 125.                        JPA_LOGGER.readingPersistenceXml(pu.getPersistenceUnitName());
 126.                        break;
 127.    
 128.                    default:
 129.                        throw unexpectedElement(reader);
 130.                }
 131.            }
 132.            PersistenceUnitMetadataHolder result = new PersistenceUnitMetadataHolder(PUs);
 133.            if (JPA_LOGGER.isTraceEnabled())
 134.                JPA_LOGGER.trace(result.toString());
 135.    
 136.            return result;
 137.        }

 139.        /**
 140.         * Parse the persistence unit definitions based on persistence_2_0.xsd.
 141.         *
 142.         *
 143.         * @param reader
 144.         * @param propertyReplacer
 145.         * @return
 146.         * @throws XMLStreamException
 147.         */
 148.        private static PersistenceUnitMetadata parsePU(XMLStreamReader reader, Version version, final Pr...
 149.            PersistenceUnitMetadata pu = new PersistenceUnitMetadataImpl();
 150.            List<String> classes = new ArrayList<String>(1);
 151.            List<String> jarFiles = new ArrayList<String>(1);
 152.            List<String> mappingFiles = new ArrayList<String>(1);
 153.            Properties properties = new Properties();
 154.    
 155.            // set defaults
 156.            pu.setTransactionType(PersistenceUnitTransactionType.JTA);
 157.            pu.setValidationMode(ValidationMode.AUTO);
 158.            pu.setSharedCacheMode(SharedCacheMode.UNSPECIFIED);
 159.            pu.setPersistenceProviderClassName(Configuration.PROVIDER_CLASS_DEFAULT);
 160.            if (version.equals(Version.JPA_1_0)) {
 161.                pu.setPersistenceXMLSchemaVersion("1.0");
 162.            } else if (version.equals(Version.JPA_2_0)) {
 163.                pu.setPersistenceXMLSchemaVersion("2.0");
 164.            } else {
 165.                pu.setPersistenceXMLSchemaVersion("2.1");
 166.            }
 167.    
 168.            final int count = reader.getAttributeCount();
 169.            for (int i = 0; i < count; i++) {
 170.                final String value = reader.getAttributeValue(i);
 171.                if (traceEnabled) {
 172.                    JPA_LOGGER.tracef("parse persistence.xml: attribute value(%d) = %s", i, value);
 173.                }
 174.                final String attributeNamespace = reader.getAttributeNamespace(i);
 175.                if (attributeNamespace != null && !attributeNamespace.isEmpty()) {
 176.                    continue;
 177.                }
 178.                final Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
 179.                switch (attribute) {
 180.                    case NAME:
 181.                        pu.setPersistenceUnitName(value);
 182.                        break;
 183.                    case TRANSACTIONTYPE:
 184.                        if (value.equalsIgnoreCase("RESOURCE_LOCAL"))
 185.                            pu.setTransactionType(PersistenceUnitTransactionType.RESOURCE_LOCAL);
 186.                        break;
 187.                    default:
 188.                        throw unexpectedAttribute(reader, i);
 189.                }
 190.            }
 191.    
 192.            // until the ending PERSISTENCEUNIT tag
 193.            while (reader.hasNext() && reader.nextTag() != END_ELEMENT) {
 194.                final Element element = Element.forName(reader.getLocalName());
 195.                if (traceEnabled) {
 196.                    JPA_LOGGER.tracef("parse persistence.xml: element=%s", element.getLocalName());
 197.                }
 198.                switch (element) {
 199.                    case CLASS:
 200.                        classes.add(getElement(reader, propertyReplacer));
 201.                        break;
 202.    
 203.                    case DESCRIPTION:
 204.                        final String description = getElement(reader, propertyReplacer);
 205.                        break;
 206.    
 207.                    case EXCLUDEUNLISTEDCLASSES:
 208.                        String text = getElement(reader, propertyReplacer);
 209.                        if (text == null || text.isEmpty()) {
 210.                            //the spec has examples where an empty
 211.                            //exclude-unlisted-classes element has the same
 212.                            //effect as setting it to true
 213.                            pu.setExcludeUnlistedClasses(true);
 214.                        } else {
 215.                            pu.setExcludeUnlistedClasses(Boolean.valueOf(text));
 216.                        }
 217.                        break;
 218.    
 219.                    case JARFILE:
 220.                        String file = getElement(reader, propertyReplacer);
 221.                        jarFiles.add(file);
 222.                        break;
 223.    
 224.                    case JTADATASOURCE:
 225.                        pu.setJtaDataSourceName(getElement(reader, propertyReplacer));
 226.                        break;
 227.    
 228.                    case NONJTADATASOURCE:
 229.                        pu.setNonJtaDataSourceName(getElement(reader, propertyReplacer));
 230.                        break;
 231.    
 232.                    case MAPPINGFILE:
 233.                        mappingFiles.add(getElement(reader, propertyReplacer));
 234.                        break;
 235.    
 236.                    case PROPERTIES:
 237.                        parseProperties(reader, properties, propertyReplacer);
 238.                        break;
 239.    
 240.                    case PROVIDER:
 241.                        pu.setPersistenceProviderClassName(getElement(reader, propertyReplacer));
 242.                        break;
 243.    
 244.                    case SHAREDCACHEMODE:
 245.                        String cm = getElement(reader, propertyReplacer);
 246.                        pu.setSharedCacheMode(SharedCacheMode.valueOf(cm));
 247.                        break;
 248.    
 249.                    case VALIDATIONMODE:
 250.                        String validationMode = getElement(reader, propertyReplacer);
 251.                        pu.setValidationMode(ValidationMode.valueOf(validationMode));
 252.                        break;
 253.    
 254.                    default:
 255.                        throw unexpectedElement(reader);
 256.                }
 257.            }
 258.            if (traceEnabled) {
 259.                JPA_LOGGER.trace("parse persistence.xml: reached ending persistence-unit tag");
 260.            }
 261.            pu.setManagedClassNames(classes);
 262.            pu.setJarFiles(jarFiles);
 263.            pu.setMappingFiles(mappingFiles);
 264.            pu.setProperties(properties);
 265.            return pu;
 266.        }

 268.        private static void parseProperties(XMLStreamReader reader, Properties properties, final Propert...
 269.    
 270.            while (reader.hasNext() && reader.nextTag() != END_ELEMENT) {
 271.                final Element element = Element.forName(reader.getLocalName());
 272.                switch (element) {
 273.                    case PROPERTY:
 274.                        final int count = reader.getAttributeCount();
 275.                        String name = null, value = null;
 276.                        for (int i = 0; i < count; i++) {
 277.                            final String attributeValue = getAttribute(reader, i, propertyReplacer);
 278.                            final String attributeNamespace = reader.getAttributeNamespace(i);
 279.                            if (attributeNamespace != null && !attributeNamespace.isEmpty()) {
 280.                                continue;
 281.                            }
 282.                            final Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i...
 283.                            switch (attribute) {
 284.                                case NAME:
 285.                                    name = attributeValue;
 286.                                    break;
 287.                                case VALUE:
 288.                                    value = attributeValue;
 289.                                    if (name != null && value != null) {
 290.                                        properties.put(name, value);
 291.                                    }
 292.                                    name = value = null;
 293.                                    break;
 294.                                default:
 295.                                    throw unexpectedAttribute(reader, i);
 296.                            }
 297.                        }
 298.                        if (reader.hasNext() && (reader.nextTag() != END_ELEMENT))
 299.                            throw unexpectedElement(reader);
 300.    
 301.                        break;
 302.                    default:
 303.                        throw unexpectedElement(reader);
 304.                }
 305.            }
 306.        }
