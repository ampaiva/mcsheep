  41.    /**
  42.     * Represents the value of a day in a month, constructed out of a {@link javax.ejb.ScheduleExpressio...
  43.     * <p/>
  44.     * <p>
  45.     * A {@link DayOfMonth} can hold an {@link Integer} or a {@link String} as its value.
  46.     * value. The various ways in which a
  47.     * {@link DayOfMonth} value can be represented are:
  48.     * <ul>
  49.     * <li>Wildcard. For example, dayOfMonth = "*"</li>
  50.     * <li>Range. Examples:
  51.     * <ul>
  52.     * <li>dayOfMonth = "1-10"</li>
  53.     * <li>dayOfMonth = "Sun-Tue"</li>
  54.     * <li>dayOfMonth = "1st-5th"</li>
  55.     * </ul>
  56.     * </li>
  57.     * <li>List. Examples:
  58.     * <ul>
  59.     * <li>dayOfMonth = "1, 12, 20"</li>
  60.     * <li>dayOfMonth = "Mon, Fri, Sun"</li>
  61.     * <li>dayOfMonth = "3rd, 1st, Last"</li>
  62.     * </ul>
  63.     * </li>
  64.     * <li>Single value. Examples:
  65.     * <ul>
  66.     * <li>dayOfMonth = "Fri"</li>
  67.     * <li>dayOfMonth = "Last"</li>
  68.     * <li>dayOfMonth = "10"</li>
  69.     * </ul>
  70.     * </li>
  71.     * </ul>
  72.     * </p>
  73.     *
  74.     * @author Jaikiran Pai
  75.     * @version $Revision: $
  76.     */
  77.    public class DayOfMonth extends IntegerBasedExpression {

 164.        public Integer getNextMatch(Calendar currentCal) {
 165.            if (this.scheduleExpressionType == ScheduleExpressionType.WILDCARD) {
 166.                return currentCal.get(Calendar.DAY_OF_MONTH);
 167.            }
 168.            int currentDayOfMonth = currentCal.get(Calendar.DAY_OF_MONTH);
 169.            SortedSet<Integer> eligibleDaysOfMonth = this.getEligibleDaysOfMonth(currentCal);
 170.            if (eligibleDaysOfMonth.isEmpty()) {
 171.                return null;
 172.            }
 173.            for (Integer hour : eligibleDaysOfMonth) {
 174.                if (currentDayOfMonth == hour) {
 175.                    return currentDayOfMonth;
 176.                }
 177.                if (hour > currentDayOfMonth) {
 178.                    return hour;
 179.                }
 180.            }
 181.            return eligibleDaysOfMonth.first();
 182.        }

 192.        private boolean hasRelativeDayOfMonth() {
 193.            if (this.relativeValues.isEmpty()) {
 194.                return false;
 195.            }

 199.        private SortedSet<Integer> getEligibleDaysOfMonth(Calendar cal) {
 200.            if (this.hasRelativeDayOfMonth() == false) {
 201.                return this.absoluteValues;
 202.            }
 203.            SortedSet<Integer> eligibleDaysOfMonth = new TreeSet<Integer>(this.absoluteValues);
 204.            for (ScheduleValue relativeValue : this.relativeValues) {
 205.                if (relativeValue instanceof SingleValue) {
 206.                    SingleValue singleValue = (SingleValue) relativeValue;
 207.                    String value = singleValue.getValue();
 208.                    Integer absoluteDayOfMonth = this.getAbsoluteDayOfMonth(cal, value);
 209.                    eligibleDaysOfMonth.add(absoluteDayOfMonth);
 210.                } else if (relativeValue instanceof RangeValue) {
 211.                    RangeValue range = (RangeValue) relativeValue;
 212.                    String start = range.getStart();
 213.                    String end = range.getEnd();
 214.    
 215.                    Integer dayOfMonthStart = null;
 216.                    // either start will be relative or end will be relative or both are relative
 217.                    if (this.isRelativeValue(start)) {
 218.                        dayOfMonthStart = this.getAbsoluteDayOfMonth(cal, start);
 219.                    } else {
 220.                        dayOfMonthStart = this.parseInt(start);
 221.                    }
 222.    
 223.                    Integer dayOfMonthEnd = null;
 224.                    if (this.isRelativeValue(end)) {
 225.                        dayOfMonthEnd = this.getAbsoluteDayOfMonth(cal, end);
 226.                    } else {
 227.                        dayOfMonthEnd = this.parseInt(end);
 228.                    }
 229.                    // validations
 230.                    this.assertValid(dayOfMonthStart);
 231.                    this.assertValid(dayOfMonthEnd);
 232.    
 233.                    // start and end are both the same. So it's just a single value
 234.                    if (dayOfMonthStart.equals(dayOfMonthEnd)) {
 235.                        eligibleDaysOfMonth.add(dayOfMonthEnd);
 236.                        continue;
 237.    
 238.                    }
 239.                    if (dayOfMonthStart > dayOfMonthEnd) {
 240.                        // In range "x-y", if x is larger than y, the range is equivalent to
 241.                        // "x-max, min-y", where max is the largest value of the corresponding attribute...
 242.                        // and min is the smallest.
 243.                        for (int i = dayOfMonthStart; i <= this.getMaxValue(); i++) {
 244.                            eligibleDaysOfMonth.add(i);
 245.                        }
 246.                        for (int i = this.getMinValue(); i <= dayOfMonthEnd; i++) {
 247.                            eligibleDaysOfMonth.add(i);
 248.                        }
 249.                    } else {
 250.                        // just keep adding from range start to range end (both inclusive).
 251.                        for (int i = dayOfMonthStart; i <= dayOfMonthEnd; i++) {
 252.                            eligibleDaysOfMonth.add(i);
 253.                        }
 254.                    }
 255.                }
 256.    
 257.            }
 258.            return eligibleDaysOfMonth;
 259.        }

 261.        private int getAbsoluteDayOfMonth(Calendar cal, String relativeDayOfMonth) {
 262.            if (relativeDayOfMonth == null || relativeDayOfMonth.trim().isEmpty()) {
 263.                throw EjbLogger.ROOT_LOGGER.relativeDayOfMonthIsNull();
 264.            }
 265.            String trimmedRelativeDayOfMonth = relativeDayOfMonth.trim();
 266.            if (trimmedRelativeDayOfMonth.equalsIgnoreCase("last")) {
 267.                int lastDayOfCurrentMonth = CalendarUtil.getLastDateOfMonth(cal);
 268.                return lastDayOfCurrentMonth;
 269.            }
 270.            if (this.isValidNegativeDayOfMonth(trimmedRelativeDayOfMonth)) {
 271.                Integer negativeRelativeDayOfMonth = Integer.parseInt(trimmedRelativeDayOfMonth);
 272.                int lastDayOfCurrentMonth = CalendarUtil.getLastDateOfMonth(cal);
 273.                return lastDayOfCurrentMonth + negativeRelativeDayOfMonth;
 274.            }
 275.            if (this.isDayOfWeekBased(trimmedRelativeDayOfMonth)) {
 276.    
 277.                String[] parts = trimmedRelativeDayOfMonth.split("\\s+");
 278.                String ordinal = parts[0];
 279.                String day = parts[1];
 280.                int dayOfWeek = DAY_OF_MONTH_ALIAS.get(day.toLowerCase(Locale.ENGLISH));
 281.    
 282.                Integer date = null;
 283.                if (ordinal.equalsIgnoreCase("last")) {
 284.                    date = CalendarUtil.getDateOfLastDayOfWeekInMonth(cal, dayOfWeek);
 285.                } else {
 286.                    int weekNumber = ORDINAL_TO_WEEK_NUMBER_MAPPING.get(ordinal.toLowerCase(Locale.ENGLI...
 287.                    date = CalendarUtil.getNthDayOfMonth(cal, weekNumber, dayOfWeek);
 288.                }
 289.    
 290.                // TODO: Rethink about this. The reason why we have this currently is to handle cases li...
 291.                // 5th Wed which may not be valid for all months (i.e. all months do not have 5 weeks). ...
 292.                // cases we set the date to last date of the month.
 293.                // This needs to be thought about a bit more in detail, to understand it's impact on oth...
 294.                if (date == null) {
 295.                    date = CalendarUtil.getLastDateOfMonth(cal);
 296.                }
 297.    
 298.                return date;
 299.            }
 300.            throw EjbLogger.ROOT_LOGGER.invalidRelativeValue(relativeDayOfMonth);
 301.        }

 303.        private boolean isValidNegativeDayOfMonth(String dayOfMonth) {
 304.            try {
 305.                Integer val = Integer.parseInt(dayOfMonth.trim());
 306.                if (val <= -1 && val >= -7) {
 307.                    return true;
 308.                }
 309.                return false;
 310.            } catch (NumberFormatException nfe) {
 311.                return false;
 312.            }
 313.    
 314.        }

 316.        private boolean isDayOfWeekBased(String relativeVal) {
 317.            String trimmedVal = relativeVal.trim();
 318.            // one or more spaces (which includes tabs and other forms of space)
 319.            Pattern p = Pattern.compile("\\s+");
 320.            String[] relativeParts = p.split(trimmedVal);
 321.            if (relativeParts == null) {
 322.                return false;
 323.            }
 324.            if (relativeParts.length != 2) {
 325.                return false;
 326.            }
 327.            String ordinal = relativeParts[0];
 328.            String dayOfWeek = relativeParts[1];
 329.            if (ordinal == null || dayOfWeek == null) {
 330.                return false;
 331.            }
 332.            String lowerCaseOrdinal = ordinal.toLowerCase(Locale.ENGLISH);
 333.            if (ORDINALS.contains(lowerCaseOrdinal) == false) {
 334.                return false;
 335.            }
 336.            String lowerCaseDayOfWeek = dayOfWeek.toLowerCase(Locale.ENGLISH);
 337.            if (DAY_OF_MONTH_ALIAS.containsKey(lowerCaseDayOfWeek) == false) {
 338.                return false;
 339.            }
 340.            return true;
 341.        }

 343.        @Override
 344.        public boolean isRelativeValue(String value) {
 345.            if (value == null) {
 346.                throw EjbLogger.ROOT_LOGGER.relativeValueIsNull();
 347.            }
 348.            if (value.equalsIgnoreCase("last")) {
 349.                return true;
 350.            }
 351.            if (this.isValidNegativeDayOfMonth(value)) {
 352.                return true;
 353.            }
 354.            if (this.isDayOfWeekBased(value)) {
 355.                return true;
 356.            }
 357.            return false;
 358.        }

 360.        @Override
 361.        protected boolean accepts(ScheduleExpressionType scheduleExprType) {
 362.            switch (scheduleExprType) {
 363.                case RANGE:
 364.                case LIST:
 365.                case SINGLE_VALUE:
 366.                case WILDCARD:
 367.                    return true;
 368.                // day-of-month doesn't support increment
 369.                case INCREMENT:
 370.                default:
 371.                    return false;
 372.            }
 373.        }

 375.        public Integer getFirstMatch(Calendar cal) {
 376.            if (this.scheduleExpressionType == ScheduleExpressionType.WILDCARD) {
 377.                return Calendar.SUNDAY;
 378.            }
 379.            SortedSet<Integer> eligibleDaysOfMonth = this.getEligibleDaysOfMonth(cal);
 380.            if (eligibleDaysOfMonth.isEmpty()) {
 381.                return null;
 382.            }
 383.            return eligibleDaysOfMonth.first();
 384.        }

 386.        @Override
 387.        protected Integer parseInt(String alias) {
 388.            try {
 389.                return super.parseInt(alias);
 390.            } catch (NumberFormatException nfe) {
 391.                if (DAY_OF_MONTH_ALIAS != null) {
 392.                    String lowerCaseAlias = alias.toLowerCase(Locale.ENGLISH);
 393.                    return DAY_OF_MONTH_ALIAS.get(lowerCaseAlias);
 394.                }
 395.            }
 396.            return null;
 397.        }
