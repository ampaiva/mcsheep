 646.                return null;
 647.    
 648.            int encodedLength = 0;
 649.            int n = 0;
 650.    
 651.            if (encodedToken[1] >= 0)
 652.                encodedLength = encodedToken[1];
 653.            else {
 654.                n = encodedToken[1] & 0x7F;
 655.                for (int i = 1; i <= n; i++) {
 656.                    encodedLength += (encodedToken[1 + i] & 0xFF) << (n - i) * 8;
 657.                }
 658.            }
 659.    
 660.            int length = encodedLength - gssUpMechOidArray.length;
 661.            byte[] encodedInitialContextToken = new byte[length];
 662.    
 663.            System.arraycopy(encodedToken, 2 + n + gssUpMechOidArray.length,
 664.                    encodedInitialContextToken, 0,
 665.                    length);
 666.            Any any;
 667.            try {
 668.                any = codec.decode_value(encodedInitialContextToken, InitialContextTokenHelper.type());
 669.            } catch (Exception e) {
 670.                return null;
 671.            }
 672.    
 673.            return InitialContextTokenHelper.extract(any);
 674.        }

 690.        /**
 691.         * <p>
 692.         * Decodes a GSS exported name that has been encoded with the GSSUP mechanism OID. See {@code cr...
 693.         * for a description of the encoding format.
 694.         * </p>
 695.         *
 696.         * @param encodedName the encoded exported name.
 697.         * @return a {@code byte[]} representing the decoded exported name.
 698.         * @see #createGSSExportedName(byte[], byte[])
 699.         */
 700.        public static byte[] decodeGssExportedName(byte[] encodedName) {
 701.            if (encodedName[0] != 0x04 || encodedName[1] != 0x01)
 702.                return null;
 703.    
 704.            int mechOidLength = (encodedName[2] & 0xFF) << 8; //MECH_OID_LEN
 705.            mechOidLength += (encodedName[3] & 0xFF);      // MECH_OID_LEN
 706.    
 707.            byte[] oidArray = new byte[mechOidLength];
 708.            System.arraycopy(encodedName, 4, oidArray, 0, mechOidLength);
 709.    
 710.            for (int i = 0; i < mechOidLength; i++) {
 711.                if (gssUpMechOidArray[i] != oidArray[i]) {
 712.                    return null;
 713.                }
 714.            }
 715.    
 716.            int offset = 4 + mechOidLength;
 717.            int nameLength = (encodedName[offset] & 0xFF) << 24;
 718.            nameLength += (encodedName[++offset] & 0xFF) << 16;
 719.            nameLength += (encodedName[++offset] & 0xFF) << 8;
 720.            nameLength += (encodedName[++offset] & 0xFF);
 721.    
 722.            byte[] name = new byte[nameLength];
 723.            System.arraycopy(encodedName, ++offset, name, 0, nameLength);
 724.    
 725.            return name;
 726.        }

 753.        public static CompoundSecMech getMatchingSecurityMech(ClientRequestInfo ri, Codec codec, short c...
