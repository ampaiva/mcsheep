 141.        @SuppressWarnings({ "unchecked", "rawtypes", "unused" })
 142.        private List<String> parseBreadCrumb(String refContent) throws Exception {
 143.

 147.            return new LinkedList(Arrays.asList(categoryIds));
 148.            
 149.            
 150.        }

 153.        private List<String> parseCategoryLineage(String lineage) throws Exception {
 154.            
 155.            String[] categoryPath = lineage.split(Constants.CATEGORY_LINEAGE_DELIMITER);
 156.            List<String> returnList = new LinkedList<String>();
 157.            for(String c : categoryPath) {
 158.                if(!StringUtils.isBlank(c)) {
 159.                    returnList.add(c);
 160.                }
 161.            }
 162.            return returnList;
 163.    
 164.        }

 166.        private String buildBreadCrumb(List<Long> ids) throws Exception {
 167.            
 168.            if(CollectionUtils.isEmpty(ids)) {
 169.                return null;
 170.            }
 171.                StringBuilder sb = new StringBuilder();
 172.                sb.append("c:");
 173.                int count = 1;
 174.                for(Long c : ids) {
 175.                    sb.append(c);
 176.                    if(count < ids.size()) {
 177.                        sb.append(",");
 178.                    }
 179.                    count++;
 180.                }
 181.            
 182.            
 183.            return sb.toString();
 184.            
 185.        }
