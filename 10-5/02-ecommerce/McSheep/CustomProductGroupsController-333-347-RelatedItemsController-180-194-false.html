  57.            
  58.            setMenu(model,request);
  59.            
  60.            Language language = (Language)request.getAttribute("LANGUAGE");
  61.            MerchantStore store = (MerchantStore)request.getAttribute(Constants.ADMIN_STORE);
  62.            
  63.            //get the product and validate it belongs to the current merchant
  64.            Product product = productService.getById(productId);

  72.            }
  73.            
  74.            
  75.            List categories = categoryService.listByStore(store,language);
  76.            
  77.            model.addAttribute("categories", categories);
  78.            model.addAttribute("product", product);

  84.        @SuppressWarnings({ "rawtypes", "unchecked" })
  85.        @PreAuthorize("hasRole('PRODUCTS')")
  86.        @RequestMapping(value="/admin/catalogue/related/paging.html", method=RequestMethod.POST, produces="application/json")
  87.        public @ResponseBody String pageRelatedItems(HttpServletRequest request, HttpServletResponse response) {
  88.            
  89.            String sProductId = request.getParameter("productId");
  90.            AjaxResponse resp = new AjaxResponse();
  91.            
  92.            try {
  93.                
  94.    
  95.                
  96.                Long productId = Long.parseLong(sProductId);
  97.                Product product = productService.getById(productId);
  98.                
  99.                Language language = (Language)request.getAttribute("LANGUAGE");
 100.                MerchantStore store = (MerchantStore)request.getAttribute(Constants.ADMIN_STORE);
 101.                
 102.                
 103.                if(product==null || product.getMerchantStore().getId().intValue()!= store.getId().intValue()) {
 104.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 105.                    resp.setErrorString("Product id is not valid");
 106.                    String returnString = resp.toJSONString();
 107.                    return returnString;
 108.                }
 109.                
 110.                
 111.    
 112.                List relationships = productRelationshipService.getByType(store, product, ProductRelationshipType.RELATED_ITEM, language);
 113.                
 114.                for(ProductRelationship relationship : relationships) {
 115.                    
 116.                    Product relatedProduct = relationship.getRelatedProduct();
 117.                    Map entry = new HashMap();
 118.                    entry.put("relationshipId", relationship.getId());
 119.                    entry.put("productId", relatedProduct.getId());
 120.                    
 121.                    ProductDescription description = relatedProduct.getDescriptions().iterator().next();
 122.                    Set descriptions = relatedProduct.getDescriptions();
 123.                    for(ProductDescription desc : descriptions) {
 124.                        if(desc.getLanguage().getId().intValue()==language.getId().intValue()) {
 125.                            description = desc;
 126.                        }
 127.                    }
 128.                    
 129.    
 130.                    entry.put("name", description.getName());
 131.                    entry.put("sku", relatedProduct.getSku());
 132.                    entry.put("available", relatedProduct.isAvailable());
 133.                    resp.addDataEntry(entry);
 134.                    
 135.                }
 136.                
 137.    
 138.                resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_SUCCESS);
 139.            
 140.            } catch (Exception e) {
 141.                LOGGER.error("Error while paging products", e);
 142.                resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 143.                resp.setErrorMessage(e);
 144.            }
 145.            
 146.            String returnString = resp.toJSONString();
 147.            return returnString;
 148.    
 149.    
 150.        }

 152.        @PreAuthorize("hasRole('PRODUCTS')")
 153.        @RequestMapping(value="/admin/catalogue/related/addItem.html", method=RequestMethod.POST, produces="application/json")
 154.        public @ResponseBody String addItem(HttpServletRequest request, HttpServletResponse response) {
 155.            
 156.            String productId = request.getParameter("productId");
 157.            String baseProductId = request.getParameter("baseProductId");
 158.            AjaxResponse resp = new AjaxResponse();
 159.            
 160.            try {
 161.                
 162.    
 163.                Long lProductId = Long.parseLong(productId);
 164.                Long lBaseProductId = Long.parseLong(baseProductId);
 165.    
 166.                MerchantStore store = (MerchantStore)request.getAttribute(Constants.ADMIN_STORE);
 167.                
 168.                Product product = productService.getById(lProductId);
 169.                
 170.                if(product==null) {
 171.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 172.                    return resp.toJSONString();
 173.                }
 174.                
 175.                if(product.getMerchantStore().getId().intValue()!=store.getId().intValue()) {
 176.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 177.                    return resp.toJSONString();
 178.                }
 179.                
 180.                Product baseProduct = productService.getById(lBaseProductId);
 181.                
 182.                if(baseProduct==null) {
 183.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 184.                    return resp.toJSONString();
 185.                }
 186.                
 187.                if(baseProduct.getMerchantStore().getId().intValue()!=store.getId().intValue()) {
 188.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 189.                    return resp.toJSONString();
 190.                }
 191.    
 192.    
 193.                ProductRelationship relationship = new ProductRelationship();
 194.                relationship.setActive(true);
 195.                relationship.setProduct(baseProduct);
 196.                relationship.setCode(ProductRelationshipType.RELATED_ITEM.name());
 197.                relationship.setStore(store);
 198.                relationship.setRelatedProduct(product);
 199.                
 200.                productRelationshipService.saveOrUpdate(relationship);
 201.                
 202.    
 203.                resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_SUCCESS);
 204.            
 205.            } catch (Exception e) {
 206.                LOGGER.error("Error while paging products", e);
 207.                resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 208.                resp.setErrorMessage(e);
 209.            }
 210.            
 211.            String returnString = resp.toJSONString();
 212.            return returnString;
 213.            
 214.        }

 216.        @PreAuthorize("hasRole('PRODUCTS')")
 217.        @RequestMapping(value="/admin/catalogue/related/removeItem.html", method=RequestMethod.POST, produces="application/json")
 218.        public @ResponseBody String removeItem(HttpServletRequest request, HttpServletResponse response) {
 219.            
 220.            String productId = request.getParameter("productId");
 221.            String baseProductId = request.getParameter("baseProductId");
 222.            AjaxResponse resp = new AjaxResponse();
 223.            
 224.            try {
 225.                
 226.    
 227.                Long lproductId = Long.parseLong(productId);
 228.                Long lBaseProductId = Long.parseLong(baseProductId);
 229.    
 230.                MerchantStore store = (MerchantStore)request.getAttribute(Constants.ADMIN_STORE);
 231.                
 232.                Product product = productService.getById(lproductId);
 233.                
 234.                if(product==null) {
 235.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 236.                    return resp.toJSONString();
 237.                }
 238.                
 239.                if(product.getMerchantStore().getId().intValue()!=store.getId().intValue()) {
 240.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 241.                    return resp.toJSONString();
 242.                }
 243.                
 244.                Product baseProduct = productService.getById(lBaseProductId);
 245.                
 246.                if(baseProduct==null) {
 247.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 248.                    return resp.toJSONString();
 249.                }
 250.                
 251.                if(baseProduct.getMerchantStore().getId().intValue()!=store.getId().intValue()) {
 252.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 253.                    return resp.toJSONString();
 254.                }
 255.                
 256.                ProductRelationship relationship = null;
 257.                List relationships = productRelationshipService.getByType(store, baseProduct, ProductRelationshipType.RELATED_ITEM);
 258.                
 259.                for(ProductRelationship r : relationships) {
 260.                    if(r.getRelatedProduct().getId().longValue()==lproductId.longValue()) {
 261.                        relationship = r;
 262.                        break;
 263.                    }
 264.                }
 265.                
 266.                if(relationship==null) {
 267.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 268.                    return resp.toJSONString();
 269.                }
 270.                
 271.                if(relationship.getStore().getId().intValue()!=store.getId().intValue()) {
 272.                    resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 273.                    return resp.toJSONString();
 274.                }
 275.    
 276.    
 277.                
 278.                
 279.                productRelationshipService.delete(relationship);
 280.                
 281.    
 282.                resp.setStatus(AjaxPageableResponse.RESPONSE_OPERATION_COMPLETED);
 283.            
 284.            } catch (Exception e) {
 285.                LOGGER.error("Error while paging products", e);
 286.                resp.setStatus(AjaxPageableResponse.RESPONSE_STATUS_FAIURE);
 287.                resp.setErrorMessage(e);
 288.            }
 289.            
 290.            String returnString = resp.toJSONString();
 291.            return returnString;
 292.            
 293.        }
