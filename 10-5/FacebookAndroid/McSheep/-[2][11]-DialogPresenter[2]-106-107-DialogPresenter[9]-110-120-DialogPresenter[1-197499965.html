 142.                throw new FacebookException("Unable to fetch the app's key-hash");
 143.            }
 144.    
 145.            // Now form the Uri
 146.            if (fallbackUrl.isRelative()) {
 147.                fallbackUrl = Utility.buildUri(
 148.                        ServerProtocol.getDialogAuthority(),
 149.                        fallbackUrl.toString(),
 150.                        webParams);
 151.            } else {
 152.                fallbackUrl = Utility.buildUri(
 153.                        fallbackUrl.getAuthority(),
 154.                        fallbackUrl.getPath(),
 155.                        webParams);
 156.            }
 157.    
 158.            Bundle intentParameters = new Bundle();
 159.            intentParameters.putString(NativeProtocol.WEB_DIALOG_URL, fallbackUrl.toString());
 160.            intentParameters.putBoolean(NativeProtocol.WEB_DIALOG_IS_FALLBACK, true);
 161.    
 162.            Intent webDialogIntent = new Intent();
 163.            NativeProtocol.setupProtocolRequestIntent(
 164.                    webDialogIntent,
 165.                    appCall.getCallId().toString(),
 166.                    feature.getAction(),
 167.                    NativeProtocol.getLatestKnownVersion(),
 168.                    intentParameters);
 169.            webDialogIntent.setClass(FacebookSdk.getApplicationContext(), FacebookActivity.class);
 170.            webDialogIntent.setAction(FacebookDialogFragment.TAG);
 171.    
 172.            appCall.setRequestIntent(webDialogIntent);
 173.        }

 175.        public static void setupAppCallForNativeDialog(
 176.                AppCall appCall,
 177.                ParameterProvider parameterProvider,
 178.                DialogFeature feature) {
 179.            Context context = FacebookSdk.getApplicationContext();
 180.            String action = feature.getAction();
 181.            int protocolVersion = getProtocolVersionForNativeDialog(feature);
 182.            if (protocolVersion == NativeProtocol.NO_PROTOCOL_AVAILABLE) {
 183.                throw new FacebookException(
 184.                        "Cannot present this dialog. This likely means that the " +
 185.                                "Facebook app is not installed.");
 186.            }
 187.    
 188.            Bundle params;
 189.            if (NativeProtocol.isVersionCompatibleWithBucketedIntent(protocolVersion)) {
 190.                // Facebook app supports the new bucketed protocol
 191.                params = parameterProvider.getParameters();
 192.            } else {
 193.                // Facebook app only supports the old flat protocol
 194.                params = parameterProvider.getLegacyParameters();
 195.            }
 196.            if (params == null) {
 197.                params = new Bundle();
 198.            }
 199.    
 200.            Intent intent = NativeProtocol.createPlatformActivityIntent(
 201.                    context,
 202.                    appCall.getCallId().toString(),
 203.                    action,
 204.                    protocolVersion,
 205.                    params);
 206.            if (intent == null) {
 207.                throw new FacebookException(
 208.                        "Unable to create Intent; this likely means the" +
 209.                                "Facebook app is not installed.");
 210.            }
 211.    
 212.            appCall.setRequestIntent(intent);
 213.        }

 215.        private static Uri getDialogWebFallbackUri(DialogFeature feature) {
 216.            String featureName = feature.name();
 217.            String action = feature.getAction();
 218.            String applicationId = FacebookSdk.getApplicationId();
 219.    
 220.            Utility.DialogFeatureConfig config =
 221.                    Utility.getDialogFeatureConfig(applicationId, action, featureName);
 222.            Uri fallbackUrl = null;
 223.            if (config != null) {
 224.                fallbackUrl = config.getFallbackUrl();
 225.            }
 226.    
 227.            return fallbackUrl;
 228.        }

 230.        public static int getProtocolVersionForNativeDialog(
 231.                DialogFeature feature) {
 232.            String applicationId = FacebookSdk.getApplicationId();
 233.            String action = feature.getAction();
 234.            int[] featureVersionSpec = getVersionSpecForFeature(applicationId, action, feature);
 235.    
 236.            return NativeProtocol.getLatestAvailableProtocolVersionForAction(
 237.                    action,
 238.                    featureVersionSpec);
 239.        }

 241.        private static int[] getVersionSpecForFeature(
 242.                String applicationId,
 243.                String actionName,
 244.                DialogFeature feature) {
 245.            // Return the value from DialogFeatureConfig if available. Otherwise, just
 246.            // default to the min-version
 247.            Utility.DialogFeatureConfig config =
 248.                    Utility.getDialogFeatureConfig(applicationId, actionName, feature.name());
 249.            if (config != null) {
 250.                return config.getVersionSpec();
 251.            } else {
 252.                return new int[]{feature.getMinVersion()};
 253.            }
