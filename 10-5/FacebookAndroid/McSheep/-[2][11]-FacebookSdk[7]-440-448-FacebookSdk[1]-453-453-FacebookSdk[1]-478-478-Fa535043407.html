 283.        private static AppLinkData createFromJson(String jsonString) {
 284.            if (jsonString  == null) {
 285.                return null;
 286.            }
 287.    
 288.            try {
 289.                // Any missing or malformed data will result in a JSONException
 290.                JSONObject appLinkArgsJson = new JSONObject(jsonString);
 291.                String version = appLinkArgsJson.getString(APPLINK_VERSION_KEY);
 292.    
 293.                JSONObject bridgeArgs = appLinkArgsJson.getJSONObject(APPLINK_BRIDGE_ARGS_KEY);
 294.                String method = bridgeArgs.getString(BRIDGE_ARGS_METHOD_KEY);
 295.                if (method.equals("applink") && version.equals("2")) {
 296.                    // We have a new deep link
 297.                    AppLinkData appLinkData = new AppLinkData();
 298.    
 299.                    appLinkData.arguments = appLinkArgsJson.getJSONObject(APPLINK_METHOD_ARGS_KEY);
 300.                    // first look for the "ref" key in the top level args
 301.                    if (appLinkData.arguments.has(METHOD_ARGS_REF_KEY)) {
 302.                        appLinkData.ref = appLinkData.arguments.getString(METHOD_ARGS_REF_KEY);
 303.                    } else if (appLinkData.arguments.has(ARGUMENTS_REFERER_DATA_KEY)) {
 304.                        // if it's not in the top level args, it could be in the "referer_data" blob
 305.                        JSONObject refererData = appLinkData.arguments.getJSONObject(
 306.                                ARGUMENTS_REFERER_DATA_KEY);
 307.                        if (refererData.has(REFERER_DATA_REF_KEY)) {
 308.                            appLinkData.ref = refererData.getString(REFERER_DATA_REF_KEY);
 309.                        }
 310.                    }
 311.    
 312.                    if (appLinkData.arguments.has(METHOD_ARGS_TARGET_URL_KEY)) {
 313.                        appLinkData.targetUri = Uri.parse(
 314.                                appLinkData.arguments.getString(METHOD_ARGS_TARGET_URL_KEY));
 315.                    }
 316.    
 317.                    appLinkData.argumentBundle = toBundle(appLinkData.arguments);
 318.    
 319.                    return appLinkData;
 320.                }
 321.            } catch (JSONException e) {
 322.                Log.d(TAG, "Unable to parse AppLink JSON", e);
 323.            } catch (FacebookException e) {
 324.                Log.d(TAG, "Unable to parse AppLink JSON", e);
 325.            }
 326.    
 327.            return null;
 328.        }

 340.        private static Bundle toBundle(JSONObject node) throws JSONException {
 341.            Bundle bundle = new Bundle();
 342.            @SuppressWarnings("unchecked")
 343.            Iterator<String> fields = node.keys();
 344.            while (fields.hasNext()) {
 345.                String key = fields.next();
 346.                Object value;
 347.                value = node.get(key);
 348.    
 349.                if (value instanceof JSONObject) {
 350.                    bundle.putBundle(key, toBundle((JSONObject) value));
 351.                } else if (value instanceof JSONArray) {
 352.                    JSONArray valueArr = (JSONArray) value;
 353.                    if (valueArr.length() == 0) {
 354.                        bundle.putStringArray(key, new String[0]);
 355.                    } else {
 356.                        Object firstNode = valueArr.get(0);
 357.                        if (firstNode instanceof JSONObject) {
 358.                            Bundle[] bundles = new Bundle[valueArr.length()];
 359.                            for (int i = 0; i < valueArr.length(); i++) {
 360.                                bundles[i] = toBundle(valueArr.getJSONObject(i));
 361.                            }
 362.                            bundle.putParcelableArray(key, bundles);
 363.                        } else if (firstNode instanceof JSONArray) {
 364.                            // we don't support nested arrays
 365.                            throw new FacebookException("Nested arrays are not supported.");
 366.                        } else { // just use the string value
 367.                            String[] arrValues = new String[valueArr.length()];
 368.                            for (int i = 0; i < valueArr.length(); i++) {
 369.                                arrValues[i] = valueArr.get(i).toString();
 370.                            }
 371.                            bundle.putStringArray(key, arrValues);
 372.                        }
 373.                    }
 374.                } else {
 375.                    bundle.putString(key, value.toString());
 376.                }
 377.            }
 378.            return bundle;
 379.        }
