  67.    /**
  68.     * This is the main Entity Bean class for SupplierOrderEJB
  69.     * It has a one-many relatioship with the LineItemEJB, and one-to-one
  70.     * relationship between ContactInfoEJB
  71.     */
  72.    
  73.    public abstract class SupplierOrderEJB implements EntityBean {
  74.    
  75.      private EntityContext context = null;
  76.    
  77.      /**
  78.       * Accessor method for purchase order ID
  79.       * @return String  the PO id
  80.       */
  81.      public abstract String getPoId();
  82.    
  83.      /**
  84.       * Setter method for PO id
  85.       * @param String the PO ID
  86.       */
  87.      public abstract void setPoId(String id);
  88.    
  89.      /**
  90.       * Accessor method for purchase order Date
  91.       * @return long  the PO Date - time from epoch
  92.       */
  93.      public abstract long getPoDate();
  94.    
  95.    
  96.      /**
  97.       * Setter method for PO Date
  98.       * @param long the PO Date - time from epoch
  99.       */
 100.      public abstract void setPoDate(long orderDate);
 101.    
 102.      /**
 103.       * Accessor method for purchase order status
 104.       * @return String  the PO status
 105.       */
 106.      public abstract String getPoStatus();
 107.    
 108.    
 109.      /**
 110.       * Setter method for PO status
 111.       * @param String the PO status
 112.       */
 113.      public abstract void setPoStatus(String stat);
 114.    
 115.      /**
 116.       * Accessor method for purchase order contact info
 117.       * @return <Code>ContactInfoLocal</Code>  the PO contactinfo
 118.       */
 119.      public abstract ContactInfoLocal getContactInfo();
 120.    
 121.      /**
 122.       * Setter method for PO contact info
 123.       * @param <Code>ContactInfoLocal</Code> the PO contact info
 124.       */
 125.      public abstract void setContactInfo(ContactInfoLocal addr);
 126.    
 127.      /**
 128.       * Accessor method for purchase order line items
 129.       * @return <Code>Collection</Code>  the PO lientems
 130.       */
 131.      public abstract Collection getLineItems();
 132.    
 133.      /**
 134.       * Setter method for PO line items
 135.       * @param <Code>Collection</Code> the PO lineitems
 136.       */
 137.      public abstract void setLineItems(Collection litems);
 138.    
 139.      /**
 140.       * Method adds a line item to existing lineitems
 141.       * @param <Code>LineItemLocal</Code> the line item details
 142.       */
 143.      public void addLineItem(LineItemLocal lItem){
 144.        getLineItems().add(lItem);
 145.      }
 146.    
 147.      /**
 148.       * the ejbCreate method
 149.       * @param <code>supplierOrder</code>
 150.       */
 151.      public String ejbCreate(SupplierOrder supplierOrder) throws CreateException {
 152.        setPoId(supplierOrder.getOrderId());
 153.        setPoDate(supplierOrder.getOrderDate().getTime());
 154.        setPoStatus(OrderStatusNames.PENDING);
 155.        return null;
 156.      }
 157.    
 158.      /**
 159.       * the ejbPostCreate method
 160.       * @param <code>supplierOrder</code>
 161.       */
 162.      public void ejbPostCreate(SupplierOrder supplierOrder)
 163.                                                  throws CreateException {
 164.      try {
 165.          ServiceLocator serviceLocator = new ServiceLocator();
 166.          ContactInfoLocalHome cinforef = (ContactInfoLocalHome)
 167.              serviceLocator.getLocalHome(JNDINames.CINFO_EJB);
 168.          ContactInfoLocal cinfoloc = (ContactInfoLocal)
 169.                          cinforef.create(supplierOrder.getShippingInfo());
 170.          setContactInfo(cinfoloc);
 171.          LineItemLocalHome lineItemref = (LineItemLocalHome)
 172.              serviceLocator.getLocalHome(JNDINames.LI_EJB);
 173.          Collection litems = supplierOrder.getLineItems();
 174.          Iterator it = litems.iterator();
 175.          while((it != null) && (it.hasNext())) {
 176.            LineItem li = (LineItem) it.next();
 177.            LineItemLocal lineItemloc = (LineItemLocal) lineItemref.create(li, 0);
 178.            addLineItem(lineItemloc);
 179.          }
 180.        } catch(ServiceLocatorException ne) {
 181.          throw new CreateException("ServiceLocator Ex while persisting PO CMR :" +
 182.                                    ne.getMessage());
 183.        }
 184.      }
 185.    
 186.      /**
 187.       * Method gets a collection of all line items for this PO. This is required
 188.       * because web tier cannot access CMR fields that return managed objects
 189.       * without a transaction
 190.       * @return <Code>Collection</Code> Colleciton of lineitems
 191.       */
 192.      public Collection getAllItems() {
 193.        Collection liColl = getLineItems();
 194.        if(liColl == null)
 195.          return(null);
 196.        ArrayList retVal = new ArrayList();
 197.        Iterator it = liColl.iterator();
 198.        while((it!=null) && (it.hasNext())) {
 199.          LineItemLocal loc = (LineItemLocal) it.next();
 200.          retVal.add(loc.getData());
 201.        }
 202.        return(retVal);
 203.      }
 204.    
 205.      public SupplierOrder getData() {
 206.        SupplierOrder supplierOrder = new SupplierOrder();
 207.        supplierOrder.setOrderId(getPoId());
 208.        supplierOrder.setOrderDate(new Date(getPoDate()));
 209.        supplierOrder.setShippingInfo(supplierOrder.getShippingInfo());
 210.        Collection lineItems = getLineItems();
 211.        for (Iterator iterator = lineItems.iterator(); iterator.hasNext();) {
 212.          LineItemLocal lineItem = (LineItemLocal) iterator.next();
 213.          supplierOrder.addLineItem(lineItem.getData());
 214.        }
 215.        return supplierOrder;
 216.      }
