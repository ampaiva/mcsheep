 347.          Collection keywords = new HashSet();
 348.          StringTokenizer st = new StringTokenizer(searchQuery);
 349.          while (st.hasMoreTokens()) {
 350.            keywords.add(st.nextToken());
 351.          }
 352.          if (keywords.isEmpty()) {
 353.            return Page.EMPTY_PAGE;
 354.          }
 355.    
 356.          Connection c = null;
 357.          PreparedStatement ps = null;
 358.          ResultSet rs = null;
 359.          Page ret = null;
 360.    
 361.          try {
 362.            c = getDataSource().getConnection();
 363.            Iterator it;
 364.            int i;
 365.            StringBuffer sb = new StringBuffer();
 366.            sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[0]);
 367.            int keywordsSize = keywords.size();
 368.            if (keywordsSize > 0) {
 369.              sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[1]);
 370.              for (i = 1; i != keywordsSize; i++) {
 371.                sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[2]);
 372.              }
 373.              sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[3]);
 374.              for (i = 1; i != keywordsSize; i++) {
 375.                sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[4]);
 376.              }
 377.              sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[5]);
 378.              for (i = 1; i != keywordsSize; i++) {
 379.                sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[6]);
 380.              }
 381.              sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[7]);
 382.            }
 383.            //System.err.println(sb.toString());
 384.            ps = c.prepareStatement(sb.toString(),
 385.                                    ResultSet.TYPE_SCROLL_INSENSITIVE,
 386.                                    ResultSet.CONCUR_READ_ONLY);
 387.            ps.setString(1, l.toString());
 388.            // The three loops are necessary because of the way the
 389.            // query was constructed.
 390.            i = 2;
 391.            for (it = keywords.iterator(); it.hasNext(); i++) {
 392.              String keyword = ((String) it.next()).toLowerCase();
 393.              ps.setString(i, "%" + keyword + "%");
 394.            }
 395.            for (it = keywords.iterator(); it.hasNext(); i++) {
 396.              String keyword = ((String) it.next()).toLowerCase();
 397.              ps.setString(i, "%" + keyword + "%");
 398.            }
 399.            for (it = keywords.iterator(); it.hasNext(); i++) {
 400.              String keyword = ((String) it.next()).toLowerCase();
 401.              ps.setString(i, "%" + keyword + "%");
 402.            }
 403.            rs = ps.executeQuery();
 404.            if (start >= 0 && rs.absolute(start+1)) {
 405.              boolean hasNext = false;
 406.              List items = new ArrayList();
 407.              do {
 408.                i = 1;
 409.                items.add(new Item(rs.getString(i++).trim(),
 410.                                   rs.getString(i++).trim(),
 411.                                   rs.getString(i++),

 415.                                   rs.getString(i++),

 417.                                   rs.getString(i++),
 418.                                   rs.getString(i++),
 419.                                   rs.getString(i++),
 420.                                   rs.getDouble(i++),
 421.                                   rs.getDouble(i++)));
 422.              } while ((hasNext = rs.next()) && (--count > 0));
 423.              ret = new Page(items, start, hasNext);
 424.            } else {
 425.              ret = Page.EMPTY_PAGE;
 426.            }
 427.    
 428.            rs.close();
 429.            ps.close();

 345.      public Page searchItems(String searchQuery, int start, int count, Locale l)
 346.        throws CatalogDAOSysException {
 347.          Collection keywords = new HashSet();
 348.          StringTokenizer st = new StringTokenizer(searchQuery);
 349.          while (st.hasMoreTokens()) {
 350.            keywords.add(st.nextToken());
 351.          }
 352.          if (keywords.isEmpty()) {
 353.            return Page.EMPTY_PAGE;
 354.          }
 355.    
 356.          Connection c = null;
 357.          PreparedStatement ps = null;
 358.          ResultSet rs = null;
 359.          Page ret = null;
 360.    
 361.          try {
 362.            c = getDataSource().getConnection();
 363.            Iterator it;
 364.            int i;
 365.            StringBuffer sb = new StringBuffer();
 366.            sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[0]);
 367.            int keywordsSize = keywords.size();
 368.            if (keywordsSize > 0) {
 369.              sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[1]);
 370.              for (i = 1; i != keywordsSize; i++) {
 371.                sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[2]);
 372.              }

 374.              for (i = 1; i != keywordsSize; i++) {
 375.                sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[4]);
 376.              }
 377.              sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[5]);

 379.                sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[6]);
 380.              }
 381.              sb.append(SEARCH_ITEMS_STATEMENT_FRAGMENTS[7]);
 382.            }
 383.            //System.err.println(sb.toString());
 384.            ps = c.prepareStatement(sb.toString(),
 385.                                    ResultSet.TYPE_SCROLL_INSENSITIVE,
 386.                                    ResultSet.CONCUR_READ_ONLY);
 387.            ps.setString(1, l.toString());
 388.            // The three loops are necessary because of the way the

 396.              String keyword = ((String) it.next()).toLowerCase();
 397.              ps.setString(i, "%" + keyword + "%");
 398.            }
 399.            for (it = keywords.iterator(); it.hasNext(); i++) {
 400.              String keyword = ((String) it.next()).toLowerCase();
 401.              ps.setString(i, "%" + keyword + "%");
 402.            }
 403.            rs = ps.executeQuery();
 404.            if (start >= 0 && rs.absolute(start+1)) {
 405.              boolean hasNext = false;
 406.              List items = new ArrayList();
 407.              do {
 408.                i = 1;
