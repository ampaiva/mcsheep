 146.        public void process (HttpServletRequest request,
 147.                              HttpServletResponse  response)
 148.            throws IOException, ServletException {
 149.    
 150.            String screenName = null;
 151.            String localeString = null;
 152.            String selectedUrl = request.getRequestURI();
 153.            if (request.getSession().getAttribute(WebKeys.CURRENT_URL) != null) {
 154.                request.getSession().removeAttribute(WebKeys.CURRENT_URL);
 155.            }
 156.    
 157.            String languageParam = request.getParameter("locale");
 158.            // The language when specified as a parameter overrides the language set in the session
 159.            if (languageParam != null) {
 160.                localeString = languageParam;
 161.            } else if (request.getSession().getAttribute(WebKeys.LOCALE) != null) {
 162.                localeString = ((Locale)request.getSession().getAttribute(WebKeys.LOCALE)).toString();
 163.            }
 164.            if (allScreens.get(localeString) == null) {
 165.               localeString = defaultLocale;
 166.            }
 167.    
 168.            // get the screen name
 169.            int lastPathSeparator = selectedUrl.lastIndexOf("/") + 1;
 170.            int lastDot = selectedUrl.lastIndexOf(".");
 171.            if (lastPathSeparator != -1 && lastDot != -1 && lastDot > lastPathSeparator) {
 172.                screenName = selectedUrl.substring(lastPathSeparator, lastDot);
 173.            }
 174.             // check if request is for the previous screen
 175.            if (screenName.equals(PREVIOUS_SCREEN)) {
 176.                String longScreenName  =  (String)request.getSession().getAttribute(WebKeys.PREVIOUS_SCR...
 177.                int lastDot2 = longScreenName.lastIndexOf(".");
 178.                if ( lastDot2 != -1 && lastDot2 > 0) {
 179.                    screenName = longScreenName.substring(0, lastDot2);
 180.                 }
 181.                // put the request attributes stored in the session in the request
 182.                if (cachePreviousScreenParameters) {
 183.                    Map previousParams = (Map)request.getSession().getAttribute(WebKeys.PREVIOUS_REQUEST...
 184.                    Map params = (Map)request.getParameterMap();
 185.                    Iterator it = previousParams.keySet().iterator();
 186.                    while (it.hasNext()) {
 187.                        String key = (String)it.next();
 188.                        Object value = previousParams.get(key);
 189.                        params.put(key,value);
 190.                    }
 191.                }
 192.                // put in the previous request attributes
 193.                if (cachePreviousScreenAttributes) {
 194.                    Map previousAttributes = (Map)request.getSession().getAttribute(WebKeys.PREVIOUS_REQ...

 197.                    while (it2.hasNext()) {
 198.                        String key = (String)it2.next();
 199.                        Object value = previousAttributes.get(key);
 200.                        request.setAttribute(key,value);
 201.                    }
 202.                }
 203.            } else {
 204.                      String extension = selectedUrl.substring(lastDot, selectedUrl.length());
 205.                      request.getSession().setAttribute(WebKeys.PREVIOUS_SCREEN, screenName + extension)...
 206.                      if (cachePreviousScreenParameters) {
 207.                          // copy all the parameters into a new map
 208.                          HashMap newParams =  new HashMap();
 209.                          Map params = (Map)request.getParameterMap();
 210.                          Iterator it = params.keySet().iterator();
 211.                         // put the request attributes stored in the session in the request
 212.                         while (it.hasNext()) {
 213.                             String key = (String)it.next();
 214.                             Object value = params.get(key);
 215.                             newParams.put(key,value);

 217.                         request.getSession().setAttribute(WebKeys.PREVIOUS_REQUEST_PARAMETERS, newParam...
 218.                      }
 219.                      if (cachePreviousScreenAttributes) {

 221.                         HashMap attributes = new HashMap();
 222.                         Enumeration e_um = request.getAttributeNames();
 223.                         while (e_um.hasMoreElements()) {
 224.                             String key = (String)e_um.nextElement();
 225.                             Object value = request.getAttribute(key);

 228.                         request.getSession().setAttribute(WebKeys.PREVIOUS_REQUEST_ATTRIBUTES, attribut...

 232.            Screen screen = null;
 233.            if (screenName != null){
 234.                Screens localeScreens = (Screens)allScreens.get(localeString);
 235.                if (localeScreens != null) screen = (Screen)localeScreens.getScreen(screenName);

 242.                if (screen != null) {
 243.                    request.setAttribute(WebKeys.CURRENT_SCREEN, screen);

 252.                }
 253.            }

 256.        private void insertTemplate(HttpServletRequest request,
 257.                                    HttpServletResponse response,
 258.                                    String templateName)
 259.            throws IOException, ServletException {
 260.    
 261.            // This method tries to wrap the request dispatcher call with-in
 262.            // a transaction for making the local EJB access efficient. However,
 263.            // it is not a fatal error, if such a transaction can not be started
 264.            // for some reason, so it handles them gracefully.
 265.    
 266.            boolean tx_started = false;
 267.            UserTransaction ut = null;
 268.    
 269.            try {
 270.                // Lookup the UserTransaction object
 271.                InitialContext ic = new InitialContext();
 272.                ut = (UserTransaction) ic.lookup("java:comp/UserTransaction");
 273.                ut.begin();         // start the transaction.
 274.                tx_started = true;
 275.            } catch (NamingException ne) {
 276.                // it should not have happened, but it is a recoverable error.
 277.                // Just dont start the transaction.
 278.                ne.printStackTrace();
 279.            } catch (NotSupportedException nse) {
 280.                // Again this is a recoverable error.
 281.                nse.printStackTrace();
 282.            } catch (SystemException se) {
 283.                // Again this is a recoverable error.
 284.                se.printStackTrace();
 285.            }
 286.    
 287.            try {
 288.                context.getRequestDispatcher(templateName).forward(request, response);
 289.            } finally {
 290.                // commit the transaction if it was started earlier successfully.
 291.                try {
 292.                    if (tx_started && ut != null) {
 293.                        ut.commit();
 294.                    }
 295.                } catch (IllegalStateException re) {
 296.                    re.printStackTrace();
 297.                } catch (RollbackException re) {
 298.                    re.printStackTrace();
 299.                } catch (HeuristicMixedException hme) {
 300.                    hme.printStackTrace();
 301.                } catch (HeuristicRollbackException hre) {
 302.                    hre.printStackTrace();
 303.                } catch (SystemException se) {
 304.                    se.printStackTrace();
 305.                }
 306.            }
 307.        }
