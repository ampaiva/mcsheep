 175.      /**
 176.       * Process the list of order approvals and update database. Send a
 177.       * PurchaseOrder to a supplier for each approved order. Also generate
 178.       * a list of approved or denied orders.
 179.       *
 180.       * @return a list of valid order approvals/denys to be sent
 181.       *         to customer relations. Or return null, if list empty
 182.       *         or no valid orders AND also the list of Purchase
 183.       *         Orders to send to supplier to fullfill the order
 184.       */
 185.      private WorkResult doWork(String xmlMessage) throws XMLDocumentException, FinderException  {
 186.    
 187.        ArrayList supplierPoList = new ArrayList();
 188.        PurchaseOrderLocal po = null;
 189.        OrderApproval approval = null;
 190.    
 191.        approval = OrderApproval.fromXML(xmlMessage, entityCatalogURL, validateXmlOrderApproval);
 192.    
 193.        //generate list of valid orders to return
 194.        OrderApproval oaMailList = new OrderApproval();
 195.        String xmlMailOrderApprovals = null;
 196.        Collection coll = approval.getOrdersList();
 197.        Iterator it = coll.iterator();
 198.    
 199.        while(it!= null && it.hasNext()) {
 200.          ChangedOrder co = (ChangedOrder) it.next();
 201.          // only PENDING orders can be updated
 202.          //if order alreay APPROVED or DENIED or COMPLETED, then order
 203.          //already processed so dont process again
 204.          String curStatus = processManager.getStatus(co.getOrderId());
 205.          if(!curStatus.equals(OrderStatusNames.PENDING)) {
 206.            continue;
 207.          }
 208.          //generate list of valid orders to return
 209.          //list contains orders to notify the customer of
 210.          //order status changes. List is sent to customer
 211.          //relations for emailing.
 212.          oaMailList.addOrder(co);
 213.    
 214.          //update process manager
 215.          //for this purchase order workflow
 216.          processManager.updateStatus(co.getOrderId(), co.getOrderStatus());
 217.    
 218.          //for all approved orders, send a PO to supplier
 219.          if(co.getOrderStatus().equals(OrderStatusNames.APPROVED)) {
 220.            po = poHome.findByPrimaryKey(co.getOrderId());
 221.            String xmlPO = getXmlPO(po, supplierOrderXDE);
 222.            supplierPoList.add(xmlPO);
 223.          }
 224.        }//end while
 225.    
 226.        xmlMailOrderApprovals = oaMailList.toXML();
 227.        return new WorkResult(xmlMailOrderApprovals, supplierPoList);
 228.      }

 249.        supplierOrderXDE.newDocument();
