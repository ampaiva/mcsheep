  92.        private Document doHttpPost(String request) {
  93.            InputStream ist = null;
  94.            OutputStream ost = null;
  95.            HttpURLConnection uc = null;
  96.    
  97.            try {
  98.                /* Initialize the HttpURLConnection */
  99.    
 100.                uc = (HttpURLConnection)(url.openConnection());
 101.                uc.setRequestMethod("POST");
 102.                uc.setDoOutput(true);
 103.                uc.setUseCaches(false);
 104.                uc.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
 105.                uc.setRequestProperty("JSESSIONID", "HAHAHAHA");
 106.                uc.connect();
 107.    
 108.                /* Send the post request */
 109.    
 110.                ost = uc.getOutputStream();
 111.                DataOutputStream out = new DataOutputStream(ost);
 112.                out.writeBytes(request);
 113.                out.flush();
 114.                out.close();
 115.                ost = null;
 116.    
 117.                /* Read and print the response */
 118.    
 119.                ist = uc.getInputStream();
 120.                BufferedReader in = new BufferedReader(new InputStreamReader(ist));
 121.                StringBuffer sb = new StringBuffer();
 122.                char[] c = new char[1];
 123.                while (in.read(c, 0, 1) == 1) {
 124.                    sb.append(c[0]);
 125.                }
 126.    
 127.                InputSource src = new InputSource(new StringReader(sb.toString()));
 128.                return documentBuilder.parse(src);
 129.            } catch (Exception e) {
 130.                if (uc != null) {
 131.                    uc.disconnect();
 132.                }
 133.                return null;
 134.            } finally { // Carefully close the input and output streams
 135.                if (ost != null) {
 136.                    try {
 137.                        ost.close();
 138.                    }
 139.                    catch (IOException ignore) {
 140.                    }
 141.                }
 142.                if (ist != null) {
 143.                    try {
 144.                        ist.close();
 145.                    }
 146.                    catch (IOException ignore) {
 147.                    }
 148.                }
 149.            }
 150.        }

 284.        private PetStoreProxy.Order parseOrder(Node orderNode) {
 285.            String id = null;
 286.            String userId = null;
 287.            Date date = null;
 288.            float amount = -1.0f;
 289.            String status = null;
 290.            NodeList children = orderNode.getChildNodes();
 291.            for(int i = 0; i < children.getLength(); i++) {
 292.                Node child = children.item(i);
 293.                if (child.getNodeType() != Node.ELEMENT_NODE) {
 294.                    continue;
 295.                }
 296.                String name = child.getNodeName();
 297.                if (name.equals("OrderId")) {
 298.                    id = getBody(child);
 299.                }
 300.                else if (name.equals("UserId")) {
 301.                    userId = getBody(child);
 302.                }
 303.                else if (name.equals("OrderDate")) {
 304.                    date = getDate(child);
 305.                }
 306.                else if (name.equals("OrderAmount")) {
 307.                    amount = getFloat(child);
 308.                }
 309.                else if (name.equals("OrderStatus")) {
 310.                    status = parseOrderStatus(getBody(child));
 311.                }
 312.            }
 313.            return new PetStoreProxy.Order(id, userId, date, amount, status);
 314.        }
