 179.            private void determineChartValues(Graphics2D g2) {
 180.                // Already checked for null in caller renderChart.
 181.                maxVal = 0f;
 182.                for (int i = 0; i < keys.length; i++) {
 183.                    if (chartModel.getOrders(keys[i]) > maxVal) {
 184.                        maxVal = chartModel.getOrders(keys[i]);
 185.                    }
 186.                }
 187.                valInterval = maxVal / (NUM_TICS - 1);
 188.    
 189.                float val = 0f;
 190.                FontMetrics fm = g2.getFontMetrics();
 191.    
 192.                // Find the largest label on the y-axis so we can determine the
 193.                // amount to indent the left and right sides.
 194.                int maxStrLen = 0;
 195.                int currStrLen = 0;
 196.                for (int i = 0; i < keys.length; i++) {
 197.                    currStrLen = fm.stringWidth(Float.toString(val));
 198.                    if (currStrLen > maxStrLen) {
 199.                        maxStrLen = currStrLen;
 200.                    }
 201.                    val += valInterval;
 202.                }
 203.    
 204.                // Find the max height of the font being used to determine the
 205.                // amount to indent the top and bottom of the chart.
 206.                int maxStrHeight = fm.getHeight();
 207.                INSET_LEFT = maxStrLen + (2 * EMPTY_SPACE) + TIC_LEN;
 208.                INSET_TOP = maxStrHeight + (2 * EMPTY_SPACE) + TIC_LEN;
 209.    
 210.                barWidth = ((float)getWidth() - 2 * INSET_LEFT)
 211.                    / (float)keys.length;
 212.            }

 214.            protected void drawAxes(Graphics2D g2) {
 215.                // Already checked for null in caller renderChart.
 216.                Color oldColor = g2.getColor();
 217.                int height = getHeight();
 218.                int width = getWidth();
 219.                float val = 0f;
 220.                FontMetrics fm = g2.getFontMetrics();
 221.                int maxStrHeight = fm.getHeight();
 222.    
 223.                g2.setColor(Color.black);
 224.    
 225.                // Draw y-axis.
 226.                g2.drawLine(INSET_LEFT, INSET_TOP, INSET_LEFT, height - INSET_TOP);
 227.    
 228.                // Draw x-axis.
 229.                g2.drawLine(INSET_LEFT, height - INSET_TOP,
 230.                    width - INSET_LEFT, height - INSET_TOP);
 231.    
 232.                // Draw tics and labels on x-axis.
 233.                int strLen = 0;
 234.                int x = INSET_LEFT + (int)(barWidth / 2);
 235.                for (int i = 0; i < keys.length; i++) {
 236.                    g2.drawLine(x, height - INSET_TOP,
 237.                        x, height - INSET_TOP + TIC_LEN);
 238.                    strLen = fm.stringWidth(keys[i]);
 239.                    g2.drawString(keys[i], x - strLen / 2,
 240.                        height - INSET_TOP + (fm.getMaxAscent()) + EMPTY_SPACE
 241.                        + TIC_LEN);
 242.                    x += (int)barWidth;
 243.                }
 244.    
 245.                //Draw tics and labels on y-axis.
 246.                int y = height - INSET_TOP;
 247.                x = INSET_LEFT;
 248.                int ticInterval = (height - (2 * INSET_TOP)) / (NUM_TICS - 1);
 249.                String str;
 250.                for (int i = 0; i < NUM_TICS - 1; i++) {
 251.                    g2.drawLine(INSET_LEFT, y, INSET_LEFT - TIC_LEN, y);
 252.                    strLen = fm.stringWidth(str = Float.toString(val));
 253.                    g2.drawString(str, x - strLen - EMPTY_SPACE - TIC_LEN,
 254.                        y + (maxStrHeight / 2));
 255.                    y -= ticInterval;
 256.                    val += valInterval;
 257.    
 258.                }
 259.                // The last tic is drawn from the known axis point due to round-off
 260.                // errors in the ticInterval that may have accumulated.  This
 261.                // doesn't look great...  A fix for this is to spread the error by
 262.                // finding the center tic first and solve the rest by divide and
 263.                // conquer.
 264.                g2.drawLine(INSET_LEFT, INSET_TOP,
 265.                    INSET_LEFT - TIC_LEN, INSET_TOP);
 266.                strLen = fm.stringWidth(str = Float.toString(val));
 267.                g2.drawString(str, x - strLen - EMPTY_SPACE - TIC_LEN,
 268.                    y + (maxStrHeight / 2));
 269.    
 270.                g2.setColor(oldColor);
 271.            }

 273.            public void renderChart(Graphics2D g2) {
 274.                keys = chartModel.getKeys();
 275.                if (keys == null) {
 276.                    // TODO: Draw a message saying we have no data.
 277.                    return;
 278.                }
 279.    
 280.                calculateTotals();
 281.                bars = new Rectangle[keys.length];
 282.                determineChartValues(g2);
 283.    
 284.                float x = INSET_LEFT + 1;
 285.                float y = getHeight() - INSET_TOP;
 286.                float maxHeight = getHeight() - (2 * INSET_TOP);
 287.                Color oldColor;
 288.                for (int i = 0; i < keys.length; i++) {
 289.                    float barHeight = (chartModel.getOrders(keys[i]) * maxHeight)
 290.                        / maxVal;
 291.                    oldColor = g2.getColor();
 292.                    g2.setColor(colorList[i % keys.length]);
 293.                    bars[i] = new Rectangle((int)x, (int)(maxHeight - barHeight
 294.                        + INSET_TOP), (int)barWidth, (int)barHeight);
 295.                    g2.fill3DRect(bars[i].x, bars[i].y, bars[i].width,
 296.                        bars[i].height, true);
 297.                    x += barWidth;
 298.                    g2.setColor(oldColor);
 299.                }
 300.                drawAxes(g2);
 301.            }

 111.        private void updateModelDates() {
 112.            Date startDate = null;
 113.            Date endDate = null;
 114.    
 115.            // Make sure that the dates entered in the text field are valid.  If
 116.            // they are not prompt the user to fix them.
 117.            try {
 118.                startDate =
 119.                    dateFormat.parse(startDateTxtField.getText());
 120.                endDate =
 121.                    dateFormat.parse(endDateTxtField.getText());
 122.            } catch (ParseException e) {
 123.                JOptionPane.showMessageDialog(this,
 124.                    PetStoreAdminClient.getString("DateFormatErrorDialog.message"),
 125.                    PetStoreAdminClient.getString("DateFormatErrorDialog.title"),
 126.                    JOptionPane.ERROR_MESSAGE);
 127.                return;
 128.            }
 129.    
 130.            // Setting the dates will automatically retrieve the data from the
 131.            // server.
 132.            pieChartModel.setDates(startDate, endDate);
 133.        }

 147.                pieChart.repaint();

 149.        }

 151.        /**
 152.         * Displays a pie chart of total sales (in $s) of a category divided into
 153.         * sales (in $s) per item of that category in a given period of days.
 154.         */
 155.        public class PieChart extends Chart {
 156.    
 157.            private Arc2D.Float[] arcs;
 158.            private String[] keys;
 159.            private JToolTip toolTip = new JToolTip();
 160.            private float[] percentages;
 161.            private int CHART_INDENT =
 162.                PetStoreAdminClient.getInteger("PieChart.indent");
 163.    
 164.            /**
 165.             * Create an instance of the PieChart class.
 166.             *
 167.             * @param pieChartModel Model to retrieve data from.
 168.             * @param width Width of the pie chart.
 169.             * @param height Height of the pie chart.
 170.             */
 171.            public PieChart(DataSource.PieChartModel pieChartModel) {
 172.                super(pieChartModel);
 173.            }
 174.    
 175.            public JToolTip createToolTip() {
 176.                toolTip.setComponent(this);
 177.                return toolTip;
 178.            }
 179.    
 180.            protected void renderLegend(Graphics2D g2) {
 181.                FontMetrics fm = g2.getFontMetrics();
 182.                int boxWH = fm.getHeight();
 183.                int y = 10;
 184.                Color oldColor = g2.getColor();
 185.                for (int i = 0; i < keys.length; i++) {
 186.                    // Draw a box.
 187.                    g2.setColor(colorList[i % keys.length]);
 188.                    g2.fillRect(10, y, boxWH, boxWH);
 189.                    // Draw the string.
 190.                    g2.setColor(Color.black);
 191.                    g2.drawString(keys[i], 10 + boxWH + 2,
 192.                        y + boxWH - fm.getDescent());
 193.                    y += boxWH + 2;
 194.                }
 195.                g2.setColor(oldColor);
 196.            }
 197.    
 198.            public void renderChart(Graphics2D g2) {
 199.                keys = chartModel.getKeys();
 200.                if (keys == null) {
 201.                    return;
 202.                }
 203.    
 204.                calculateTotals();
 205.                percentages = new float[keys.length];
 206.    
 207.                for (int i = 0; i < keys.length; i++) {
 208.                    percentages[i] =
 209.                        chartModel.getRevenue(keys[i]) / totalRevenue * 100f;
 210.                }
 211.    
 212.                int width = getWidth() - CHART_INDENT;
 213.                int height = getHeight() - CHART_INDENT;
 214.                float pieWH = (float)width;
 215.                float startX = 20f;
 216.                float startY = 20f;
 217.    
 218.                // Keep the perspective ratio on the pie chart 1:1
 219.                if (width < height) {
 220.                    pieWH = (float)width;
 221.                    startY += (float)((height - width) / 2);
 222.                } else if (height < width) {
 223.                    pieWH = (float)height;
 224.                    startX += (float)((width - height) / 2);
 225.                }
 226.    
 227.                float angle = 0f;
 228.                Color oldColor;
 229.                arcs = new Arc2D.Float[percentages.length];
 230.                for (int i = 0; i < percentages.length; i++) {
 231.                    // Convert percentage to coordinate system
 232.                    // (percent * 360 / 100).
 233.                    float extent = percentages[i] * 360f / 100f;
 234.                    arcs[i] = new Arc2D.Float(startX, startY, pieWH, pieWH,
 235.                            angle, extent, Arc2D.PIE);
 236.                    angle += extent;
 237.    
 238.                    oldColor = g2.getColor();
 239.                    g2.setColor(colorList[i % percentages.length]);
 240.                    g2.fill(arcs[i]);
 241.                    g2.setColor(Color.black);
 242.                    g2.draw(arcs[i]);
 243.                    g2.setColor(oldColor);
 244.                }
 245.                renderLegend(g2);
 246.            }
 247.    
 248.            public boolean contains(int x, int y) {
 249.                boolean result = false;
 250.                if (arcs == null) {
 251.                    return result;
 252.                }
 253.    
 254.                for (int i = 0; i < arcs.length; i++) {
 255.                    if (arcs[i].contains(x, y)) {
 256.                        setToolTipText(keys[i] + " ("
 257.                            + df.format(percentages[i]) + "%)");
 258.                        result = true;
 259.                    }
 260.                }
 261.                return result;
 262.            }

 276.        }
