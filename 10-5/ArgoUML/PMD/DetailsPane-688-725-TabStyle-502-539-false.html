 303.        public StylePanel findPanelFor(Class targetClass) {
 304.            Class panelClass = null;
 305.            TabFigTarget p = panels.get(targetClass);
 306.            if (p == null) {
 307.                Class newClass = targetClass;
 308.                while (newClass != null && panelClass == null) {
 309.                    panelClass = panelClassFor(newClass);
 310.                    newClass = newClass.getSuperclass();
 311.                }
 312.                if (panelClass == null) {
 313.                    return null;
 314.                }
 315.                try {
 316.                    p = (TabFigTarget) panelClass.newInstance();
 317.                } catch (IllegalAccessException ignore) {
 318.                    LOG.error(ignore);
 319.                    return null;
 320.                } catch (InstantiationException ignore) {
 321.                    LOG.error(ignore);
 322.                    return null;
 323.                }
 324.                panels.put(targetClass, p);
 325.            }
 326.            LOG.debug("found style for " + targetClass.getName() + "("
 327.                    + p.getClass() + ")");
 328.            return (StylePanel) p;
 329.    
 330.        }

 332.        /**
 333.         * Get the class for the required stylepanel.
 334.         *
 335.         * @param targetClass the class of the current selected target.
 336.         * @return the panel class for the class given or
 337.         * null if none available.
 338.         */
 339.        public Class panelClassFor(Class targetClass) {
 340.            if (targetClass == null) {
 341.                return null;
 342.            }
 343.    
 344.            StringNamespace classNs = (StringNamespace) StringNamespace
 345.                    .parse(targetClass);
 346.    
 347.            StringNamespace baseNs = (StringNamespace) StringNamespace.parse(
 348.                    "org.argouml.ui.", Namespace.JAVA_NS_TOKEN);
 349.    
 350.            StringNamespaceElement targetClassElement =
 351.                (StringNamespaceElement) classNs.peekNamespaceElement();
 352.    
 353.            LOG.debug("Attempt to find style panel for: " + classNs);
 354.    
 355.            classNs.popNamespaceElement();
 356.    
 357.            String[] bases = new String[] {
 358.                    classNs.toString(), baseNs.toString() 
 359.            };
 360.            for (String stylePanelName : stylePanelNames) {
 361.                for (String baseName : bases) {
 362.                    String name = baseName + "." + stylePanelName
 363.                            + targetClassElement;
 364.                    Class cls = loadClass(name);
 365.                    if (cls != null) {
 366.                        return cls;
 367.                    }
 368.                }
 369.            }
 370.            return null;
 371.        }

 377.            } catch (ClassNotFoundException ignore) {
 378.                LOG.debug("ClassNotFoundException. Could not find class:"
 379.                        + name);
 380.            }
 381.            return null;
 382.        }
