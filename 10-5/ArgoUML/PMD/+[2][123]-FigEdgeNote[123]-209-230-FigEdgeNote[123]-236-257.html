  63.    /**
  64.     * Class to display a UML note connection to a annotated model element.
  65.     * <p>
  66.     * 
  67.     * The owner of this fig is always a CommentEdge. Because it is different from
  68.     * most every other FigEdge in ArgoUML, it doesn't subclass FigEdgeModelElement.
  69.     * 
  70.     * @author Andreas Rueckert a_rueckert@gmx.net
  71.     * @author jaap.branderhorst@xs4all.nl
  72.     */
  73.    public class FigEdgeNote extends FigEdgePoly
  74.            implements ArgoFig, DiagramElement, Owned, IItemUID, PropertyChangeListener {
  75.    
  76.        private static final Logger LOG = Logger.getLogger(FigEdgeNote.class);
  77.    
  78.        private Object comment;
  79.        private Object annotatedElement;
  80.    
  81.        private DiagramSettings settings;
  82.    
  83.        private ItemUID itemUid;
  84.        
  85.        /**
  86.         * @param element owning CommentEdge object. This is a special case since it
  87.         *            is not a UML element.
  88.         * @param theSettings render settings
  89.         */
  90.        public FigEdgeNote(Object element, DiagramSettings theSettings) {
  91.            // element will normally be null when called from PGML parser
  92.            // It will get it's source & destination set later in attachEdges
  93.            super();
  94.            settings = theSettings;
  95.    
  96.            if (element != null) {
  97.                super.setOwner(element);
  98.            } else {
  99.                super.setOwner(new CommentEdge());
 100.            }
 101.            
 102.            setBetweenNearestPoints(true);
 103.            getFig().setLineWidth(LINE_WIDTH);
 104.            getFig().setDashed(true);
 105.            
 106.            // Unfortunately the Fig and it's associated CommentEdge will not be
 107.            // fully initialized yet here if we're being loaded from a PGML file.
 108.            // The remainder of the initialization will happen when 
 109.            // set{Dest|Source}FigNode are called from PGMLStackParser.attachEdges()
 110.        }
 111.    
 112.        /*
 113.         * @see org.tigris.gef.presentation.FigEdge#setFig(org.tigris.gef.presentation.Fig)
 114.         */
 115.        @Override
 116.        public void setFig(Fig f) {
 117.            LOG.info("Setting the internal fig to " + f);
 118.            super.setFig(f);
 119.            getFig().setDashed(true);
 120.        }
 121.    
 122.    
 123.        /*
 124.         * @see java.lang.Object#toString()
 125.         */
 126.        @Override
 127.        public String toString() {
 128.            return Translator.localize("misc.comment-edge");
 129.        }
 130.    
 131.        /*
 132.         * Listen for a RemoveAssociationEvent between the comment
 133.         * and the annotated element. When recieved delete the CommentEdge
 134.         * and this FigEdgeNote.
 135.         * @see org.argouml.uml.diagram.ui.FigEdgeModelElement#modelChanged(java.beans.PropertyChangeEve...
 136.         */
 137.        protected void modelChanged(PropertyChangeEvent e) {
 138.            if (e instanceof RemoveAssociationEvent
 139.                    && e.getOldValue() == annotatedElement) {
 140.                removeFromDiagram();
 141.            }
 142.        }
 143.        
 144.        /*
 145.         * @see org.tigris.gef.presentation.Fig#getTipString(java.awt.event.MouseEvent)
 146.         */
 147.        @Override
 148.        public String getTipString(MouseEvent me) {
 149.            return "Comment Edge"; // TODO: get tip string from comment
 150.        }
 151.    
 152.    
 153.        /*
 154.         * @see java.beans.PropertyChangeListener#propertyChange(java.beans.PropertyChangeEvent)
 155.         */
 156.        @Override
 157.        public void propertyChange(PropertyChangeEvent pve) {
 158.            modelChanged(pve);
 159.        }
 160.    
 161.     
 162.    
 163.        /*
 164.         * @see org.tigris.gef.presentation.Fig#removeFromDiagram()
 165.         */
 166.        @Override
 167.        public final void removeFromDiagram() {
 168.            if (comment != null) {
 169.                removeElementListener(comment);
 170.            }
 171.    
 172.            super.removeFromDiagram();
 173.            damage();
 174.        }
 175.     
 176.    
 177.        /**
 178.         * Returns the source of the edge. The source is the owner of the
 179.         * node the edge travels from in a binary relationship. For
 180.         * instance: for a classifierrole, this is the sender.
 181.         * @return MModelElement
 182.         */
 183.        protected Object getSource() {
 184.            Object theOwner = getOwner();
 185.            if (theOwner != null) {
 186.                return ((CommentEdge) theOwner).getSource();
 187.            }
 188.            return null;
 189.        }
 190.        /**
 191.         * Returns the destination of the edge. The destination is the
 192.         * owner of the node the edge travels to in a binary
 193.         * relationship. For instance: for a classifierrole, this is the
 194.         * receiver.
 195.         * @return Object
 196.         */
 197.        protected Object getDestination() {
 198.            Object theOwner = getOwner();
 199.            if (theOwner != null) {
 200.                return ((CommentEdge) theOwner).getDestination();
 201.            }
 202.            return null;
 203.        }
 204.        
 205.        /*
 206.         * @see org.tigris.gef.presentation.FigEdge#setDestFigNode(org.tigris.gef.presentation.FigNode)
 207.         */
 208.        @Override
 209.        public void setDestFigNode(FigNode fn) {
 210.            // When this is called from PGMLStackParser.attachEdges, we finished
 211.            // the initialization of owning pseudo element (CommentEdge)
 212.            if (fn != null && Model.getFacade().isAComment(fn.getOwner())) {
 213.                Object oldComment = comment;
 214.                if (oldComment != null) {
 215.                    removeElementListener(oldComment);
 216.                }
 217.                comment = fn.getOwner();
 218.                if (comment != null) {
 219.                    addElementListener(comment);
 220.                }
 221.                
 222.                ((CommentEdge) getOwner()).setComment(comment);
 223.            } else if (fn != null 
 224.                    && !Model.getFacade().isAComment(fn.getOwner())) {
 225.                annotatedElement = fn.getOwner();
 226.                ((CommentEdge) getOwner()).setAnnotatedElement(annotatedElement);
 227.            }
 228.    
 229.            super.setDestFigNode(fn);
 230.        }
 231.    
 232.        /*
 233.         * @see org.tigris.gef.presentation.FigEdge#setSourceFigNode(org.tigris.gef.presentation.FigNode...
 234.         */
 235.        @Override
 236.        public void setSourceFigNode(FigNode fn) {
 237.            // When this is called from PGMLStackParser.attachEdges, we finished
 238.            // the initialization of owning pseudo element (CommentEdge)
 239.            if (fn != null && Model.getFacade().isAComment(fn.getOwner())) {
 240.                Object oldComment = comment;
 241.                if (oldComment != null) {
 242.                    removeElementListener(oldComment);
 243.                }
 244.                comment = fn.getOwner();
 245.                if (comment != null) {
 246.                    addElementListener(comment);
 247.                }
 248.                ((CommentEdge) getOwner()).setComment(comment);
 249.            } else if (fn != null 
 250.                    && !Model.getFacade().isAComment(fn.getOwner())) {
 251.                annotatedElement = fn.getOwner();
 252.                ((CommentEdge) getOwner()).setAnnotatedElement(annotatedElement);
 253.            }
 254.            super.setSourceFigNode(fn);
 255.        }
 256.        
 257.        private void addElementListener(Object element) {
 258.            Model.getPump().addModelEventListener(this, element);
 259.        }
 260.        
 261.        private void removeElementListener(Object element) {
 262.            Model.getPump().removeModelEventListener(this, element);
 263.        }

 268.            return ArgoFigUtil.getProject(this);
 269.        }
 270.    
 271.        public DiagramSettings getSettings() {
 272.            return settings;
 273.        }

 276.      
 277.        }

 282.            // unimplemented
 283.        }
 284.    
 285.        public void setSettings(DiagramSettings theSettings) {
 286.            settings = theSettings;
 287.        }

 290.         * Setter for the UID

 292.         */
 293.        public void setItemUID(ItemUID newId) {
 294.            itemUid = newId;
 295.        }
 296.

 298.         * Getter for the UID
 299.         * @return the UID

 301.        public ItemUID getItemUID() {
 302.            return itemUid;
 303.        }

 305.    
 306.        /**
 307.         * Setting the owner of the Fig must be done in the constructor and not
 308.         * changed afterwards for all ArgoUML figs.
 309.         * 
 310.         * @param owner owning UML element
 311.         * @deprecated for 0.27.3 by tfmorris. Set owner in constructor. This method
 312.         *             is implemented in GEF, so we'll leave this implementation
 313.         *             here to block any attempts to use it within ArgoUML.
 314.         */
 315.        @SuppressWarnings("deprecation")
 316.        @Deprecated
 317.        public void setOwner(Object owner) {
 318.            if (owner != getOwner()) {
 319.                throw new UnsupportedOperationException(
 320.                        "Owner must be set in constructor and left unchanged");
 321.            }
 322.        }
