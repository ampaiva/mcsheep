 142.         * Changed <em>lay</em> from <em>LayerPerspective</em> to
 143.         * <em>LayerPerspectiveMutable</em>.  This class is a child of
 144.         * <em>LayerPerspective</em> and was implemented to correct some
 145.         * difficulties in changing the model. <em>lay</em> is used mainly
 146.         * in <em>LayerManager</em>(GEF) to control the adding, changing and
 147.         * deleting layers on the diagram...<p>

 150.         * @author psager@tigris.org Jan. 24, 2002

 153.            if (!Model.getFacade().isANamespace(handle)) {
 154.                LOG.error(
 155.                    "Illegal argument. Object " + handle + " is not a namespace");
 156.                throw new IllegalArgumentException(
 157.                    "Illegal argument. Object " + handle + " is not a namespace");
 158.            }
 159.            Object m = handle;
 160.            boolean init = (null == getNamespace());
 161.            super.setNamespace(m);
 162.            DeploymentDiagramGraphModel gm = createGraphModel();
 163.            gm.setHomeModel(m);
 164.            if (init) {
 165.                LayerPerspective lay =
 166.                    new LayerPerspectiveMutable(Model.getFacade().getName(m), gm);
 167.                DeploymentDiagramRenderer rend = new DeploymentDiagramRenderer();
 168.                lay.setGraphNodeRenderer(rend);
 169.                lay.setGraphEdgeRenderer(rend);
 170.                setLayer(lay);
 171.            }
 172.        }

 579.        public DiagramElement createDiagramElement(
 580.                final Object modelElement,
 581.                final Rectangle bounds) {
 582.            
 583.            FigNodeModelElement figNode = null;
 584.            
 585.            DiagramSettings settings = getDiagramSettings();
 586.            
 587.            if (Model.getFacade().isANode(modelElement)) {
 588.                figNode = new FigMNode(modelElement, bounds, settings);
 589.            } else if (Model.getFacade().isAAssociation(modelElement)) {
 590.                figNode =
 591.                    createNaryAssociationNode(modelElement, bounds, settings);
 592.            } else if (Model.getFacade().isANodeInstance(modelElement)) {
 593.                figNode = new FigNodeInstance(modelElement, bounds, settings);
 594.            } else if (Model.getFacade().isAComponent(modelElement)) {
 595.                figNode = new FigComponent(modelElement, bounds, settings);
 596.            } else if (Model.getFacade().isAComponentInstance(modelElement)) {
 597.                figNode = new FigComponentInstance(modelElement, bounds, settings);
 598.            } else if (Model.getFacade().isAClass(modelElement)) {
 599.                figNode = new FigClass(modelElement, bounds, settings);
 600.            } else if (Model.getFacade().isAInterface(modelElement)) {
 601.                figNode = new FigInterface(modelElement, bounds, settings);
 602.            } else if (Model.getFacade().isAObject(modelElement)) {
 603.                figNode = new FigObject(modelElement, bounds, settings);
 604.            } else if (Model.getFacade().isAActor(modelElement)) {
 605.                figNode = new FigActor(modelElement, bounds, settings);
 606.            } else if (Model.getFacade().isAComment(modelElement)) {
 607.                figNode = new FigComment(modelElement, bounds, settings);
 608.            }
 609.            
 610.            if (figNode != null) {
 611.                LOG.debug("Model element " + modelElement + " converted to " 
 612.                        + figNode);
 613.            } else {
 614.                LOG.debug("Dropped object NOT added " + figNode);
 615.            }
 616.            return figNode;
 617.        }

 477.        @Override
 478.        public boolean doesAccept(Object objectToAccept) {

 480.                return true;

 482.                return true;
 483.            } else if (Model.getFacade().isAComment(objectToAccept)) {

 488.                    return false;
 489.                } else if (Model.getFacade().isASubsystem(objectToAccept)) {
 490.                    return false;
 491.                }
 492.                return true;
 493.            }
 494.            return false;

 498.        @Override
 499.        public DiagramElement drop(Object droppedObject, Point location) {
 500.            DiagramElement figNode = null;
 501.           
 502.            // If location is non-null, convert to a rectangle that we can use
 503.            Rectangle bounds = null;
 504.            if (location != null) {
 505.                bounds = new Rectangle(location.x, location.y, 0, 0);
 506.            }
 507.    
 508.            return createDiagramElement(droppedObject, bounds);
 509.        }

 511.        public DiagramElement createDiagramElement(
 512.                final Object modelElement,
 513.                final Rectangle bounds) {
 514.            
 515.            FigNodeModelElement figNode = null;
 516.            
 517.            DiagramSettings settings = getDiagramSettings();
 518.            
 519.            if (Model.getFacade().isAActor(modelElement)) {
 520.                figNode = new FigActor(modelElement, bounds, settings);
 521.            } else if (Model.getFacade().isAUseCase(modelElement)) {
 522.                figNode = new FigUseCase(modelElement, bounds, settings);
 523.            } else if (Model.getFacade().isAComment(modelElement)) {
 524.                figNode = new FigComment(modelElement, bounds, settings);
 525.            } else if (Model.getFacade().isAPackage(modelElement)) {
 526.                if (!Model.getFacade().isAModel(modelElement)
 527.                        && !Model.getFacade().isASubsystem(modelElement)) {
 528.                    /* If we do not exclude a Model here, then dropping the 
 529.                     * Model on a UseCase diagram causes a package 
 530.                     * to be drawn. */
 531.                    figNode = new FigPackage(modelElement, bounds, settings);
 532.                }
 533.            }
 534.            if (figNode != null) {
 535.                LOG.debug("Model element " + modelElement + " converted to " 
 536.                        + figNode);
 537.            } else {
 538.                LOG.debug("Dropped object NOT added " + figNode);
 539.            }
 540.            return figNode;
 541.        }
