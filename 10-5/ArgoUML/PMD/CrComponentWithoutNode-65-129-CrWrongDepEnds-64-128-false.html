 101.        @Override
 102.        public boolean stillValid(ToDoItem i, Designer dsgr) {
 103.        if (!isActive()) {
 104.                return false;
 105.            }
 106.        ListSet offs = i.getOffenders();
 107.        UMLDeploymentDiagram dd = (UMLDeploymentDiagram) offs.get(0);
 108.        //if (!predicate(dm, dsgr)) return false;
 109.        ListSet newOffs = computeOffenders(dd);
 110.        boolean res = offs.equals(newOffs);
 111.        return res;
 112.        }
 113.    
 114.        /**
 115.         * If there are deps that are going from inside a FigComponent to
 116.         * inside a FigComponentInstance the returned vector-set is not
 117.         * null.  Then in the vector-set are the UMLDeploymentDiagram and
 118.         * all FigDependencies with this characteristic and their
 119.         * FigObjects described over the supplier and client.
 120.         *
 121.         * @param dd the diagram to check
 122.         * @return the set of offenders
 123.         */
 124.        public ListSet computeOffenders(UMLDeploymentDiagram dd) {
 125.        Collection figs = dd.getLayer().getContents();
 126.        ListSet offs = null;
 127.            for (Object obj : figs) {
 128.            if (!(obj instanceof FigDependency)) {
 129.                    continue;
 130.                }
 131.            FigDependency figDependency = (FigDependency) obj;
 132.            if (!(Model.getFacade().isADependency(figDependency.getOwner()))) {
 133.                    continue;
 134.                }
 135.            Object dependency = figDependency.getOwner();
 136.            Collection suppliers = Model.getFacade().getSuppliers(dependency);
 137.            int count = 0;
 138.            if (suppliers != null) {
 139.                    for (Object moe : suppliers) {
 140.                if (Model.getFacade().isAObject(moe)) {
 141.                Object objSup = moe;
 142.                if (Model.getFacade().getElementResidences(objSup)
 143.                        != null
 144.                    && (Model.getFacade().getElementResidences(objSup)
 145.                            .size() > 0)) {
 146.                    count += 2;
 147.                            }
 148.                if (Model.getFacade().getComponentInstance(objSup)
 149.                        != null) {
 150.                    count++;
 151.                            }
 152.                }
 153.            }
 154.            }
 155.            Collection clients = Model.getFacade().getClients(dependency);
 156.            if (clients != null && (clients.size() > 0)) {
 157.                    for (Object moe : clients) {
 158.                if (Model.getFacade().isAObject(moe)) {
 159.                Object objCli = moe;
 160.                if (Model.getFacade().getElementResidences(objCli)
 161.                        != null
 162.                    && (Model.getFacade().getElementResidences(objCli)
 163.                            .size() > 0)) {
 164.                    count += 2;
 165.                            }
 166.                if (Model.getFacade().getComponentInstance(objCli)
 167.                        != null) {
 168.                    count++;
 169.                            }
 170.                }
 171.            }
 172.            }
 173.            if (count == 3) {
 174.            if (offs == null) {
 175.                offs = new ListSet();
 176.                offs.add(dd);
 177.            }
 178.            offs.add(figDependency);
 179.            offs.add(figDependency.getSourcePortFig());
 180.            offs.add(figDependency.getDestPortFig());
 181.            }
 182.        }
 183.        return offs;
 184.        }
