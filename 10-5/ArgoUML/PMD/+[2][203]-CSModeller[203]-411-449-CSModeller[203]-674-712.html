  94.         * @param monitor    Progress monitor dialog displayed while parsing files
  95.         * @param startCount starting file number
  96.         */
  97.        public void model(List cNodes, ProgressMonitor monitor, int startCount) {
  98.    
  99.            int count = startCount;
 100.            phase = 0;
 101.            for (Object obj : cNodes) {
 102.                if (monitor.isCanceled()) {
 103.                    monitor.updateSubTask(
 104.                            Translator.localize("dialog.import.cancelled"));
 105.                    return;
 106.                }
 107.                cu = (CompilationUnitNode) obj;
 108.                addNamespace(cu.DefaultNamespace);
 109.                addNamespaceNodes(cu.Namespaces);
 110.                monitor.updateProgress(count++);
 111.            }
 112.            phase++;
 113.            for (Object obj : cNodes) {
 114.                if (monitor.isCanceled()) {
 115.                    monitor.updateSubTask(
 116.                            Translator.localize("dialog.import.cancelled"));
 117.                    return;
 118.                }
 119.                cu = (CompilationUnitNode) obj;
 120.                addNamespace(cu.DefaultNamespace);
 121.                addNamespaceNodes(cu.Namespaces);
 122.                monitor.updateProgress(count++);
 123.            }
 124.        }

 157.                } else if (phase == 1) {

 159.                    addProperties(cn, parent);
 160.                    addMethods(cn, parent);
 161.                    buildGeneralization(cn, ns);

 163.            }
 164.            for (InterfaceNode cn : ns.Interfaces) {
 165.                if (phase == 0) {
 166.                    addInterface(cn.Modifiers, 
 167.                            cn.Name.Identifier[cn.Name.Identifier.length - 1], 
 168.                            parent);
 169.                } else if (phase == 1) {
 170.                    addMethods(cn, parent);
 171.                }
 172.            }
 173.        }
