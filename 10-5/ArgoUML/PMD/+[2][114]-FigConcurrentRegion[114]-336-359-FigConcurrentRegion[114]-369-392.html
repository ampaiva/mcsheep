 407.            calcBounds(); //_x = x; _y = y; _w = w; _h = h;

 548.        @Override
 549.        protected void updateLayout(UmlChangeEvent event) {
 550.            if (!"container".equals(event.getPropertyName()) 
 551.                    && !"isConcurrent".equals(event.getPropertyName())) {
 552.                super.updateLayout(event);
 553.            }
 554.            final String eName = event.getPropertyName();
 555.            /*
 556.             * A Concurrent region cannot have incoming or outgoing transitions so
 557.             * incoming or outgoing transitions are redirected to its concurrent
 558.             * composite state container.
 559.             */
 560.            // TODO: This comparison is very suspect, it should use equals
 561.            // method. The code within the block is in fact never executed.
 562.            // I hesitate to change this now as it will trigger code has never been
 563.            // used before and am not aware of any problems that it usage may
 564.            // introduce.
 565.            // I do think that we need to be able to find a different way to
 566.            // implement the intent here which seems to be to correct edge drawings
 567.            // that should actually not be allowed - Bob
 568.            if (eName == "incoming" || eName == "outgoing") {
 569.                final Object owner = getOwner();
 570.                final Collection transactions = (Collection) event.getNewValue();
 571.                if (!transactions.isEmpty()) {
 572.                    final Object transition = transactions.iterator().next();
 573.                    if (eName == "incoming") {
 574.                        if (Model.getFacade().isATransition(transition)) {
 575.                            Model.getCommonBehaviorHelper().setTarget(transition,
 576.                                    Model.getFacade().getContainer(owner));
 577.                        }
 578.                    } else {
 579.                        if (Model.getFacade().isATransition(transition)) {
 580.                            Model.getStateMachinesHelper().setSource(transition,
 581.                                    Model.getFacade().getContainer(owner));
 582.                        }
 583.                    }
 584.                }
 585.            }
 586.        }

 218.        /**
 219.         * Override setBounds to keep shapes looking right. <p>
 220.         *
 221.         * When resized by this way, it only changes the height and the
 222.         * adjacent region's height.
 223.         *
 224.         * {@inheritDoc}
 225.         */
 226.        @Override
 227.        protected void setStandardBounds(int x, int y, int w, int h) {
 228.            if (getNameFig() == null) {
 229.                return;
 230.            }
 231.            Rectangle oldBounds = getBounds();
 232.            Dimension nameDim = getNameFig().getMinimumSize();
 233.            int adjacentindex = -1;
 234.            List regionsList = null;
 235.            int index = 0;
 236.            if (getEnclosingFig() != null) {
 237.                x = oldBounds.x;
 238.                w = oldBounds.width;
 239.                FigCompositeState f = ((FigCompositeState) getEnclosingFig());
 240.                regionsList = f.getEnclosedFigs();
 241.                index = regionsList.indexOf(this);
 242.    
 243.                /* if curHandle.index is 0 or 2,
 244.                 * the adjacent region is the previous region
 245.                 * but if it is 5 or 7, the adjacent region is the next region.
 246.                 * curHandle.index show which corner of the bound we are dragging.
 247.                 */
 248.                if (((curHandle.index == 0) || (curHandle.index == 2))
 249.                        && index > 0) {
 250.                    adjacentindex = index - 1;
 251.                }
 252.                if (((curHandle.index == 5) || (curHandle.index == 7))
 253.                        && (index < (regionsList.size() - 1))) {
 254.                    adjacentindex = index + 1;
 255.                }
 256.                if (h <= getMinimumSize().height) {
 257.                    if (h <= oldBounds.height) {
 258.                        h = oldBounds.height;
 259.                        y = oldBounds.y;
 260.                    }
 261.                }
 262.    
 263.                /* We aren't able to resize neither the top bound
 264.                 * from the first region nor
 265.                 * the bottom bound from the last region.
 266.                 */
 267.    
 268.                if (adjacentindex == -1) {
 269.                    x = oldBounds.x;
 270.                    y = oldBounds.y;
 271.                    h = oldBounds.height;
 272.    
 273.                    /*The group must be resized if a text field exceed the bounds*/
 274.                    if (w > f.getBounds().width) {
 275.                        Rectangle fR = f.getBounds();
 276.                        f.setBounds(fR.x, fR.y, w + 6, fR.height);
 277.                    }
 278.                } else {
 279.                    int hIncrement = oldBounds.height - h;
 280.                    FigConcurrentRegion adjacentFig =
 281.                        ((FigConcurrentRegion)
 282.                            regionsList.get(adjacentindex));
 283.                    if ((adjacentFig.getBounds().height + hIncrement)
 284.                            <= adjacentFig.getMinimumSize().height) {
 285.                        y = oldBounds.y;
 286.                        h = oldBounds.height;
 287.                    } else {
 288.                        if ((curHandle.index == 0) || (curHandle.index == 2)) {
 289.                            ((FigConcurrentRegion) regionsList.
 290.                                get(adjacentindex)).setBounds(0, hIncrement);
 291.                        }
 292.                        if ((curHandle.index == 5) || (curHandle.index == 7)) {
 293.                            ((FigConcurrentRegion) regionsList.
 294.                                get(adjacentindex)).setBounds(-hIncrement,
 295.                                        hIncrement);
 296.                        }
 297.                    }
 298.                }
 299.            }
 300.    
 301.            dividerline.setShape(x, y, x + w, y);
 302.            getNameFig().setBounds(x + MARGIN,
 303.                    y + SPACE_TOP,
 304.                    w - 2 * MARGIN,
 305.                    nameDim.height);
 306.            getInternal().setBounds(
 307.                    x + MARGIN,
 308.                    y + nameDim.height + SPACE_TOP + SPACE_MIDDLE,
 309.                    w - 2 * MARGIN,
 310.                    h - nameDim.height - SPACE_TOP - SPACE_MIDDLE - SPACE_BOTTOM);
 311.            getBigPort().setBounds(x, y, w, h);
 312.            cover.setBounds(x, y, w, h);
 313.    
 314.            calcBounds(); //_x = x; _y = y; _w = w; _h = h;
 315.            updateEdges();
 316.            firePropChange("bounds", oldBounds, getBounds());
 317.        }

 548.        @Override
 549.        protected void updateLayout(UmlChangeEvent event) {
 550.            if (!"container".equals(event.getPropertyName()) 
 551.                    && !"isConcurrent".equals(event.getPropertyName())) {
 552.                super.updateLayout(event);
 553.            }
 554.            final String eName = event.getPropertyName();
 555.            /*
 556.             * A Concurrent region cannot have incoming or outgoing transitions so
 557.             * incoming or outgoing transitions are redirected to its concurrent
 558.             * composite state container.
 559.             */
 560.            // TODO: This comparison is very suspect, it should use equals
 561.            // method. The code within the block is in fact never executed.
 562.            // I hesitate to change this now as it will trigger code has never been
 563.            // used before and am not aware of any problems that it usage may
 564.            // introduce.
 565.            // I do think that we need to be able to find a different way to
 566.            // implement the intent here which seems to be to correct edge drawings
 567.            // that should actually not be allowed - Bob
 568.            if (eName == "incoming" || eName == "outgoing") {
 569.                final Object owner = getOwner();
 570.                final Collection transactions = (Collection) event.getNewValue();
 571.                if (!transactions.isEmpty()) {
 572.                    final Object transition = transactions.iterator().next();
 573.                    if (eName == "incoming") {
 574.                        if (Model.getFacade().isATransition(transition)) {
 575.                            Model.getCommonBehaviorHelper().setTarget(transition,
 576.                                    Model.getFacade().getContainer(owner));
 577.                        }
 578.                    } else {
 579.                        if (Model.getFacade().isATransition(transition)) {
 580.                            Model.getStateMachinesHelper().setSource(transition,
 581.                                    Model.getFacade().getContainer(owner));
 582.                        }
 583.                    }
 584.                }
 585.            }
 586.        }
