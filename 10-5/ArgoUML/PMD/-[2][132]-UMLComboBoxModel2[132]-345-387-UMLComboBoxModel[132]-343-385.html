 364.                        fireListEvents = false;
 365.                        first = index;
 366.                        last = index;
 367.                    } else { // in middle of interval
 368.                        last++;
 369.                    }
 370.                }
 371.            }
 372.            fireListEvents = true;

 635.         * call to isValidElement.  This method can be overriden by
 636.         * subclasses if they cannot determine if it is a valid event just
 637.         * by checking the changed element.

 642.        protected boolean isValidEvent(PropertyChangeEvent e) {
 643.            boolean valid = false;
 644.            if (!(getChangedElement(e) instanceof Collection)) {
 645.                if ((e.getNewValue() == null && e.getOldValue() != null)
 646.                        // Don't try to test this if we're removing the element
 647.                        || isValidElement(getChangedElement(e))) {

 651.                Collection col = (Collection) getChangedElement(e);
 652.                if (!col.isEmpty()) {
 653.                    valid = true;
 654.                    for (Object o : col) {
 655.                        if (!isValidElement(o)) {
 656.                            valid = false;
 657.                            break;
 658.                        }
 659.                    }
 660.                } else {
 661.                    if (e.getOldValue() instanceof Collection
 662.                        && !((Collection) e.getOldValue()).isEmpty()) {
 663.                        valid = true;
 664.                    }
 665.                }
 666.            }
