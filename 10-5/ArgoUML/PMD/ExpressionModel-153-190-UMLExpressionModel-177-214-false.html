  86.         * since the event's only (read-only) state is the expression.  The source
  87.         * of events generated here is always "this".
  88.         */
  89.        protected transient ChangeEvent changeEvent = null;
  90.    
  91.        /**
  92.         * The constructor.
  93.         *
  94.         * @param target the UML element
  95.         * @param name the name of the property
  96.         */
  97.        public UMLExpressionModel(Object target, String name) {
  98.            this.target = target;
  99.            propertyName = name;
 100.    //        mustRefresh = true;
 101.            startListeningForModelChanges();
 102.        }
 103.    
 104.        protected void startListeningForModelChanges() {
 105.        if (Model.getFacade().isAUMLElement(target)) {
 106.            Model.getPump().addModelEventListener(this, target,
 107.                propertyName);
 108.        }
 109.        LOG.debug(">>Start listening for UML changes...");
 110.        }
 111.    
 112.        protected void stopListeningForModelChanges() {
 113.        if (Model.getFacade().isAUMLElement(target)) {
 114.            Model.getPump().removeModelEventListener(this, target,
 115.                        propertyName);
 116.        }
 117.        LOG.debug(">>Stop listening for UML changes...");
 118.        }
 119.    
 120.        public void propertyChange(PropertyChangeEvent e) {
 121.        if (propertyName.equals(e.getPropertyName())) {
 122.            if (rememberExpression != e.getNewValue()) {
 123.            fireStateChanged();
 124.            LOG.debug(">>UML expression changed.");
 125.            } else {
 126.            /* This should not happen. */
 127.            LOG.debug(">>Got an event for a modelchange that we inflicted ourselves...");
 128.            }
 129.        }
 130.        }
 131.    
 132.        protected Object getTarget() {
 133.            return target;
 134.        }

 137.         * @return the expression
 138.         */
 139.        public abstract Object getExpression();

 142.         * @param expr the expression
 143.         */
 144.        public abstract void setExpression(Object expr);

 146.        /**
 147.         * @return a new expression with given language and body
 148.         */
 149.        public abstract Object newExpression(String language, String body);
 150.    
 151.        /**
 152.         * @return the language of the expression
 153.         */
 154.        public String getLanguage() {
 155.        Object expression = getExpression();
 156.            if (expression == null) {
 157.                return EMPTYSTRING;
 158.            }
 159.            String language = null;
 160.            try {
 161.                language = Model.getDataTypesHelper().getLanguage(expression);
 162.            } catch (Exception ex) {
 163.                // if it has no language, then we assume it is not an instance of
 164.                // OpaqueExpression, so we leave it null and handle elsewhere
 165.            }
 166.            return language;
 167.        }
 168.    
 169.        /**
 170.         * @return The body text of the expression.
 171.         */
 172.        public String getBody() {
 173.            Object expression = getExpression();
 174.            if (expression == null) {
 175.                return EMPTYSTRING;
 176.            }
 177.            return Model.getFacade().getBody(expression).toString();
 178.        }
 179.    
 180.        /**
 181.         * @param lang the language of the expression
 182.         */
 183.        public void setLanguage(String lang) {
 184.    
 185.        Object expression = getExpression();
 186.            boolean mustChange = true;
 187.            if (expression != null) {
 188.                String oldValue =
 189.                    Model.getDataTypesHelper().getLanguage(expression);
 190.                if (oldValue != null && oldValue.equals(lang)) {
 191.                    mustChange = false;
 192.                }
 193.            }
 194.            if (mustChange) {
 195.                String body = EMPTYSTRING;
 196.                if (expression != null
 197.                        && Model.getDataTypesHelper().getBody(expression) != null) {
 198.                    body = Model.getDataTypesHelper().getBody(expression);
 199.                }
 200.    
 201.                setExpression(lang, body);
 202.            }
 203.        }
 204.    
 205.        /**
 206.         * @param body the body text of the expression
 207.         */
 208.        public void setBody(String body) {
 209.    
 210.        Object expression = getExpression();
 211.            boolean mustChange = true;
 212.            if (expression != null) {
 213.                Object oldValue = Model.getFacade().getBody(expression).toString();
 214.                if (oldValue != null && oldValue.equals(body)) {
 215.                    mustChange = false;
 216.                }
 217.            }
 218.            if (mustChange) {
 219.                String lang = null;
 220.                if (expression != null) {
 221.                    lang = Model.getDataTypesHelper().getLanguage(expression);
 222.                }
 223.                if (lang == null) {
 224.                    lang = EMPTYSTRING;
 225.                }
 226.    
 227.                setExpression(lang, body);
 228.            }
 229.        }
 230.    
 231.        /**
 232.         * This is only called if we already know that the values differ.
 233.         *
 234.         * @param lang the language of the expression
 235.         * @param body the body text of the expression
 236.         */
 237.        private void setExpression(String lang, String body) {
 238.        assert lang != null;
 239.        assert body != null;
 240.    
 241.            // Expressions are DataTypes, not independent model elements
 242.            // be careful not to reuse them
 243.        rememberExpression = getExpression();
 244.        stopListeningForModelChanges();
 245.        if (rememberExpression != null) {
 246.            Model.getUmlFactory().delete(rememberExpression);
 247.        }
 248.        if (lang.length() == 0 && body.length()==0) {
 249.            rememberExpression = null;
 250.        } else {
 251.            rememberExpression = newExpression(lang, body);
 252.        }
 253.        setExpression(rememberExpression);
 254.        startListeningForModelChanges();
 255.        }

 257.        /**
 258.         * Adds a ChangeListener.
 259.         * The change listeners are run each
 260.         * time the expression changes.
 261.         *
 262.         * @param l the ChangeListener to add
 263.         * @see #removeChangeListener
 264.         */
 265.        public void addChangeListener(ChangeListener l) {
 266.            listenerList.add(ChangeListener.class, l);
 267.            LOG.debug(">>Add listener");
 268.        }
 269.    
 270.        /**
 271.         * Removes a ChangeListener.
 272.         *
 273.         * @param l the ChangeListener to remove
 274.         * @see #addChangeListener
 275.         */
 276.        public void removeChangeListener(ChangeListener l) {
 277.            listenerList.remove(ChangeListener.class, l);
 278.            LOG.debug(">>Remove listener");
 279.        }
 280.    
 281.        /**
 282.         * Runs each ChangeListener's
 283.         * stateChanged method.
 284.         *
 285.         * @see #setRangeProperties
 286.         * @see EventListenerList
 287.         */
 288.        protected void fireStateChanged() {
 289.        LOG.debug(">>Fire state changed to listeners.");
 290.            Object[] listeners = listenerList.getListenerList();
 291.            for (int i = listeners.length - 2; i >= 0; i -=2 ) {
 292.                if (listeners[i] == ChangeListener.class) {
 293.                    if (changeEvent == null) {
 294.                        changeEvent = new ChangeEvent(this);
 295.                    }
 296.                    ((ChangeListener)listeners[i+1]).stateChanged(changeEvent);
 297.                }
 298.            }
