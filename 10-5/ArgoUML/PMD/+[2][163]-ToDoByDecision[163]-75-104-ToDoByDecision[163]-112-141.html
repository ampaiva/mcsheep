 117.            for (Decision dec : Designer.theDesigner().getDecisionModel()
 118.                    .getDecisionList()) {
 119.            int nMatchingItems = 0;
 120.            path[1] = dec;
 121.                for (ToDoItem item : items) {
 122.            if (!item.supports(dec)) {
 123.                        continue;
 124.                    }
 125.            nMatchingItems++;
 126.            }
 127.            if (nMatchingItems == 0) {
 128.                    continue;
 129.                }
 130.            int[] childIndices = new int[nMatchingItems];
 131.            Object[] children = new Object[nMatchingItems];
 132.            nMatchingItems = 0;
 133.                for (ToDoItem item : items) {
 134.            if (!item.supports(dec)) {
 135.                        continue;
 136.                    }
 137.            childIndices[nMatchingItems] = getIndexOfChild(dec, item);
 138.            children[nMatchingItems] = item;
 139.            nMatchingItems++;
 140.            }
 141.            fireTreeNodesInserted(this, path, childIndices, children);
 142.        }
 143.        }
