 704.        /**
 705.         * Once the diagram has loaded we build the previous/next links between
 706.         * any swimlanes.
 707.         */
 708.        @Override
 709.        public void postLoad() {
 710.        FigPartition previous = null;
 711.    
 712.        // Create a map of partitions keyed by x coordinate
 713.        HashMap map = new HashMap();
 714.        
 715.        Iterator it = new ArrayList(getLayer().getContents()).iterator();
 716.        while (it.hasNext()) {
 717.            Fig f = (Fig) it.next();
 718.            if (f instanceof FigPartition) {
 719.            map.put(Integer.valueOf(f.getX()), f);
 720.            }
 721.        }
 722.        
 723.        // Sort the x coordinates into order
 724.        List xList = new ArrayList(map.keySet());
 725.            Collections.sort(xList);
 726.        
 727.            // Link the previous/next reference of the swimlanes
 728.            // according to the x order.
 729.        it = xList.iterator();
 730.        while (it.hasNext()) {
 731.            Fig f = (Fig) map.get(it.next());
 732.            if (f instanceof FigPartition) {
 733.            FigPartition fp = (FigPartition) f;
 734.            if (previous != null) {
 735.                previous.setNextPartition(fp);
 736.            }
 737.            fp.setPreviousPartition(previous);
 738.            fp.setNextPartition(null);
 739.            previous = fp; 
 740.            }
 741.        }
 742.        }
 743.    
 744.        /**
 745.         * Extends basic functionality to handle logic for enclosement of states
 746.         * within a swimlane.
 747.         * @param enclosed The FigNode enclosed.
 748.         * @param oldEncloser The previous encloser (null if none)
 749.         * @param newEncloser The encloser (null if none)
 750.         */
 751.        public void encloserChanged(
 752.                FigNode enclosed, FigNode oldEncloser, FigNode newEncloser) {
 753.        
 754.        if (oldEncloser == null && newEncloser == null) {
 755.            return;
 756.        }
 757.        
 758.        if (enclosed instanceof FigStateVertex
 759.            || enclosed instanceof FigObjectFlowState) {
 760.            changePartition(enclosed);
 761.        }
 762.        }
 763.        
 764.        /**
 765.         * Extends basic functionality to handle logic for enclosement of states
 766.         * within a swimlane.
 767.         * @param enclosed The FigNode enclosed.
 768.         */
 769.        private void changePartition(FigNode enclosed) {
 770.        
 771.        assert enclosed != null;
 772.        
 773.        Object state = enclosed.getOwner();
 774.        ActivityGraphsHelper activityGraph = Model.getActivityGraphsHelper();
 775.        
 776.            for (Object f : getLayer().getContentsNoEdges()) {
 777.                if (f instanceof FigPartition) {
 778.                FigPartition fig = (FigPartition) f;
 779.                Object partition = fig.getOwner();
 780.                if (fig.getBounds().intersects(enclosed.getBounds())) {
 781.                        activityGraph.addContent(partition, state);
 782.                } else if (isStateInPartition(state, partition)) {
 783.                        activityGraph.removeContent(partition, state);
 784.                }
 785.                }
 786.            }
 787.        }

 791.        }

 819.        public DiagramElement createDiagramElement(
 820.                final Object modelElement,
 821.                final Rectangle bounds) {
 822.            
 823.            FigNodeModelElement figNode = null;
 824.            
 825.            DiagramSettings settings = getDiagramSettings();
 826.            
 827.            if (Model.getFacade().isAPartition(modelElement)) {
 828.                figNode = new FigPartition(modelElement, bounds, settings);
 829.            } else if (Model.getFacade().isAActionState(modelElement)) {
 830.                figNode = new FigActionState(modelElement, bounds, settings);
 831.            } else if (Model.getFacade().isACallState(modelElement)) {
 832.                figNode = new FigCallState(modelElement, bounds, settings);
 833.            } else if (Model.getFacade().isAObjectFlowState(modelElement)) {
 834.                figNode = new FigObjectFlowState(modelElement, bounds, settings);
 835.            } else if (Model.getFacade().isASubactivityState(modelElement)) {
 836.                figNode = new FigSubactivityState(modelElement, bounds, settings);
 837.            } else if (Model.getFacade().isAFinalState(modelElement)) {
 838.                figNode = new FigFinalState(modelElement, bounds, settings);
 839.            } else if (Model.getFacade().isAPseudostate(modelElement)) {
 840.                Object kind = Model.getFacade().getKind(modelElement);
 841.                if (kind == null) {
 842.                    LOG.warn("found a null type pseudostate");
 843.                    return null;
 844.                }
 845.                if (kind.equals(Model.getPseudostateKind().getInitial())) {
 846.                    figNode = new FigInitialState(modelElement, bounds, settings);
 847.                } else if (kind.equals(
 848.                        Model.getPseudostateKind().getChoice())) {
 849.                    figNode = new FigBranchState(modelElement, bounds, settings);
 850.                } else if (kind.equals(
 851.                        Model.getPseudostateKind().getJunction())) {
 852.                    figNode = new FigJunctionState(modelElement, bounds, settings);
 853.                } else if (kind.equals(
 854.                        Model.getPseudostateKind().getFork())) {
 855.                    figNode = new FigForkState(modelElement, bounds, settings);
 856.                } else if (kind.equals(
 857.                        Model.getPseudostateKind().getJoin())) {
 858.                    figNode = new FigJoinState(modelElement, bounds, settings);
 859.                } else {
 860.                    LOG.warn("found a type not known");
 861.                }
 862.            } else if (Model.getFacade().isAComment(modelElement)) {
 863.                figNode = new FigComment(modelElement, bounds, settings);
 864.            }
 865.            
 866.            if (figNode != null) {
 867.                LOG.debug("Model element " + modelElement + " converted to " 
 868.                        + figNode);
 869.            } else {
 870.                LOG.debug("Dropped object NOT added " + figNode);
 871.            }
 872.            return figNode;
 873.        }

  52.    /**
  53.     * Diagram for UML2 State Machine diagram
  54.     * @author Bob Tarling
  55.     */
  56.    public class UMLStateDiagram extends UMLDiagram implements StateDiagram {
  57.    
  58.        /**
  59.         * Construct a State Diagram. Default constructor used by PGML parser during
  60.         * diagram load. It should not be used by other callers.
  61.         * @deprecated only for use by PGML parser
  62.         */
  63.        @Deprecated
  64.        public UMLStateDiagram() {
  65.            super(new StateDiagramGraphModel());
  66.        }
  67.        
  68.        /**
  69.         * Construct a new State Machine diagram.
  70.         * 
  71.         * @param name the name of the new diagram
  72.         * @param machine the owner of the diagram which will be a state machine
  73.         */
  74.        public UMLStateDiagram(String name, Object machine) {
  75.            super(name, machine, new StateDiagramGraphModel());
  76.            if (name == null || name.trim().length() == 0) {
  77.                name = getLabelName() + Model.getFacade().getName(machine);
  78.                try {
  79.                    setName(name);
  80.                } catch (PropertyVetoException pve) {
  81.                    // nothing we can do about veto, so just ignore it
  82.                }
  83.            }
  84.            initialize(machine);
  85.        }
  86.    
  87.        @Override
  88.        public void initialize(Object o) {
  89.    
  90.            setNamespace(o);
  91.    
  92.            StateDiagramGraphModel gm = createGraphModel();
  93.            gm.setHomeModel(namespace);
  94.            StateDiagramRenderer rend = new StateDiagramRenderer();
  95.    
  96.            LayerPerspective lay = new LayerPerspectiveMutable(
  97.                    Model.getFacade().getName(namespace), gm);
  98.            lay.setGraphNodeRenderer(rend);
  99.            lay.setGraphEdgeRenderer(rend);
 100.            setLayer(lay);
 101.    
 102.            Model.getPump().addModelEventListener(this, namespace, 
 103.                    new String[] {"remove"});
 104.        }
 105.    
 106.        private StateDiagramGraphModel createGraphModel() {
 107.            if ((getGraphModel() instanceof StateDiagramGraphModel)) {
 108.                return (StateDiagramGraphModel) getGraphModel();
 109.            } else {
 110.                return new StateDiagramGraphModel();
 111.            }
 112.        }
 113.    
 114.        /*
 115.         * @see org.argouml.uml.diagram.ui.UMLDiagram#propertyChange(java.beans.PropertyChangeEvent)
 116.         */
 117.        @Override
 118.        public void propertyChange(PropertyChangeEvent evt) {
 119.            if ((evt.getSource() == namespace)
 120.                    && (evt instanceof DeleteInstanceEvent)) {
 121.                Model.getPump().removeModelEventListener(this, 
 122.                        namespace, new String[] {"remove"});
 123.                if (getProject() != null) {
 124.                    getProject().moveToTrash(this);
 125.                } else {
 126.                    DiagramFactory.getInstance().removeDiagram(this);
 127.                }
 128.            }
 129.        }

 131.    
 132.        protected Object[] getUmlActions() {
 133.            
 134.            ArrayList actions = new ArrayList();
 135.            
 136.            actions.add(getNodeAction(Model.getMetaTypes().getState()));
 137.            actions.add(getEdgeAction(Model.getMetaTypes().getTransition()));
 138.            actions.add(null);
 139.            actions.add(getPseudoAction(Model.getPseudostateKind().getInitial()));
 140.            actions.add(getNodeAction(Model.getMetaTypes().getFinalState()));
 141.            actions.add(getPseudoAction(
 142.                    Model.getPseudostateKind().getEntryPoint()));
 143.            actions.add(getPseudoAction(Model.getPseudostateKind().getExitPoint()));
 144.            actions.add(getPseudoAction(Model.getPseudostateKind().getJunction()));
 145.            actions.add(getPseudoAction(Model.getPseudostateKind().getChoice()));
 146.            actions.add(getPseudoAction(Model.getPseudostateKind().getFork()));
 147.            actions.add(getPseudoAction(Model.getPseudostateKind().getJoin()));
 148.            actions.add(getPseudoAction(
 149.                    Model.getPseudostateKind().getShallowHistory()));
 150.            actions.add(getPseudoAction(
 151.                    Model.getPseudostateKind().getDeepHistory()));
 152.            
 153.            return actions.toArray();
 154.        }

 209.        public DiagramElement createDiagramElement(
 210.                final Object modelElement,
 211.                final Rectangle bounds) {
 212.            
 213.            FigNodeModelElement figNode = null;
 214.            
 215.            DiagramSettings settings = getDiagramSettings();
 216.            
 217.            if (Model.getFacade().isAFinalState(modelElement)) {
 218.                figNode = new FigFinalState(modelElement, bounds, settings);
 219.            } else if (Model.getFacade().isAState(modelElement)) {
 220.                figNode = new FigVertex(modelElement, bounds, settings);
 221.            } else if (Model.getFacade().isAComment(modelElement)) {
 222.                figNode = new FigComment(modelElement, bounds, settings);
 223.            } else if (Model.getFacade().isAPseudostate(modelElement)) {
 224.                Object kind = Model.getFacade().getKind(modelElement);
 225.                if (kind == null) {
 226.                    return null;
 227.                }
 228.                if (kind.equals(Model.getPseudostateKind().getInitial())) {
 229.                    figNode = new FigInitialState(modelElement, bounds, settings);
 230.                } else if (kind.equals(
 231.                        Model.getPseudostateKind().getEntryPoint())) {
 232.                    figNode = 
 233.                            new FigEntryPoint(modelElement, bounds, settings);
 234.                } else if (kind.equals(
 235.                        Model.getPseudostateKind().getExitPoint())) {
 236.                    figNode = new FigExitPoint(modelElement, bounds, settings);
 237.                } else if (kind.equals(
 238.                        Model.getPseudostateKind().getChoice())) {
 239.                    figNode = new FigBranchState(modelElement, bounds, settings);
 240.                } else if (kind.equals(
 241.                        Model.getPseudostateKind().getJunction())) {
 242.                    figNode = new FigJunctionState(modelElement, bounds, settings);
 243.                } else if (kind.equals(
 244.                        Model.getPseudostateKind().getFork())) {
 245.                    figNode = new FigForkState(modelElement, bounds, settings);
 246.                } else if (kind.equals(
 247.                        Model.getPseudostateKind().getJoin())) {
 248.                    figNode = new FigJoinState(modelElement, bounds, settings);
 249.                } else if (kind.equals(
 250.                        Model.getPseudostateKind().getShallowHistory())) {
 251.                    figNode = new FigShallowHistoryState(modelElement, bounds, 
 252.                            settings);
 253.                } else if (kind.equals(
 254.                        Model.getPseudostateKind().getDeepHistory())) {
 255.                    figNode = new FigDeepHistoryState(modelElement, bounds, 
 256.                            settings);
 257.                }
 258.            }
 259.            return figNode;
 260.        }
