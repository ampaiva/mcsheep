 201.        public void handleSettingsTabRefresh() {
 202.            if (scope == Argo.SCOPE_APPLICATION) {
 203.                useGuillemots.setSelected(getBoolean(
 204.                        Notation.KEY_USE_GUILLEMOTS));
 205.                notationLanguage.setSelectedItem(Notation.getConfiguredNotation());
 206.                showAssociationNames.setSelected(Configuration.getBoolean(
 207.                        Notation.KEY_SHOW_ASSOCIATION_NAMES, true));
 208.                showVisibility.setSelected(getBoolean(
 209.                        Notation.KEY_SHOW_VISIBILITY));
 210.                showInitialValue.setSelected(getBoolean(
 211.                        Notation.KEY_SHOW_INITIAL_VALUE));
 212.                showProperties.setSelected(getBoolean(
 213.                        Notation.KEY_SHOW_PROPERTIES));
 214.                /*
 215.                 * The next one defaults to TRUE, to stay compatible with older
 216.                 * ArgoUML versions that did not have this setting:
 217.                 */
 218.                showTypes.setSelected(Configuration.getBoolean(
 219.                        Notation.KEY_SHOW_TYPES, true));
 220.                showMultiplicity.setSelected(getBoolean(
 221.                        Notation.KEY_SHOW_MULTIPLICITY));
 222.                showStereotypes.setSelected(getBoolean(
 223.                        Notation.KEY_SHOW_STEREOTYPES));
 224.                /*
 225.                 * The next one defaults to TRUE, despite that this is
 226.                 * NOT compatible with older ArgoUML versions
 227.                 * (before 0.24) that did
 228.                 * not have this setting - see issue 1395 for the rationale:
 229.                 */
 230.                showSingularMultiplicities.setSelected(Configuration.getBoolean(
 231.                        Notation.KEY_SHOW_SINGULAR_MULTIPLICITIES, true));
 232.            }
 233.            if (scope == Argo.SCOPE_PROJECT) {
 234.                assert p != null;
 235.                ProjectSettings ps = p.getProjectSettings();
 236.                NotationSettings ns = ps.getNotationSettings();
 237.    
 238.                notationLanguage.setSelectedItem(Notation.findNotation(
 239.                        ps.getNotationLanguage()));
 240.                useGuillemots.setSelected(ps.getUseGuillemotsValue());
 241.                showAssociationNames.setSelected(ns.isShowAssociationNames());
 242.                showVisibility.setSelected(ns.isShowVisibilities());
 243.                showMultiplicity.setSelected(ns.isShowMultiplicities());
 244.                showInitialValue.setSelected(ns.isShowInitialValues());
 245.                showProperties.setSelected(ns.isShowProperties());
 246.                showTypes.setSelected(ns.isShowTypes());
 247.                showStereotypes.setSelected(ps.getShowStereotypesValue());
 248.                showSingularMultiplicities.setSelected(
 249.                        ns.isShowSingularMultiplicities());
 250.            }
 251.        }

 264.         * @see org.argouml.ui.GUISettingsTabInterface#handleSettingsTabSave()
 265.         */
 266.        public void handleSettingsTabSave() {
 267.            if (scope == Argo.SCOPE_APPLICATION) {
 268.                Notation.setDefaultNotation(
 269.                        (NotationName) notationLanguage.getSelectedItem());
 270.                Configuration.setBoolean(Notation.KEY_USE_GUILLEMOTS,
 271.                        useGuillemots.isSelected());
 272.                Configuration.setBoolean(Notation.KEY_SHOW_ASSOCIATION_NAMES,
 273.                        showAssociationNames.isSelected());
 274.                Configuration.setBoolean(Notation.KEY_SHOW_VISIBILITY,
 275.                        showVisibility.isSelected());
 276.                Configuration.setBoolean(Notation.KEY_SHOW_MULTIPLICITY,
 277.                        showMultiplicity.isSelected());
 278.                Configuration.setBoolean(Notation.KEY_SHOW_INITIAL_VALUE,
 279.                        showInitialValue.isSelected());
 280.                Configuration.setBoolean(Notation.KEY_SHOW_PROPERTIES,
 281.                        showProperties.isSelected());
 282.                Configuration.setBoolean(Notation.KEY_SHOW_TYPES,
 283.                        showTypes.isSelected());
 284.                Configuration.setBoolean(Notation.KEY_SHOW_STEREOTYPES,
 285.                        showStereotypes.isSelected());
 286.                Configuration.setBoolean(Notation.KEY_SHOW_SINGULAR_MULTIPLICITIES,
 287.                        showSingularMultiplicities.isSelected());

 290.                assert p != null;

 292.                NotationSettings ns = ps.getNotationSettings();
 293.                NotationName nn = (NotationName) notationLanguage.getSelectedItem();
 294.                if (nn != null) {
 295.                    ps.setNotationLanguage(nn.getConfigurationValue());
 296.                }
 297.                ps.setUseGuillemots(useGuillemots.isSelected());
 298.                ns.setShowAssociationNames(showAssociationNames.isSelected());
 299.                ns.setShowVisibilities(showVisibility.isSelected());
 300.                ns.setShowMultiplicities(showMultiplicity.isSelected());
 301.                ns.setShowInitialValues(showInitialValue.isSelected());
 302.                ns.setShowProperties(showProperties.isSelected());
 303.                ns.setShowTypes(showTypes.isSelected());
 304.                ps.setShowStereotypes(showStereotypes.isSelected());
 305.                ns.setShowSingularMultiplicities(
 306.                        showSingularMultiplicities.isSelected());
 307.            }
 308.        }

 215.        /*
 216.         * @see org.argouml.ui.GUISettingsTabInterface#handleSettingsTabRefresh()
 217.         */
 218.        public void handleSettingsTabRefresh() {
 219.            if (scope == Argo.SCOPE_APPLICATION) {
 220.                showBoldNames.setSelected(getBoolean(
 221.                        DiagramAppearance.KEY_SHOW_BOLD_NAMES));
 222.                selectedDiagramFontName = DiagramAppearance.getInstance()
 223.                        .getConfiguredFontName();
 224.                selectedDiagramFontSize = Configuration
 225.                        .getInteger(DiagramAppearance.KEY_FONT_SIZE);
 226.                /*
 227.                 * The next one defaults to TRUE, despite that this is
 228.                 * NOT compatible with older ArgoUML versions
 229.                 * (before 0.28?) that did
 230.                 * not have this setting - see issue 535
 231.                 */
 232.                hideBidirectionalArrows.setSelected(Configuration.getBoolean(
 233.                        DiagramAppearance.KEY_HIDE_BIDIRECTIONAL_ARROWS, true));
 234.                defaultShadowWidth.setSelectedIndex(Configuration.getInteger(
 235.                        DiagramAppearance.KEY_DEFAULT_SHADOW_WIDTH, 1));
 236.            }
 237.            if (scope == Argo.SCOPE_PROJECT) {
 238.                assert p != null;
 239.                ProjectSettings ps = p.getProjectSettings();
 240.                DiagramSettings ds = ps.getDefaultDiagramSettings();
 241.    
 242.                showBoldNames.setSelected(ds.isShowBoldNames());
 243.                selectedDiagramFontName = ds.getFontName();
 244.                selectedDiagramFontSize = ds.getFontSize();
 245.                hideBidirectionalArrows.setSelected(!
 246.                        ds.isShowBidirectionalArrows());
 247.                defaultShadowWidth.setSelectedIndex(
 248.                        ds.getDefaultShadowWidth());
 249.            }
 250.        }
 251.    
 252.        /**
 253.         * Get a boolean from the configuration.
 254.         *
 255.         * @param key a notation key.
 256.         * @return a boolean
 257.         */
 258.        protected static boolean getBoolean(ConfigurationKey key) {
 259.            return Configuration.getBoolean(key, false);
 260.        }

 262.        /*
 263.         * @see org.argouml.ui.GUISettingsTabInterface#handleSettingsTabSave()
 264.         */
 265.        public void handleSettingsTabSave() {
 266.            if (scope == Argo.SCOPE_APPLICATION) {
 267.                Configuration.setBoolean(DiagramAppearance.KEY_SHOW_BOLD_NAMES,
 268.                        showBoldNames.isSelected());
 269.                Configuration.setString(DiagramAppearance.KEY_FONT_NAME,
 270.                        selectedDiagramFontName);
 271.                Configuration.setInteger(DiagramAppearance.KEY_FONT_SIZE,
 272.                        selectedDiagramFontSize);
 273.                Configuration.setBoolean(
 274.                        DiagramAppearance.KEY_HIDE_BIDIRECTIONAL_ARROWS,
 275.                        hideBidirectionalArrows.isSelected());            
 276.                Configuration.setInteger(DiagramAppearance.KEY_DEFAULT_SHADOW_WIDTH,
 277.                        defaultShadowWidth.getSelectedIndex());
 278.            }
 279.            if (scope == Argo.SCOPE_PROJECT) {
 280.                assert p != null;
 281.                ProjectSettings ps = p.getProjectSettings();
 282.                final DiagramSettings ds = ps.getDefaultDiagramSettings();
 283.    
 284.                ds.setShowBoldNames(showBoldNames.isSelected());
 285.                ds.setFontName(selectedDiagramFontName);
 286.                ds.setFontSize(selectedDiagramFontSize);
 287.                ds.setDefaultShadowWidth(defaultShadowWidth.getSelectedIndex());
 288.                ds.setShowBidirectionalArrows(
 289.                        !hideBidirectionalArrows.isSelected());
 290.    
 291.                /* Update the diagram, and other users of these settings: */
 292.                Memento memento = new Memento() {
 293.                    public void redo() {
 294.                        ds.notifyOfChangedSettings();
 295.                    }
 296.    
 297.                    public void undo() {
 298.                        ds.notifyOfChangedSettings();
 299.                    }
 300.                };
 301.                // TODO: Undo should be managed externally or we should be given 
 302.                // an Undo manager to use (the project's) rather than using a 
 303.                // global one
 304.    //            if (DiagramUndoManager.getInstance().isGenerateMementos()) {
 305.    //                DiagramUndoManager.getInstance().addMemento(memento);
 306.    //            }
 307.                memento.redo();
 308.            }
 309.        }
