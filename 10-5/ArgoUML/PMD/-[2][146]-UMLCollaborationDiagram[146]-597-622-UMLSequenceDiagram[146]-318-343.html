 159.        /**

 165.         * Changed <em>lay</em> from <em>LayerPerspective</em> to

 167.         * <em>LayerPerspective</em> and was implemented to correct some

 175.        public void setNamespace(Object handle) {
 176.            if (!Model.getFacade().isANamespace(handle)) {
 177.                LOG.error(
 178.                    "Illegal argument. Object " + handle + " is not a namespace");
 179.                throw new IllegalArgumentException(
 180.                    "Illegal argument. Object " + handle + " is not a namespace");
 181.            }
 182.            super.setNamespace(handle);
 183.            CollabDiagramGraphModel gm = createGraphModel();
 184.            gm.setCollaboration(handle);
 185.            LayerPerspective lay =
 186.                new LayerPerspectiveMutable(Model.getFacade().getName(handle), gm);
 187.            CollabDiagramRenderer rend = new CollabDiagramRenderer(); // singleton
 188.            lay.setGraphNodeRenderer(rend);
 189.            lay.setGraphEdgeRenderer(rend);
 190.            setLayer(lay);
 191.        }

 254.    
 255.            super.postLoad();
 256.    
 257.            if (getNamespace() == null) {
 258.                throw new IllegalStateException(
 259.                        "The namespace of the collaboration diagram is not set");

 262.            Collection messages;
 263.            Iterator msgIterator;
 264.            Collection ownedElements =
 265.                Model.getFacade().getOwnedElements(getNamespace());
 266.            Iterator oeIterator = ownedElements.iterator();
 267.            Layer lay = getLayer();
 268.            while (oeIterator.hasNext()) {
 269.                Object me = oeIterator.next();
 270.                if (Model.getFacade().isAAssociationRole(me)) {
 271.                    messages = Model.getFacade().getMessages(me);
 272.                    msgIterator = messages.iterator();

 275.                        FigMessage figMessage =
 276.                            (FigMessage) lay.presentationFor(message);
 277.                        if (figMessage != null) {
 278.                            figMessage.addPathItemToFigAssociationRole(lay);
 279.                        }
 280.                    }

 282.            }

 271.         * @param classifierRole
 272.         * @param location The position where to put the new fig.
 273.         * @return
 274.         */
 275.        private FigClassifierRole makeNewFigCR(Object classifierRole, 
 276.                Point location) {
 277.            if (classifierRole != null) {
 278.                Rectangle bounds = new Rectangle();
 279.                
 280.                // Y position of the new CR should match existing CRs Y position
 281.                for (Fig fig : (List<Fig>) getLayer().getContentsNoEdges()) {
 282.                    if (fig instanceof FigClassifierRole) {
 283.                        bounds.y = fig.getY();
 284.                        bounds.height = fig.getHeight();
 285.                        break;
 286.                    }
 287.                }
 288.                if (location != null) {
 289.                    if (bounds.y == 0) {
 290.                        bounds.y = location.y;
 291.                    }
 292.                    bounds.x = location.x;
 293.                }
 294.    
 295.                FigClassifierRole newCR = new FigClassifierRole(classifierRole,
 296.                        bounds, getDiagramSettings());
 297.                getGraphModel().getNodes().add(newCR.getOwner());
 298.                
 299.                return newCR;
 300.            }
 301.            return null;

 355.        /**
 356.         * Ensure that all elements represented in this diagram are part of this
 357.         * diagrams collaboration
 358.         */
 359.        public void postLoad() {
 360.            super.postLoad();
 361.            
 362.            final Facade facade = Model.getFacade();                   
 363.            LOG.info("doing postLoad on " + getName());
 364.            // See issue 5811. We have collaborationroles, associationroles
 365.            // and messages and actions saved to the incorrect interaction and
 366.            // and collaboration. If we detect this circumstance at load then
 367.            // move the model elements and delete the empty collaborations
 368.            // and interactions.
 369.            final Object collaboration = getCollaboration();
 370.            Object correctInteraction = null;
 371.            for (final Fig f : getLayer().getContents()) {
 372.                final Object modelElement = f.getOwner();
 373.                if (f instanceof FigMessage) {
 374.                    final Object interaction = facade.getInteraction(modelElement);
 375.                    final Object context = facade.getContext(interaction);
 376.                    if (context == collaboration) {
 377.                        correctInteraction = interaction;
 378.                    }
 379.                }
 380.            }
 381.            if (correctInteraction != null) {
 382.                final CollaborationsHelper collabHelper =
 383.                    Model.getCollaborationsHelper();
 384.                for (final Fig f : getLayer().getContents()) {
 385.                    if (f instanceof FigMessage) {
 386.                        final Object message = f.getOwner();
 387.                        LOG.info("Checking message " + f.getOwner());
 388.                        final Object interaction = facade.getInteraction(message);
 389.                        final Object context = facade.getContext(interaction);
 390.                        final Object action = facade.getAction(message);
 391.                        if (context != collaboration) {
 392.                            LOG.warn("namespace of interaction does not match "
 393.                                    + "collaboration - moving "
 394.                                    + message + " to " + correctInteraction);
 395.                            collabHelper.addMessage(correctInteraction, message);
 396.                            Model.getCoreHelper().setNamespace(
 397.                                    action, collaboration);
 398.                        }
 399.                    } else if (f instanceof FigClassifierRole) {
 400.                        final Object cr = f.getOwner();
 401.                        final Object namespace = facade.getNamespace(cr);
 402.                        if (namespace != collaboration) {
 403.                            LOG.warn("namespace of classifierrole does not match "
 404.                                    + "collaboration - moving "
 405.                                    + cr + " to " + collaboration);
 406.                            
 407.                            Model.getCoreHelper().setNamespace(
 408.                                    cr, collaboration);
 409.                            
 410.                            Collection associationEndRoles =
 411.                                facade.getAssociationEnds(cr);
 412.                            for (Object assEndRole : associationEndRoles) {
 413.                                Object assRole = facade.getAssociation(assEndRole);
 414.                                if (facade.getNamespace(assRole) != collaboration) {
 415.                                    Model.getCoreHelper().setNamespace(
 416.                                            assRole, collaboration);
 417.                                }
 418.                            }
 419.                        }
 420.                    }
 421.                }
 422.            }
