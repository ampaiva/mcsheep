 432.                    if (i2 >= str.length()) {
 433.                        i1 = i2;
 434.                        break;
 435.                    }
 436.    
 437.                    if (i2 > i1) {
 438.                        String ent = str.substring(i1, i2);
 439.                        if ("proc".equals(ent)) {
 440.                            sb.append('%');
 441.                        } else {
 442.                            try {
 443.                                sb.append((char) Integer.parseInt(ent));
 444.                            } catch (NumberFormatException nfe) {
 445.                                // TODO: handle parse error
 446.                            }
 447.                        }
 448.                    }
 449.                    i1 = i2 + 1;
 450.                }
 451.            }
 452.            if (i2 > i1) {
 453.                sb.append(str.substring(i1, i2));
 454.            }
 455.            return sb.toString();
 456.        }

 458.        /**
 459.         * Utility method to encode a String in a way that allows it to be
 460.         * saved properly in an XML file and regained filtering out any noice
 461.         * that an XML framework might have seen fit to add.
 462.         *
 463.         * TODO: Why are we doing this ourselves?  Surely encoding information
 464.         * for XML serialization is a well known task - tfm
 465.         * I have never understood why this is being done. I think we should remove
 466.         * any usage - bob
 467.         * 
 468.         * @param    str    The String to encode.
 469.         * @return    The encoded String.
 470.         */
 471.        public static String encode(String str) {
 472.        StringBuffer sb;
 473.        int i1, i2;
 474.        char c;
 475.    
 476.        if (str == null) {
 477.                return null;
 478.            }
 479.        sb = new StringBuffer();
 480.        for (i1 = 0, i2 = 0; i2 < str.length(); i2++) {
 481.            c = str.charAt(i2);
 482.            if (c == '%') {
 483.            if (i2 > i1) {
 484.                        sb.append(str.substring(i1, i2));
 485.                    }
 486.            sb.append("%proc;");
 487.            i1 = i2 + 1;
 488.            } else if (c < 0x28
 489.                    ||  (c >= 0x3C && c <= 0x40 && c != 0x3D && c != 0x3F)
 490.                    ||  (c >= 0x5E && c <= 0x60 && c != 0x5F)
 491.                    ||   c >= 0x7B) {
 492.            if (i2 > i1) {
 493.                        sb.append(str.substring(i1, i2));
 494.                    }
 495.            sb.append("%" + Integer.toString(c) + ";");
 496.            i1 = i2 + 1;
 497.            }
 498.        }
 499.        if (i2 > i1) {
 500.                sb.append(str.substring(i1, i2));
 501.            }
 502.    
 503.        //cat.debug("encode:\n" + str + "\n -> " + sb.toString());
 504.        return sb.toString();
 505.        }
 506.    }

 407.         *

 409.         * @return    A copy of the original String.
 410.         */
 411.        public static String decode(String str) {
 412.            if (str == null) {
 413.                return null;
 414.            }
 415.    
 416.            StringBuffer sb;
 417.            int i1, i2;
 418.            char c;
 419.    
 420.            sb = new StringBuffer();
 421.            for (i1 = 0, i2 = 0; i2 < str.length(); i2++) {
 422.                c = str.charAt(i2);
