 398.                newValue.add(value);
 399.                value = newValue;
 400.            } else if (!prop.isMultivalued() && value instanceof Collection) {
 401.                Collection col = (Collection) value;
 402.                if (col.isEmpty()) {
 403.                    value = null; // or getDefaultValueFor(prop)?
 404.                } else {
 405.                    // too bad, we choose to take the first value
 406.                    value = col.iterator().next();
 407.                }
 408.            }
 409.            // workaround for missing ability to parse "*"
 410.            if (value instanceof Collection) {
 411.                Collection newValue = new ArrayList();
 412.                for (Object v : ((Collection) value)) {
 413.                    newValue.add(postprocessPropertyValue(prop, v));
 414.                }
 415.                value = newValue;
 416.            } else {
 417.                value = postprocessPropertyValue(prop, value);
 418.            }
 419.            // ready to set the value finally
 420.            Stereotype stereotype = (Stereotype) prop.eContainer();
 421.            UMLUtil.setTaggedValue(elem, stereotype, prop.getName(), value);
 422.        }

 427.                    && "UnlimitedNatural".equals(prop.getType().getName())
 428.                    && "*".equals(value)) {
 429.                value = LiteralUnlimitedNatural.UNLIMITED;
 430.            }

 377.    
 378.        /*
 379.         * Sets the value of an element#s property (tagged value). This method
 380.         * makes sure that a Collection of values is set if and only if the
 381.         * property is multivalued (upper multiplicity value greater 1), so passing
 382.         * a collection is safe.
 383.         * 
 384.         * @see org.argouml.model.ExtensionMechanismsHelper#setValueOfTag(java.lang.Object, java.lang.Ob...
 385.         */
 386.        public void setTaggedValue(Object handle, Object property, Object value) {
 387.            if (!(handle instanceof Element)) {
 388.                return;
 389.            }
 390.            if (!(property instanceof Property)) {
 391.                return;
 392.            }
 393.            Element elem = (Element) handle;
 394.            Property prop = (Property) property;
 395.            // consider the property multiplicity
 396.            if (prop.isMultivalued() && !(value instanceof Collection)) {
 397.                Collection newValue = new ArrayList();
 398.                newValue.add(value);
 399.                value = newValue;
 400.            } else if (!prop.isMultivalued() && value instanceof Collection) {
 401.                Collection col = (Collection) value;
 402.                if (col.isEmpty()) {
 403.                    value = null; // or getDefaultValueFor(prop)?
 404.                } else {
 405.                    // too bad, we choose to take the first value
 406.                    value = col.iterator().next();
 407.                }
 408.            }
 409.            // workaround for missing ability to parse "*"
 410.            if (value instanceof Collection) {
 411.                Collection newValue = new ArrayList();
 412.                for (Object v : ((Collection) value)) {
 413.                    newValue.add(postprocessPropertyValue(prop, v));
 414.                }
 415.                value = newValue;
 416.            } else {
 417.                value = postprocessPropertyValue(prop, value);
 418.            }
 419.            // ready to set the value finally
 420.            Stereotype stereotype = (Stereotype) prop.eContainer();
 421.            UMLUtil.setTaggedValue(elem, stereotype, prop.getName(), value);
 422.        }

 424.        private Object postprocessPropertyValue(Property prop, Object value) {
 425.            // workaround for missing ability to parse "*"
 426.            if (prop.getType() != null
 427.                    && "UnlimitedNatural".equals(prop.getType().getName())
 428.                    && "*".equals(value)) {
 429.                value = LiteralUnlimitedNatural.UNLIMITED;
 430.            }
 431.            return value;
 432.        }
