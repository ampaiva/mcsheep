 190.         *         if the lengths differ more than 2, this length difference
 191.         */
 192.        public int countDiffs(String s1, String s2) {
 193.        int len = Math.min(s1.length(), s2.length());
 194.        int count = Math.abs(s1.length() - s2.length());
 195.        if (count > 2) {
 196.            return count;
 197.        }
 198.        for (int i = 0; i < len; i++) {
 199.            if (s1.charAt(i) != s2.charAt(i)) {
 200.                count++;
 201.            }
 202.        }
 203.        return count;
 204.        }

 207.         * @param s the given string
 208.         * @return the string s with all non-letters/digits stripped off
 209.         */
 210.        public String strip(String s) {
 211.        StringBuffer res = new StringBuffer(s.length());
 212.        int len = s.length();
 213.        for (int i = 0; i < len; i++) {
 214.            char c = s.charAt(i);
 215.            if (Character.isLetterOrDigit(c)) {
 216.                res.append(Character.toLowerCase(c));
 217.            } else if (c == ']' && i > 1 && s.charAt(i - 1) == '[') {
 218.            res.append("[]");
 219.            }
 220.        }
 221.        return res.toString();
 222.        }
