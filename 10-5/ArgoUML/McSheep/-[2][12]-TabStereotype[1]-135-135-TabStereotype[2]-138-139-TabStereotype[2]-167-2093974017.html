 299.             * @since 20110215

 301.            private boolean isValidEventRemove(PropertyChangeEvent e) {
 302.                boolean valid = false;
 303.                if (!(getChangedElement(e) instanceof Collection)) {
 304.                   
 305.                    if ((e.getNewValue() != null && e.getOldValue() == null) 
 306.                            || isValidElement(getChangedElement(e))) {
 307.                        valid = true; // we tried to remove a value
 308.                    }
 309.                } else {
 310.                    Collection col = (Collection) getChangedElement(e);
 311.                    Iterator it = col.iterator();
 312.                    if (!col.isEmpty()) {
 313.                        valid = true;
 314.                        while (it.hasNext()) {
 315.                            Object o = it.next();
 316.                            if (!isValidElement(o)) {
 317.                                valid = false;
 318.                                break;
 319.                            }
 320.                        }
 321.                    } else {
 322.                        if (e.getOldValue() instanceof Collection
 323.                            && !((Collection) e.getOldValue()).isEmpty()) {
 324.                            valid = true;
 325.                        }
 326.                    }
 327.                }
 328.                return valid;
 329.            }

 343.                if (e instanceof AttributeChangeEvent) {
 344.                   // ignored this event
 345.                } else if (e instanceof AddAssociationEvent) {
 346.                    //Remove the element from Available List
 347.                    boolean valid = false;
 348.                    if (!(getChangedElement(e) instanceof Collection)) {

 350.                    } else {

 352.                        Iterator it = col.iterator();
 353.                        valid = true;
 354.                        while (it.hasNext()) {
 355.                            Object o = it.next();
 356.                            if (!contains(o)) {
 357.                                valid = false;
 358.                                break;
 359.                            }
 360.                        }
 361.                    }
 362.                    if (valid) {
 363.                        Object o = getChangedElement(e);
 364.                        if (o instanceof Collection) {
 365.                            Iterator it = ((Collection) o).iterator();
 366.                            while (it.hasNext()) {
 367.                                Object o3 = it.next();
 368.                                removeElement(o3);
 369.                            }
 370.                        } else {
 371.                            removeElement(o);
 372.                        }
 373.                    }
 374.                    
 375.                   
 376.                    
 377.                } else if (e instanceof RemoveAssociationEvent) {
 378.                    if (isValidEventRemove(e)) {
 379.                        Object o = getChangedElement(e);
 380.                        if (o instanceof Collection) {
 381.                            ArrayList tempList = new ArrayList((Collection) o);
 382.                            Iterator it = tempList.iterator();
 383.                            while (it.hasNext()) {
 384.                                Object o2 = it.next();
 385.                                addElement(o2);
 386.                            }
 387.                        } else {
 388.                            addElement(o); 
 389.                        }
 390.                    }
 391.                    
 392.                }
 393.            }

 355.        /**
 356.         * Returns the choices a user can make.
 357.         * @return List of choices
 358.         */
 359.        private List getChoices() {
 360.            List result = new ArrayList();
 361.            for (int index : choicesList.getSelectedIndices()) {
 362.                result.add(choicesList.getModel().getElementAt(index));
 363.            }
 364.            return result;
 365.        }

 372.        private List getSelectedChoices() {

 378.        }

 380.        /**
 381.         * Returns the elements of the right-hand "selected" list.  Note
 382.         * that these are not the elements selected (ie highlighted) in the
 383.         * Swing sense, but rather the entire contents of the list containing
 384.         * the user selections.
 385.         *
 386.         * @return a Vector of selected elements.
 387.         */
 388.        public Vector getSelected() {
 389.            Vector result = new Vector();
 390.            ListModel list = selectedList.getModel();
 391.            for (int i = 0; i < list.getSize(); i++) {
 392.                result.add(list.getElementAt(i));
 393.            }
 394.            return result;
 395.        }

 399.         * 
 400.         */
 401.        private void addSelection() {
 402.            List theChoices = getChoices();
 403.            if (exclusive) {
 404.                ((SortedListModel) choicesList.getModel()).removeAll(theChoices);

 410.        /**
 411.         * Removes the selected elements in the selected list and adds them to the
 412.         * choices list. The GUI will be updated by the ListModel listeners.
 413.         */
 414.        private void removeSelection() {
 415.            List theChoices = getSelectedChoices();
 416.            ((SortedListModel) selectedList.getModel()).removeAll(theChoices);
 417.            if (exclusive) {
 418.                ((SortedListModel) choicesList.getModel()).addAll(theChoices);
 419.            }
 420.        }
