 124.                    chooser = new JFileChooser();
 125.                }
 126.    
 127.                Object[] s = {defaultName };
 128.                chooser.setDialogTitle(
 129.                        Translator.messageFormat("filechooser.save-graphics", s));
 130.                // Only specified format are allowed.
 131.                chooser.setAcceptAllFileFilterUsed(false);
 132.                sgm.setFileChooserFilters(chooser, defaultName);
 133.    
 134.                String fn = Configuration.getString(
 135.                        SaveGraphicsManager.KEY_SAVE_GRAPHICS_PATH);
 136.                if (fn.length() > 0) {
 137.                    chooser.setSelectedFile(new File(fn));
 138.                }
 139.    
 140.                int retval = chooser.showSaveDialog(ArgoFrame.getFrame());
 141.                if (retval == JFileChooser.APPROVE_OPTION) {
 142.                    File theFile = chooser.getSelectedFile();
 143.                    if (theFile != null) {
 144.                        String path = theFile.getPath();
 145.                        Configuration.setString(
 146.                                SaveGraphicsManager.KEY_SAVE_GRAPHICS_PATH,
 147.                                path);
 148.    
 149.                        theFile = new File(theFile.getParentFile(),
 150.                                sgm.fixExtension(theFile.getName()));
 151.                        String suffix = sgm.getFilterFromFileName(theFile.getName())
 152.                            .getSuffix();
 153.                        return doSave(theFile, suffix, true);
 154.                    }
 155.                }
 156.            } catch (OutOfMemoryError e) {
 157.                ExceptionDialog ed = new ExceptionDialog(ArgoFrame.getFrame(),
 158.                    "You have run out of memory. "
 159.                    + "Close down ArgoUML and restart with a larger heap size.", e);
 160.                ed.setModal(true);
 161.                ed.setVisible(true);
 162.            } catch (Exception e) {
 163.                ExceptionDialog ed = 
 164.                    new ExceptionDialog(ArgoFrame.getFrame(), e);
 165.                ed.setModal(true);
 166.                ed.setVisible(true);
 167.                LOG.error("Got some exception", e);
 168.            }
 169.    
 170.            return false;
 171.        }
 172.    
 173.        /**
 174.         * Actually do the saving of the graphics file.
 175.         *
 176.         * @return true if it was successful.
 177.         * @param theFile is the file that we are writing to
 178.         * @param suffix is the suffix. Used for deciding what format the file
 179.         * shall have.
 180.         * @param useUI is true if we are supposed to use the UI e.g. to warn
 181.         *              the user that we are replacing an old file.
 182.         */
 183.        private boolean doSave(File theFile,
 184.                   String suffix, boolean useUI)
 185.        throws FileNotFoundException, IOException {
 186.    
 187.            SaveGraphicsManager sgm = SaveGraphicsManager.getInstance();
 188.            SaveGraphicsAction cmd = null;
 189.    
 190.            cmd = sgm.getSaveActionBySuffix(suffix);
 191.            if (cmd == null) {
 192.                return false;
 193.            }
 194.    
 195.            if (useUI) {
 196.                updateStatus(Translator.localize(
 197.                        "statusmsg.bar.save-graphics-status-writing",
 198.                        new Object[] {theFile}));
 199.            }
 200.        if (theFile.exists() && useUI) {
 201.            int response = JOptionPane.showConfirmDialog(
 202.                    ArgoFrame.getFrame(),
 203.                    Translator.messageFormat("optionpane.confirm-overwrite",
 204.                            new Object[] {theFile}),
 205.                    Translator.localize("optionpane.confirm-overwrite-title"),
 206.                    JOptionPane.YES_NO_OPTION);
 207.            if (response != JOptionPane.YES_OPTION) {
 208.            return false;
 209.            }
 210.        }
 211.        FileOutputStream fo = new FileOutputStream(theFile);
 212.        cmd.setStream(fo);
 213.            cmd.setScale(Configuration.getInteger(
 214.                    SaveGraphicsManager.KEY_GRAPHICS_RESOLUTION, 1));
 215.            try {
 216.                cmd.actionPerformed(null);
 217.            } finally {
 218.                fo.close();
 219.            }
 220.            if (useUI) {
 221.                updateStatus(Translator.localize(
 222.                        "statusmsg.bar.save-graphics-status-wrote", 
 223.                        new Object[] {theFile}));
 224.            }
 225.        return true;
 226.        }
 227.        
 228.        private void updateStatus(String status) {
 229.            ArgoEventPump.fireEvent(
 230.                    new ArgoStatusEvent(ArgoEventTypes.STATUS_TEXT,
 231.                    this, status));
 232.        }

 235.        /**
 236.         * Execute this action from the command line.
 237.         *
 238.         * TODO: The underlying GEF library relies on Acme that doesn't allow
 239.         * us to create these files unless there is a window showing. For this
 240.         * reason I have had to split the performing of commands in
 241.         * {@link org.argouml.application.Main#main(String[])} so that we can,
 242.         * by not supplying the -batch option, run these commands
 243.         * with the window showing. Hopefully this can eventually be fixed.
 244.         *
 245.         * @see org.argouml.application.api.CommandLineInterface#doCommand(String)
 246.         * @param argument is the file name that we save to.
 247.         * @return true if it is OK.
 248.         */
 249.        public boolean doCommand(String argument) {
 250.        File file = new File(argument);
 251.        String suffix = SuffixFilter.getExtension(file);
 252.        if (suffix == null) {
 253.            return false;
 254.        }
 255.    
 256.        try {
 257.            return doSave(file, suffix, false);
 258.        } catch (FileNotFoundException e) {
 259.            LOG.error("File not found error when writing.", e);
 260.        } catch (IOException e) {
 261.            LOG.error("IO error when writing.", e);
 262.        }
 263.        return false;
 264.        }
