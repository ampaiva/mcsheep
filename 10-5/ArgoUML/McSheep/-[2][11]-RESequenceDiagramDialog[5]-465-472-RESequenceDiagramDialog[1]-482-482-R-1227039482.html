 792.        /**
 793.         * Gets or builds a classifier role from a type. The type is a classifier
 794.         * name, either fully qualified (with whole package path) or not.
 795.         * Also ensures that there is an association to the actual classifier.<p>
 796.         * TODO: Hide this method in the implementation of a to be defined
 797.         * interface.
 798.         * TODO: objName is not used. Are there plans for this?
 799.         */
 800.        private Object getClassifierFromModel(
 801.                final String type,
 802.                final String objName) {
 803.            Object theClassifier = null;
 804.            int pos = type.lastIndexOf(".");
 805.            if (pos != -1) {
 806.                // full package path given, so let's get it from the model
 807.                Object namespace = model;
 808.                pos = 0;
 809.                StringTokenizer st = new StringTokenizer(type, ".");
 810.                while (st.hasMoreTokens()) {
 811.                    String s = st.nextToken();
 812.                    pos += s.length();
 813.                    Object element = Model.getFacade().lookupIn(namespace, s);
 814.                    if (element == null) {
 815.                        // package/classifier is missing, so create one
 816.                        if (st.hasMoreTokens()) {
 817.                            // must be a package
 818.                            element =
 819.                                Model.getModelManagementFactory()
 820.                                    .buildPackage(s);
 821.                        } else {
 822.                            // must be a classifier, let's assume a class
 823.                            element = Model.getCoreFactory().buildClass(s);
 824.                        }
 825.                        Model.getCoreHelper().setNamespace(element, namespace);
 826.                        Model.getCoreHelper().addOwnedElement(namespace, element);
 827.                    }
 828.                    namespace = element;
 829.                    pos++;
 830.                }
 831.                theClassifier = namespace;
 832.            } else {
 833.                // classifier without package information given
 834.                // first, let's look in the namespace of the actual classifier
 835.                Object namespace = Model.getFacade().getNamespace(classifier);
 836.                theClassifier = Model.getFacade().lookupIn(namespace, type);
 837.                if (!Model.getFacade().isAClassifier(theClassifier)) {
 838.                    theClassifier = null;
 839.                    // let's search for it in the imports (component dependencies)
 840.                    Collection sdeps =
 841.                        Model.getFacade().getSupplierDependencies(classifier);
 842.                    Iterator iter1 = sdeps != null ? sdeps.iterator() : null;
 843.                    while (theClassifier == null
 844.                            && iter1 != null
 845.                            && iter1.hasNext()) {
 846.                        Object dep = iter1.next();
 847.                        if (Model.getFacade().isADependency(dep)) {
 848.                            Collection clients = Model.getFacade().getClients(dep);
 849.                            Iterator iter2 =
 850.                                clients != null ? clients.iterator() : null;
 851.                            while (theClassifier == null
 852.                                    && iter2 != null
 853.                                    && iter2.hasNext()) {
 854.                                Object comp = iter2.next();
 855.                                if (Model.getFacade().isAComponent(comp)) {
 856.                                    theClassifier = permissionLookup(comp, type);
 857.                                }
 858.                            }
 859.                        }
 860.                    }
 861.                }
 862.            }
 863.            if (theClassifier == null) {
 864.                // not found any matching classifier, so create one, and put it
 865.                // into the namespace of the actual classifier
 866.                theClassifier = Model.getCoreFactory().buildClass(type);
 867.                Object namespace = Model.getFacade().getNamespace(classifier);
 868.                Model.getCoreHelper().setNamespace(theClassifier, namespace);
 869.                Model.getCoreHelper().addOwnedElement(namespace, theClassifier);
 870.            }
 871.            ensureDirectedAssociation(classifier, theClassifier);
 872.            return theClassifier;
 873.        }
 874.    
 875.        /**
 876.         * Checks if there is a directed association between two classifiers, and
 877.         * creates one if necessary.<p>
 878.         * TODO: Hide this method in the implementation of a to be defined
 879.         * interface.
 880.         */
 881.        private void ensureDirectedAssociation(Object fromCls, Object toCls) {
 882.            String fromName = Model.getFacade().getName(fromCls);
 883.            String toName = Model.getFacade().getName(toCls);
 884.            Object assocEnd = null;
 885.            for (Iterator i =
 886.                    Model.getFacade().getAssociationEnds(toCls).iterator();
 887.                 i.hasNext();) {
 888.                Object ae = i.next();
 889.                Object assoc = Model.getFacade().getAssociation(ae);
 890.                if (Model.getFacade().getConnections(assoc).size() == 2
 891.                        && Model.getFacade().getType(
 892.                                Model.getFacade().getNextEnd(ae)) == fromCls
 893.                        && Model.getFacade().getName(ae) == null
 894.                        && Model.getFacade().isNavigable(ae)) {
 895.                    assocEnd = ae;
 896.                }
 897.            }
 898.            if (assocEnd == null) {
 899.                String assocName = fromName + " -> " + toName;
 900.                Modeller.buildDirectedAssociation(assocName, fromCls, toCls);
 901.            }
 902.        }

 904.        /**
 905.         * Get the classifier with the given name from a permission
 906.         * (null if not found).<p>
 907.         * TODO: Hide this method in the implementation of a to be defined
 908.         * interface.
 909.         */
 910.        private Object permissionLookup(Object comp, String clsName) {
 911.            Object theClassifier = null;
 912.            // TODO: This could use the new CoreHelper.getPackageImports()
 913.            Collection cdeps = Model.getFacade().getClientDependencies(comp);
 914.            // TODO: Do we really need to test for null here?
 915.            // We should get empty collections.
 916.            Iterator iter1 = cdeps != null ? cdeps.iterator() : null;
 917.            while (theClassifier == null && iter1 != null && iter1.hasNext()) {
 918.                Object perm = iter1.next();
 919.                if (Model.getFacade().isAPackageImport(perm)) {
 920.                    Collection suppliers = Model.getFacade().getSuppliers(perm);
 921.                    // TODO: Do we really need to test for null here?
 922.                    // We should get empty collections.
 923.                    Iterator iter2 =
 924.                        suppliers != null ? suppliers.iterator() : null;
 925.                    while (theClassifier == null
 926.                            && iter2 != null
 927.                            && iter2.hasNext()) {
 928.                        Object elem = iter2.next();
 929.                        // TODO: I'm not sure what this is trying to do, but it
 930.                        // probably isn't what it thinks it is.  The supplier to
 931.                        // an import is going to be a Package, which is not a
 932.                        // Classifier.  Perhaps this intends to process the
 933.                        // ownedElements of the Package. - tfm - 20070803
 934.                        if (Model.getFacade().isAClassifier(elem)
 935.                             && clsName.equals(Model.getFacade().getName(elem))) {
 936.                            theClassifier = elem;
 937.                        }
 938.                    }
 939.                }
 940.            }
 941.            return theClassifier;
 942.        }

 215.                 * The next one defaults to TRUE, to stay compatible with older
 216.                 * ArgoUML versions that did not have this setting:
 217.                 */
 218.                showTypes.setSelected(Configuration.getBoolean(
 219.                        Notation.KEY_SHOW_TYPES, true));
 220.                showMultiplicity.setSelected(getBoolean(
 221.                        Notation.KEY_SHOW_MULTIPLICITY));
 222.                showStereotypes.setSelected(getBoolean(
 223.                        Notation.KEY_SHOW_STEREOTYPES));
 224.                /*
 225.                 * The next one defaults to TRUE, despite that this is
 226.                 * NOT compatible with older ArgoUML versions
 227.                 * (before 0.24) that did
 228.                 * not have this setting - see issue 1395 for the rationale:
 229.                 */
 230.                showSingularMultiplicities.setSelected(Configuration.getBoolean(
 231.                        Notation.KEY_SHOW_SINGULAR_MULTIPLICITIES, true));
 232.            }
 233.            if (scope == Argo.SCOPE_PROJECT) {
 234.                assert p != null;
 235.                ProjectSettings ps = p.getProjectSettings();
 236.                NotationSettings ns = ps.getNotationSettings();
 237.    
 238.                notationLanguage.setSelectedItem(Notation.findNotation(
 239.                        ps.getNotationLanguage()));
 240.                useGuillemots.setSelected(ps.getUseGuillemotsValue());
 241.                showAssociationNames.setSelected(ns.isShowAssociationNames());
 242.                showVisibility.setSelected(ns.isShowVisibilities());
 243.                showMultiplicity.setSelected(ns.isShowMultiplicities());
 244.                showInitialValue.setSelected(ns.isShowInitialValues());
 245.                showProperties.setSelected(ns.isShowProperties());
 246.                showTypes.setSelected(ns.isShowTypes());
 247.                showStereotypes.setSelected(ps.getShowStereotypesValue());
 248.                showSingularMultiplicities.setSelected(
 249.                        ns.isShowSingularMultiplicities());
 250.            }
 251.        }

 263.        /*
 264.         * @see org.argouml.ui.GUISettingsTabInterface#handleSettingsTabSave()
 265.         */
 266.        public void handleSettingsTabSave() {
 267.            if (scope == Argo.SCOPE_APPLICATION) {
 268.                Notation.setDefaultNotation(
 269.                        (NotationName) notationLanguage.getSelectedItem());
 270.                Configuration.setBoolean(Notation.KEY_USE_GUILLEMOTS,
 271.                        useGuillemots.isSelected());
 272.                Configuration.setBoolean(Notation.KEY_SHOW_ASSOCIATION_NAMES,
 273.                        showAssociationNames.isSelected());
 274.                Configuration.setBoolean(Notation.KEY_SHOW_VISIBILITY,
 275.                        showVisibility.isSelected());
 276.                Configuration.setBoolean(Notation.KEY_SHOW_MULTIPLICITY,
 277.                        showMultiplicity.isSelected());
 278.                Configuration.setBoolean(Notation.KEY_SHOW_INITIAL_VALUE,
 279.                        showInitialValue.isSelected());
 280.                Configuration.setBoolean(Notation.KEY_SHOW_PROPERTIES,
 281.                        showProperties.isSelected());
 282.                Configuration.setBoolean(Notation.KEY_SHOW_TYPES,
 283.                        showTypes.isSelected());
 284.                Configuration.setBoolean(Notation.KEY_SHOW_STEREOTYPES,
 285.                        showStereotypes.isSelected());
 286.                Configuration.setBoolean(Notation.KEY_SHOW_SINGULAR_MULTIPLICITIES,
 287.                        showSingularMultiplicities.isSelected());
 288.            }
 289.            if (scope == Argo.SCOPE_PROJECT) {
 290.                assert p != null;
 291.                ProjectSettings ps = p.getProjectSettings();
 292.                NotationSettings ns = ps.getNotationSettings();
 293.                NotationName nn = (NotationName) notationLanguage.getSelectedItem();
 294.                if (nn != null) {
 295.                    ps.setNotationLanguage(nn.getConfigurationValue());
 296.                }
 297.                ps.setUseGuillemots(useGuillemots.isSelected());
 298.                ns.setShowAssociationNames(showAssociationNames.isSelected());
 299.                ns.setShowVisibilities(showVisibility.isSelected());
 300.                ns.setShowMultiplicities(showMultiplicity.isSelected());
 301.                ns.setShowInitialValues(showInitialValue.isSelected());
 302.                ns.setShowProperties(showProperties.isSelected());
 303.                ns.setShowTypes(showTypes.isSelected());
 304.                ps.setShowStereotypes(showStereotypes.isSelected());
 305.                ns.setShowSingularMultiplicities(
 306.                        showSingularMultiplicities.isSelected());
 307.            }
 308.        }
