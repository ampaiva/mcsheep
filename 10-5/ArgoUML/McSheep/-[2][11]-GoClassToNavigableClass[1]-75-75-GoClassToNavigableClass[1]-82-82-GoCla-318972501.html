 238.            end0 = Model.getFacade().getGeneral(gen);
 239.            end1 = Model.getFacade().getSpecific(gen);
 240.        } else if (Model.getFacade().isADependency(edge)) {
 241.            Collection clients = Model.getFacade().getClients(edge);
 242.            Collection suppliers = Model.getFacade().getSuppliers(edge);
 243.            if (clients == null || clients.isEmpty() 
 244.                    || suppliers == null || suppliers.isEmpty()) {
 245.                    return false;
 246.                }
 247.            end0 = clients.iterator().next();
 248.            end1 = suppliers.iterator().next();
 249.        } else if (edge instanceof CommentEdge) {
 250.            end0 = ((CommentEdge) edge).getSource();
 251.            end1 = ((CommentEdge) edge).getDestination();
 252.        } else {
 253.            return false;       
 254.            }
 255.            
 256.            // Both ends must be defined and nodes that are on the graph already.
 257.            if (end0 == null || end1 == null) {
 258.                LOG.error("Edge rejected. Its ends are not attached to anything");
 259.                return false;
 260.            }
 261.            
 262.            if (!containsNode(end0)
 263.                    && !containsEdge(end0)) {
 264.                LOG.error("Edge rejected. Its source end is attached to " + end0
 265.                        + " but this is not in the graph model");
 266.                return false;
 267.            }
 268.            if (!containsNode(end1)
 269.                    && !containsEdge(end1)) {
 270.                LOG.error("Edge rejected. Its destination end is attached to "
 271.                        + end1 + " but this is not in the graph model");
 272.                return false;
 273.            }
 274.            
 275.            return true;
 276.        }
 277.    
 278.    
 279.        /*
 280.         * Add the given node to the graph, if valid.
 281.         *
 282.         * @see org.tigris.gef.graph.MutableGraphModel#addNode(java.lang.Object)
 283.         */
 284.        @Override
 285.        public void addNode(Object node) {
 286.        LOG.debug("adding MClassifierRole node!!");
 287.        if (!canAddNode(node)) {
 288.                return;
 289.            }
 290.        getNodes().add(node);
 291.        // TODO: assumes public, user pref for default visibility?
 292.        if (Model.getFacade().isAClassifier(node)) {
 293.            Model.getCoreHelper().addOwnedElement(getHomeModel(), node);
 294.            // ((MClassifier)node).setNamespace(_collab.getNamespace());
 295.        }
 296.    
 297.        fireNodeAdded(node);
 298.        }
 299.    
 300.        /*
 301.         * Add the given edge to the graph, if valid.
 302.         *
 303.         * @see org.tigris.gef.graph.MutableGraphModel#addEdge(java.lang.Object)
 304.         */
 305.        @Override
 306.        public void addEdge(Object edge) {
 307.            LOG.debug("adding class edge!!!!!!");
 308.            if (!canAddEdge(edge)) {
 309.                return;
 310.            }
 311.            getEdges().add(edge);
 312.            // TODO: assumes public
 313.            if (Model.getFacade().isAModelElement(edge)
 314.            && Model.getFacade().getNamespace(edge) == null) {
 315.                Model.getCoreHelper().addOwnedElement(getHomeModel(), edge);
 316.            }
 317.            fireEdgeAdded(edge);
 318.        }
 319.    
 320.        /*
 321.         * @see org.tigris.gef.graph.MutableGraphModel#addNodeRelatedEdges(java.lang.Object)
 322.         */
 323.        @Override
 324.        public void addNodeRelatedEdges(Object node) {
 325.            super.addNodeRelatedEdges(node);
 326.    
 327.        if (Model.getFacade().isAClassifier(node)) {
 328.            Collection ends = Model.getFacade().getAssociationEnds(node);
 329.            for (Object end : ends) {
 330.            if (canAddEdge(Model.getFacade().getAssociation(end))) {
 331.                        addEdge(Model.getFacade().getAssociation(end));
 332.                    }
 333.            }
 334.        }
 335.        if (Model.getFacade().isAGeneralizableElement(node)) {
 336.            Collection generalizations = 
 337.                Model.getFacade().getGeneralizations(node);
 338.            for (Object generalization : generalizations) {
 339.            if (canAddEdge(generalization)) {
 340.                addEdge(generalization);
 341.                return;
 342.            }
 343.            }
 344.            Collection specializations = Model.getFacade().getSpecializations(node);
 345.            for (Object specialization : specializations) {
 346.            if (canAddEdge(specialization)) {
 347.                addEdge(specialization);
 348.                return;
 349.            }
 350.            }
 351.        }
 352.        if (Model.getFacade().isAModelElement(node)) {
 353.            Collection dependencies =
 354.            new ArrayList(Model.getFacade().getClientDependencies(node));
 355.            dependencies.addAll(Model.getFacade().getSupplierDependencies(node));
 356.            for (Object dependency : dependencies) {
 357.            if (canAddEdge(dependency)) {
 358.                addEdge(dependency);
 359.                return;
 360.            }
 361.            }
 362.        }
 363.        }

 366.        /*
 367.         * Return true if the two given ports can be connected by a
 368.         * kind of edge to be determined by the ports.
 369.         *

 376.            && (Model.getFacade().isAClassifierRole(toP))) {

 378.            }

 380.        }

 388.        public void vetoableChange(PropertyChangeEvent pce) {
 389.        //throws PropertyVetoException
 390.    
 391.        if ("ownedElement".equals(pce.getPropertyName())) {
 392.            List oldOwned = (List) pce.getOldValue();
 393.            Object eo = /*(MElementImport)*/ pce.getNewValue();
 394.            Object me = Model.getFacade().getModelElement(eo);
 395.            if (oldOwned.contains(eo)) {
 396.            LOG.debug("model removed " + me);
 397.            if (Model.getFacade().isAClassifier(me)) {
 398.                        removeNode(me);
 399.                    }
 400.            if (Model.getFacade().isAMessage(me)) {
 401.                        removeNode(me);
 402.                    }
 403.            if (Model.getFacade().isAAssociation(me)) {
 404.                        removeEdge(me);
 405.                    }
 406.            } else {
 407.            LOG.debug("model added " + me);
 408.            }
 409.        }
 410.        }
