 152.        public void setNamespace(Object handle) {
 153.            if (!Model.getFacade().isANamespace(handle)) {
 154.                LOG.error(
 155.                    "Illegal argument. Object " + handle + " is not a namespace");
 156.                throw new IllegalArgumentException(
 157.                    "Illegal argument. Object " + handle + " is not a namespace");
 158.            }
 159.            Object m = handle;
 160.            boolean init = (null == getNamespace());
 161.            super.setNamespace(m);
 162.            DeploymentDiagramGraphModel gm = createGraphModel();
 163.            gm.setHomeModel(m);
 164.            if (init) {
 165.                LayerPerspective lay =
 166.                    new LayerPerspectiveMutable(Model.getFacade().getName(m), gm);
 167.                DeploymentDiagramRenderer rend = new DeploymentDiagramRenderer();
 168.                lay.setGraphNodeRenderer(rend);
 169.                lay.setGraphEdgeRenderer(rend);
 170.                setLayer(lay);
 171.            }
 172.        }
