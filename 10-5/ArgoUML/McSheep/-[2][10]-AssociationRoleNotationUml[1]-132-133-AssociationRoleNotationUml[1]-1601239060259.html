  74.         * @param callState the UML CallState
  75.         */
  76.        public CallStateNotationUml(Object callState) {
  77.            super(callState);

  81.         * @see org.argouml.notation.providers.NotationProvider#parse(java.lang.Object, java.lang.String...
  82.         */
  83.        public void parse(Object modelElement, String text) {
  84.            try {
  85.                parseCallState(modelElement, text);
  86.            } catch (ParseException pe) {
  87.                String msg = "statusmsg.bar.error.parsing.callstate";
  88.                Object[] args = {pe.getLocalizedMessage(),
  89.                                 Integer.valueOf(pe.getErrorOffset()), };
  90.                ArgoEventPump.fireEvent(new ArgoHelpEvent(
  91.                        ArgoEventTypes.HELP_CHANGED, this,
  92.                        Translator.messageFormat(msg, args)));
  93.            }
  94.        }

  96.        protected Object parseCallState(Object callState, String s1)
  97.            throws ParseException {
  98.    
  99.            String s = s1.trim();
 100.    
 101.            int a = s.indexOf("(");
 102.            int b = s.indexOf(")");
 103.            if (((a < 0) && (b >= 0)) || ((b < 0) && (a >= 0)) || (b < a)) {
 104.                throw new ParseException("No matching brackets () found.", 0);
 105.            }
 106.    
 107.            /* First we decode the string: */
 108.            String newClassName = null;
 109.            String newOperationName = null;
 110.            StringTokenizer tokenizer = new StringTokenizer(s, "(");
 111.            while (tokenizer.hasMoreTokens()) {
 112.                String nextToken = tokenizer.nextToken().trim();
 113.                if (nextToken.endsWith(")")) {
 114.                    newClassName = nextToken.substring(0, nextToken.length() - 1);
 115.                } else {
 116.                    newOperationName = nextToken.trim();
 117.                }
 118.            }
 119.    
 120.            /* Secondly we check the previous model structure: */
 121.            String oldOperationName = null;
 122.            String oldClassName = null;
 123.            Object entry = Model.getFacade().getEntry(callState);
 124.            Object operation = null;
 125.            Object clazz = null;
 126.            if (Model.getFacade().isACallAction(entry)) {
 127.                operation = Model.getFacade().getOperation(entry);
 128.                if (Model.getFacade().isAOperation(operation)) {
 129.                    oldOperationName = Model.getFacade().getName(operation);
 130.                    clazz = Model.getFacade().getOwner(operation);
 131.                    oldClassName = Model.getFacade().getName(clazz);
 132.                }
 133.            }
 134.            
 135.            /* And 3rd, we adapt the model: */
 136.            boolean found = false;
 137.            if ((newClassName != null) 
 138.                    && newClassName.equals(oldClassName)
 139.                    && (newOperationName != null) 
 140.                    && !newOperationName.equals(oldOperationName)) {
 141.                // Same class, other operation
 142.                for ( Object op : Model.getFacade().getOperations(clazz)) {
 143.                    if (newOperationName.equals(
 144.                            Model.getFacade().getName(op))) {
 145.                        Model.getCommonBehaviorHelper().setOperation(entry, op);
 146.                        found = true;
 147.                        break;
 148.                    }
 149.                }
 150.                if (!found) {
 151.                    throw new ParseException(
 152.                            "Operation " + newOperationName 
 153.                            + " not found in " + newClassName + ".", 0);
 154.                }
 155.    
 156.            } else if ((newClassName != null) 
 157.                    && !newClassName.equals(oldClassName)
 158.                    && (newOperationName != null)) {
 159.                // Other class
 160.                Object model = 
 161.                    ProjectManager.getManager().getCurrentProject().getRoot();
 162.                Collection c =
 163.                    Model.getModelManagementHelper().
 164.                        getAllModelElementsOfKind(model,
 165.                                    Model.getMetaTypes().getClassifier());
 166.                Iterator i = c.iterator();
 167.                Object classifier = null;
 168.                while (i.hasNext()) {
 169.                    Object cl = i.next();
 170.                    String cn = Model.getFacade().getName(cl);
 171.                    if (cn.equals(newClassName)) {
 172.                        classifier = cl;
 173.                        break;
 174.                    }
 175.                }
 176.                if (classifier == null) {
 177.                    throw new ParseException(
 178.                            "Classifier " + newClassName + " not found.", 0);
 179.                }
 180.                // We found the classifier, now go find the operation:
 181.                if (classifier != null) {
 182.                    Collection ops = Model.getFacade().getOperations(classifier);
 183.                    Iterator io = ops.iterator();
 184.                    while (io.hasNext()) {
 185.                        Object op = io.next();
 186.                        if (newOperationName.equals(
 187.                                Model.getFacade().getName(op))) {
 188.                            /* Here we located the new classifier 
 189.                             * with its operation. */
 190.                            found = true;
 191.                            if (!Model.getFacade().isACallAction(entry)) {
 192.                                entry = Model.getCommonBehaviorFactory()
 193.                                    .buildCallAction(op, "ca");
 194.                                Model.getStateMachinesHelper().setEntry(
 195.                                        callState, entry);
 196.                            } else {
 197.                                Model.getCommonBehaviorHelper().setOperation(
 198.                                        entry, op);
 199.                            }
 200.                            break;
 201.                        }
 202.                    }
 203.                    if (!found) {
 204.                        throw new ParseException(
 205.                                "Operation " + newOperationName 
 206.                                + " not found in " + newClassName + ".", 0);
 207.                    }
 208.                }
 209.            }
 210.            if (!found) {
 211.                throw new ParseException(
 212.                        "Incompatible input found.", 0);
 213.            }
 214.    
 215.            return callState;
 216.        }

 225.        private String toString(Object modelElement) {
 226.            String ret = "";
 227.            Object action = Model.getFacade().getEntry(modelElement);
 228.            if (Model.getFacade().isACallAction(action)) {
 229.                Object operation = Model.getFacade().getOperation(action);
 230.                if (operation != null) {
 231.                    Object n = Model.getFacade().getName(operation);
 232.                    if (n != null) {
 233.                        ret = (String) n;
 234.                    }
 235.                    n =
 236.                        Model.getFacade().getName(
 237.                            Model.getFacade().getOwner(operation));
 238.                    if (n != null && !n.equals("")) {
 239.                        ret += "\n(" + (String) n + ")";
 240.                    }
 241.                }
 242.            }
 243.    
 244.            if (ret == null) {
 245.                return "";
 246.            }
 247.            return ret;
 248.        }
