 337.                                Model.getAggregationKind().getComposite());
 338.                        Model.getCoreHelper().setAggregation1(ae1,
 339.                                Model.getAggregationKind().getNone());
 340.                        break;
 341.    
 342.                    case 1:
 343.    
 344.                        // Start is a shared aggregation of end
 345.    
 346.                        Model.getCoreHelper().setAggregation1(ae0,
 347.                                Model.getAggregationKind().getAggregate());
 348.                        Model.getCoreHelper().setAggregation1(ae1,
 349.                                Model.getAggregationKind().getNone());
 350.                        break;
 351.    
 352.                    case 2:
 353.    
 354.                        // End is a composite aggregation of start
 355.    
 356.                        Model.getCoreHelper().setAggregation1(ae0,
 357.                                Model.getAggregationKind().getNone());
 358.                        Model.getCoreHelper().setAggregation1(ae1,
 359.                                Model.getAggregationKind().getComposite());
 360.                        break;
 361.    
 362.                    case 3:
 363.    
 364.                        // End is a shared aggregation of start
 365.                        Model.getCoreHelper().setAggregation1(ae0,
 366.                                Model.getAggregationKind().getNone());
 367.                        Model.getCoreHelper().setAggregation1(ae1,
 368.                                Model.getAggregationKind().getAggregate());
 369.                        break;
 370.    
 371.                    case 4:
 372.    
 373.                        // No aggregation
 374.                        Model.getCoreHelper().setAggregation1(ae0,
 375.                                Model.getAggregationKind().getNone());
 376.                        Model.getCoreHelper().setAggregation1(ae1,
 377.                                Model.getAggregationKind().getNone());
 378.                        break;
 379.    
 380.                    default:
 381.                    }
 382.                } catch (Exception pve) {
 383.    
 384.                    // Someone took our association away.
 385.    
 386.                    LOG.error("WizAssocComposite: could not set " + "aggregation.",
 387.                            pve);
 388.                }
 389.    
 390.            default:
 391.            }
 392.        }
 393.    
 394.        /**
 395.         * Determine if we have sufficient information to finish.
 396.         * <p>
 397.         * We can't finish if our parent
 398.         * {@link org.argouml.cognitive.critics.Wizard} can't finish.
 399.         * <p>
 400.         * We can finish if we're on step 0.
 401.         * <p>
 402.         * We can finish if we're on step 1 and have made a choice.
 403.         * <p>
 404.         * 
 405.         * @return <code>true</code> if we can finish, otherwise
 406.         *         <code>false</code>.
 407.         * @see org.argouml.cognitive.critics.Wizard
 408.         */
 409.        @Override
 410.        public boolean canFinish() {
 411.    
 412.            // Can't finish if our parent can't
 413.    
 414.            if (!super.canFinish()) {
 415.                return false;
 416.            }
 417.    
 418.            // Can finish if it's step 0
 419.    
 420.            if (getStep() == 0) {
 421.                return true;
 422.            }
 423.    
 424.            // Can finish if we're on step1 and have actually made a choice
 425.    
 426.            if ((getStep() == 1) && (step1Choice != null)
 427.                    && (step1Choice.getSelectedIndex() != -1)) {
 428.                return true;
 429.            }
 430.    
 431.            // Otherwise we can't finish
 432.    
 433.            return false;
 434.        }
 435.    
 436.    }

  66.            Translator.localize("critics.WizNavigable-option2");

 130.        public void doAction(int oldStep) {
 131.        LOG.debug("doAction " + oldStep);

 134.            int choice = -1;

 144.                new ArrayList(Model.getFacade().getConnections(asc)).get(0);

 148.                    choice == 0 || choice == 2);
