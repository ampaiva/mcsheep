 243.    class MyLocale {
 244.        private Locale myLocale;
 245.    
 246.        /**
 247.         * The constructor.
 248.         *
 249.         * @param locale the Locale
 250.         */
 251.        MyLocale(Locale locale) {
 252.            myLocale = locale;
 253.        }
 254.    
 255.        /**
 256.         * @return returns the locale
 257.         */
 258.        Locale getLocale() {
 259.            return myLocale;
 260.        }
 261.    
 262.        /*
 263.         * @see java.lang.Object#toString()
 264.         */
 265.        public String toString() {
 266.            StringBuffer displayString = new StringBuffer(myLocale.toString());
 267.            displayString.append(" (");
 268.            displayString.append(myLocale.getDisplayLanguage(myLocale));
 269.            if (myLocale.getDisplayCountry(myLocale) != null 
 270.                    && myLocale.getDisplayCountry(myLocale).length() > 0) {
 271.                displayString.append(" ");
 272.                displayString.append(myLocale.getDisplayCountry(myLocale));
 273.            }
 274.            displayString.append(")");
 275.            if (myLocale.equals(Translator.getSystemDefaultLocale())) {
 276.                displayString.append(" - Default");
 277.            }
 278.            return displayString.toString();
 279.        }
 280.    
 281.        static Collection<MyLocale> getLocales() {
 282.            Collection<MyLocale> c = new ArrayList<MyLocale>();
 283.            for (Locale locale : Arrays.asList(Translator.getLocales())) {
 284.                c.add(new MyLocale(locale));
 285.            }
 286.            return c;
 287.        }
 288.    
 289.        static MyLocale getDefault(Collection<MyLocale> c) {
 290.            Locale locale = Locale.getDefault();
 291.            for (MyLocale ml : c) {
 292.                if (locale.equals(ml.getLocale())) {
 293.                    return ml;
 294.                }
 295.            }
 296.            return null;
 297.        }

 319.         * @see org.argouml.ui.GUISettingsTabInterface#handleSettingsTabSave()
 320.         */
 321.        public void handleSettingsTabSave() {
 322.            if (getActionAlreadyAssigned(ShortcutMgr
 323.                    .decodeKeyStroke(shortcutField.getText())) != null) {
 324.                // conflict detected: showing a warning to the user, instead of
 325.                // saving shortcuts
 326.                JOptionPane.showMessageDialog(this, 
 327.                        Translator.localize(
 328.                                "optionpane.shortcut-save-conflict"),
 329.                        Translator.localize(
 330.                                "optionpane.shortcut-save-conflict-title"),
 331.                              JOptionPane.WARNING_MESSAGE);
 332.            } else {
 333.                // saving shortcuts
 334.                ShortcutMgr.saveShortcuts(actions);
 335.            }
 336.        }
 337.    
 338.        /**
 339.         * This is called every time a row is selected. It just updates the fields
 340.         * and the buttons status
 341.         * 
 342.         * @see javax.swing.event.ListSelectionListener#valueChanged(javax.swing.event.ListSelectionEven...
 343.         */
 344.        public void valueChanged(ListSelectionEvent lse) {
 345.            if (lse.getValueIsAdjusting()) {
 346.                return;
 347.            }
 348.            Object src = lse.getSource();
 349.            if (src != table.getSelectionModel() || table.getSelectedRow() == -1) {
 350.                return;
 351.            }
 352.            // if a shortcut has been select then we have to check if the actual
 353.            // action is in conflict with other ones
 354.            if (!noneButton.isSelected()) {
 355.                ActionWrapper oldAction = getActionAlreadyAssigned(ShortcutMgr
 356.                        .decodeKeyStroke(shortcutField.getText()));
 357.                if (oldAction != null) {
 358.                    // this shortcut was already been assigned to another action;
 359.                    // let's pop-up a message for the user
 360.                    String t = MessageFormat.format(Translator
 361.                            .localize("optionpane.conflict-shortcut"),
 362.                            new Object[] {shortcutField.getText(),
 363.                                    oldAction.getActionName() });
 364.                    int response = JOptionPane.showConfirmDialog(this, t, t,
 365.                            JOptionPane.YES_NO_OPTION);
 366.                    switch (response) {
 367.                    case JOptionPane.YES_OPTION:
 368.                        oldAction.setCurrentShortcut(null);
 369.                        // blanking the old action's shortcut..
 370.                        // and now refreshing the table.
 371.                        table.setValueAt(oldAction, -1, -1);
 372.                        break;
 373.                    case JOptionPane.NO_OPTION:
 374.                        // re-selecting the old row, without changing the target -
 375.                        // and without throwing another ListSelectionEvent!
 376.                        table.getSelectionModel().removeListSelectionListener(this);
 377.                        table.getSelectionModel().setSelectionInterval(
 378.                                lastRowSelected, lastRowSelected);
 379.                        table.getSelectionModel().addListSelectionListener(this);
 380.                        return;
 381.                    }
 382.                }
 383.            }
 384.            // let's change the target
 385.            setTarget(actions[table.getSelectedRow()]);
 386.    
 387.            lastRowSelected = table.getSelectedRow();
 388.        }
 389.    
 390.        /**
 391.         * Verifies if the given keyStroke has already been assigned
 392.         * to another action 
 393.         * 
 394.         * @param keyStroke
 395.         *            the KeyStroke to be checked
 396.         * @return the Action that has already been assigned
 397.         */
 398.        public ActionWrapper getActionAlreadyAssigned(KeyStroke keyStroke) {
 399.            for (int i = 0; i < actions.length; i++) {
 400.                if (actions[i].getCurrentShortcut() != null
 401.                        && actions[i].getCurrentShortcut().equals(keyStroke)
 402.                        && !actions[i].getActionName().equals(
 403.                                target.getActionName())) {
 404.                    return actions[i];
 405.                }
 406.            }
 407.            // duplicate shortcut not found; let's check for duplicates
 408.            KeyStroke duplicate = ShortcutMgr.getDuplicate(keyStroke);
 409.            if (duplicate != null) {
 410.                // there's a duplicate: let's recheck if there is a conflict
 411.                for (int i = 0; i < actions.length; i++) {
 412.                    if (actions[i].getCurrentShortcut() != null
 413.                            && actions[i].getCurrentShortcut().equals(duplicate)
 414.                            && !actions[i].getActionName().equals(
 415.                                    target.getActionName())) {
 416.                        return actions[i];
 417.                    }
 418.                }            
 419.            }
 420.            return null;
 421.        }
 422.    
 423.        /**
 424.         * This is called every time a panel button (custom / default / none) is
 425.         * pressed.
 426.         * 
 427.         * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
 428.         */
 429.        public void actionPerformed(ActionEvent e) {
 430.            resetKeyStrokeConflict();
 431.    
 432.            if (e.getSource() == customButton) {
 433.                setKeyStrokeValue(ShortcutMgr.decodeKeyStroke(shortcutField
 434.                        .getText()));
 435.                shortcutField.setEnabled(true);
 436.                shortcutField.requestFocus();
 437.            } else if (e.getSource() == defaultButton) {
 438.                setKeyStrokeValue(target.getDefaultShortcut());
 439.                shortcutField.setEnabled(false);
 440.                checkShortcutAlreadyAssigned(target.getDefaultShortcut());
 441.            } else if (e.getSource() == noneButton) {
 442.                setKeyStrokeValue(null);
 443.                shortcutField.setEnabled(false);
 444.            }
 445.        }
 446.    
 447.        /**
 448.         * This method simply reset the "conflict" gui, by blanking the warning
 449.         * label and resetting the shortcut's background color
 450.         *
 451.         */
 452.        private void resetKeyStrokeConflict() {
 453.            this.warningLabel.setText(" ");
 454.            this.shortcutField.setBackground(shortcutFieldDefaultBg);
 455.        }
 456.    
 457.        /**
 458.         * Updates the GUI with the given new KeyStroke
 459.         * 
 460.         * @param newKeyStroke      the KeyStroke to be set
 461.         */
 462.        private void setKeyStrokeValue(KeyStroke newKeyStroke) {
 463.            String formattedKeyStroke = KeyEventUtils.formatKeyStroke(newKeyStroke);
 464.            // updating the shortcut field
 465.            shortcutField.setText(formattedKeyStroke);
 466.            // updating the table data
 467.            table.getModel().setValueAt(newKeyStroke, table.getSelectedRow(), 1);
 468.            // updating the actions
 469.            actions[table.getSelectedRow()].setCurrentShortcut(newKeyStroke);
 470.            table.repaint();
 471.        }
 472.    
 473.        /**
 474.         * Listener method for ShortcutChangedEvents
 475.         * 
 476.         * @see org.argouml.ui.cmd.ShortcutChangedListener#shortcutChange(org.argouml.ui.cmd.ShortcutCha...
 477.         */
 478.        public void shortcutChange(ShortcutChangedEvent event) {
 479.            checkShortcutAlreadyAssigned(event.getKeyStroke());
 480.            setKeyStrokeValue(event.getKeyStroke());
 481.            this.selectedContainer.repaint();
 482.        }

 484.        /**
 485.         * Update the GUI, if there is a conflict for the given key stroke
 486.         * 
 487.         * @param newKeyStroke      the key stroke to be checked
 488.         */
 489.        private void checkShortcutAlreadyAssigned(KeyStroke newKeyStroke) {
 490.            ActionWrapper oldAction = getActionAlreadyAssigned(newKeyStroke);
 491.            if (oldAction != null) {
 492.                // the shortcut has already been assigned to another action!
 493.                this.shortcutField.setBackground(Color.YELLOW);
 494.                this.warningLabel.setText(MessageFormat.format(Translator
 495.                        .localize("misc.shortcuts.conflict"),
 496.                        new Object[] {KeyEventUtils.formatKeyStroke(oldAction
 497.                                .getCurrentShortcut()), 
 498.                                oldAction.getActionName()}));
 499.            } else {
 500.                resetKeyStrokeConflict();
 501.            }
 502.        }

 504.        /**
 505.         * Table model for the table with actions and shortcuts.
 506.         */
 507.        class ShortcutTableModel extends AbstractTableModel {
 508.    
 509.            /**
 510.             * Constructor.
 511.             */
 512.            public ShortcutTableModel() {
 513.                elements = new Object[actions.length][3];
 514.    
 515.                for (int i = 0; i < elements.length; i++) {
 516.                    ActionWrapper currentAction = actions[i];
 517.                    elements[i][0] = currentAction.getActionName();
 518.                    elements[i][1] = currentAction.getCurrentShortcut();
 519.                    elements[i][2] = currentAction.getDefaultShortcut();
 520.                }
 521.            }
 522.    
 523.            /*
 524.             * @see javax.swing.table.TableModel#getColumnCount()
 525.             */

 527.                return columnNames.length;
 528.            }

 530.            /*
 531.             * @see javax.swing.table.TableModel#getColumnName(int)

 536.            }

 539.             * @see javax.swing.table.TableModel#getRowCount()
 540.             */
 541.            public int getRowCount() {
 542.                return elements.length;
 543.            }
 544.    
 545.            /*
 546.             * @see javax.swing.table.TableModel#getValueAt(int, int)
 547.             */
 548.            public Object getValueAt(int row, int col) {
 549.                return elements[row][col];
 550.            }
 551.    
 552.            /**
 553.             * Sets the value in the cell at <code>col</code> and
 554.             * <code>row</code> to <code>ob</code>. If <code>ob</code> is
 555.             * an Action instance, then the <code>col</code> and <code>row</code>
 556.             * parameters are ignored, and the ob Action is searched among the 
 557.             * elements and the actions.
 558.             * 
 559.             * @see javax.swing.table.TableModel#setValueAt( java.lang.Object, int,
 560.             *      int)
 561.             */
 562.            @Override
 563.            public void setValueAt(Object ob, int row, int col) {
 564.                // if the given object is a KeyStroke instance, then we ca
 565.                if (ob instanceof ActionWrapper) {
 566.                    ActionWrapper newValueAction = (ActionWrapper) ob;
 567.                    for (int i = 0; i < elements.length; i++) {
 568.                        if (elements[i][0].equals(newValueAction.getActionName())) {
 569.                            elements[i][1] = newValueAction.getCurrentShortcut();
 570.                            repaint();
 571.                            break;
 572.                        }
 573.                    }
 574.                    // let's update also the actions array
 575.                    for (int i = 0; i < actions.length; i++) {
 576.                        if (actions[i].getKey().equals(newValueAction.getKey())) {
 577.                            actions[i].setCurrentShortcut(newValueAction
 578.                                    .getCurrentShortcut());
 579.                            break;
 580.                        }
 581.                    }
 582.    
 583.                } else {
 584.                    elements[row][col] = ob;
 585.                }
 586.            }
 587.    
 588.            /*
 589.             * @see javax.swing.table.TableModel#getColumnClass(int)
 590.             */
 591.            @Override
 592.            public Class<?> getColumnClass(int col) {
 593.                switch (col) {
 594.                case 0:
 595.                    return String.class;
 596.                case 1:
 597.                    return KeyStroke.class;
 598.                case 2:
 599.                    return KeyStroke.class;
 600.                default:
 601.                    return null;
 602.                }
 603.            }

 611.            }

 613.        }
