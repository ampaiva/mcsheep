 317.        private void loadModel() {
 318.            synchronized (packageLock) {
 319.                if (profilePackages == null) {
 320.                    try {
 321.                        if (modelFile != null) {
 322.                            profilePackages = new FileModelLoader()

 326.                                    .loadModel(reference);

 328.                    } catch (ProfileException e1) {
 329.                        LOG.error("Exception loading profile "
 330.                                + reference.getPath(), e1);
 331.                        profilePackages = Collections.emptySet();
 332.                        return;
 333.                    }
 334.                }
 335.    
 336.                Collection packagesInProfile = filterPackages(profilePackages);
 337.    
 338.                for (Object obj : packagesInProfile) {
 339.                    // if there is only one package in the model, 
 340.                    // we should suppose it's the profile model, 
 341.                    // if there is more than one, we take the ones
 342.                    // marked as <<profile>>
 343.                    if (Model.getFacade().isAModelElement(obj)
 344.                            && (Model.getFacade().isAProfile(obj)
 345.                                    || (packagesInProfile.size() == 1))) {
 346.    
 347.                        // load profile name
 348.                        String name = Model.getFacade().getName(obj);
 349.                        if (name != null) {
 350.                            displayName = name;
 351.                        } else {
 352.                            if (displayName == null) {
 353.                                displayName = Translator
 354.                                        .localize("misc.profile.unnamed");
 355.                            }
 356.                        }
 357.                        LOG.info("profile " + displayName);
 358.    
 359.                        loadDependentProfiles(obj);
 360.                    }
 361.                }
 362.    
 363.                loadFigNodes(packagesInProfile);
 364.            }
 365.        }

 400.                                + "\" has a dependency named \"" + dependencyName

 473.        private CrOCL generateCriticFromComment(Object critique) {
 474.            String ocl = "" + Model.getFacade().getBody(critique);
 475.            String headline = null;
 476.            String description = null;
 477.            int priority = ToDoItem.HIGH_PRIORITY;
 478.            List<Decision> supportedDecisions = new ArrayList<Decision>();
 479.            List<String> knowledgeTypes = new ArrayList<String>();
 480.            String moreInfoURL = null;
 481.    
 482.            Collection tags = Model.getFacade().getTaggedValuesCollection(critique);
 483.            boolean i18nFound = false;
 484.            
 485.            for (Object tag : tags) {
 486.                if (Model.getFacade().getTag(tag).toLowerCase().equals("i18n")) {
 487.                    i18nFound = true;
 488.                    String i18nSource = Model.getFacade().getValueOfTag(tag);
 489.                    headline = Translator.localize(i18nSource + "-head");
 490.                    description = Translator.localize(i18nSource + "-desc");
 491.                    moreInfoURL = Translator.localize(i18nSource + "-moreInfoURL");
 492.                } else if (!i18nFound
 493.                        && Model.getFacade().getTag(tag).toLowerCase().equals(
 494.                                "headline")) {
 495.                    headline = Model.getFacade().getValueOfTag(tag);
 496.                } else if (!i18nFound
 497.                        && Model.getFacade().getTag(tag).toLowerCase().equals(
 498.                                "description")) {
 499.                    description = Model.getFacade().getValueOfTag(tag);
 500.                } else if (Model.getFacade().getTag(tag).toLowerCase().equals(
 501.                        "priority")) {
 502.                    priority = str2Priority(Model.getFacade().getValueOfTag(tag));
 503.                } else if (Model.getFacade().getTag(tag).toLowerCase().equals(
 504.                        "supporteddecision")) {
 505.                    String decStr = Model.getFacade().getValueOfTag(tag);
 506.    
 507.                    StringTokenizer st = new StringTokenizer(decStr, ",;:");
 508.    
 509.                    while (st.hasMoreTokens()) {
 510.                        Decision decision = str2Decision(st.nextToken().trim()
 511.                                .toLowerCase());
 512.                        
 513.                        if (decision != null) {
 514.                            supportedDecisions.add(decision);
 515.                        }
 516.                    }
 517.                } else if (Model.getFacade().getTag(tag).toLowerCase().equals(
 518.                        "knowledgetype")) {
 519.                    String ktStr = Model.getFacade().getValueOfTag(tag);
 520.    
 521.                    StringTokenizer st = new StringTokenizer(ktStr, ",;:");
 522.    
 523.                    while (st.hasMoreTokens()) {
 524.                        String knowledge = str2KnowledgeType(st.nextToken().trim()
 525.                                .toLowerCase());
 526.                        
 527.                        if (knowledge != null) {
 528.                            knowledgeTypes.add(knowledge);
 529.                        }
 530.                    }
 531.                } else if (!i18nFound
 532.                        && Model.getFacade().getTag(tag).toLowerCase().equals(
 533.                                "moreinfourl")) {
 534.                    moreInfoURL = Model.getFacade().getValueOfTag(tag);
 535.                }
 536.    
 537.            }
 538.    
 539.            LOG.debug("OCL-Critic: " + ocl);
 540.    
 541.            try {
 542.                return new CrOCL(ocl, headline, description, priority,
 543.                        supportedDecisions, knowledgeTypes, moreInfoURL);
 544.            } catch (InvalidOclException e) {
 545.                LOG.error("Invalid OCL in XMI!", e);
 546.    
 547.                return null;
 548.            }
 549.    
 550.        }

 314.         * @param manager the profile manager which will be used to resolve
 315.         *        dependencies
 316.         */
 317.        private void loadModel() {
 318.            synchronized (packageLock) {
 319.                if (profilePackages == null) {
 320.                    try {
 321.                        if (modelFile != null) {
 322.                            profilePackages = new FileModelLoader()
 323.                                    .loadModel(reference);
 324.                        } else {
 325.                            profilePackages = new URLModelLoader()
 326.                                    .loadModel(reference);
 327.                        }
 328.                    } catch (ProfileException e1) {
 329.                        LOG.error("Exception loading profile "
 330.                                + reference.getPath(), e1);
 331.                        profilePackages = Collections.emptySet();
 332.                        return;
 333.                    }
 334.                }
 335.    
 336.                Collection packagesInProfile = filterPackages(profilePackages);
 337.    
 338.                for (Object obj : packagesInProfile) {
 339.                    // if there is only one package in the model, 
 340.                    // we should suppose it's the profile model, 
 341.                    // if there is more than one, we take the ones
 342.                    // marked as <<profile>>
 343.                    if (Model.getFacade().isAModelElement(obj)
 344.                            && (Model.getFacade().isAProfile(obj)
 345.                                    || (packagesInProfile.size() == 1))) {
 346.    
 347.                        // load profile name
 348.                        String name = Model.getFacade().getName(obj);
 349.                        if (name != null) {
 350.                            displayName = name;
 351.                        } else {
 352.                            if (displayName == null) {
 353.                                displayName = Translator
 354.                                        .localize("misc.profile.unnamed");
 355.                            }
 356.                        }
 357.                        LOG.info("profile " + displayName);
 358.    
 359.                        loadDependentProfiles(obj);
 360.                    }
 361.                }
 362.    
 363.                loadFigNodes(packagesInProfile);
 364.            }
 365.        }

 473.        private CrOCL generateCriticFromComment(Object critique) {
 474.            String ocl = "" + Model.getFacade().getBody(critique);
 475.            String headline = null;
 476.            String description = null;
 477.            int priority = ToDoItem.HIGH_PRIORITY;
 478.            List<Decision> supportedDecisions = new ArrayList<Decision>();
 479.            List<String> knowledgeTypes = new ArrayList<String>();
 480.            String moreInfoURL = null;
 481.    
 482.            Collection tags = Model.getFacade().getTaggedValuesCollection(critique);
 483.            boolean i18nFound = false;
 484.            
 485.            for (Object tag : tags) {
 486.                if (Model.getFacade().getTag(tag).toLowerCase().equals("i18n")) {
 487.                    i18nFound = true;
 488.                    String i18nSource = Model.getFacade().getValueOfTag(tag);
 489.                    headline = Translator.localize(i18nSource + "-head");
 490.                    description = Translator.localize(i18nSource + "-desc");
 491.                    moreInfoURL = Translator.localize(i18nSource + "-moreInfoURL");
 492.                } else if (!i18nFound
 493.                        && Model.getFacade().getTag(tag).toLowerCase().equals(
 494.                                "headline")) {
 495.                    headline = Model.getFacade().getValueOfTag(tag);
 496.                } else if (!i18nFound
 497.                        && Model.getFacade().getTag(tag).toLowerCase().equals(
 498.                                "description")) {
 499.                    description = Model.getFacade().getValueOfTag(tag);
 500.                } else if (Model.getFacade().getTag(tag).toLowerCase().equals(
 501.                        "priority")) {
 502.                    priority = str2Priority(Model.getFacade().getValueOfTag(tag));
 503.                } else if (Model.getFacade().getTag(tag).toLowerCase().equals(
 504.                        "supporteddecision")) {
 505.                    String decStr = Model.getFacade().getValueOfTag(tag);
 506.    
 507.                    StringTokenizer st = new StringTokenizer(decStr, ",;:");
 508.    
 509.                    while (st.hasMoreTokens()) {
 510.                        Decision decision = str2Decision(st.nextToken().trim()
 511.                                .toLowerCase());
 512.                        
 513.                        if (decision != null) {
 514.                            supportedDecisions.add(decision);
 515.                        }
 516.                    }
 517.                } else if (Model.getFacade().getTag(tag).toLowerCase().equals(
 518.                        "knowledgetype")) {
 519.                    String ktStr = Model.getFacade().getValueOfTag(tag);
 520.    
 521.                    StringTokenizer st = new StringTokenizer(ktStr, ",;:");
 522.    
 523.                    while (st.hasMoreTokens()) {
 524.                        String knowledge = str2KnowledgeType(st.nextToken().trim()
 525.                                .toLowerCase());
 526.                        
 527.                        if (knowledge != null) {
 528.                            knowledgeTypes.add(knowledge);
 529.                        }
 530.                    }
 531.                } else if (!i18nFound
 532.                        && Model.getFacade().getTag(tag).toLowerCase().equals(
 533.                                "moreinfourl")) {
 534.                    moreInfoURL = Model.getFacade().getValueOfTag(tag);
 535.                }
 536.    
 537.            }
 538.    
 539.            LOG.debug("OCL-Critic: " + ocl);
 540.    
 541.            try {
 542.                return new CrOCL(ocl, headline, description, priority,
 543.                        supportedDecisions, knowledgeTypes, moreInfoURL);
 544.            } catch (InvalidOclException e) {
 545.                LOG.error("Invalid OCL in XMI!", e);
 546.    
 547.                return null;
 548.            }
 549.    
 550.        }

 552.        private String str2KnowledgeType(String token) {
 553.            String knowledge = null;
 554.            
 555.            if (token.equals("completeness")) {
 556.                knowledge = Critic.KT_COMPLETENESS;
 557.            }
 558.            if (token.equals("consistency")) {
 559.                knowledge = Critic.KT_CONSISTENCY;
 560.            }
 561.            if (token.equals("correctness")) {
 562.                knowledge = Critic.KT_CORRECTNESS;
 563.            }
 564.            if (token.equals("designers")) {
 565.                knowledge = Critic.KT_DESIGNERS;
 566.            }
 567.            if (token.equals("experiencial")) {
 568.                knowledge = Critic.KT_EXPERIENCIAL;
 569.            }
 570.            if (token.equals("optimization")) {
 571.                knowledge = Critic.KT_OPTIMIZATION;
 572.            }
 573.            if (token.equals("organizational")) {
 574.                knowledge = Critic.KT_ORGANIZATIONAL;
 575.            }
 576.            if (token.equals("presentation")) {
 577.                knowledge = Critic.KT_PRESENTATION;
 578.            }
 579.            if (token.equals("semantics")) {
 580.                knowledge = Critic.KT_SEMANTICS;

 758.        private FigNodeDescriptor loadImage(String stereotype, File f)
 759.            throws IOException {
 760.            FigNodeDescriptor descriptor = new FigNodeDescriptor();
 761.            descriptor.length = (int) f.length();
 762.            descriptor.src = f.getPath();
 763.            descriptor.stereotype = stereotype;
 764.    
 765.            BufferedInputStream bis = new BufferedInputStream(
 766.                    new FileInputStream(f));
 767.    
 768.            byte[] buf = new byte[descriptor.length];
 769.            try {
 770.                bis.read(buf);
 771.            } catch (IOException e) {
 772.                e.printStackTrace();
 773.            }
 774.    
 775.            descriptor.img = new ImageIcon(buf).getImage();
 776.    
 777.            return descriptor;
 778.        }
