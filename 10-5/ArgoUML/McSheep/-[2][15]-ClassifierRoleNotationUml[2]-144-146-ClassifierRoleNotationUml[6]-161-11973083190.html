  56.    /**
  57.     * The notation for an Enumeration Literal. <p>
  58.     * 
  59.     * The supported notation is: <pre>
  60.     *     [ "<<" stereotype ">>" ] name [ ";" name ]*
  61.     * </pre>
  62.     * This means:<ul><li> 
  63.     * The name is not optional (i.e. leaving it blank 
  64.     * means deletion of the literal). <li>
  65.     * Multiple literals may be entered at once 
  66.     * by separating the names with a semicolon. <li> 
  67.     * Extra literals are inserted after the one being parsed. <li>
  68.     * A stereotype may precede the name of any literal. </ul><p>
  69.     * 
  70.     * As explained in issue 5000, the real implemented BNF is something like: 
  71.     * <pre>
  72.     *     ["<<" stereotype ["," stereotype]* ">>"] name [ ";" ["<<" stereotype ["," stereotype]* ">>"] ...
  73.     * </pre>
  74.     * 
  75.     * @author Michiel
  76.     */
  77.    public class EnumerationLiteralNotationUml extends EnumerationLiteralNotation {
  78.    
  79.        /**
  80.         * The constructor.
  81.         * 
  82.         * @param enumLiteral the UML element
  83.         */
  84.        public EnumerationLiteralNotationUml(Object enumLiteral) {
  85.            super(enumLiteral);
  86.        }
  87.    
  88.        @Override
  89.        public String getParsingHelp() {
  90.            return "parsing.help.fig-enumeration-literal";
  91.        }
  92.    
  93.        @Override
  94.        public void parse(Object modelElement, String text) {
  95.            try {
  96.                parseEnumerationLiteralFig(
  97.                        Model.getFacade().getEnumeration(modelElement),
  98.                        modelElement, text);
  99.            } catch (ParseException pe) {
 100.                String msg = "statusmsg.bar.error.parsing.enumeration-literal";
 101.                Object[] args = {
 102.                    pe.getLocalizedMessage(),
 103.                    Integer.valueOf(pe.getErrorOffset()),
 104.                };
 105.                ArgoEventPump.fireEvent(new ArgoHelpEvent(
 106.                        ArgoEventTypes.HELP_CHANGED, this,
 107.                    Translator.messageFormat(msg, args)));
 108.            }
 109.        }
 110.        
 111.        /**
 112.         * Parse a string representing one or more ";" separated 
 113.         * enumeration literals.
 114.         * 
 115.         * @param enumeration the enumeration that the literal belongs to
 116.         * @param literal the literal on which the editing will happen
 117.         * @param text the string to parse
 118.         * @throws ParseException for invalid input - so that the right 
 119.         * message may be shown to the user
 120.         */
 121.        protected void  parseEnumerationLiteralFig(
 122.                Object enumeration, Object literal, String text) 
 123.            throws ParseException {
 124.            
 125.            if (enumeration == null || literal == null) {
 126.                return;
 127.            }
 128.            Project project = ProjectManager.getManager().getCurrentProject();
 129.    
 130.            ParseException pex = null;
 131.            int start = 0;
 132.            int end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 133.            
 134.            if (end == -1) {
 135.                /* No text. We may remove the literal. */
 136.                project.moveToTrash(literal);
 137.                return;
 138.            }
 139.            String s = text.substring(start, end).trim();
 140.            if (s.length() == 0) {
 141.                /* No non-white chars in text? remove literal! */
 142.                project.moveToTrash(literal);
 143.                return;
 144.            }
 145.            parseEnumerationLiteral(s, literal);
 146.    
 147.            int i = Model.getFacade().getEnumerationLiterals(enumeration)
 148.                .indexOf(literal);
 149.            // check for more literals (';' separated):
 150.            start = end + 1;
 151.            end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 152.            while (end > start && end <= text.length()) {
 153.                s = text.substring(start, end).trim();
 154.                if (s.length() > 0) {
 155.                    // yes, there are more:
 156.                    Object newLiteral = 
 157.                        Model.getCoreFactory().createEnumerationLiteral();
 158.                    if (newLiteral != null) {
 159.                        try {
 160.                            if (i != -1) {
 161.                                Model.getCoreHelper().addLiteral(
 162.                                        enumeration, ++i, newLiteral);
 163.                            } else {
 164.                                Model.getCoreHelper().addLiteral(
 165.                                        enumeration, 0, newLiteral);
 166.                            }
 167.                            parseEnumerationLiteral(s, newLiteral);
 168.                        } catch (ParseException ex) {
 169.                            if (pex == null) {
 170.                                pex = ex;
 171.                            }
 172.                        }
 173.                    }
 174.                }
 175.                start = end + 1;
 176.                end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 177.            }
 178.            if (pex != null) {
 179.                throw pex;
 180.            }
 181.        }
 182.    
 183.        protected void parseEnumerationLiteral(String text, Object literal) 
 184.            throws ParseException {
 185.            text = text.trim();
 186.            if (text.length() == 0) {
 187.                return;
 188.            }
 189.            // strip any trailing semi-colons
 190.            if (text.charAt(text.length() - 1) == ';') {
 191.                text = text.substring(0, text.length() - 2);
 192.            }
 193.            MyTokenizer st;
 194.    
 195.            String name = null;
 196.            StringBuilder stereotype = null;
 197.            String token;
 198.    
 199.            try {
 200.                st = new MyTokenizer(text, "<<,\u00AB,\u00BB,>>");
 201.                while (st.hasMoreTokens()) {
 202.                    token = st.nextToken();
 203.    
 204.                    if ("<<".equals(token) || "\u00AB".equals(token)) {
 205.                        if (stereotype != null) {
 206.                            String msg = 
 207.                                "parsing.error.model-element-name.twin-stereotypes";
 208.                            throw new ParseException(Translator.localize(msg),
 209.                                    st.getTokenIndex());
 210.                        }
 211.    
 212.                        stereotype = new StringBuilder();
 213.                        while (true) {
 214.                            token = st.nextToken();
 215.                            if (">>".equals(token) || "\u00BB".equals(token)) {
 216.                                break;
 217.                            }
 218.                            stereotype.append(token);
 219.                        }
 220.                    } else {
 221.                        if (name != null) {
 222.                            String msg = 
 223.                                "parsing.error.model-element-name.twin-names";
 224.                            throw new ParseException(Translator.localize(msg), 
 225.                                    st.getTokenIndex());
 226.                        }
 227.    
 228.                        name = token;
 229.                    }
 230.                }
 231.            } catch (NoSuchElementException nsee) {
 232.                String msg = 
 233.                    "parsing.error.model-element-name.unexpected-name-element";
 234.                throw new ParseException(Translator.localize(msg),
 235.                        text.length());
 236.            } catch (ParseException pre) {
 237.                throw pre;
 238.            }
 239.    
 240.            if (name != null) {
 241.                name = name.trim();
 242.            }
 243.            if (name != null) {
 244.                Model.getCoreHelper().setName(literal, name);
 245.            }
 246.    
 247.            StereotypeUtility.dealWithStereotypes(literal, stereotype, false);
 248.    
 249.            return;
 250.        }
 251.    
 252.        private String toString(Object modelElement, boolean useGuillemets) {
 253.            String nameStr = "";
 254.            /* Heuristic algorithm: do not show stereotypes if there is no name. */
 255.            if (Model.getFacade().getName(modelElement) != null) {
 256.                nameStr = NotationUtilityUml.generateStereotype(modelElement, 
 257.                        useGuillemets);
 258.                if (nameStr.length() > 0) {
 259.                    nameStr += " ";
 260.                }
 261.                nameStr += Model.getFacade().getName(modelElement).trim();
 262.            }
 263.            return nameStr;
 264.        }
 265.    
 266.        @Override
 267.        public String toString(Object modelElement, NotationSettings settings) {
 268.            return toString(modelElement, settings.isUseGuillemets());
 269.        }
