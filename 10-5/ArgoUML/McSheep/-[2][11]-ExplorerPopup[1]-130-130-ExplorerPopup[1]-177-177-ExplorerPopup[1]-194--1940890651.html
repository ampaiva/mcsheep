 254.        /*
 255.         * @see org.tigris.gef.graph.MutableGraphModel#canAddEdge(java.lang.Object)
 256.         */
 257.        @Override
 258.        public boolean canAddEdge(Object edge) {
 259.            if (edge == null) {
 260.                return false;
 261.            }
 262.            if (containsEdge(edge)) {
 263.                return false;
 264.            }
 265.            Object sourceModelElement = null;
 266.            Object destModelElement = null;
 267.            if (Model.getFacade().isAAssociation(edge)) {
 268.                Collection conns = Model.getFacade().getConnections(edge);
 269.                if (conns.size() < 2) {
 270.                    LOG.error("Association rejected. Must have at least 2 ends");
 271.                    return false;
 272.                }
 273.                Iterator iter = conns.iterator();
 274.                Object associationEnd0 = iter.next();
 275.                Object associationEnd1 = iter.next();
 276.                if (associationEnd0 == null || associationEnd1 == null) {
 277.                    LOG.error("Association rejected. An end is null");
 278.                    return false;
 279.                }
 280.                sourceModelElement = Model.getFacade().getType(associationEnd0);
 281.                destModelElement = Model.getFacade().getType(associationEnd1);
 282.            } else if (Model.getFacade().isAAssociationEnd(edge)) {
 283.                sourceModelElement = Model.getFacade().getAssociation(edge);
 284.                destModelElement = Model.getFacade().getType(edge);
 285.                if (sourceModelElement == null || destModelElement == null) {
 286.                    LOG.error("Association end rejected. An end is null");
 287.                    return false;
 288.                }
 289.    
 290.                if (!containsEdge(sourceModelElement) 
 291.                        && !containsNode(sourceModelElement)) {
 292.                    LOG.error("Association end rejected. The source model element ("
 293.                            + sourceModelElement.getClass().getName()
 294.                            + ") must be on the diagram");
 295.                    return false;
 296.                }
 297.    
 298.                if (!containsNode(destModelElement)) {
 299.                    LOG.error("Association end rejected. "
 300.                            + "The destination model element must be "
 301.                            + "on the diagram.");
 302.                    return false;
 303.                }
 304.    
 305.            } else if (Model.getFacade().isAGeneralization(edge)) {
 306.                sourceModelElement = Model.getFacade().getSpecific(edge);
 307.                destModelElement = Model.getFacade().getGeneral(edge);
 308.            } else if (Model.getFacade().isADependency(edge)) {
 309.                Collection clients = Model.getFacade().getClients(edge);
 310.                Collection suppliers = Model.getFacade().getSuppliers(edge);
 311.                if (clients == null || suppliers == null) {
 312.                    return false;
 313.                }
 314.                sourceModelElement = clients.iterator().next();
 315.                destModelElement = suppliers.iterator().next();
 316.            } else if (Model.getFacade().isALink(edge)) {
 317.                Collection roles = Model.getFacade().getConnections(edge);
 318.                if (roles.size() < 2) {
 319.                    return false;
 320.                }
 321.                Iterator iter = roles.iterator();
 322.                Object linkEnd0 = iter.next();
 323.                Object linkEnd1 = iter.next();
 324.                if (linkEnd0 == null || linkEnd1 == null) {
 325.                    return false;
 326.                }
 327.                sourceModelElement = Model.getFacade().getInstance(linkEnd0);
 328.                destModelElement = Model.getFacade().getInstance(linkEnd1);
 329.            } else if (edge instanceof CommentEdge) {
 330.                sourceModelElement = ((CommentEdge) edge).getSource();
 331.                destModelElement = ((CommentEdge) edge).getDestination();
 332.            } else if (Model.getFacade().isADirectedRelationship(edge)) {
 333.                Collection sources = Model.getFacade().getSources(edge);
 334.                Collection targets = Model.getFacade().getTargets(edge);
 335.                if (sources.size() == 1 && targets.size() == 1) {
 336.                    sourceModelElement = sources.iterator().next();
 337.                    destModelElement = targets.iterator().next();
 338.                } else {
 339.                    LOG.error("Edge rejected. More than one source or target for a DirectedRelationship"...
 340.                    return false;
 341.                }
 342.            } else {
 343.                return false;
 344.            }
 345.    
 346.            if (sourceModelElement == null || destModelElement == null) {
 347.                LOG.error("Edge rejected. Its ends are not attached to anything");
 348.                return false;
 349.            }
 350.    
 351.            if (!containsNode(sourceModelElement)
 352.                    && !containsEdge(sourceModelElement)) {
 353.                LOG.error("Edge rejected. Its source end is attached to "
 354.                        + sourceModelElement
 355.                        + " but this is not in the graph model");
 356.                return false;
 357.            }
 358.            if (!containsNode(destModelElement)
 359.                    && !containsEdge(destModelElement)) {
 360.                LOG.error("Edge rejected. Its destination end is attached to "
 361.                        + destModelElement 
 362.                        + " but this is not in the graph model");
 363.                return false;
 364.            }
 365.    
 366.            return true;
 367.        }

 391.        /**
 392.         * Add the given edge to the graph, if of the correct type.
 393.         * Throws IllegalArgumentException if edge is null or either of its
 394.         * ends are null.
 395.         *
 396.         * @param edge the edge to be added
 397.         */
 398.        @Override
 399.        public void addEdge(Object edge) {
 400.            if (edge == null) {
 401.                throw new IllegalArgumentException("Cannot add a null edge");
 402.            }
 403.    
 404.            if (getDestPort(edge) == null || getSourcePort(edge) == null) {
 405.                throw new IllegalArgumentException(
 406.                        "The source and dest port should be provided on an edge");
 407.            }
 408.    
 409.            if (LOG.isInfoEnabled()) {
 410.                LOG.info("Adding an edge of type "
 411.                       + edge.getClass().getName()
 412.                       + " to class diagram.");
 413.            }

 623.    
 624.        if (Model.getFacade().isAInterface(newNode)

 626.            return;
 627.            }
 628.    
 629.            // cast the params
 630.        Object edgeAssoc = edge;
 631.

 633.        Object theOtherEnd = null;

 636.            if (isSource) {

 645.    
 646.            //set the new end type!
 647.            Model.getCoreHelper().setType(theEnd, newNode);
 648.        }
