 142.                // check that the end is a Classifier, rather than its child
 143.                // ClassifierRole, since we have effectively eliminated that
 144.                // possiblity in rejecting AssociationRoles above.
 145.    
 146.            Object type = Model.getFacade().getType(ae);
 147.    
 148.                if (Model.getFacade().isAInterface(type)) {
 149.                    haveInterfaceEnd = true;
 150.                } else if (Model.getFacade().isNavigable(ae)) {
 151.                    otherEndNavigable = true;
 152.                }
 153.    
 154.                // We can give up looking if we've hit both criteria
 155.    
 156.                if (haveInterfaceEnd && otherEndNavigable) {
 157.                    return PROBLEM_FOUND;
 158.                }
 159.            }
 160.    
 161.            // If we drop out we didn't meet both criteria, and all is well.
 162.    
 163.            return NO_PROBLEM;
 164.        }
