 422.            Collection cmnt = new ArrayList();
 423.            if (Model.getFacade().isAComment(node)) {
 424.                cmnt.addAll(Model.getFacade().getAnnotatedElements(node));
 425.            }
 426.            // TODO: Comments are on Element in UML 2.x
 427.            if (Model.getFacade().isAModelElement(node)) {
 428.                cmnt.addAll(Model.getFacade().getComments(node));
 429.            }
 430.            Iterator iter = cmnt.iterator();
 431.            while (iter.hasNext()) {
 432.                Object ae = iter.next();
 433.                CommentEdge ce = new CommentEdge(node, ae);
 434.                if (canAddEdge(ce)) {
 435.                    addEdge(ce);
 436.                }
 437.            }
 438.        }
 439.    
 440.        /**
 441.         * Create an edge of the given type and connect it to the
 442.         * given nodes.
 443.         *
 444.         * @param edgeType       the UML object type of the connection
 445.         * @param fromElement    the UML object for the "from" element
 446.         * @param fromStyle      the aggregationkind for the connection
 447.         *                       in case of an association
 448.         * @param toElement      the UML object for the "to" element
 449.         * @param toStyle        the aggregationkind for the connection
 450.         *                       in case of an association
 451.         * @param unidirectional for association and associationrole
 452.         * @param namespace      the namespace to use if it can't be determined
 453.         * @return               the newly build connection (UML object)
 454.         */
 455.        protected Object buildConnection(
 456.                Object edgeType,
 457.                Object fromElement,
 458.                Object fromStyle,
 459.                Object toElement,
 460.                Object toStyle,
 461.                Object unidirectional,
 462.                Object namespace) {
 463.    
 464.            Object connection = null;
 465.            if (edgeType == CommentEdge.class) {
 466.                connection =
 467.                    buildCommentConnection(fromElement, toElement);
 468.            } else {
 469.                try {
 470.                    connection =
 471.                        Model.getUmlFactory().buildConnection(
 472.                                edgeType,
 473.                                fromElement,
 474.                                fromStyle,
 475.                                toElement,
 476.                                toStyle,
 477.                                unidirectional,
 478.                                namespace);
 479.                LOG.info("Created " + connection + " between " 
 480.                        + fromElement + " and " + toElement);
 481.                } catch (UmlException ex) {
 482.                    // fail silently as we expect users to accidentally drop
 483.                    // on to wrong component
 484.                } catch (IllegalArgumentException iae) {
 485.                    // idem, e.g. for a generalization with leaf/root object
 486.                    // TODO: but showing the message in the statusbar would help
 487.                // TODO: IllegalArgumentException should not be used for
 488.                // events we expect to happen. We need a different way of
 489.                // catching well-formedness rules.
 490.                LOG.warn("IllegalArgumentException caught", iae);
 491.                }
 492.            }
 493.            return connection;
 494.        }
 495.    
 496.        /**
 497.         * Builds the model behind a connection between a comment and
 498.         * the annotated modelelement.
 499.         *
 500.         * @param from The comment or annotated element.
 501.         * @param to The comment or annotated element.
 502.         * @return A commentEdge representing the model behind the connection
 503.         *         between a comment and an annotated modelelement.
 504.         */
 505.        public CommentEdge buildCommentConnection(Object from, Object to) {
 506.            if (from == null || to == null) {
 507.                throw new IllegalArgumentException("Either fromNode == null "
 508.                                           + "or toNode == null");
 509.            }
 510.            Object comment = null;
 511.            Object annotatedElement = null;
 512.            if (Model.getFacade().isAComment(from)) {
 513.                comment = from;
 514.                annotatedElement = to;
 515.            } else {
 516.                if (Model.getFacade().isAComment(to)) {
 517.                    comment = to;
 518.                    annotatedElement = from;
 519.                } else {
 520.                    return null;
 521.                }
 522.            }
 523.    
 524.            CommentEdge connection = new CommentEdge(from, to);
 525.            Model.getCoreHelper().addAnnotatedElement(comment, annotatedElement);
 526.            return connection;
 527.    
 528.        }
 529.    
 530.        /**
 531.         * Checks if some type of edge is valid to connect two
 532.         * types of node.
 533.         *
 534.         * @param edgeType  the UML object type of the connection
 535.         * @param fromElement     the UML object type of the "from"
 536.         * @param toElement       the UML object type of the "to"
 537.         * @return true if valid
 538.         */
 539.        protected boolean isConnectionValid(
 540.                Object edgeType,
 541.                Object fromElement,
 542.                Object toElement) {
 543.    
 544.            if (!nodes.contains(fromElement) || !nodes.contains(toElement)) {
 545.                // The connection is not valid unless both nodes are
 546.                // in this graph model.
 547.                return false;
 548.            }
 549.    
 550.            if (edgeType.equals(CommentEdge.class)) {
 551.                return ((Model.getFacade().isAComment(fromElement)
 552.                       && Model.getFacade().isAModelElement(toElement))
 553.                     || (Model.getFacade().isAComment(toElement)
 554.                       && Model.getFacade().isAModelElement(fromElement)));
 555.            }
 556.            return Model.getUmlFactory().isConnectionValid(
 557.                    edgeType,
 558.                    fromElement,
 559.                    toElement,
 560.                    true);
 561.        }

 231.        /**
 232.         * Construct and add a new edge of the given kind and connect
 233.         * the given ports.
 234.         *
 235.         * @param fromPort   The originating port to connect
 236.         *
 237.         * @param toPort     The destination port to connect
 238.         *
 239.         * @param edgeType  The type of edge to create. This is one of the types
 240.         *                  returned by the methods of
 241.         *                  <code>org.argouml.model.MetaTypes</code>
 242.         *
 243.         * @return           The type of edge created (the same as
 244.         *                   <code>edgeClass</code> if we succeeded,
 245.         *                   <code>null</code> otherwise)
 246.         */
 247.        public Object connect(Object fromPort, Object toPort, Object edgeType) {
 248.            // If this was an association then there will be relevant
 249.            // information to fetch out of the mode arguments.  If it
 250.            // not an association then these will be passed forward
 251.            // harmlessly as null.
 252.            Editor curEditor = Globals.curEditor();
 253.            ModeManager modeManager = curEditor.getModeManager();
 254.            Mode mode = modeManager.top();
 255.            Dictionary args = mode.getArgs();
 256.            Object style = args.get("aggregation");
 257.            Boolean unidirectional = (Boolean) args.get("unidirectional");
 258.            Object model = getProject().getModel();
 259.    
 260.            // Create the UML connection of the given type between the
 261.            // given model elements.
 262.            // default aggregation (none)
 263.            Object connection =
 264.                buildConnection(
 265.                    edgeType, fromPort, style, toPort,
 266.                    null, unidirectional,
 267.                    model);
 268.    
 269.            if (connection == null) {
 270.                if (LOG.isDebugEnabled()) {
 271.                    LOG.debug("Cannot make a " + edgeType
 272.                            + " between a " + fromPort.getClass().getName()
 273.                            + " and a " + toPort.getClass().getName());
 274.                }
 275.                return null;
 276.            }
 277.    
 278.            addEdge(connection);
 279.            if (LOG.isDebugEnabled()) {

 407.                    new ArrayList(Model.getFacade().getClientDependencies(node));

 440.        /**
 441.         * Create an edge of the given type and connect it to the
 442.         * given nodes.
 443.         *
 444.         * @param edgeType       the UML object type of the connection
 445.         * @param fromElement    the UML object for the "from" element
 446.         * @param fromStyle      the aggregationkind for the connection
 447.         *                       in case of an association
 448.         * @param toElement      the UML object for the "to" element
 449.         * @param toStyle        the aggregationkind for the connection
 450.         *                       in case of an association
 451.         * @param unidirectional for association and associationrole
 452.         * @param namespace      the namespace to use if it can't be determined
 453.         * @return               the newly build connection (UML object)
 454.         */
 455.        protected Object buildConnection(
 456.                Object edgeType,
 457.                Object fromElement,
 458.                Object fromStyle,
 459.                Object toElement,
 460.                Object toStyle,
 461.                Object unidirectional,
 462.                Object namespace) {

 466.                connection =
 467.                    buildCommentConnection(fromElement, toElement);
 468.            } else {
 469.                try {
 470.                    connection =
 471.                        Model.getUmlFactory().buildConnection(
 472.                                edgeType,
 473.                                fromElement,
 474.                                fromStyle,
 475.                                toElement,
 476.                                toStyle,

 481.                } catch (UmlException ex) {
 482.                    // fail silently as we expect users to accidentally drop
 483.                    // on to wrong component
 484.                } catch (IllegalArgumentException iae) {
 485.                    // idem, e.g. for a generalization with leaf/root object

 489.                // catching well-formedness rules.

 494.        }

 534.         * @param edgeType  the UML object type of the connection
 535.         * @param fromElement     the UML object type of the "from"

 539.        protected boolean isConnectionValid(
 540.                Object edgeType,
 541.                Object fromElement,
 542.                Object toElement) {

 544.            if (!nodes.contains(fromElement) || !nodes.contains(toElement)) {

 554.                       && Model.getFacade().isAModelElement(fromElement)));
 555.            }
 556.            return Model.getUmlFactory().isConnectionValid(
 557.                    edgeType,

 559.                    toElement,
 560.                    true);
 561.        }
