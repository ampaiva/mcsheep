 185.        public String createTable(TableDefinition tableDefinition) {
 186.            StringBuffer sb = new StringBuffer();
 187.            sb.append("CREATE TABLE ");
 188.            sb.append(tableDefinition.getName()).append(" (");
 189.            sb.append(LINE_SEPARATOR);
 190.    
 191.            Iterator it = tableDefinition.getColumnDefinitions().iterator();
 192.            while (it.hasNext()) {
 193.                ColumnDefinition colDef = (ColumnDefinition) it.next();
 194.                sb.append("    ");
 195.                sb.append(colDef.getName()).append(" ");
 196.                sb.append(colDef.getDatatype());
 197.                Boolean nullable = colDef.getNullable();
 198.                if (nullable != null) {
 199.                    if (nullable.equals(Boolean.FALSE)) {
 200.                        sb.append(" ").append("NOT NULL");
 201.                    }
 202.                }
 203.                sb.append(",").append(LINE_SEPARATOR);
 204.            }
 205.    
 206.            StringBuffer sbPk = new StringBuffer();
 207.            it = tableDefinition.getPrimaryKeyFields().iterator();
 208.            while (it.hasNext()) {
 209.                String primaryKeyField = (String) it.next();
 210.                if (sbPk.length() > 0) {
 211.                    sbPk.append(", ");
 212.                }
 213.                sbPk.append(primaryKeyField);
 214.            }
 215.    
 216.            sb.append("CONSTRAINT PK").append(primaryKeyCounter);
 217.            sb.append(" PRIMARY KEY (").append(sbPk).append(")");
 218.            sb.append(LINE_SEPARATOR);
 219.    
 220.            sb.append(");").append(LINE_SEPARATOR).append(LINE_SEPARATOR);
 221.    
 222.            primaryKeyCounter++;
 223.    
 224.            return sb.toString();
 225.        }
