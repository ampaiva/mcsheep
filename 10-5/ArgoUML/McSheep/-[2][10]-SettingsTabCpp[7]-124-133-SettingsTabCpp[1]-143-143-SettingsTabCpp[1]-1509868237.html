  87.    /**
  88.     * This is the main class for the Swing importer framework.  It extends 
  89.     * ImportCommon which contains all the GUI independent pieces of the import
  90.     * framework.  
  91.     * <p>
  92.     * The Service Providers Interface (SPI) to the individual language importers
  93.     * is defined in such a way that they can be completely GUI independent as
  94.     * well, receiving lists of source files and settings for the import and 
  95.     * reporting progress via progress monitor API.
  96.     * 
  97.     * It provides JPanels for tailoring the import run in the FileChooser.
  98.     * <p>
  99.     * 
 100.     * The Import run is started by calling doFile(Project, File)
 101.     * <p>
 102.     * 
 103.     * Supports recursive search in folder for source files with matching 
 104.     * extensions.
 105.     * <p>
 106.     * 
 107.     * There are three levels of detail for import:
 108.     * <p>
 109.     * 
 110.     * <ol>
 111.     * <li> 0 - classifiers only
 112.     * <li> 1 - classifiers plus feature specifications
 113.     * <li> 2 - full import, feature detail (ie. operations with methods)
 114.     * </ol>
 115.     * 
 116.     * @author Andreas Rueckert a_rueckert@gmx.net
 117.     * @author Tom Morris <tfmorris@gmail.com>
 118.     */
 119.    public class Import extends ImportCommon implements ImportSettings {

 511.         * abstract class FileImportSupport used to provide).

 513.        private JComponent getChooser() {
 514.            String directory = Globals.getLastDirectory();
 515.    
 516.            final JFileChooser chooser = new ImportFileChooser(this, directory);

 520.    
 521.            if (current != null) {
 522.                updateFilters(chooser, null, current.getSuffixFilters());
 523.            }
 524.            return chooser;
 525.        }
 526.    
 527.        private static void updateFilters(JFileChooser chooser,
 528.                SuffixFilter[] oldFilters, SuffixFilter[] newFilters) {
 529.            if (oldFilters != null) {
 530.                for (int i = 0; i < oldFilters.length; i++) {
 531.                    chooser.removeChoosableFileFilter(oldFilters[i]);
 532.                }
 533.            }
 534.            if (newFilters != null) {
 535.                for (int i = 0; i < newFilters.length; i++) {
 536.                    chooser.addChoosableFileFilter(newFilters[i]);
 537.                }
 538.                if (newFilters.length > 0) {
 539.                    chooser.setFileFilter(newFilters[0]);
 540.                }
 541.            }
 542.        }

 602.            private void initChooser() {
 603.                setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
 604.                setMultiSelectionEnabled(true);
 605.                setSelectedFile(getCurrentDirectory());
 606.            }

 608.            /*
 609.             * @see javax.swing.JFileChooser#approveSelection()
 610.             */
 611.            @Override
 612.            public void approveSelection() {
 613.                File[] files = getSelectedFiles();
 614.                File dir = getCurrentDirectory();
 615.                if (files.length == 0) {
 616.                    files = new File[] {dir};
 617.                }
 618.                if (files.length == 1) {
 619.                    File file = files[0];
 620.                    if (file != null && file.isDirectory()) {
 621.                        dir = file;
 622.                    } else {
 623.                        dir = file.getParentFile();
 624.                    }
 625.                }
 626.                theImport.setSelectedFiles(files);
 627.                try {
 628.                    theImport.setSelectedSuffixFilter(
 629.                            (SuffixFilter) getFileFilter());
 630.                } catch (Exception e) {
 631.                    // this is because of the (senseless?) "All files" FileFilter
 632.                    theImport.setSelectedSuffixFilter(null);
 633.                }
 634.                Globals.setLastDirectory(dir.getPath());
 635.                theImport.disposeDialog();
 636.    
 637.                theImport.doFile();
 638.            }

 650.        /**
 651.         * Extended configuration panel for file import.  Built based on settings
 652.         * requested by the specific language importer.
 653.         */
 654.        class ConfigPanelExtension extends JPanel {
 655.    
 656.    
 657.            /**
 658.             * Construct the configuration extension panel.
 659.             * @param settings A list of settings requested by the language importer
 660.             */
 661.            public ConfigPanelExtension(final List<Setting> settings) {
 662.    
 663.                setLayout(new GridBagLayout());
 664.                
 665.                if (settings == null || settings.size() == 0) {
 666.                    JLabel label = new JLabel("No settings for this importer");
 667.                    add(label, createGridBagConstraints(true, false, false));
 668.                    add(new JPanel(), createGridBagConstraintsFinal()); 
 669.                    return;
 670.                }
 671.    
 672.                for (Setting setting : settings) {
 673.                    if (setting instanceof UniqueSelection2) {
 674.                        JLabel label = new JLabel(setting.getLabel());
 675.                        add(label, createGridBagConstraints(true, false, false));
 676.    
 677.                        final UniqueSelection2 us = (UniqueSelection2) setting;
 678.                        ButtonGroup group = new ButtonGroup();
 679.                        int count = 0;
 680.                        for (String option : us.getOptions()) {
 681.                            JRadioButton button = new JRadioButton(option);
 682.                            final int index = count++;
 683.                            if (us.getDefaultSelection() == index) {
 684.                                button.setSelected(true);
 685.                            }
 686.                            group.add(button);
 687.                            button.addActionListener(new ActionListener() {
 688.                                public void actionPerformed(ActionEvent e) {
 689.                                    us.setSelection(index);
 690.                                }
 691.                            });
 692.                            add(button, createGridBagConstraints(false, false,
 693.                                    false));
 694.                        }
 695.                    } else if (setting instanceof UserString2) {
 696.                        JLabel label = new JLabel(setting.getLabel());
 697.                        add(label, createGridBagConstraints(true, false, false));
 698.                        final UserString2 us = (UserString2) setting;
 699.                        final JTextField text = 
 700.                            new JTextField(us.getDefaultString());
 701.                        text.addFocusListener(new FocusListener() {
 702.                            public void focusGained(FocusEvent e) { } 
 703.                            public void focusLost(FocusEvent e) {
 704.                                us.setUserString(text.getText());           
 705.                            }
 706.                            
 707.                        });
 708.                        add(text, createGridBagConstraints(true, false, false));
 709.                    } else if (setting instanceof BooleanSelection2) {
 710.                        final BooleanSelection2 bs = (BooleanSelection2) setting;
 711.                        final JCheckBox button = new JCheckBox(setting.getLabel());
 712.                        button.setEnabled(bs.isSelected());
 713.                        button.addActionListener(new ActionListener() {
 714.                            public void actionPerformed(ActionEvent e) {
 715.                                bs.setSelected(button.isSelected());
 716.                            }
 717.                        });
 718.                        add(button, createGridBagConstraints(true, false, false));
 719.                    } else if (setting instanceof PathSelection) {
 720.                        JLabel label = new JLabel(setting.getLabel());
 721.                        add(label, createGridBagConstraints(true, false, false));
 722.                        PathSelection ps = (PathSelection) setting;
 723.                        // TODO: Need to add FileChooser 
 724.                        JTextField text = new JTextField(ps.getDefaultPath());
 725.                        add(text, createGridBagConstraints(true, false, false));
 726.                        // TODO: Update setting
 727.                    } else if (setting instanceof PathListSelection) {
 728.                        PathListSelection pls = (PathListSelection) setting;
 729.                        add(new ImportClasspathDialog(pls),
 730.                                createGridBagConstraints(true, false, false));
 731.                    } else {
 732.                        throw new RuntimeException("Unknown setting type requested "
 733.                                + setting);
 734.                    }
 735.                }
 736.                add(new JPanel(), createGridBagConstraintsFinal()); 
 737.            }
 738.    
 739.    
 740.            /**
 741.             * Create a GridBagConstraints object to use with the layout.
 742.             * 
 743.             * @param topInset true to use a top inset 
 744.             * @param bottomInset true to use a bottom inset
 745.             * @param fill true to fill (horizontally)
 746.             * @return the grid bag constraints
 747.             */
 748.            private GridBagConstraints createGridBagConstraints(boolean topInset,
 749.                    boolean bottomInset, boolean fill) {
 750.                GridBagConstraints gbc = new GridBagConstraints();
 751.                gbc.gridx = GridBagConstraints.RELATIVE;
 752.                gbc.gridy = GridBagConstraints.RELATIVE;
 753.                gbc.gridwidth = GridBagConstraints.REMAINDER;
 754.                gbc.gridheight = 1;
 755.                gbc.weightx = 1.0;
 756.                gbc.weighty = 0.0;
 757.                gbc.anchor = GridBagConstraints.NORTHWEST;
 758.                gbc.fill = fill ? GridBagConstraints.HORIZONTAL
 759.                        : GridBagConstraints.NONE;
 760.                gbc.insets = 
 761.                    new Insets(
 762.                            topInset ? 5 : 0, 
 763.                                    5, 
 764.                                    bottomInset ? 5 : 0, 
 765.                                            5);
 766.                gbc.ipadx = 0;
 767.                gbc.ipady = 0;
 768.                return gbc;
 769.            }
 770.    
 771.            /**
 772.             * A GridBagConstraints for the last item to take up the rest of the
 773.             * space.
 774.             * 
 775.             * @return the GridBagConstraints object
 776.             */
 777.            private GridBagConstraints createGridBagConstraintsFinal() {
 778.                GridBagConstraints gbc = createGridBagConstraints(false, true,
 779.                        false);
 780.                gbc.gridheight = GridBagConstraints.REMAINDER;
 781.                gbc.weighty = 1.0;
 782.                return gbc;
 783.            }
 784.    
 785.        }
