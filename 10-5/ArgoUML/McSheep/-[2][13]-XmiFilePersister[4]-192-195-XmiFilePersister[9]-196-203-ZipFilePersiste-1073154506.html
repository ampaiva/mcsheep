 319.        /**

 326.            
 327.            if (argoString != null) {

 330.                ArgoParser parser = new ArgoParser();
 331.                try {

 338.            }

 341.                InputStream inputStream = new ByteArrayInputStream(pgml.getBytes());
 342.                MemberFilePersister persister = 
 343.                // TODO: Cyclic dependency between PersistanceManager and here
 344.                    PersistenceManager.getInstance()
 345.                            .getDiagramMemberFilePersister();
 346.                // possibly use the following instead
 347.    //            MemberFilePersister persister = new DiagramMemberFilePersister();
 348.                persister.load(project, inputStream);
 349.            }
 350.            if (todoString != null) {
 351.                LOG.info("Parsing todoString " + todoString.length());
 352.                InputStream inputStream = 
 353.                    new ByteArrayInputStream(todoString.getBytes());
 354.                MemberFilePersister persister = null;
 355.                persister = new TodoListMemberFilePersister();
 356.                persister.load(project, inputStream);
 357.            } else {
 358.                project.addMember(new ProjectMemberTodoList("", project));
 359.            }
 360.        }

  62.    /**
  63.     * To persist to and from zipped xmi file storage.
  64.     *
  65.     * @author Bob Tarling
  66.     * @author Ludovic Maître
  67.     */
  68.    class ZipFilePersister extends XmiFilePersister {
  69.        /**
  70.         * Logger.
  71.         */
  72.        private static final Logger LOG =
  73.            Logger.getLogger(ZipFilePersister.class);
  74.    
  75.        /**
  76.         * The constructor.
  77.         */
  78.        public ZipFilePersister() {
  79.        }
  80.    
  81.        /*
  82.         * @see org.argouml.persistence.AbstractFilePersister#getExtension()
  83.         */

  85.            return "zip";
  86.        }
  87.    
  88.        /*
  89.         * @see org.argouml.persistence.AbstractFilePersister#getDesc()
  90.         */
  91.        protected String getDesc() {
  92.            return Translator.localize("combobox.filefilter.zip");
  93.        }
  94.        
  95.        /*
  96.         * @see org.argouml.persistence.XmiFilePersister#isSaveEnabled()
  97.         */
  98.        public boolean isSaveEnabled() {
  99.            return true;
 100.        }
 101.    
 102.        /**
 103.         * It is being considered to save out individual xmi's from individuals
 104.         * diagrams to make it easier to modularize the output of Argo.
 105.         *
 106.         * @param file
 107.         *            The file to write.
 108.         * @param project
 109.         *            the project to save
 110.         * @throws SaveException
 111.         *             when anything goes wrong
 112.         *
 113.         * @see org.argouml.persistence.ProjectFilePersister#save(
 114.         *      org.argouml.kernel.Project, java.io.File)
 115.         */
 116.        public void doSave(Project project, File file) throws SaveException {
 117.    
 118.            LOG.info("Receiving file '" + file.getName() + "'");
 119.    
 120.            File lastArchiveFile = new File(file.getAbsolutePath() + "~");
 121.            File tempFile = null;
 122.    
 123.            try {
 124.                tempFile = createTempFile(file);
 125.            } catch (FileNotFoundException e) {
 126.                throw new SaveException(
 127.                        "Failed to archive the previous file version", e);
 128.            } catch (IOException e) {
 129.                throw new SaveException(
 130.                        "Failed to archive the previous file version", e);
 131.            }
 132.    
 133.            OutputStream bufferedStream = null;
 134.            try {
 135.                //project.setFile(file);
 136.    
 137.                ZipOutputStream stream =
 138.                    new ZipOutputStream(new FileOutputStream(file));
 139.                String fileName = file.getName();
 140.                ZipEntry xmiEntry =
 141.                    new ZipEntry(fileName.substring(0, fileName.lastIndexOf(".")));
 142.                stream.putNextEntry(xmiEntry);
 143.                bufferedStream = new BufferedOutputStream(stream);
 144.    
 145.                int size = project.getMembers().size();
 146.                for (int i = 0; i < size; i++) {
 147.                    ProjectMember projectMember =
 148.                        project.getMembers().get(i);
 149.                    if (projectMember.getType().equalsIgnoreCase("xmi")) {
 150.                        if (LOG.isInfoEnabled()) {
 151.                            LOG.info("Saving member of type: "
 152.                                  + (project.getMembers()
 153.                                        .get(i)).getType());
 154.                        }
 155.                        MemberFilePersister persister
 156.                            = new ModelMemberFilePersister();
 157.                        persister.save(projectMember, bufferedStream);
 158.                    }
 159.                }
 160.                stream.close();
 161.                // if save did not raise an exception
 162.                // and name+"#" exists move name+"#" to name+"~"
 163.                // this is the correct backup file
 164.                if (lastArchiveFile.exists()) {
 165.                    lastArchiveFile.delete();
 166.                }
 167.                if (tempFile.exists() && !lastArchiveFile.exists()) {
 168.                    tempFile.renameTo(lastArchiveFile);
 169.                }
 170.                if (tempFile.exists()) {
 171.                    tempFile.delete();
 172.                }
 173.            } catch (Exception e) {
 174.                LOG.error("Exception occured during save attempt", e);
 175.                try {
 176.                    bufferedStream.close();
 177.                } catch (IOException ex) {
 178.                    // If we get a 2nd error, just ignore it
 179.                }
 180.    
 181.                // frank: in case of exception
 182.                // delete name and mv name+"#" back to name if name+"#" exists
 183.                // this is the "rollback" to old file
 184.                file.delete();
 185.                tempFile.renameTo(file);
 186.                // we have to give a message to user and set the system to unsaved!
 187.                throw new SaveException(e);
 188.            }
 189.    
 190.            try {
 191.                bufferedStream.close();
 192.            } catch (IOException ex) {
 193.                LOG.error("Failed to close save output writer", ex);
 194.            }
 195.        }
 196.    
 197.        /*
 198.         * @see org.argouml.persistence.ProjectFilePersister#doLoad(java.io.File)
 199.         */
 200.        public Project doLoad(File file)
 201.            throws OpenException {
 202.    
 203.            LOG.info("Receiving file '" + file.getName() + "'");
 204.    
 205.            try {
 206.                Project p = ProjectFactory.getInstance().createProject();
 207.                String fileName = file.getName();
 208.                String extension =
 209.                    fileName.substring(
 210.                            fileName.indexOf('.'),
 211.                            fileName.lastIndexOf('.'));
 212.                InputStream stream = openZipStreamAt(file.toURI().toURL(),
 213.                        extension);
 214.    
 215.                // TODO: What progressMgr is to be used here? Where does
 216.                //       it come from?
 217.                InputSource is =
 218.                    new InputSource(
 219.                        new XmiInputStream(stream, this, 100000, null));
 220.                is.setSystemId(file.toURI().toURL().toExternalForm());
 221.    
 222.                ModelMemberFilePersister modelPersister =
 223.                    new ModelMemberFilePersister();
 224.                
 225.                modelPersister.readModels(is);
 226.                // TODO Handle multiple top level packages
 227.                Object model = modelPersister.getCurModel();
 228.                Model.getUmlHelper().addListenersToModel(model);
 229.                p.setUUIDRefs(modelPersister.getUUIDRefs());
 230.                p.addMember(model);
 231.                parseXmiExtensions(p);
 232.                modelPersister.registerDiagrams(p);
 233.    
 234.                p.setRoot(model);
 235.                p.setRoots(modelPersister.getElementsRead());
 236.                ProjectManager.getManager().setSaveEnabled(false);
 237.                return p;
 238.            } catch (IOException e) {
 239.                throw new OpenException(e);
 240.            }
 241.    
 242.        }
 243.    
 244.        /**
 245.         * Open a ZipInputStream to the first file found with a given extension.
 246.         *
 247.         * @param url
 248.         *            The URL of the zip file.
 249.         * @param ext
 250.         *            The required extension.
 251.         * @return the zip stream positioned at the required location.
 252.         * @throws IOException
 253.         *             if there is a problem opening the file.
 254.         */
 255.        private ZipInputStream openZipStreamAt(URL url, String ext)
 256.            throws IOException {
 257.            ZipInputStream zis = new ZipInputStream(url.openStream());
 258.            ZipEntry entry = zis.getNextEntry();
 259.            while (entry != null && !entry.getName().endsWith(ext)) {
 260.                entry = zis.getNextEntry();
 261.            }
 262.            return zis;
 263.        }

 271.        public boolean hasAnIcon() {
 272.            return false;
