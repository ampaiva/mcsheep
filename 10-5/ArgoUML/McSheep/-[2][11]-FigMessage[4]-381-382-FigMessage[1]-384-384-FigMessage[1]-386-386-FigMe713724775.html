 484.         * Determines the direction of the message arrow. Determination of the type
 485.         * of arrow happens in modelChanged.
 486.         */
 487.        public void updateArrow() {
 488.          Object mes = getOwner(); // Message
 489.        if (mes == null || getLayer() == null) {
 490.            return;
 491.        }
 492.        Object sender = Model.getFacade().getSender(mes); // ClassifierRole
 493.        Object receiver = Model.getFacade().getReceiver(mes); // ClassifierRole
 494.        Fig senderPort = getLayer().presentationFor(sender);
 495.        Fig receiverPort = getLayer().presentationFor(receiver);
 496.        if (senderPort == null || receiverPort == null) {
 497.            return;
 498.        }
 499.        int sx = senderPort.getX();
 500.        int sy = senderPort.getY();
 501.        int rx = receiverPort.getX();
 502.        int ry = receiverPort.getY();
 503.        if (sx < rx && Math.abs(sy - ry) <= Math.abs(sx - rx)) { // east
 504.            setArrow(EAST);
 505.        } else if (sx > rx && Math.abs(sy - ry) <= Math.abs(sx - rx)) { // west
 506.            setArrow(WEST);
 507.        } else if (sy < ry) { // south
 508.            setArrow(SOUTH);
 509.        } else {
 510.            setArrow(NORTH);
 511.        }
 512.        }

 483.        /**
 484.         * Determines the direction of the message arrow. Determination of the type
 485.         * of arrow happens in modelChanged.
 486.         */
 487.        public void updateArrow() {
 488.          Object mes = getOwner(); // Message
 489.        if (mes == null || getLayer() == null) {
 490.            return;
 491.        }
 492.        Object sender = Model.getFacade().getSender(mes); // ClassifierRole
 493.        Object receiver = Model.getFacade().getReceiver(mes); // ClassifierRole
 494.        Fig senderPort = getLayer().presentationFor(sender);
 495.        Fig receiverPort = getLayer().presentationFor(receiver);
 496.        if (senderPort == null || receiverPort == null) {
 497.            return;
 498.        }
 499.        int sx = senderPort.getX();
 500.        int sy = senderPort.getY();
 501.        int rx = receiverPort.getX();
 502.        int ry = receiverPort.getY();
 503.        if (sx < rx && Math.abs(sy - ry) <= Math.abs(sx - rx)) { // east
 504.            setArrow(EAST);
 505.        } else if (sx > rx && Math.abs(sy - ry) <= Math.abs(sx - rx)) { // west
 506.            setArrow(WEST);
 507.        } else if (sy < ry) { // south
 508.            setArrow(SOUTH);
 509.        } else {
 510.            setArrow(NORTH);
 511.        }
 512.        }
