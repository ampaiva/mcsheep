 277.         *
 278.         * @param sb the StringBuffer to append to.
 279.         * @param prefix the prefix to each line.

 282.         * @return the number of pending empty lines.
 283.         */
 284.        private static int appendComment(StringBuffer sb, String prefix,
 285.                          String comment, int nlprefix) {
 286.        int nlcount = 0;
 287.    
 288.        for (; nlprefix > 0; nlprefix--) {
 289.            if (prefix != null) {
 290.                sb.append(prefix);
 291.            }
 292.            sb.append(LINE_SEPARATOR);
 293.            nlcount++;
 294.        }

 297.            return nlcount;
 298.        }

 301.                             "",
 302.                             MyTokenizer.LINE_SEPARATOR);

 307.            if (prefix != null) {
 308.                sb.append(prefix);
 309.                    }

 311.            sb.append(LINE_SEPARATOR);

 314.            if (prefix != null) {
 315.                sb.append(prefix);
 316.                    }
 317.            sb.append(LINE_SEPARATOR);
 318.            nlcount++;
 319.            } else {
 320.            nlcount++;
 321.            }
 322.        }

 324.        return nlcount;

  68.         */
  69.        public static String getDocs(Object o, String indent) {
  70.            return getDocs(o, indent, "/** ", " *  ", " */");
  71.        }
  72.    
  73.        /**
  74.         * @param o the ModelElement
  75.         * @param indent the current indentation for new lines
  76.         * @param header is the first line
  77.         * @param prefix is inserted at every line before the doc
  78.         * @param footer is the closing line
  79.         * @return the string that represents the documentation
  80.         *         for the given ModelElement
  81.         */
  82.        public static String getDocs(Object o, String indent, String header,
  83.                     String prefix, String footer) {
  84.            String sResult = defaultFor(o, indent);
  85.    
  86.            if (Model.getFacade().isAModelElement(o)) {
  87.                Iterator iter = Model.getFacade().getTaggedValues(o);
  88.                if (iter != null) {
  89.                    while (iter.hasNext()) {
  90.                        Object tv = iter.next();
  91.                        String tag = Model.getFacade().getTagOfTag(tv);
  92.                        if (Argo.DOCUMENTATION_TAG.equals(tag) 
  93.                                || Argo.DOCUMENTATION_TAG_ALT.equals(tag)) {
  94.                            sResult = Model.getFacade().getValueOfTag(tv);
  95.                            // give priority to standard documentation tag
  96.                            if (Argo.DOCUMENTATION_TAG.equals(tag)) {
  97.                                break;
  98.                            }
  99.                        }
 100.                    }
 101.                }
 102.            }
 103.    
 104.            if (sResult == null) {
 105.                return "(No comment)";
 106.            }
 107.    
 108.        StringBuffer result = new StringBuffer();
 109.        if (header != null) {
 110.            result.append(header).append(LINE_SEPARATOR);
 111.        }
 112.    
 113.        if (indent != null) {
 114.            if (prefix != null) {
 115.            prefix = indent + prefix;
 116.            }
 117.    
 118.            if (footer != null) {
 119.            footer = indent + footer;
 120.            }
 121.        }
 122.    
 123.        appendComment(result, prefix, sResult, 0);
 124.    
 125.        if (footer != null) {
 126.            result.append(footer);
 127.        }
 128.    
 129.            return result.toString();
 130.        }

 133.         * @param o the ModelElement. If it is not a ModelElement,
 134.         *          then you'll get a IllegalArgumentException
 135.         * @param s the string representing the documentation
 136.         */
 137.        public static void setDocs(Object o, String s) {
 138.            Object taggedValue =
 139.                    Model.getFacade().getTaggedValue(o, Argo.DOCUMENTATION_TAG);
 140.            if (taggedValue == null) {
 141.                taggedValue =
 142.                        Model.getExtensionMechanismsFactory().buildTaggedValue(
 143.                                Argo.DOCUMENTATION_TAG, s);
 144.                Model.getExtensionMechanismsHelper().addTaggedValue(o, taggedValue);
 145.            } else {
 146.                Model.getExtensionMechanismsHelper().setValueOfTag(taggedValue, s);
 147.            }
 148.        }

 150.        /**
 151.         * Determine whether documentation is associated with the given
 152.         * element or not.
 153.         *
 154.         * Added 2001-10-05 STEFFEN ZSCHALER for use by
 155.         * org.argouml.language.java.generator.CodeGenerator
 156.         *
 157.         * @param o The given element.
 158.         * @return true if the given element has docs.
 159.         */
 160.        public static boolean hasDocs(Object o) {
 161.            if (Model.getFacade().isAModelElement(o)) {
 162.                Iterator i = Model.getFacade().getTaggedValues(o);
 163.    
 164.                if (i != null) {
 165.                    while (i.hasNext()) {
 166.                        Object tv = i.next();
 167.                        String tag = Model.getFacade().getTagOfTag(tv);
 168.                        String value = Model.getFacade().getValueOfTag(tv);
 169.                        if ((Argo.DOCUMENTATION_TAG.equals(tag) 
 170.                                || Argo.DOCUMENTATION_TAG_ALT.equals(tag))
 171.                            && value != null && value.trim().length() > 0) {
 172.                            return true;
 173.                        }
 174.                    }
 175.                }
 176.            }
 177.            return false;
 178.        }

 180.        /**
 181.         * Generate default documentation.
 182.         *
 183.         * @param o the ModelElement
 184.         * @param indent the current indentation string for new lines
 185.         * @return the default documentation
 186.         */
 187.        public static String defaultFor(Object o, String indent) {
 188.        if (Model.getFacade().isAClass(o)) {
 189.                // TODO: Needs localization
 190.            return " A class that represents ...\n\n"
 191.            + indent + " @see OtherClasses\n"
 192.            + indent + " @author your_name_here";
 193.        }
 194.        if (Model.getFacade().isAAttribute(o)) {
 195.    
 196.            return " An attribute that represents ...";
 197.        }
 198.    
 199.        if (Model.getFacade().isAOperation(o)) {
 200.            return " An operation that does...\n\n"
 201.            + indent + " @param firstParam a description of this parameter";
 202.        }
 203.        if (Model.getFacade().isAInterface(o)) {
 204.            return " An interface defining operations expected of ...\n\n"
 205.            + indent + " @see OtherClasses\n"
 206.            + indent + " @author your_name_here";
 207.        }
 208.        if (Model.getFacade().isAModelElement(o)) {
 209.            return "\n";
 210.        }
 211.    
 212.        return null;
 213.        }

 231.        /**
 232.         * Get the comments (the notes in a diagram) for a modelelement.
 233.         *
 234.         * @return a string with the comments.
 235.         * @param o The given modelelement.
 236.         * @param header is the comment header.
 237.         * @param prefix is the comment prefix (on every line).
 238.         * @param footer is the comment footer.
 239.         */
 240.        public static String getComments(Object o,
 241.                         String header, String prefix,
 242.                         String footer) {
 243.        StringBuffer result = new StringBuffer();
 244.        if (header != null) {
 245.            result.append(header).append(LINE_SEPARATOR);
 246.        }
 247.    
 248.        if (Model.getFacade().isAUMLElement(o)) {
 249.            Collection comments = Model.getFacade().getComments(o);
 250.            if (!comments.isEmpty()) {
 251.            int nlcount = 2;
 252.            for (Iterator iter = comments.iterator(); iter.hasNext();) {
 253.                Object c = iter.next();
 254.                String s = Model.getFacade().getName(c);
 255.                nlcount = appendComment(result,
 256.                            prefix,
 257.                            s,
 258.                            nlcount > 1 ? 0 : 1);
 259.            }
 260.            } else {
 261.            return "";
 262.            }
 263.        } else {
 264.            return "";
 265.        }
 266.    
 267.        if (footer != null) {
 268.            result.append(footer).append(LINE_SEPARATOR);
 269.        }
 270.    
 271.        return result.toString();
 272.        }
 273.    
 274.        /**
 275.         * Append a string to sb which is chopped into lines and each line
 276.         * prefixed with prefix.
 277.         *
 278.         * @param sb the StringBuffer to append to.
 279.         * @param prefix the prefix to each line.
 280.         * @param comment the text to reformat.
 281.         * @param nlprefix the number of empty lines to prefix the comment with.
 282.         * @return the number of pending empty lines.
 283.         */
 284.        private static int appendComment(StringBuffer sb, String prefix,
 285.                          String comment, int nlprefix) {
 286.        int nlcount = 0;
 287.    
 288.        for (; nlprefix > 0; nlprefix--) {
 289.            if (prefix != null) {
 290.                sb.append(prefix);
 291.            }
 292.            sb.append(LINE_SEPARATOR);
 293.            nlcount++;
 294.        }
 295.    
 296.        if (comment == null) {
 297.            return nlcount;
 298.        }
 299.    
 300.        MyTokenizer tokens = new MyTokenizer(comment,
 301.                             "",
 302.                             MyTokenizer.LINE_SEPARATOR);
 303.    
 304.        while (tokens.hasMoreTokens()) {
 305.            String s = tokens.nextToken();
 306.            if (!s.startsWith("\r") && !s.startsWith("\n")) {
 307.            if (prefix != null) {
 308.                sb.append(prefix);
 309.                    }
 310.            sb.append(s);
 311.            sb.append(LINE_SEPARATOR);
 312.            nlcount = 0;
 313.            } else if (nlcount > 0) {
 314.            if (prefix != null) {
 315.                sb.append(prefix);
 316.                    }
 317.            sb.append(LINE_SEPARATOR);
 318.            nlcount++;
 319.            } else {
 320.            nlcount++;
 321.            }
 322.        }
 323.    
 324.        return nlcount;
 325.        }
