 127.         * @see tudresden.ocl.check.types.Type#navigateQualified(
 128.         *         java.lang.String, tudresden.ocl.check.types.Type[])
 129.         */
 130.        public Type navigateQualified(String name, Type[] qualifiers)
 131.        throws OclTypeException {
 132.    
 133.        if (classifier == null) {
 134.            throw new OclTypeException("attempting to access features of Void");
 135.        }
 136.    
 137.    
 138.        if (qualifiers != null) {
 139.            throw new OclTypeException("qualified associations "
 140.                           + "not supported yet!");
 141.        }
 142.    
 143.        Type type = Basic.navigateAnyQualified(name, this, qualifiers);
 144.        if (type != null)  {
 145.            return type;
 146.        }
 147.    
 148.        Object foundAssocType = null, foundAttribType = null; // MClassifiers
 149.        boolean isSet = false, isSequence = false; // cannot be Bag
 150.    
 151.        // first search for appropriate attributes
 152.        Collection attributes =
 153.            Model.getCoreHelper().getAttributesInh(classifier);
 154.        Iterator iter = attributes.iterator();
 155.        while (iter.hasNext() && foundAttribType == null) {
 156.            Object attr = iter.next();
 157.            if (Model.getFacade().getName(attr).equals(name)) {
 158.            foundAttribType = Model.getFacade().getType(attr);
 159.            }
 160.        }
 161.    
 162.        // look for associations
 163.        Collection associationEnds =
 164.            Model.getCoreHelper().getAssociateEndsInh(classifier);
 165.        Iterator asciter = associationEnds.iterator();
 166.        while (asciter.hasNext() && foundAssocType == null) {
 167.            Object ae = asciter.next(); //MAssociationEnd
 168.            if (Model.getFacade().getName(ae) != null
 169.            && name.equals(Model.getFacade().getName(ae))) {
 170.    
 171.            foundAssocType = Model.getFacade().getType(ae);
 172.            } else if (Model.getFacade().getName(ae) == null
 173.                   || Model.getFacade().getName(ae).equals("")) {
 174.    
 175.            String oppositeName =
 176.                Model.getFacade().getName(Model.getFacade().getType(ae));
 177.            if (oppositeName != null) {
 178.    
 179.                String lowerOppositeName =
 180.                oppositeName.substring(0, 1).toLowerCase();
 181.                lowerOppositeName += oppositeName.substring(1);
 182.                if (lowerOppositeName.equals(name)) {
 183.                    foundAssocType = Model.getFacade().getType(ae);
 184.                }
 185.            }
 186.            }
 187.            if (foundAssocType != null) {
 188.            Object multiplicity = Model.getFacade().getMultiplicity(ae);
 189.            if (multiplicity != null
 190.                && (Model.getFacade().getUpper(multiplicity) > 1
 191.                || Model.getFacade().getUpper(multiplicity)
 192.                               == -1)) {
 193.                if (Model.getExtensionMechanismsHelper().hasStereotype(ae, 
 194.                        "ordered")) {
 195.                            isSequence = true;
 196.                        } else {
 197.                            isSet = true;
 198.                        }
 199.            }
 200.            }
 201.        }
 202.    
 203.        if (foundAssocType != null && foundAttribType != null) {
 204.            throw new OclTypeException("cannot access feature " + name
 205.                           + " of classifier " + toString()
 206.                           + " because both an attribute and "
 207.                           + "an association end of this name "
 208.                           + "where found");
 209.        }
 210.    
 211.        Object foundType;
 212.        if (foundAssocType == null) {
 213.            foundType = foundAttribType;
 214.        } else {
 215.            foundType = foundAssocType;
 216.        }
 217.    
 218.        if (foundType == null) {
 219.            throw new OclTypeException("attribute " + name
 220.                           + " not found in classifier "
 221.                           + toString());
 222.        }
 223.    
 224.        Type result = getOclRepresentation(foundType);
 225.    
 226.        if (isSet) {
 227.            result =
 228.            new tudresden.ocl.check.types.Collection(
 229.                    tudresden.ocl.check.types.Collection.SET,
 230.                result);
 231.        }
 232.        if (isSequence) {
 233.            result =
 234.            new tudresden.ocl.check.types.Collection(
 235.                tudresden.ocl.check.types.Collection.SEQUENCE,
 236.                result);
 237.        }
 238.    
 239.        return result;
 240.        }

 243.         * @see tudresden.ocl.check.types.Type2#navigateParameterizedQuery(
 244.         *         java.lang.String, tudresden.ocl.check.types.Type[])

 247.        throws OclTypeException {
 248.        return internalNavigateParameterized(name, qualifiers, true);
 249.        }

 252.         * @see tudresden.ocl.check.types.Type#navigateParameterized(

 255.        public Type navigateParameterized (String name, Type[] qualifiers)
 256.        throws OclTypeException {
 257.        return internalNavigateParameterized(name, qualifiers, false);
 258.        }

 260.        private Type internalNavigateParameterized(final String name,
 261.                                                  final Type[] params,
 262.                                                  boolean fCheckIsQuery)
 263.        throws OclTypeException {
 264.    
 265.        if (classifier == null) {
 266.            throw new OclTypeException("attempting to access features of Void");
 267.        }
 268.    
 269.        Type type = Basic.navigateAnyParameterized(name, params);
 270.        if (type != null) {
 271.            return type;
 272.        }
 273.    
 274.        Object foundOp = null; //MOperation
 275.        java.util.Collection operations =
 276.                    Model.getFacade().getOperations(classifier);
 277.        Iterator iter = operations.iterator();
 278.        while (iter.hasNext() && foundOp == null) {
 279.            Object op = iter.next();
 280.            if (operationMatchesCall(op, name, params)) {
 281.            foundOp = op;
 282.            }
 283.        }
 284.    
 285.        if (foundOp == null) {
 286.            throw new OclTypeException("operation " + name
 287.                           + " not found in classifier "
 288.                           + toString());
 289.        }
 290.    
 291.        if (fCheckIsQuery) {
 292.            /* Query checking added 05/21/01, sz9 */
 293.            if (!Model.getFacade().isQuery(foundOp)) {
 294.            throw new OclTypeException("Non-query operations cannot "
 295.                           + "be used in OCL expressions. ("
 296.                           + name + ")");
 297.            }
 298.        }
 299.    
 300.            Collection returnParams = 
 301.                Model.getCoreHelper().getReturnParameters(foundOp);
 302.            Object rp;
 303.            if (returnParams.size() == 0) {
 304.                rp = null;
 305.            } else {
 306.                rp = returnParams.iterator().next();
 307.            } 
 308.            if (returnParams.size() > 1)  {
 309.                LOG.warn("OCL compiler only handles one return parameter"
 310.                        + " - Found " + returnParams.size()
 311.                        + " for " + Model.getFacade().getName(foundOp));
 312.            }
 313.    
 314.        if (rp == null || Model.getFacade().getType(rp) == null) {
 315.            LOG.warn("WARNING: supposing return type void!");
 316.            return new ArgoAny(null);
 317.        }
 318.        Object returnType = Model.getFacade().getType(rp);
 319.    
 320.        return getOclRepresentation(returnType);
 321.        }
 322.    
 323.        /*
 324.         * @see tudresden.ocl.check.types.Type#conformsTo(tudresden.ocl.check.types.Type)
 325.         */
 326.        public boolean conformsTo(Type type) {
 327.        if (type instanceof ArgoAny) {
 328.            ArgoAny other = (ArgoAny) type;
 329.            return equals(type)
 330.            || Model.getCoreHelper()
 331.                .getAllSupertypes(classifier).contains(other.classifier);
 332.        }
 333.            return false;
 334.        }

 346.        }

 377.        protected Type getOclRepresentation(Object foundType) {
 378.        Type result = null;
 379.    
 380.        if (Model.getFacade().getName(foundType).equals("int")
 381.            || Model.getFacade().getName(foundType).equals("Integer")) {
 382.            result = Basic.INTEGER;
 383.        }
 384.    
 385.        if (Model.getFacade().getName(foundType).equals("float")
 386.            || Model.getFacade().getName(foundType).equals("double")) {
 387.            result = Basic.REAL;
 388.        }
 389.    
 390.        if (Model.getFacade().getName(foundType).equals("bool")
 391.            || Model.getFacade().getName(foundType).equals("Boolean")
 392.            || Model.getFacade().getName(foundType).equals("boolean")) {
 393.            result = Basic.BOOLEAN;
 394.        }
 395.    
 396.        if (Model.getFacade().getName(foundType).equals("String")) {
 397.            result = Basic.STRING;
 398.        }
 399.    
 400.        if (result == null) {
 401.            result = new ArgoAny(foundType);
 402.        }
 403.    
 404.        return result;
 405.        }

 407.        /**
 408.         * @param operation The operation.
 409.         * @param callName The name that we are trying to match.
 410.         * @param callParams The parameters that we are trying to match.
 411.         * @return <code>true</code> if the given Operation names and parameters
 412.         *           match the given name and parameters.
 413.         */
 414.        protected boolean operationMatchesCall(Object operation,
 415.                           String callName,
 416.                           Type[] callParams) {
 417.        if (!callName.equals(Model.getFacade().getName(operation))) {
 418.            return false;
 419.        }
 420.    
 421.            Collection operationParameters =
 422.                    Model.getFacade().getParameters(operation);
 423.        if (!Model.getFacade().isReturn(
 424.                            operationParameters.iterator().next())) {
 425.            LOG.warn(
 426.                    "ArgoFacade$ArgoAny expects the first operation parameter "
 427.            + "to be the return type; this isn't the case"
 428.            );
 429.        }
 430.        if (!(Model.getFacade().isReturn(operationParameters.iterator().next())
 431.              && operationParameters.size() == (callParams.length + 1))) {
 432.            return false;
 433.        }
 434.        Iterator paramIter = operationParameters.iterator();
 435.        paramIter.next(); // skip first parameter == return type
 436.        int index = 0;
 437.        while (paramIter.hasNext()) {
 438.            Object nextParam = paramIter.next();
 439.            Object paramType =
 440.                        Model.getFacade().getType(nextParam); //MClassifier
 441.            Type operationParam = getOclRepresentation(paramType);
 442.            if (!callParams[index].conformsTo(operationParam)) {
 443.            return false;
 444.            }
 445.            index++;
 446.        }
 447.        return true;
 448.        }
