 165.                        try {
 166.                            parseAttribute(s, newAttribute);
 167.                            /* If the 1st attribute is static, 
 168.                             * then the new ones, too. */
 169.                            Model.getCoreHelper().setStatic(
 170.                                    newAttribute,
 171.                                    Model.getFacade().isStatic(attribute));
 172.                        } catch (ParseException ex) {
 173.                            if (pex == null) {
 174.                                pex = ex;
 175.                            }
 176.                        }
 177.                    }
 178.                }
 179.                start = end + 1;
 180.                end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 181.            }
 182.            if (pex != null) {
 183.                throw pex;
 184.            }
 185.        }
 186.    
 187.        /**
 188.         * Parse a line on the form:<pre>
 189.         *      [/] visibility name [: type-expression] [= initial-value]
 190.         * </pre>
 191.         *
 192.         * <ul>
 193.         * <li>The / for derived is optional but has to be the first non-white 
 194.         * character.
 195.         * <li>If only one of visibility and name is given, then it is assumed to
 196.         * be the name and the visibility is left unchanged.
 197.         * <li>Type and initial value can be given in any order.
 198.         * <li>Properties can be given between any element in the form<pre>
 199.         *      {[name] [= [value]] [, ...]}
 200.         * </pre>
 201.         * <li>Multiplicity can be given between any element except after the
 202.         * initial-value and before the type or end (to allow java-style array
 203.         * indexing in the initial value). It must be given on form [multiplicity]
 204.         * with the square brackets included.
 205.         * <li>Stereotypes can be given between any element except after the
 206.         * initial-value and before the type or end (to allow java-style bit-shifts
 207.         * in the initial value). It must be given on form
 208.         * <<stereotype1,stereotype2,stereotype3>>.
 209.         * </ul>
 210.         *
 211.         * The following properties are recognized to have special meaning:
 212.         * frozen.<p>
 213.         *
 214.         * This syntax is compatible with the UML 1.3 spec.
 215.         *
 216.         * (formerly: visibility name [multiplicity] : type-expression =
 217.         * initial-value {property-string} ) (2nd formerly: [visibility] [keywords]
 218.         * type name [= init] [;] )
 219.         *
 220.         * @param text    The String to parse.
 221.         * @param attribute The attribute to modify to comply 
 222.         *                           with the instructions in s.
 223.         * @throws ParseException
 224.         *             when it detects an error in the attribute string. See also
 225.         *             ParseError.getErrorOffset().
 226.         */
 227.        protected void parseAttribute(
 228.                String text,
 229.                Object attribute) throws ParseException {
 230.            StringBuilder multiplicity = null;
 231.            String name = null;
 232.            List<String> properties = null;
 233.            StringBuilder stereotype = null; // This is null as until
 234.                                    // the first stereotype declaration is seen.
 235.                                    // After that it is non-null.
 236.            String token;
 237.            String type = null;
 238.            StringBuilder value = null;
 239.            String visibility = null;
 240.            boolean derived = false;
 241.            boolean hasColon = false;
 242.            boolean hasEq = false;
 243.            int multindex = -1;
 244.            MyTokenizer st;
 245.    
 246.            text = text.trim();
 247.            /* Handle Derived: */
 248.            if (text.length() > 0 && "/".indexOf(text.charAt(0)) >= 0) {
 249.                derived = true;
 250.                text = text.substring(1);
 251.                text = text.trim();
 252.            }
 253.    
 254.            /* Handle Visibility: */
 255.            if (text.length() > 0 
 256.                    && NotationUtilityUml.VISIBILITYCHARS.indexOf(text.charAt(0)) 
 257.                        >= 0) {
 258.                visibility = text.substring(0, 1);
 259.                text = text.substring(1);
 260.            }
 261.    
 262.            try {
 263.                st = new MyTokenizer(text, 
 264.                        " ,\t,<<,\u00AB,\u00BB,>>,[,],:,=,{,},\\,",
 265.                        NotationUtilityUml.attributeCustomSep);
 266.                while (st.hasMoreTokens()) {
 267.                    token = st.nextToken();
 268.                    if (" ".equals(token) || "\t".equals(token)
 269.                            || ",".equals(token)) {
 270.                        if (hasEq) {
 271.                            value.append(token);
 272.                        }
 273.                    } else if ("<<".equals(token) || "\u00AB".equals(token)) {
 274.                        if (hasEq) {
 275.                            value.append(token);
 276.                        } else {
 277.                            if (stereotype != null) {
 278.                                String msg = 
 279.                                    "parsing.error.attribute.two-sets-stereotypes";
 280.                                throw new ParseException(Translator.localize(msg),
 281.                                        st.getTokenIndex());
 282.                            }
 283.                            stereotype = new StringBuilder();
 284.                            while (true) {
 285.                                token = st.nextToken();
 286.                                if (">>".equals(token) || "\u00BB".equals(token)) {
 287.                                    break;
 288.                                }
 289.                                stereotype.append(token);
 290.                            }
 291.                        }
 292.                    } else if ("[".equals(token)) {
 293.                        if (hasEq) {
 294.                            value.append(token);
 295.                        } else {
 296.                            if (multiplicity != null) {
 297.                                String msg = 
 298.                                    "parsing.error.attribute.two-multiplicities";
 299.                                throw new ParseException(Translator.localize(msg),
 300.                                        st.getTokenIndex());
 301.                            }
 302.                            multiplicity = new StringBuilder();
 303.                            multindex = st.getTokenIndex() + 1;
 304.                            while (true) {
 305.                                token = st.nextToken();
 306.                                if ("]".equals(token)) {
 307.                                    break;
 308.                                }
 309.                                multiplicity.append(token);
 310.                            }
 311.                        }
 312.                    } else if ("{".equals(token)) {
 313.                        StringBuilder propname = new StringBuilder();
 314.                        String propvalue = null;
 315.    
 316.                        if (properties == null) {
 317.                            properties = new ArrayList<String>();
 318.                        }
 319.                        while (true) {
 320.                            token = st.nextToken();
 321.                            if (",".equals(token) || "}".equals(token)) {
 322.                                if (propname.length() > 0) {
 323.                                    properties.add(propname.toString());
 324.                                    properties.add(propvalue);
 325.                                }
 326.                                propname = new StringBuilder();
 327.                                propvalue = null;
 328.    
 329.                                if ("}".equals(token)) {
 330.                                    break;
 331.                                }
 332.                            } else if ("=".equals(token)) {
 333.                                if (propvalue != null) {
 334.                                    String msg = 
 335.                                        "parsing.error.attribute.prop-two-values";
 336.                                    Object[] args = {propvalue};
 337.    
 338.                                    throw new ParseException(Translator.localize(
 339.                                            msg, args), st.getTokenIndex());
 340.                                }
 341.                                propvalue = "";
 342.                            } else if (propvalue == null) {
 343.                                propname.append(token);
 344.                            } else {
 345.                                propvalue += token;
 346.                            }
 347.                        }
 348.                        if (propname.length() > 0) {
 349.                            properties.add(propname.toString());
 350.                            properties.add(propvalue);
 351.                        }
 352.                    } else if (":".equals(token)) {
 353.                        hasColon = true;
 354.                        hasEq = false;
 355.                    } else if ("=".equals(token)) {
 356.                        if (value != null) {
 357.                            String msg = 
 358.                                "parsing.error.attribute.two-default-values";
 359.                            throw new ParseException(Translator.localize(msg), st
 360.                                    .getTokenIndex());
 361.                        }
 362.                        value = new StringBuilder();
 363.                        hasColon = false;
 364.                        hasEq = true;
 365.                    } else {
 366.                        if (hasColon) {
 367.                            if (type != null) {
 368.                                String msg = "parsing.error.attribute.two-types";
 369.                                throw new ParseException(Translator.localize(msg),
 370.                                        st.getTokenIndex());
 371.                            }
 372.                            if (token.length() > 0
 373.                                    && (token.charAt(0) == '\"'
 374.                                        || token.charAt(0) == '\'')) {
 375.                                String msg = "parsing.error.attribute.quoted";
 376.                                throw new ParseException(Translator.localize(msg),
 377.                                        st.getTokenIndex());
 378.                            }
 379.                            if (token.length() > 0 && token.charAt(0) == '(') {
 380.                                String msg = "parsing.error.attribute.is-expr";
 381.                                throw new ParseException(Translator.localize(msg),
 382.                                        st.getTokenIndex());
 383.                            }
 384.                            type = token;
 385.                        } else if (hasEq) {
 386.                            value.append(token);
 387.                        } else {
 388.                            if (name != null && visibility != null) {
 389.                                String msg = "parsing.error.attribute.extra-text";
 390.                                throw new ParseException(Translator.localize(msg),
 391.                                        st.getTokenIndex());
 392.                            }
 393.                            if (token.length() > 0
 394.                                    && (token.charAt(0) == '\"'
 395.                                        || token.charAt(0) == '\'')) {
 396.                                String msg = "parsing.error.attribute.name-quoted";
 397.                                throw new ParseException(Translator.localize(msg),
 398.                                        st.getTokenIndex());
 399.                            }
 400.                            if (token.length() > 0 && token.charAt(0) == '(') {
 401.                                String msg = "parsing.error.attribute.name-expr";
 402.                                throw new ParseException(Translator.localize(msg),
 403.                                        st.getTokenIndex());
 404.                            }
 405.    
 406.                            if (name == null
 407.                                    && visibility == null
 408.                                    && token.length() > 1
 409.                                    && NotationUtilityUml.VISIBILITYCHARS
 410.                                            .indexOf(token.charAt(0)) >= 0) {
 411.                                visibility = token.substring(0, 1);
 412.                                token = token.substring(1);
 413.                            }
 414.    
 415.                            if (name != null) {
 416.                                visibility = name;
 417.                                name = token;
 418.                            } else {
 419.                                name = token;
 420.                            }
 421.                        }
 422.                    }
 423.                }
 424.            } catch (NoSuchElementException nsee) {
 425.                String msg = "parsing.error.attribute.unexpected-end-attribute";
 426.                throw new ParseException(Translator.localize(msg), text.length());
 427.            } 
 428.            // catch & rethrow is not necessary if we don't do nothing (penyaskito)
 429.            // catch (ParseException pre) {
 430.            //      throw pre;
 431.            // }
 432.    
 433.            if (LOG.isDebugEnabled()) {
 434.                LOG.debug("ParseAttribute [name: " + name 
 435.                        + " visibility: " + visibility 
 436.                        + " type: " + type + " value: " + value 
 437.                        + " stereo: " + stereotype
 438.                        + " mult: " + multiplicity);
 439.                if (properties != null) {
 440.                    for (int i = 0; i + 1 < properties.size(); i += 2) {
 441.                        LOG.debug("\tProperty [name: " + properties.get(i) + " = "
 442.                                + properties.get(i + 1) + "]");
 443.                    }
 444.                }
 445.            }
 446.    
 447.            dealWithDerived(attribute, derived);
 448.            dealWithVisibility(attribute, visibility);
 449.            dealWithName(attribute, name);
 450.            dealWithType(attribute, type);
 451.            dealWithValue(attribute, value);
 452.            dealWithMultiplicity(attribute, multiplicity, multindex);
 453.            dealWithProperties(attribute, properties);
 454.            StereotypeUtility.dealWithStereotypes(attribute, stereotype, true);
 455.        }

 531.        @Override
 532.        public String toString(Object modelElement, NotationSettings settings) {

 538.    
 539.        /*
 540.         * Generates a string representation for the provided
 541.         * attribute. The string representation will be of the form:
 542.         *          visibility name [multiplicity] : type-expression =
 543.         *                          initial-value {property-string}
 544.         * Depending on settings in Notation, visibility, multiplicity,
 545.         * type-expression, initial value and properties are shown/not shown.
 546.         */
 547.        private String toString(Object modelElement, boolean useGuillemets, 
 548.                boolean showVisibility, boolean showMultiplicity, boolean showTypes,
 549.                boolean showInitialValues, boolean showProperties) {
 550.            try {
 551.                String derived = "";
 552.                Object tv = Model.getFacade().getTaggedValue(modelElement, 
 553.                        Facade.DERIVED_TAG);
 554.                if (tv != null) {
 555.                    String tag = Model.getFacade().getValueOfTag(tv);
 556.                    if ("true".equalsIgnoreCase(tag)) {
 557.                        derived = "/";
 558.                    }
 559.                }
 560.                
 561.                String stereo = NotationUtilityUml.generateStereotype(modelElement, 
 562.                        useGuillemets);
 563.                String name = Model.getFacade().getName(modelElement);
 564.                String multiplicity = generateMultiplicity(
 565.                        Model.getFacade().getMultiplicity(modelElement));
 566.                String type = ""; // fix for loading bad projects
 567.                if (Model.getFacade().getType(modelElement) != null) {
 568.                    type = Model.getFacade().getName(
 569.                            Model.getFacade().getType(modelElement));
 570.                }
 571.    
 572.                StringBuilder sb = new StringBuilder(20);
 573.                sb.append(derived);
 574.    
 575.                if ((stereo != null) && (stereo.length() > 0)) {
 576.                    sb.append(stereo).append(" ");
 577.                }
 578.                if (showVisibility) {
 579.                    String visibility = NotationUtilityUml
 580.                            .generateVisibility2(modelElement);
 581.                    if (visibility != null && visibility.length() > 0) {
 582.                        sb.append(visibility);
 583.                    }
 584.                }
 585.                if ((name != null) && (name.length() > 0)) {
 586.                    sb.append(name).append(" ");
 587.                }
 588.                if ((multiplicity != null)
 589.                        && (multiplicity.length() > 0)
 590.                        && showMultiplicity) {
 591.                    sb.append("[").append(multiplicity).append("]").append(" ");
 592.                }
 593.                if ((type != null) && (type.length() > 0)
 594.                        /*
 595.                         * The "show types" defaults to TRUE, to stay compatible
 596.                         * with older ArgoUML versions that did not have this
 597.                         * setting:
 598.                         */
 599.                        && showTypes) {
 600.                    sb.append(": ").append(type).append(" ");
 601.                }
 602.                if (showInitialValues) {
 603.                    Object iv = Model.getFacade().getInitialValue(modelElement);
 604.                    if (iv != null) {
 605.                        String initialValue = 
 606.                            (String) Model.getFacade().getBody(iv);
 607.                        if (initialValue != null && initialValue.length() > 0) {
 608.                            sb.append(" = ").append(initialValue).append(" ");
 609.                        }
 610.                    }
 611.                }
 612.                if (showProperties) {
 613.                    String changeableKind = "";
 614.                    if (Model.getFacade().isReadOnly(modelElement)) {
 615.                        changeableKind = "frozen";
 616.                    }
 617.                    if (Model.getFacade().getChangeability(modelElement) != null) {
 618.                        if (Model.getChangeableKind().getAddOnly().equals(
 619.                                Model.getFacade().getChangeability(modelElement))) {
 620.                            changeableKind = "addOnly";
 621.                        }
 622.                    }
 623.                    StringBuilder properties = new StringBuilder();
 624.                    if (changeableKind.length() > 0) {
 625.                        properties.append("{ ").append(changeableKind).append(" }");
 626.                    }
 627.    
 628.                    if (properties.length() > 0) {
 629.                        sb.append(properties);
 630.                    }
 631.                }
 632.                return sb.toString().trim();
 633.            } catch (InvalidElementException e) {
 634.                // The element was deleted while we were processing it
 635.                return "";
 636.            }
 637.        }

 640.            if (m == null || "1".equals(Model.getFacade().toString(m))) {
 641.                return "";
 642.            }
 643.            return Model.getFacade().toString(m);
 644.        }

  98.        /**
  99.         * Parse a string representing one ore more ';' separated operations. The
 100.         * case that a String or char contains a ';' (e.g. in an initializer) is
 101.         * handled, but not other occurences of ';'.
 102.         *
 103.         * @param classifier  Classifier The classifier the operation(s) belong to
 104.         * @param operation   Operation The operation on which the editing happened
 105.         * @param text The string to parse
 106.         * @throws ParseException for invalid input
 107.         */
 108.        public void parseOperationFig(
 109.                Object classifier,
 110.                Object operation,
 111.                String text) throws ParseException {
 112.    
 113.            if (classifier == null || operation == null) {
 114.                return;
 115.            }
 116.            ParseException pex = null;
 117.            int start = 0;
 118.            int end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 119.            Project currentProject =
 120.                ProjectManager.getManager().getCurrentProject();
 121.            if (end == -1) {
 122.                //no text? remove op!
 123.                currentProject.moveToTrash(operation);
 124.                return;
 125.            }
 126.            String s = text.substring(start, end).trim();
 127.            if (s.length() == 0) {
 128.                //no non-whitechars in text? remove op!
 129.                currentProject.moveToTrash(operation);
 130.                return;
 131.            }
 132.            parseOperation(s, operation);
 133.            int i = Model.getFacade().getFeatures(classifier).indexOf(operation);
 134.            // check for more operations (';' separated):
 135.            start = end + 1;
 136.            end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 137.            while (end > start && end <= text.length()) {
 138.                s = text.substring(start, end).trim();
 139.                if (s.length() > 0) {
 140.                    // yes, there are more:
 141.                    Object returnType = currentProject.getDefaultReturnType();
 142.                    Object newOp =
 143.                        Model.getCoreFactory()
 144.                            .buildOperation(classifier, returnType);
 145.                    if (newOp != null) {
 146.                        try {
 147.                            parseOperation(s, newOp);
 148.                            //newOp.setOwnerScope(op.getOwnerScope()); //
 149.                            //not needed in case of operation
 150.                            if (i != -1) {
 151.                                Model.getCoreHelper().addFeature(
 152.                                        classifier, ++i, newOp);
 153.                            } else {
 154.                                Model.getCoreHelper().addFeature(
 155.                                        classifier, newOp);
 156.                            }
 157.                        } catch (ParseException ex) {
 158.                            if (pex == null) {
 159.                                pex = ex;
 160.                            }
 161.                        }
 162.                    }
 163.                }
 164.                start = end + 1;
 165.                end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 166.            }
 167.            if (pex != null) {
 168.                throw pex;
 169.            }
 170.        }

 172.    
 173.        /**
 174.         * Parse a line of text and aligns the Operation to the specification
 175.         * given. The line should be on the following form:<ul>
 176.         * <li> visibility name (parameter list) : return-type-expression
 177.         * {property-string}
 178.         * </ul>
 179.         *
 180.         * All elements are optional and, if left unspecified, will preserve their
 181.         * old values.<p>
 182.         * 
 183.         * <em>Stereotypes</em> can be given between any element in the line on the
 184.         * form: <<stereotype1,stereotype2,stereotype3>><p>
 185.         *
 186.         * The following properties are recognized to have special meaning:
 187.         * abstract, concurrency, concurrent, guarded, leaf, query, root and
 188.         * sequential.<p>
 189.         *
 190.         * This syntax is compatible with the UML 1.3 spec.<p>
 191.         *
 192.         * (formerly visibility name (parameter list) : return-type-expression
 193.         * {property-string} ) (formerly 2nd: [visibility] [keywords] returntype
 194.         * name(params)[;] )
 195.         *
 196.         * @param s   The String to parse.
 197.         * @param op  The Operation to adjust to the specification in s.
 198.         * @throws ParseException
 199.         *             when it detects an error in the attribute string. See also
 200.         *             ParseError.getErrorOffset().
 201.         */
 202.        public void parseOperation(String s, Object op) throws ParseException {
 203.            MyTokenizer st;
 204.            boolean hasColon = false;
 205.            String name = null;
 206.            String parameterlist = null;
 207.            StringBuilder stereotype = null;
 208.            String token;
 209.            String type = null;
 210.            String visibility = null;
 211.            List<String> properties = null;
 212.            int paramOffset = 0;
 213.    
 214.            s = s.trim();
 215.            if (s.length() > 0 
 216.                    && NotationUtilityUml.VISIBILITYCHARS.indexOf(s.charAt(0)) 
 217.                        >= 0) {
 218.                visibility = s.substring(0, 1);
 219.                s = s.substring(1);
 220.            }
 221.    
 222.            try {
 223.                st = new MyTokenizer(s, " ,\t,<<,\u00AB,\u00BB,>>,:,=,{,},\\,",
 224.                        NotationUtilityUml.operationCustomSep);
 225.                while (st.hasMoreTokens()) {
 226.                    token = st.nextToken();
 227.                    if (" ".equals(token) || "\t".equals(token)
 228.                            || ",".equals(token)) {
 229.                        continue; // Do nothing
 230.                    } else if ("<<".equals(token) || "\u00AB".equals(token)) {
 231.                        if (stereotype != null) {
 232.                            parseError("operation.stereotypes", 
 233.                                    st.getTokenIndex());
 234.                        }
 235.                        stereotype = new StringBuilder();
 236.                        while (true) {
 237.                            token = st.nextToken();
 238.                            if (">>".equals(token) || "\u00BB".equals(token)) {
 239.                                break;
 240.                            }
 241.                            stereotype.append(token);
 242.                        }
 243.                    } else if ("{".equals(token)) {
 244.                        properties = tokenOpenBrace(st, properties);
 245.                    } else if (":".equals(token)) {
 246.                        hasColon = true;
 247.                    } else if ("=".equals(token)) {
 248.                        parseError("operation.default-values", st.getTokenIndex());
 249.                    } else if (token.charAt(0) == '(' && !hasColon) {
 250.                        if (parameterlist != null) {
 251.                            parseError("operation.two-parameter-lists", 
 252.                                    st.getTokenIndex());
 253.                        }
 254.    
 255.                        parameterlist = token;
 256.                    } else {
 257.                        if (hasColon) {
 258.                            if (type != null) {
 259.                                parseError("operation.two-types", 
 260.                                        st.getTokenIndex());
 261.                            }
 262.    
 263.                            if (token.length() > 0
 264.                                    && (token.charAt(0) == '\"'
 265.                                        || token.charAt(0) == '\'')) {
 266.                                parseError("operation.type-quoted",
 267.                                        st.getTokenIndex());
 268.                            }
 269.    
 270.                            if (token.length() > 0 && token.charAt(0) == '(') {
 271.                                parseError("operation.type-expr", 
 272.                                        st.getTokenIndex());
 273.                            }
 274.    
 275.                            type = token;
 276.                        } else {
 277.                            if (name != null && visibility != null) {
 278.                                parseError("operation.extra-text",
 279.                                        st.getTokenIndex());
 280.                            }
 281.    
 282.                            if (token.length() > 0
 283.                                    && (token.charAt(0) == '\"'
 284.                                        || token.charAt(0) == '\'')) {
 285.                                parseError("operation.name-quoted",
 286.                                        st.getTokenIndex());
 287.                            }
 288.    
 289.                            if (token.length() > 0 && token.charAt(0) == '(') {
 290.                                parseError("operation.name-expr", 
 291.                                        st.getTokenIndex());
 292.                            }
 293.    
 294.                            if (name == null
 295.                                    && visibility == null
 296.                                    && token.length() > 1
 297.                                    && NotationUtilityUml.VISIBILITYCHARS.indexOf(
 298.                                            token.charAt(0))
 299.                                                        >= 0) {
 300.                                visibility = token.substring(0, 1);
 301.                                token = token.substring(1);
 302.                            }
 303.    
 304.                            if (name != null) {
 305.                                visibility = name;
 306.                                name = token;
 307.                            } else {
 308.                                name = token;
 309.                            }
 310.                        }
 311.                    }
 312.                } // end while loop
 313.            } catch (NoSuchElementException nsee) {
 314.                parseError("operation.unexpected-end-operation", 
 315.                        s.length());
 316.            } catch (ParseException pre) {
 317.                throw pre;
 318.            }
 319.            
 320.            if (parameterlist != null) {
 321.                // parameterlist is guaranteed to contain at least "("
 322.                if (parameterlist.charAt(parameterlist.length() - 1) != ')') {
 323.                    parseError("operation.parameter-list-incomplete",
 324.                            paramOffset + parameterlist.length() - 1);
 325.                }
 326.    
 327.                paramOffset++;
 328.                parameterlist = parameterlist.substring(1,
 329.                        parameterlist.length() - 1);
 330.                NotationUtilityUml.parseParamList(op, parameterlist, paramOffset);
 331.            }
 332.    
 333.            if (visibility != null) {
 334.                Model.getCoreHelper().setVisibility(op,
 335.                        NotationUtilityUml.getVisibility(visibility.trim()));
 336.            }
 337.    
 338.            if (name != null) {
 339.                Model.getCoreHelper().setName(op, name.trim());
 340.            } else if (Model.getFacade().getName(op) == null
 341.                    || "".equals(Model.getFacade().getName(op))) {
 342.                Model.getCoreHelper().setName(op, "anonymous");
 343.            }
 344.    
 345.            if (type != null) {
 346.                Object ow = Model.getFacade().getOwner(op);
 347.                Object ns = null;
 348.                if (ow != null && Model.getFacade().getNamespace(ow) != null) {
 349.                    ns = Model.getFacade().getNamespace(ow);
 350.                } else {
 351.                    ns = Model.getFacade().getRoot(op);
 352.                }
 353.                Object mtype = NotationUtilityUml.getType(type.trim(), ns);
 354.                setReturnParameter(op, mtype);
 355.            }
 356.    
 357.            if (properties != null) {
 358.                NotationUtilityUml.setProperties(op, properties, 
 359.                        NotationUtilityUml.operationSpecialStrings);
 360.            }
 361.    
 362.            // Don't create a stereotype for <<signal>> on a Reception
 363.            // but create any other parsed stereotypes as needed
 364.            if (!Model.getFacade().isAReception(op) 
 365.                    || !RECEPTION_KEYWORD.equals(stereotype.toString())) {
 366.                StereotypeUtility.dealWithStereotypes(op, stereotype, true);
 367.            }
 368.        }
 369.    
 370.        /**
 371.         * Convenience method to signal a parser error.
 372.         * 
 373.         * @param message
 374.         *            string containing error message literal. It will be appended
 375.         *            to the base "parser.error." and localized.
 376.         * @param offset
 377.         *            offset to where error occurred
 378.         * @throws ParseException
 379.         */
 380.        private void parseError(String message, int offset)
 381.            throws ParseException {
 382.    
 383.            throw new ParseException(
 384.                    Translator.localize("parsing.error." + message), 
 385.                    offset);
 386.        }
 387.    
 388.        /**
 389.         * Parse tokens following an open brace (properties).
 390.         * 
 391.         * @param st tokenizer being used
 392.         * @param properties current properties list
 393.         * @return updated list of properties
 394.         * @throws ParseException
 395.         */
 396.        private List<String> tokenOpenBrace(MyTokenizer st, List<String> properties)
 397.            throws ParseException {
 398.            String token;
 399.            StringBuilder propname = new StringBuilder();
 400.            String propvalue = null;
 401.    
 402.            if (properties == null) {
 403.                properties = new ArrayList<String>();
 404.            }
 405.            while (true) {
 406.                token = st.nextToken();
 407.                if (",".equals(token) || "}".equals(token)) {
 408.                    if (propname.length() > 0) {
 409.                        properties.add(propname.toString());
 410.                        properties.add(propvalue);
 411.                    }
 412.                    propname = new StringBuilder();
 413.                    propvalue = null;
 414.    
 415.                    if ("}".equals(token)) {
 416.                        break;
 417.                    }
 418.                } else if ("=".equals(token)) {
 419.                    if (propvalue != null) {
 420.                        String msg = 
 421.                            "parsing.error.operation.prop-stereotypes";
 422.                        Object[] args = {propname};
 423.                        throw new ParseException(
 424.                                Translator.localize(msg, 
 425.                                args), 
 426.                                st.getTokenIndex());
 427.                    }
 428.                    propvalue = "";
 429.                } else if (propvalue == null) {
 430.                    propname.append(token);
 431.                } else {
 432.                    propvalue += token;
 433.                }
 434.            }
 435.            if (propname.length() > 0) {
 436.                properties.add(propname.toString());
 437.                properties.add(propvalue);
 438.            }
 439.            return properties;
 440.        }

 456.                    param = p;
 457.                    break;
 458.                }
 459.            }
 460.            while (it.hasNext()) {

 471.            }
 472.            Model.getCoreHelper().setType(param, type);
 473.        }
