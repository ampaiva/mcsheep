  92.            // warn about inherited name conflicts, different critic?
  93.            Iterator features = str.iterator();
  94.            while (features.hasNext()) {
  95.                Object o = features.next();
  96.    
  97.                if (!(Model.getFacade().isAStructuralFeature(o))) {
  98.                    continue;
  99.                }
 100.    
 101.                Object sf = /*(MStructuralFeature)*/ o;
 102.    
 103.                String sfName = Model.getFacade().getName(sf);
 104.                if ("".equals(sfName)) {
 105.                    continue;
 106.                }
 107.    
 108.                String nameStr = sfName;
 109.                if (nameStr.length() == 0) {
 110.                    continue;
 111.                }
 112.    
 113.                namesSeen.add(nameStr);
 114.    
 115.            }
 116.    
 117.            Collection assocEnds = Model.getFacade().getAssociationEnds(cls);
 118.    
 119.            // warn about inherited name conflicts, different critic?
 120.            Iterator myEnds = assocEnds.iterator();
 121.            while (myEnds.hasNext()) {
 122.                Object myAe = /*(MAssociationEnd)*/ myEnds.next();
 123.                Object asc =
 124.                    /*(MAssociation)*/
 125.                    Model.getFacade().getAssociation(myAe);
 126.                Collection conn = Model.getFacade().getConnections(asc);
 127.    
 128.                if (Model.getFacade().isAAssociationRole(asc)) {
 129.                    conn = Model.getFacade().getConnections(asc);
 130.                }
 131.                if (conn == null) {
 132.                    continue;
 133.                }
 134.    
 135.                Iterator ascEnds = conn.iterator();
 136.                while (ascEnds.hasNext()) {
 137.                    Object ae = /*(MAssociationEnd)*/ ascEnds.next();
 138.                    if (Model.getFacade().getType(ae) == cls) {
 139.                        continue;
 140.                    }
 141.                    String aeName = Model.getFacade().getName(ae);
 142.                    if ("".equals(aeName)) {
 143.                        continue;
 144.                    }
 145.                    String aeNameStr = aeName;
 146.                    if (aeNameStr == null || aeNameStr.length() == 0) {
 147.                        continue;
 148.                    }
 149.    
 150.                    if (namesSeen.contains(aeNameStr)) {
 151.                        return PROBLEM_FOUND;
 152.                    }
 153.                }
 154.            }
 155.            return NO_PROBLEM;
 156.        }
