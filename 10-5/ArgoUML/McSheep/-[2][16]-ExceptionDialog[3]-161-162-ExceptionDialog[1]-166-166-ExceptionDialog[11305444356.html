  67.        private JLabel northLabel;
  68.        private boolean aborted = false;
  69.    
  70.        /**
  71.         * The constructor.
  72.         */
  73.        ProblemsDialog(Frame frame, String errors) {
  74.            super(frame);
  75.            setResizable(true);
  76.            setModal(true);
  77.            setTitle(Translator.localize("dialog.title.import-problems"));
  78.    
  79.            Dimension scrSize = Toolkit.getDefaultToolkit().getScreenSize();
  80.            getContentPane().setLayout(new BorderLayout(0, 0));
  81.    
  82.            // the introducing label
  83.            northLabel =
  84.                new JLabel(Translator.localize("label.import-problems"));
  85.            getContentPane().add(northLabel, BorderLayout.NORTH);
  86.    
  87.            // the text box containing the problem messages
  88.            JEditorPane textArea = new JEditorPane();
  89.            textArea.setText(errors);
  90.            JPanel centerPanel = new JPanel(new BorderLayout());
  91.            centerPanel.add(new JScrollPane(textArea));
  92.            centerPanel.setPreferredSize(new Dimension(600, 200));
  93.            getContentPane().add(centerPanel);
  94.    
  95.            // continue and abort buttons
  96.            continueButton = new JButton(Translator.localize("button.continue"));
  97.            abortButton = new JButton(Translator.localize("button.abort"));
  98.            JPanel bottomPanel = new JPanel();
  99.            bottomPanel.add(continueButton);
 100.            bottomPanel.add(abortButton);
 101.            getContentPane().add(bottomPanel, BorderLayout.SOUTH);
 102.            continueButton.requestFocusInWindow();
 103.    
 104.            // listeners
 105.            continueButton.addActionListener(this);
 106.            abortButton.addActionListener(this);
 107.            addWindowListener(new WindowAdapter() {
 108.                public void windowClosing(WindowEvent evt) {
 109.                    disposeDialog();
 110.                }
 111.            });
 112.    
 113.            pack();
 114.            Dimension contentPaneSize = getContentPane().getSize();
 115.            setLocation(scrSize.width / 2 - contentPaneSize.width / 2,
 116.                    scrSize.height / 2 - contentPaneSize.height / 2);
 117.        }
 118.    
 119.        /*
 120.         * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
 121.         */
 122.        public void actionPerformed(ActionEvent e) {
 123.            if (e.getSource().equals(abortButton)) {
 124.                aborted = true;
 125.            }
 126.            disposeDialog();
 127.        }
 128.    
 129.        /**
 130.         * Returns whether the Abort button was pressed.
 131.         * @return aborted
 132.         */
 133.        public boolean isAborted() {
 134.            return aborted;
 135.        }
 136.    
 137.        private void disposeDialog() {
 138.            setVisible(false);
 139.            dispose();
 140.        }
 141.    
 142.        /**
 143.         * The UID.
 144.         */
 145.        private static final long serialVersionUID = -9221358976863603143L;
 146.    }
