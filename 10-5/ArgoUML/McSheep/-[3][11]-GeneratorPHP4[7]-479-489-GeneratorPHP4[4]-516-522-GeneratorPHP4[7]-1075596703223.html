 389.                                       .getDefaultValue(objParameter) != null) {

 546.        private String generateClassifierAssociations(Object element) {
 547.            StringBuilder text = new StringBuilder();
 548.    
 549.            if (Model.getFacade().isAClass(element)) {
 550.                text.append(INDENT).append("// --- ASSOCIATIONS ---\n");
 551.    
 552.                for (Object assocEnd : Model.getFacade().getAssociationEnds(
 553.                        element)) {

 564.        /**
 565.         * Return the opposite navigable end or null.  If this associationEnd
 566.         * is a part of a binary association and the opposite end is navigable, 
 567.         * return that end, otherwise return null.
 568.         */
 569.        private Object getNavigableOppositeEnd(Object assEnd) {
 570.            Collection otherEnds = Model.getFacade().getOtherAssociationEnds(
 571.                    assEnd);
 572.            Object returnValue = null;
 573.            if (otherEnds.size() == 1) {

 775.

 824.        /*
 825.         * Generates the default value for a type
 826.         *
 827.         * @param modelElement classifier representing a type
 828.         * @param sDefault     default value
 829.         * @param bCast        prefix value with cast statement
 830.         *
 831.         * @return default value with explicite cast (if needed)
 832.         */
 833.        private final String generateDefaultValue(Object modelElement,
 834.                String sDefault, boolean bCast) {
 835.            if (modelElement == null) {
 836.                return null;
 837.            }
 838.    
 839.            if (!Model.getFacade().isAClassifier(modelElement)) {
 840.                throw new ClassCastException(modelElement.getClass()
 841.                        + " has wrong object type, Classifier required");
 842.            }
 843.    
 844.            String sType = convertType(modelElement);
 845.            if (sType == null) {
 846.                return "null";
 847.            } else if (sType.equals("string")) {
 848.                String sReturn = bCast ? "(string) " : "";
 849.                if (sDefault != null) {
 850.                    int iFirstApos = sDefault.indexOf("'");
 851.                    while (iFirstApos != -1) {
 852.                        sDefault = sDefault.substring(0, iFirstApos)
 853.                                + "\\" + sDefault.substring(iFirstApos);
