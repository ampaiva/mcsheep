 175.        /**
 176.         * Fire an event to all members of the listener list.
 177.         *
 178.         * @param propertyName The name of the event.
 179.         * @param oldValue The old value.
 180.         * @param newValue The new value.
 181.         */
 182.        void firePropertyChanged(String propertyName,
 183.                                         Object oldValue, Object newValue) {
 184.            // Guaranteed to return a non-null array
 185.            Object[] listeners = listenerList.getListenerList();
 186.            // Process the listeners last to first, notifying
 187.            // those that are interested in this event
 188.            for (int i = listeners.length - 2; i >= 0; i -= 2) {
 189.                if (listeners[i] == PropertyChangeListener.class) {
 190.                    // Lazily create the event:
 191.                    if (event == null) {
 192.                        event =
 193.                            new PropertyChangeEvent(
 194.                                this,
 195.                                propertyName,
 196.                                oldValue,
 197.                                newValue);
 198.                    }
 199.                    ((PropertyChangeListener) listeners[i + 1]).propertyChange(
 200.                        event);
 201.                }
 202.            }
 203.            event = null;
 204.        }

 176.         * Fire an event to all members of the listener list.
 177.         *
 178.         * @param propertyName The name of the event.
 179.         * @param oldValue The old value.
 180.         * @param newValue The new value.
 181.         */
 182.        void firePropertyChanged(String propertyName,
 183.                                         Object oldValue, Object newValue) {
 184.            // Guaranteed to return a non-null array
 185.            Object[] listeners = listenerList.getListenerList();
 186.            // Process the listeners last to first, notifying
 187.            // those that are interested in this event
 188.            for (int i = listeners.length - 2; i >= 0; i -= 2) {
 189.                if (listeners[i] == PropertyChangeListener.class) {
 190.                    // Lazily create the event:
 191.                    if (event == null) {
 192.                        event =
 193.                            new PropertyChangeEvent(
 194.                                this,
 195.                                propertyName,
 196.                                oldValue,
 197.                                newValue);
 198.                    }
 199.                    ((PropertyChangeListener) listeners[i + 1]).propertyChange(
 200.                        event);
 201.                }
 202.            }
 203.            event = null;
 204.        }

 217.         *             projects will be able to be open at a time, so all code
 218.         *             should be prepared to deal with multiple projects and should
 219.         *             require a Project to be passed as an argument if they need
 220.         *             access.

 230.                    ArgoDiagram activeDiagram = diagrams.get(0);
 231.                    currentProject.setActiveDiagram(activeDiagram);
 232.                }
 233.            }
 234.            notifyProjectAdded(newProject, oldProject);
 235.        }

 564.        /**
 565.         * Called when the model subsystem creates a command.
 566.         * We must add this to the UndoManager.
 567.         *
 568.         * @param command the command.
 569.         * @return result of the command, if any
 570.         * @see org.argouml.model.ModelCommandCreationObserver#execute(ModelCommand)
 571.         */
 572.        public Object execute(final ModelCommand command) {
 573.            setSaveEnabled(true);
 574.            AbstractCommand wrappedCommand = new AbstractCommand() {
 575.                private ModelCommand modelCommand = command;
 576.                public void undo() {
 577.                    modelCommand.undo();
 578.                }
 579.                public boolean isUndoable() {
 580.                    return modelCommand.isUndoable();
 581.                }
 582.                public boolean isRedoable() {
 583.                    return modelCommand.isRedoable();
 584.                }
 585.                public Object execute() {
 586.                    return modelCommand.execute();
 587.                }
 588.                public String toString() {
 589.                    return modelCommand.toString();
 590.                }
 591.            };
 592.            Project p = getCurrentProject();
 593.            if (p != null) {
 594.                return getCurrentProject().getUndoManager().execute(wrappedCommand);
 595.            } else {
 596.                return wrappedCommand.execute();
 597.            }
 598.        }
