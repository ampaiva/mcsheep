 647.        /**
 648.         * Class to filter XML declaration and DOCTYPE declaration from an output
 649.         * stream to allow use as nested XML files.
 650.         * 
 651.         * @author Tom Morris
 652.         */
 653.        class XmlFilterOutputStream extends FilterOutputStream {
 654.    
 655.            private CharsetDecoder decoder;
 656.    
 657.            private boolean headerProcessed = false;
 658.    
 659.            private static final int BUFFER_SIZE = 120;
 660.    
 661.            /**
 662.             * The following three fields make up a doubly mapped buffer with two
 663.             * sets of pointers (the ByteBuffer objects), one for input and one for
 664.             * output. Bytes are written into the buffer using outBB which advances
 665.             * the current position. The following ASCII art shows what the buffer
 666.             * looks like.<code>
 667.             *           vp       vl
 668.             * xxxxxx++++..........
 669.             *       ^p  ^l
 670.             * </code> vp & vl are the output (buffer writing) position and limit,
 671.             * respectively and show the free space that can receive bytes. The ^p
 672.             * and ^l pointers show the input (buffer reading) position and limit.
 673.             * They point to valid bytes which have not yet been converted to
 674.             * characters. The xxx bytes have been both written in and read back
 675.             * out. The +++ bytes have been written into the buffer, but not read
 676.             * out yet.
 677.             */
 678.            private byte[] bytes = new byte[BUFFER_SIZE * 2];
 679.    
 680.            private ByteBuffer outBB = ByteBuffer.wrap(bytes);
 681.    
 682.            private ByteBuffer inBB = ByteBuffer.wrap(bytes);
 683.    
 684.            // Buffer containing characters which have been decoded from the bytes
 685.            // in inBB.
 686.            private CharBuffer outCB = CharBuffer.allocate(BUFFER_SIZE);
 687.    
 688.            // RegEx pattern for XML declaration and, optionally, DOCTYPE
 689.            // Backslashes are doubled up - one for Java, one for Regex
 690.            private final Pattern xmlDeclarationPattern = Pattern
 691.                    .compile("\\s*<\\?xml.*\\?>\\s*(<!DOCTYPE.*>\\s*)?");
 692.    
 693.            /**
 694.             * Construct a filtered output stream using the given character set
 695.             * name.
 696.             * 
 697.             * @param outputStream source output stream to filter
 698.             * @param charsetName name of character set to use for encoding
 699.             */
 700.            public XmlFilterOutputStream(OutputStream outputStream,
 701.                    String charsetName) {
 702.                this(outputStream, Charset.forName(charsetName));
 703.            }
 704.    
 705.            /**
 706.             * Construct a filtered output stream using the given character set.
 707.             * 
 708.             * @param outputStream source output stream to filter
 709.             * @param charset character set to use for encoding
 710.             */
 711.            public XmlFilterOutputStream(OutputStream outputStream, Charset charset) {
 712.                super(outputStream);
 713.                decoder = charset.newDecoder();
 714.                decoder.onMalformedInput(CodingErrorAction.REPORT);
 715.                decoder.onUnmappableCharacter(CodingErrorAction.REPORT);
 716.                startEntry();
 717.            }
 718.    
 719.            /**
 720.             * Reset processing for the beginning of an entry.
 721.             */
 722.            public void startEntry() {
 723.                headerProcessed = false;
 724.                resetBuffers();
 725.            }
 726.    
 727.            private void resetBuffers() {
 728.                inBB.limit(0);
 729.                outBB.position(0);
 730.                outCB.position(0);
 731.            }
 732.    
 733.            @Override
 734.            public void write(byte[] b, int off, int len) throws IOException {
 735.                if ((off | len | (b.length - (len + off)) | (off + len)) < 0) {
 736.                    throw new IndexOutOfBoundsException();
 737.                }
 738.    
 739.                if (headerProcessed) {
 740.                    out.write(b, off, len);
 741.                } else {
 742.                    // TODO: Make this more efficient for large I/Os
 743.                    for (int i = 0; i < len; i++) {
 744.                        write(b[off + i]);
 745.                    }
 746.                }
 747.    
 748.            }
 749.    
 750.            @Override
 751.            public void write(int b) throws IOException {
 752.    
 753.                if (headerProcessed) {
 754.                    out.write(b);
 755.                } else {
 756.                    outBB.put((byte) b);
 757.                    inBB.limit(outBB.position());
 758.                    // Convert from bytes back to characters
 759.                    CoderResult result = decoder.decode(inBB, outCB, false);
 760.                    if (result.isError()) {
 761.                        throw new RuntimeException(
 762.                                "Unknown character decoding error");
 763.                    }
 764.    
 765.                    // This will have problems if the smallest possible
 766.                    // data segment is smaller than the size of the buffer
 767.                    // needed for regex matching
 768.                    if (outCB.position() == outCB.limit()) {
 769.                        processHeader();
 770.                    }
 771.    
 772.                }
 773.            }
 774.    
 775.            private void processHeader() throws IOException {
 776.                headerProcessed = true;
 777.                outCB.position(0); // rewind our character buffer
 778.    
 779.                Matcher matcher = xmlDeclarationPattern.matcher(outCB);
 780.                // Remove anything that matches our pattern
 781.                String headerRemainder = matcher.replaceAll("");
 782.                int index = headerRemainder.length() - 1;
 783.                if (headerRemainder.charAt(index) == '\0') {
 784.                    // Remove null characters at the end
 785.                    do {
 786.                        index--;
 787.                    } while (index >= 0 && headerRemainder.charAt(index) == '\0');
 788.                    headerRemainder = headerRemainder.substring(0, index + 1);
 789.                }
 790.    
 791.                // Reencode the remaining characters as bytes again
 792.                ByteBuffer bb = decoder.charset().encode(headerRemainder);
 793.    
 794.                // and write them to our output stream
 795.                byte[] outBytes = new byte[bb.limit()];
 796.                bb.get(outBytes);
 797.                out.write(outBytes, 0, outBytes.length);
 798.    
 799.                // Write any left over bytes in the input buffer
 800.                // (perhaps from a partially decoded character)
 801.                if (inBB.remaining() > 0) {
 802.                    out.write(inBB.array(), inBB.position(), inBB.remaining());
 803.                    inBB.position(0);
 804.                    inBB.limit(0);
 805.                }
 806.            }
 807.    
 808.            /**
 809.             * This method has no effect to keep sub-writers from closing it
 810.             * accidently. The master can use the method {@link #reallyClose()} to
 811.             * actually close the underlying stream.
 812.             */
 813.            @Override
 814.            public void close() throws IOException {
 815.                flush();
 816.            }
 817.    
 818.            /**
 819.             * Close the stream.
 820.             * 
 821.             * @throws IOException
 822.             */
 823.            public void reallyClose() throws IOException {
 824.                out.close();
 825.            }
 826.    
 827.            /**
 828.             * Flush the stream. If the stream is flushed before the header is
 829.             * completely processed, whatever has been written so far will get
 830.             * processed before the flush.
 831.             */
 832.            @Override
 833.            public void flush() throws IOException {
 834.                if (!headerProcessed) {
 835.                    processHeader();
 836.                }
 837.                out.flush();
 838.            }
 839.    
 840.        }

 154.        public Collection parse(InputSource inputSource, boolean readOnly)
 155.            throws UmlException {
 156.            
 157.            Collection<RefObject> newElements = Collections.emptyList();
 158.    
 159.            String extentBase = inputSource.getPublicId();
 160.            if (extentBase == null) {
 161.                extentBase = inputSource.getSystemId();
 162.            }
 163.            if (extentBase == null) {
 164.                extentBase = MDRModelImplementation.MODEL_EXTENT_NAME;
 165.            }
 166.            String extentName = extentBase;
 167.            UmlPackage extent = 
 168.                (UmlPackage) modelImpl.getRepository().getExtent(extentName);
 169.            int serial = 1;
 170.            while (extent != null) {
 171.                extentName = extentBase + " " + serial;
 172.                serial++;
 173.                extent = (UmlPackage) modelImpl.getRepository().getExtent(
 174.                        extentName);
 175.            }
 176.    
 177.            extent = (UmlPackage) modelImpl.createExtent(extentName, readOnly);
 178.            if (extent == null) {
 179.                LOG.error("Failed to create extent " + extentName);
 180.            }
 181.            
 182.            try {
 183.                LOG.info("Loading to extent '" + extentName + "' " + extent);
 184.    
 185.                InputConfig config = new InputConfig();
 186.                config.setUnknownElementsListener(this);
 187.                config.setUnknownElementsIgnored(true);
 188.    
 189.                String pId = inputSource.getPublicId();
 190.                String sId = modelImpl.getPublic2SystemIds().get(pId);
 191.                if (sId != null) {
 192.                    if (sId.equals(inputSource.getSystemId())) {
 193.                        LOG.info("Attempt to reread profile - ignoring - "
 194.                                + "publicId = \"" + pId + "\";  systemId = \""
 195.                                + sId + "\".");
 196.                        return Collections.emptySet();
 197.                    } else {
 198.                        throw new UmlException("Profile with the duplicate publicId "
 199.                                + "is being loaded! publicId = \"" + pId
 200.                                + "\"; existing systemId = \""
 201.                                + modelImpl.getPublic2SystemIds().get(pId)
 202.                                + "\"; new systemId = \"" + sId + "\".");
 203.                    }
 204.                }
 205.                resolver = new XmiReferenceResolverImpl(new RefPackage[] {extent},
 206.                        config, modelImpl.getObjectToId(), 
 207.                        modelImpl.getPublic2SystemIds(), modelImpl.getIdToObject(), 
 208.                        modelImpl.getSearchPath(), 
 209.                        readOnly, 
 210.                        inputSource.getPublicId(), inputSource.getSystemId(),
 211.                        modelImpl);
 212.                config.setReferenceResolver(resolver);
 213.                config.setHeaderConsumer(this);
 214.                
 215.                XMIReader xmiReader =
 216.                        XMIReaderFactory.getDefault().createXMIReader(config);
 217.    
 218.                /*
 219.                 * MDR has a hardcoded printStackTrace on all exceptions,
 220.                 * even if they're caught, which is unsightly, so we handle
 221.                 * unknown elements ourselves rather than letting MDR throw
 222.                 * an exception for us to catch.
 223.                 * 
 224.                 * org/netbeans/lib/jmi/util/Logger.java
 225.                 * 
 226.                 * This can be uses to disable logging.  Default output is 
 227.                 * System.err
 228.                 * setProperty("org.netbeans.lib.jmi.Logger.fileName", "")
 229.                 *              org.netbeans.mdr.Logger
 230.                 * 
 231.                 * The property org.netbeans.lib.jmi.Logger controls the minimum 
 232.                 * severity level for logging
 233.                 */
 234.                // Turn off NetBeans logging to System.err
 235.    //            System.setProperty("org.netbeans.lib.jmi.Logger.fileName", "");
 236.                // Set minimum severity level for MDR
 237.    //            System.setProperty("org.netbeans.lib.jmi.Logger", 
 238.    //                    Integer.toString(ErrorManager.INFORMATIONAL));
 239.                InputConfig config2 = (InputConfig) xmiReader.getConfiguration();
 240.                config2.setUnknownElementsListener(this);
 241.                config2.setUnknownElementsIgnored(true);
 242.                unknownElement = false;
 243.                uml13 = false;
 244.                ignoredElementCount = 0;
 245.    
 246.                // Disable event delivery during model load
 247.                modelImpl.getModelEventPump().stopPumpingEvents();
 248.    
 249.                try {
 250.                    String systemId = inputSource.getSystemId();
 251.                    // If we've got a streaming input, copy it to make sure we'll
 252.                    // be able to rewind it if necessary
 253.                    if (inputSource.getByteStream() != null
 254.                            || inputSource.getCharacterStream() != null) {
 255.                        File file = copySource(inputSource);
 256.                        systemId = file.toURI().toURL().toExternalForm();
 257.                        String publicId = inputSource.getPublicId();
 258.                        inputSource = new InputSource(systemId);
 259.                        inputSource.setPublicId(publicId);                    
 260.                    }
 261.                    MDRepository repository = modelImpl.getRepository();
 262.                    
 263.                    // Use a transaction to avoid the performance penalty (3x) of 
 264.                    // MDR's autocommit mode
 265.                    repository.beginTrans(true);
 266.    
 267.                    newElements = xmiReader.read(inputSource.getByteStream(),
 268.                            systemId, extent);
 269.                    
 270.                    // If a UML 1.3 file, attempt to upgrade it to UML 1.4
 271.                    if (uml13) {
 272.                        // Roll back transaction from first attempt & start new one
 273.                        repository.endTrans(true);
 274.                        repository.beginTrans(true);
 275.    
 276.                        // Clear the associated ID maps & reset starting collection
 277.                        resolver.clearIdMaps();
 278.    
 279.                        newElements = convertAndLoadUml13(inputSource.getSystemId(),
 280.                                extent, xmiReader, inputSource);
 281.                    }
 282.    
 283.                    // Commit our transaction
 284.                    repository.endTrans();
 285.                } catch (Throwable e) {
 286.                    // Roll back transaction to remove any partial results read
 287.                    try {
 288.                        modelImpl.getRepository().endTrans(true);
 289.                    } catch (Throwable e2) {
 290.                        // Ignore any error.  The transaction may already have
 291.                        // been unwound as part of exception processing by MDR
 292.                    }
 293.                    if (e instanceof MalformedXMIException) {
 294.                        throw (MalformedXMIException) e;
 295.                    } else if (e instanceof IOException) {
 296.                        throw (IOException) e;
 297.                    } else {
 298.                        // We shouldn't get here, but just in case...
 299.                        // We want a wide exception catcher to make sure our
 300.                        // transaction always gets ended
 301.                        e.printStackTrace();
 302.                        throw new MalformedXMIException();
 303.                    }
 304.                } finally {
 305.                    modelImpl.getModelEventPump().startPumpingEvents();
 306.                }
 307.    
 308.                if (unknownElement) {
 309.                    modelImpl.deleteExtent(extent);
 310.                    throw new XmiException("Unknown element in XMI file : "
 311.                            + unknownElementName);
 312.                }
 313.    
 314.                if (ignoredElementCount > 0) {
 315.                    LOG.warn("Ignored one or more elements from list "
 316.                            + ignoredElements);
 317.                }
 318.    
 319.            } catch (MalformedXMIException e) {
 320.                // If we can find a nested SAX exception, it will have information
 321.                // on the line number, etc.
 322.                ErrorManager.Annotation[] annotations = 
 323.                    ErrorManager.getDefault().findAnnotations(e);
 324.                for (ErrorManager.Annotation annotation : annotations) {
 325.                    Throwable throwable = annotation.getStackTrace();
 326.                    if (throwable instanceof SAXParseException) {
 327.                        SAXParseException spe = (SAXParseException) throwable;
 328.                        throw new XmiException(spe.getMessage(), spe.getPublicId(),
 329.                                spe.getSystemId(), spe.getLineNumber(), 
 330.                                spe.getColumnNumber(), e);
 331.                    } else if (throwable instanceof SAXException) {
 332.                        SAXException se = (SAXException) throwable;
 333.                        Exception e1 = se.getException();
 334.                        if (e1 instanceof org.argouml.model.XmiReferenceRuntimeException) {
 335.                            String href = 
 336.                                ((org.argouml.model.XmiReferenceRuntimeException) e1)
 337.                                    .getReference();
 338.                            throw new org.argouml.model.XmiReferenceException(href,
 339.                                    e);
 340.                        }
 341.                        throw new XmiException(se.getMessage(), se);
 342.                    }
 343.                }
 344.                modelImpl.deleteExtent(extent);
 345.                throw new XmiException(e);
 346.            } catch (IOException e) {
 347.                try {
 348.                    modelImpl.deleteExtent(extent);
 349.                } catch (InvalidObjectException e2) {
 350.                    // Ignore if the extent never got created or has been deleted
 351.                }
 352.                throw new XmiException(e);
 353.            }
 354.    
 355.            return newElements;
 356.        }

 393.        /*
 394.         * @see org.argouml.model.XmiReader#getXMIUUIDToObjectMap()
 395.         */
 396.        public Map<String, Object> getXMIUUIDToObjectMap() {
 397.            if (resolver != null) {
 398.                // Give the resolver.getIdToObjectMap() entries 
 399.                // priority over entries with the same UUID from 
 400.                // resolver.getIdToObjectMaps() because entries
 401.                // in resolver.getIdToObjectMaps() are historic.
 402.                HashMap<String, Object> globalXmiIdToObjectMap = 
 403.                    new HashMap<String, Object>(resolver.getIdToObjectMap());
 404.    
 405.                Map<String, Map<String, Object>> idToObjectMaps = 
 406.                    resolver.getIdToObjectMaps();
 407.                Set<Entry<String,Map<String,Object>>> entrySet = null;
 408.                // I think that the synchronized access to idToObjectMaps is 
 409.                // required in order to respect the thread safe nature of the
 410.                // object.
 411.                // FIXME: maybe this should be moved into XmiReferenceResolverImpl,
 412.                // because it depends on internal implementation details of it.
 413.                synchronized (idToObjectMaps) {
 414.                    entrySet =
 415.                        new HashSet<Entry<String,Map<String,Object>>>(
 416.                                idToObjectMaps.entrySet());
 417.                    for (Entry<String, Map<String, Object>> entry : entrySet) {
 418.                        entry.setValue(new HashMap<String,Object>(entry.getValue()));
 419.                    }
 420.                }
 421.                for (Entry<String, Map<String, Object>> entry : entrySet) {
 422.                    String xmiIdPrefix =
 423.                        entry.getKey().startsWith(getTempXMIFileURIPrefix()) ? "" :

 441.        private static final String STYLE_PATH =
 442.            "/org/argouml/model/mdr/conversions/";
 443.    
 444.        /*
 445.         * A near clone of this code works fine outside of ArgoUML, but throws a
 446.         * null pointer exception during the transform when run within ArgoUML I
 447.         * think it's something to do with the class libraries being used, but I
 448.         * can't figure out what, so I've done a simpler, less efficient stepwise
 449.         * translation below in serialTransform
 450.         */
 451.        private InputSource chainedTransform(String[] styles, InputSource input)
 452.            throws XmiException {
 453.            SAXTransformerFactory stf =
 454.                (SAXTransformerFactory) TransformerFactory.newInstance();
 455.    
 456.            // TODO: Reconfigure exception handling to distinguish between errors
 457.            // that are possible due to bad input data and those that represent
 458.            // unexpected processing errors.
 459.            try {
 460.                // Set up reader to be first filter in chain
 461.                SAXParserFactory spf = SAXParserFactory.newInstance();
 462.                SAXParser parser = spf.newSAXParser();
 463.                XMLReader last = parser.getXMLReader();
 464.    
 465.                // Create filter for each style sheet and chain to previous
 466.                // filter/reader
 467.                for (int i = 0; i < styles.length; i++) {
 468.                    String xsltFileName = STYLE_PATH + styles[i];
 469.                    URL xsltUrl = getClass().getResource(xsltFileName);
 470.                    if (xsltUrl == null) {
 471.                        throw new IOException("Error opening XSLT style sheet : "
 472.                                + xsltFileName);
 473.                    }
 474.                    StreamSource xsltStreamSource =
 475.                        new StreamSource(xsltUrl.openStream());
 476.                    xsltStreamSource.setSystemId(xsltUrl.toExternalForm());
 477.                    XMLFilter filter = stf.newXMLFilter(xsltStreamSource);
 478.    
 479.                    filter.setParent(last);
 480.                    last = filter;
 481.                }
 482.    
 483.                SAXSource transformSource = new SAXSource(last, input);
 484.    
 485.                // Create temporary file for output
 486.                // TODO: we should be able to chain this directly to XMI reader
 487.                File tmpFile = File.createTempFile(TEMP_XMI_FILE_PREFIX, ".xmi");
 488.                tmpFile.deleteOnExit();
 489.                StreamResult result =
 490.                    new StreamResult(
 491.                        new FileOutputStream(tmpFile));
 492.    
 493.                Transformer transformer = stf.newTransformer();
 494.                transformer.transform(transformSource, result);
 495.    
 496.                return new InputSource(new FileInputStream(tmpFile));
 497.    
 498.            } catch (SAXException e) {
 499.                throw new XmiException(e);
 500.            } catch (ParserConfigurationException e) {
 501.                throw new XmiException(e);
 502.            } catch (IOException e) {
 503.                throw new XmiException(e);
 504.            } catch (TransformerConfigurationException e) {
 505.                throw new XmiException(e);
 506.            } catch (TransformerException e) {
 507.                throw new XmiException(e);
 508.            }
 509.    
 510.        }

 512.        private InputSource serialTransform(String[] styles, InputSource input)
 513.            throws UmlException {
 514.            SAXSource myInput = new SAXSource(input);
 515.            SAXTransformerFactory stf =
 516.                (SAXTransformerFactory) TransformerFactory.newInstance();
 517.            try {
 518.    
 519.                for (int i = 0; i < styles.length; i++) {
 520.                    // Set up source for style sheet
 521.                    String xsltFileName = STYLE_PATH + styles[i];
 522.                    if (LOG.isInfoEnabled()) {
 523.                        LOG.info("Transforming with " + xsltFileName);
 524.                    }
 525.                    URL xsltUrl = getClass().getResource(xsltFileName);
 526.                    if (xsltUrl == null) {
 527.                        throw new UmlException("Error opening XSLT style sheet : "
 528.                                + xsltFileName);
 529.                    }
 530.                    StreamSource xsltStreamSource =
 531.                        new StreamSource(xsltUrl.openStream());
 532.                    xsltStreamSource.setSystemId(xsltUrl.toExternalForm());
 533.    
 534.                    // Create & set up temporary output file
 535.                    File tmpOutFile = File.createTempFile(TEMP_XMI_FILE_PREFIX, ".xmi");
 536.                    tmpOutFile.deleteOnExit();
 537.                    StreamResult result =
 538.                        new StreamResult(new FileOutputStream(
 539.                            tmpOutFile));
 540.    
 541.                    // Create transformer and do transformation
 542.                    Transformer transformer = stf.newTransformer(xsltStreamSource);
 543.                    transformer.transform(myInput, result);
 544.    
 545.                    LOG.info("Wrote converted XMI file - " + tmpOutFile
 546.                            + " converted using : " + xsltFileName);
 547.    
 548.                    // Set up for next iteration
 549.                    myInput =
 550.                        new SAXSource(new InputSource(new FileInputStream(
 551.                            tmpOutFile)));
 552.                    myInput.setSystemId(tmpOutFile.toURI().toURL().toExternalForm());
 553.                }
 554.                return myInput.getInputSource();
 555.            } catch (IOException e) {
 556.                throw new UmlException(e);
 557.            } catch (TransformerConfigurationException e) {
 558.                throw new UmlException(e);
 559.            } catch (TransformerException e) {
 560.                throw new UmlException(e);
 561.            }
 562.    
 563.        }

 565.        private File copySource(InputSource input) throws IOException {
 566.            byte[] buf = new byte[2048];
 567.            int len;
 568.            
 569.            // Create & set up temporary output file
 570.            File tmpOutFile = File.createTempFile(TEMP_XMI_FILE_PREFIX, ".xmi");
 571.            tmpOutFile.deleteOnExit();
 572.            FileOutputStream out = new FileOutputStream(tmpOutFile);
 573.            
 574.            // TODO: Bob says - Coding by use of side effect here.
 575.            // Maybe this should be done in a clearer way but it fixes
 576.            // http://argouml.tigris.org/issues/show_bug.cgi?id=4978
 577.            // It seems that when loading an XMI that is not contained in a zip
 578.            // file then the InputStream given as the argument to this method
 579.            // can't be reused as it is at the end of the stream. In that case
 580.            // systemId appears to be none-null at this stage.
 581.            // So if systemId is not null we recreate the InputSource.
 582.            String systemId = input.getSystemId();
 583.            if (systemId != null) {
 584.                input = new InputSource(new URL(systemId).openStream());

 588.            
 589.            while ((len = in.read(buf)) >= 0) {
 590.                out.write(buf, 0, len);
 591.            }
 592.            out.close();
 593.    
 594.            LOG.debug("Wrote copied XMI file to " + tmpOutFile);
