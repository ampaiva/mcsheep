 243.    class MyLocale {
 244.        private Locale myLocale;
 245.    
 246.        /**
 247.         * The constructor.
 248.         *
 249.         * @param locale the Locale
 250.         */
 251.        MyLocale(Locale locale) {
 252.            myLocale = locale;
 253.        }
 254.    
 255.        /**
 256.         * @return returns the locale
 257.         */
 258.        Locale getLocale() {
 259.            return myLocale;
 260.        }
 261.    
 262.        /*
 263.         * @see java.lang.Object#toString()
 264.         */
 265.        public String toString() {
 266.            StringBuffer displayString = new StringBuffer(myLocale.toString());
 267.            displayString.append(" (");
 268.            displayString.append(myLocale.getDisplayLanguage(myLocale));
 269.            if (myLocale.getDisplayCountry(myLocale) != null 
 270.                    && myLocale.getDisplayCountry(myLocale).length() > 0) {
 271.                displayString.append(" ");
 272.                displayString.append(myLocale.getDisplayCountry(myLocale));
 273.            }
 274.            displayString.append(")");
 275.            if (myLocale.equals(Translator.getSystemDefaultLocale())) {
 276.                displayString.append(" - Default");
 277.            }
 278.            return displayString.toString();
 279.        }
 280.    
 281.        static Collection<MyLocale> getLocales() {
 282.            Collection<MyLocale> c = new ArrayList<MyLocale>();
 283.            for (Locale locale : Arrays.asList(Translator.getLocales())) {
 284.                c.add(new MyLocale(locale));
 285.            }
 286.            return c;
 287.        }
 288.    
 289.        static MyLocale getDefault(Collection<MyLocale> c) {
 290.            Locale locale = Locale.getDefault();
 291.            for (MyLocale ml : c) {
 292.                if (locale.equals(ml.getLocale())) {
 293.                    return ml;
 294.                }
 295.            }
 296.            return null;
 297.        }

 134.        /**
 135.         * Create a Property for the position of the given tab pane, selecting
 136.         * the current display value from the user properties file.
 137.         */
 138.        private Property createProperty(String text, String[] positions,
 139.                        Class tab) {
 140.            ConfigurationKey key = makeKey(tab);
 141.            String currentValue = Configuration.getString(key, "South");
 142.            return new Property(Translator.localize(text), String.class,
 143.                    currentValue, positions);
 144.        }

 146.        private void loadPosition(Property position, Class tab) {
 147.            ConfigurationKey key = makeKey(tab);
 148.            position.setCurrentValue(Configuration.getString(key, "South"));
 149.        }
 150.    
 151.        private void savePosition(Property position, Class tab) {
 152.            ConfigurationKey key = makeKey(tab);
 153.            Configuration.setString(key, position.getCurrentValue().toString());
 154.        }
 155.    
 156.        private ConfigurationKey makeKey(Class tab) {
 157.            String className = tab.getName();
 158.            String shortClassName =
 159.            className.substring(className.lastIndexOf('.') + 1).toLowerCase();
 160.            ConfigurationKey key = Configuration.makeKey("layout", shortClassName);
 161.            return key;
 162.        }
