  63.    /**
  64.     * <p>A class that facilitates access to the UML profile for C++ 
  65.     * (CppUmlProfile.xmi). 
  66.     * This is done by facilitating the use of the stereotypes, built-ins 
  67.     * and other model elements of the profile in the model which is being used.
  68.     * </p>
  69.     * <p>Both the generator and the importer must use the same profile, 
  70.     * if not we are going to make future RTE very difficult. 
  71.     * Also, the users of the module are going to be
  72.     * confused. 
  73.     * The main point in favor of this is that there is no open source UML
  74.     * profile for C++. 
  75.     * Lets be pioneers here ;-)
  76.     * </p>
  77.     * 
  78.     * @author Luis Sergio Oliveira (euluis)
  79.     * @since 0.25.4
  80.     * @see Issue 3771 (http://argouml.tigris.org/issues/show_bug.cgi?id=3771)
  81.     */

 120.        private void initBuiltInTypes() {
 121.            builtInTypes = new HashSet();
 122.            Collection allDataTypes = getCoreHelper().getAllDataTypes(profile);
 123.            for (Object dt : allDataTypes) {
 124.                builtInTypes.add(getFacade().getName(dt));
 125.            }
 126.        }

 128.        /**
 129.         * Checks if the given type is a C++ builtin type.
 130.         * 
 131.         * @param typeName name of the type to check
 132.         * @return true if typeName is a builtin type, false otherwise
 133.         */
 134.        public boolean isBuiltIn(String typeName) {
 135.            if (builtInTypes == null) {
 136.                initBuiltInTypes();
 137.            }
 138.            typeName = trimAndEnsureOneSpaceOnlyBetweenTokens(typeName);
 139.            return builtInTypes.contains(typeName);
 140.        }
 141.    
 142.        static String trimAndEnsureOneSpaceOnlyBetweenTokens(String typeName) {
 143.            return typeName.trim().replaceAll("\\s+", " ");
 144.        }
 145.    
 146.        /**
 147.         * Retrieves the given builtin type model element representation as a
 148.         * DataType.
 149.         * 
 150.         * @param typeName name of the type
 151.         * @return the model element that models the C++ builtin type
 152.         */
 153.        public Object getBuiltIn(String typeName) {
 154.            assert isBuiltIn(typeName) : "Must be a C++ built in!";
 155.            Object builtinType = findDataType(typeName, profile);
 156.            if (builtinType == null) {
 157.                builtinType = getCoreFactory().buildDataType(typeName, 
 158.                    models.iterator().next());
 159.            }
 160.            // copy the documentation from the profile if it exists
 161.            Object profileDT = findDataType(typeName, profile);
 162.            Object dtDocuTV = getFacade().getTaggedValue(profileDT, 
 163.                    TV_NAME_DOCUMENTATION);
 164.            if (dtDocuTV != null) {
 165.                Object tdDocumentation = getTagDefinition(TV_NAME_DOCUMENTATION);
 166.                Object modelDTDocuTV = getExtensionMechanismsFactory().
 167.                    buildTaggedValue(tdDocumentation, 
 168.                        new String[] {getFacade().getValueOfTag(dtDocuTV)});
 169.                getExtensionMechanismsHelper().addTaggedValue(builtinType, 
 170.                        modelDTDocuTV);
 171.            }
 172.            return builtinType;
 173.        }

 175.        public static Collection getModels() {
 176.            Collection models = new ArrayList();
 177.            for (Object rootModelElement : getFacade().getRootElements()) {
 178.                if (getFacade().isAModel(rootModelElement)) {
 179.                    models.add(rootModelElement);
 180.                }
 181.            }
 182.            return models;
 183.        }

 185.        public static Collection getEditableModels() {
 186.            Collection editableModels = new ArrayList();
 187.            for (Object rootModelElement : getFacade().getRootElements()) {
 188.                if (getFacade().isAModel(rootModelElement) && 
 189.                    !getModelManagementHelper().isReadOnly(rootModelElement)) {
 190.                    editableModels.add(rootModelElement);
 191.                }
 192.            }
 193.            return editableModels;
 194.        }

 196.        public static Object getTagDefinition(String tdName) {
 197.            // look for it in all the models
 198.            Collection models = getModels();
 199.            for (Object model : models) {
 200.                Collection tagDefinitions = getModelManagementHelper().
 201.                    getAllModelElementsOfKindWithModel(model, 
 202.                        Model.getMetaTypes().getTagDefinition());
 203.                for (Object td : tagDefinitions) {
 204.                    if (tdName.equals(getFacade().getName(td))) {
 205.                        return td;
 206.                    }
 207.                }
 208.            }
 209.            // create it in an editable model if one is found
 210.            Collection editableModels = getEditableModels();
 211.            Iterator it = editableModels.iterator();
 212.            if (it.hasNext()) {
 213.                return getExtensionMechanismsFactory().buildTagDefinition(tdName, 
 214.                    null, it.next());
 215.            }
 216.            throw new IllegalStateException("Unable to find an editable model.");
 217.        }
 218.    
 219.        static Object findDataType(String typeName, Object model2) {
 220.            Collection dataTypes = getCoreHelper().getAllDataTypes(model2);
 221.            for (Object dt : dataTypes) {
 222.                if (getFacade().getName(dt).equals(typeName)) {
 223.                    return dt;
 224.                }
 225.            }
 226.            return null;
 227.        }

 237.        protected BaseProfile(Collection<Object> theModels, Object profileModel) {

 241.            assert profileModel != null : 
 242.                "The profileModel must be non-null."; //$NON-NLS-1$
 243.            profile = profileModel;
 244.        }

 246.        /**
 247.         * @return the Collection containing the profile models.
 248.         */
 249.        static Collection loadProfileModels() {
 250.            ProfileModelLoader profileModelLoader = new ResourceModelLoader(
 251.                    BaseProfile.class);
 252.            Collection elements;
 253.            try {
 254.                elements = profileModelLoader.loadModel(PROFILE_REFERENCE);
 255.            } catch (ProfileException e) {
 256.                throw new RuntimeException(e);
 257.            }
 258.            return elements;
 259.        }

 261.        protected Object getCppStereotypeInModel(String stereotypeName) {
 262.            Object cppStereotype = null;
 263.            for (Object model : models) {
 264.                cppStereotype = getStereotype(model, stereotypeName);
 265.                if (cppStereotype != null) {
 266.                    break;
 267.                }
 268.            }
 269.            if (cppStereotype == null) {
 270.                cppStereotype = getStereotype(profile, stereotypeName);
 271.            }
 272.            return cppStereotype;
 273.        }
 274.    
 275.        private Object getStereotype(Object aModel, String stereotypeName) {
 276.            Collection stereotypes = getExtensionMechanismsHelper().getStereotypes(
 277.                    aModel);
 278.            for (Object stereotype : stereotypes) {
 279.                if (stereotypeName.equals(getFacade().getName(stereotype))) {
 280.                    return stereotype;
 281.                }
 282.            }
 283.            return null;
 284.        }
 285.    
 286.        protected Object getTagDefinition(String stereoName, String tdName) {
 287.            Object stereo = getCppStereotypeInModel(stereoName);
 288.            assertModelElementContainedInModels(stereo);
 289.            Collection tagDefinitions = getFacade().getTagDefinitions(stereo);
 290.            for (Object tagDefinition : tagDefinitions) {
 291.                if (tdName.equals(getFacade().getName(tagDefinition))) {
 292.                    return tagDefinition;
 293.                }
 294.            }
 295.            return null;
 296.        }

 298.        public void applyStereotype(String stereoName, Object modelElement) {
 299.            assertModelElementContainedInModels(modelElement);
 300.            Object stereo = getCppStereotypeInModel(stereoName);
 301.            getCoreHelper().addStereotype(modelElement, stereo);
 302.        }

 304.        @SuppressWarnings("serial")
 305.        private void assertModelElementContainedInModels(Object modelElement) {
 306.            boolean contained = false;
 307.            ArrayList<Object> modelsAndProfile = new ArrayList<Object>(models) {
 308.                { add(profile); }
 309.            };
 310.            for (Object model : modelsAndProfile) {
 311.                contained = model.equals(getFacade().getRoot(modelElement));
 312.                if (contained) {
 313.                    break;
 314.                }
 315.            }
 316.            assert contained : "model element (" + modelElement + ") not contained "
 317.                    + "in models.";
 318.        }

 328.            getExtensionMechanismsHelper().setDataValues(tv, new String[] {tvv});
 329.            getExtensionMechanismsHelper().addTaggedValue(me, tv);
 330.        }

  63.    /**
  64.     * <p>A class that facilitates access to the UML profile for C++ 
  65.     * (CppUmlProfile.xmi). 
  66.     * This is done by facilitating the use of the stereotypes, built-ins 
  67.     * and other model elements of the profile in the model which is being used.
  68.     * </p>
  69.     * <p>Both the generator and the importer must use the same profile, 
  70.     * if not we are going to make future RTE very difficult. 
  71.     * Also, the users of the module are going to be
  72.     * confused. 
  73.     * The main point in favor of this is that there is no open source UML
  74.     * profile for C++. 
  75.     * Lets be pioneers here ;-)
  76.     * </p>
  77.     * 
  78.     * @author Luis Sergio Oliveira (euluis)
  79.     * @since 0.25.4
  80.     * @see Issue 3771 (http://argouml.tigris.org/issues/show_bug.cgi?id=3771)
  81.     */
  82.    public class BaseProfile {
  83.    
  84.        private static final Logger LOG = Logger.getLogger(BaseProfile.class);
  85.    
  86.        /**
  87.         * The name of the documentation Tagged Value. 
  88.         */
  89.        public static final String TV_NAME_DOCUMENTATION = "documentation";
  90.    
  91.        static final String PROFILE_FILE_NAME = 
  92.            "/org/argouml/language/cpp/profile/CppUmlProfile.xmi";
  93.    
  94.        static {
  95.            URL profileURL = null;
  96.            try {
  97.                profileURL = new URL(
  98.                    "http://argouml-cpp.tigris.org/profile/CppUmlProfile.xmi");
  99.            } catch (MalformedURLException e) {
 100.                LOG.error("Exception", e);
 101.            }
 102.            PROFILE_REFERENCE = new ProfileReference(PROFILE_FILE_NAME, 
 103.                profileURL);
 104.        }
 105.        
 106.        static final ProfileReference PROFILE_REFERENCE;
 107.        
 108.        private Collection<Object> models;
 109.        
 110.        /**
 111.         * The C++ UML profile as loaded from CppUmlProfile.xmi.
 112.         */
 113.        private Object profile;
 114.    
 115.        /**
 116.         * Set of built in types.
 117.         */
 118.        private Set builtInTypes;
 119.    
 120.        private void initBuiltInTypes() {
 121.            builtInTypes = new HashSet();
 122.            Collection allDataTypes = getCoreHelper().getAllDataTypes(profile);
 123.            for (Object dt : allDataTypes) {
 124.                builtInTypes.add(getFacade().getName(dt));
 125.            }
 126.        }
 127.    
 128.        /**
 129.         * Checks if the given type is a C++ builtin type.
 130.         * 
 131.         * @param typeName name of the type to check
 132.         * @return true if typeName is a builtin type, false otherwise
 133.         */
 134.        public boolean isBuiltIn(String typeName) {
 135.            if (builtInTypes == null) {
 136.                initBuiltInTypes();
 137.            }
 138.            typeName = trimAndEnsureOneSpaceOnlyBetweenTokens(typeName);
 139.            return builtInTypes.contains(typeName);
 140.        }
 141.    
 142.        static String trimAndEnsureOneSpaceOnlyBetweenTokens(String typeName) {
 143.            return typeName.trim().replaceAll("\\s+", " ");
 144.        }
 145.    
 146.        /**
 147.         * Retrieves the given builtin type model element representation as a
 148.         * DataType.
 149.         * 
 150.         * @param typeName name of the type
 151.         * @return the model element that models the C++ builtin type
 152.         */
 153.        public Object getBuiltIn(String typeName) {
 154.            assert isBuiltIn(typeName) : "Must be a C++ built in!";
 155.            Object builtinType = findDataType(typeName, profile);
 156.            if (builtinType == null) {
 157.                builtinType = getCoreFactory().buildDataType(typeName, 
 158.                    models.iterator().next());
 159.            }
 160.            // copy the documentation from the profile if it exists
 161.            Object profileDT = findDataType(typeName, profile);
 162.            Object dtDocuTV = getFacade().getTaggedValue(profileDT, 
 163.                    TV_NAME_DOCUMENTATION);
 164.            if (dtDocuTV != null) {
 165.                Object tdDocumentation = getTagDefinition(TV_NAME_DOCUMENTATION);
 166.                Object modelDTDocuTV = getExtensionMechanismsFactory().
 167.                    buildTaggedValue(tdDocumentation, 
 168.                        new String[] {getFacade().getValueOfTag(dtDocuTV)});
 169.                getExtensionMechanismsHelper().addTaggedValue(builtinType, 
 170.                        modelDTDocuTV);
 171.            }
 172.            return builtinType;
 173.        }
 174.        
 175.        public static Collection getModels() {
 176.            Collection models = new ArrayList();
 177.            for (Object rootModelElement : getFacade().getRootElements()) {
 178.                if (getFacade().isAModel(rootModelElement)) {
 179.                    models.add(rootModelElement);
 180.                }
 181.            }
 182.            return models;
 183.        }
 184.        
 185.        public static Collection getEditableModels() {
 186.            Collection editableModels = new ArrayList();
 187.            for (Object rootModelElement : getFacade().getRootElements()) {
 188.                if (getFacade().isAModel(rootModelElement) && 
 189.                    !getModelManagementHelper().isReadOnly(rootModelElement)) {
 190.                    editableModels.add(rootModelElement);
 191.                }
 192.            }
 193.            return editableModels;
 194.        }
 195.    
 196.        public static Object getTagDefinition(String tdName) {
 197.            // look for it in all the models
 198.            Collection models = getModels();
 199.            for (Object model : models) {
 200.                Collection tagDefinitions = getModelManagementHelper().
 201.                    getAllModelElementsOfKindWithModel(model, 
 202.                        Model.getMetaTypes().getTagDefinition());
 203.                for (Object td : tagDefinitions) {
 204.                    if (tdName.equals(getFacade().getName(td))) {
 205.                        return td;
 206.                    }
 207.                }
 208.            }
 209.            // create it in an editable model if one is found
 210.            Collection editableModels = getEditableModels();
 211.            Iterator it = editableModels.iterator();
 212.            if (it.hasNext()) {
 213.                return getExtensionMechanismsFactory().buildTagDefinition(tdName, 
 214.                    null, it.next());
 215.            }
 216.            throw new IllegalStateException("Unable to find an editable model.");
 217.        }
 218.    
 219.        static Object findDataType(String typeName, Object model2) {
 220.            Collection dataTypes = getCoreHelper().getAllDataTypes(model2);
 221.            for (Object dt : dataTypes) {
 222.                if (getFacade().getName(dt).equals(typeName)) {
 223.                    return dt;
 224.                }
 225.            }
 226.            return null;
 227.        }
 228.    
 229.        Object getProfile() {
 230.            return profile;
 231.        }
 232.    
 233.        protected BaseProfile(Collection<Object> theModels) {
 234.            this(theModels,loadProfileModels().iterator().next());
 235.        }
 236.    
 237.        protected BaseProfile(Collection<Object> theModels, Object profileModel) {
 238.            models = theModels;
 239.            assert models.size() > 0 : 
 240.                "There must be at least one user model."; //$NON-NLS-1$
 241.            assert profileModel != null : 
 242.                "The profileModel must be non-null."; //$NON-NLS-1$
 243.            profile = profileModel;
 244.        }
 245.    
 246.        /**
 247.         * @return the Collection containing the profile models.
 248.         */
 249.        static Collection loadProfileModels() {
 250.            ProfileModelLoader profileModelLoader = new ResourceModelLoader(
 251.                    BaseProfile.class);
 252.            Collection elements;
 253.            try {
 254.                elements = profileModelLoader.loadModel(PROFILE_REFERENCE);
 255.            } catch (ProfileException e) {
 256.                throw new RuntimeException(e);
 257.            }
 258.            return elements;
 259.        }
 260.    
 261.        protected Object getCppStereotypeInModel(String stereotypeName) {
 262.            Object cppStereotype = null;
 263.            for (Object model : models) {
 264.                cppStereotype = getStereotype(model, stereotypeName);
 265.                if (cppStereotype != null) {
 266.                    break;
 267.                }
 268.            }
 269.            if (cppStereotype == null) {
 270.                cppStereotype = getStereotype(profile, stereotypeName);
 271.            }
 272.            return cppStereotype;
 273.        }
 274.    
 275.        private Object getStereotype(Object aModel, String stereotypeName) {
 276.            Collection stereotypes = getExtensionMechanismsHelper().getStereotypes(
 277.                    aModel);
 278.            for (Object stereotype : stereotypes) {
 279.                if (stereotypeName.equals(getFacade().getName(stereotype))) {
 280.                    return stereotype;
 281.                }
 282.            }
 283.            return null;
 284.        }
 285.    
 286.        protected Object getTagDefinition(String stereoName, String tdName) {
 287.            Object stereo = getCppStereotypeInModel(stereoName);
 288.            assertModelElementContainedInModels(stereo);
 289.            Collection tagDefinitions = getFacade().getTagDefinitions(stereo);
 290.            for (Object tagDefinition : tagDefinitions) {
 291.                if (tdName.equals(getFacade().getName(tagDefinition))) {
 292.                    return tagDefinition;
 293.                }
 294.            }
 295.            return null;
 296.        }
 297.    
 298.        public void applyStereotype(String stereoName, Object modelElement) {
 299.            assertModelElementContainedInModels(modelElement);
 300.            Object stereo = getCppStereotypeInModel(stereoName);
 301.            getCoreHelper().addStereotype(modelElement, stereo);
 302.        }
 303.    
 304.        @SuppressWarnings("serial")
 305.        private void assertModelElementContainedInModels(Object modelElement) {
 306.            boolean contained = false;
 307.            ArrayList<Object> modelsAndProfile = new ArrayList<Object>(models) {
 308.                { add(profile); }
 309.            };
 310.            for (Object model : modelsAndProfile) {
 311.                contained = model.equals(getFacade().getRoot(modelElement));
 312.                if (contained) {
 313.                    break;
 314.                }
 315.            }
 316.            assert contained : "model element (" + modelElement + ") not contained "
 317.                    + "in models.";
 318.        }
 319.    
 320.        public void applyTaggedValue(String stereoName, String tdName, 
 321.                Object me, String tvv) {
 322.            assertModelElementContainedInModels(me);
 323.            assert getFacade().getStereotypes(me).contains(
 324.                    getCppStereotypeInModel(stereoName));
 325.            Object td = getTagDefinition(stereoName, tdName);
 326.            Object tv = getExtensionMechanismsFactory().createTaggedValue();
 327.            getExtensionMechanismsHelper().setType(tv, td);
 328.            getExtensionMechanismsHelper().setDataValues(tv, new String[] {tvv});
 329.            getExtensionMechanismsHelper().addTaggedValue(me, tv);
 330.        }
 331.        
 332.        /**
 333.         * Applies a newlly created tagged value with value tvv related to the
 334.         * tag definition with name tdName to the given model element me.
 335.         *
 336.         * @param tdName the name of the tag definition for which to create a
 337.         * tagged value
 338.         * @param me the model element to which to apply the tagged value that will
 339.         * be created
 340.         * @param tvv the value that will be set in the tagged value
 341.         */
 342.        public void applyTaggedValue(String tdName, Object me, String tvv) {
 343.            assertModelElementContainedInModels(me);
 344.            Object td = getTagDefinition(tdName);
 345.            Object tv = getExtensionMechanismsFactory().createTaggedValue();
 346.            getExtensionMechanismsHelper().setType(tv, td);
 347.            getExtensionMechanismsHelper().setDataValues(tv, new String[] {tvv});
 348.            getExtensionMechanismsHelper().addTaggedValue(me, tv);
 349.        }
