 400.         * Load all the existing rules from the perspective manager
 401.         * for presentation. These will be presented as the library of rules
 402.         * the user may pick from.
 403.         */
 404.        private void loadLibrary() {
 405.            List<PerspectiveRule> rulesLib = new ArrayList<PerspectiveRule>();
 406.            rulesLib.addAll(PerspectiveManager.getInstance().getRules());
 407.            Collections.sort(rulesLib, new Comparator<PerspectiveRule>() {
 408.                public int compare(PerspectiveRule o1, PerspectiveRule o2) {
 409.                    return o1.toString().compareTo(o2.toString());
 410.                }
 411.            });
 412.            // remove the ones already selected (if a perspective is selected)
 413.            ExplorerPerspective selPers =
 414.                (ExplorerPerspective) perspectiveList.getSelectedValue();
 415.            if (selPers != null) {
 416.                for (PerspectiveRule persRule : selPers.getList()) {
 417.                    for (PerspectiveRule libRule : rulesLib) {
 418.                        if (libRule.toString().equals(persRule.toString())) {
 419.                            rulesLib.remove(libRule);
 420.                            break;
 421.                        }
 422.                    }
 423.                }
 424.            }
 425.            // add them
 426.            ruleLibraryListModel.clear();
 427.            for (PerspectiveRule rule : rulesLib) {
 428.                ruleLibraryListModel.addElement(rule);

 655.                loadLibrary();

 304.
