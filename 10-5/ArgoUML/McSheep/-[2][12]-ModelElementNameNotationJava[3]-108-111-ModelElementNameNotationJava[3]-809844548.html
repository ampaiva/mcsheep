  54.    /**
  55.     * The UML notation for an AssociationRole.
  56.     * 
  57.     * @author michiel
  58.     */
  59.    public class AssociationRoleNotationUml extends AssociationRoleNotation {
  60.    
  61.        /**
  62.         * The constructor.
  63.         * 
  64.         * @param role the given association-role
  65.         */
  66.        public AssociationRoleNotationUml(Object role) {
  67.            super(role);
  68.        }
  69.    
  70.        /*
  71.         * @see org.argouml.notation.providers.NotationProvider#getParsingHelp()
  72.         */
  73.        public String getParsingHelp() {
  74.            return "parsing.help.fig-association-role";
  75.        }
  76.    
  77.        /*
  78.         * @see org.argouml.notation.providers.NotationProvider#parse(java.lang.Object, java.lang.String...
  79.         */
  80.        public void parse(Object modelElement, String text) {
  81.            try {
  82.                parseRole(modelElement, text);
  83.            } catch (ParseException pe) {
  84.                String msg = "statusmsg.bar.error.parsing.association-role";
  85.                Object[] args = {
  86.                    pe.getLocalizedMessage(),
  87.                    Integer.valueOf(pe.getErrorOffset()),
  88.                };
  89.                ArgoEventPump.fireEvent(new ArgoHelpEvent(
  90.                        ArgoEventTypes.HELP_CHANGED, this,
  91.                    Translator.messageFormat(msg, args)));
  92.            }
  93.        }
  94.    
  95.        /**
  96.         * Parse the string that represents an AssociationRole: <pre>
  97.         *     ["/" name] [":" name_of_the_base_association]
  98.         * </pre>
  99.         * 
 100.         * @param role   The AssociationRole <em>text</em> describes.
 101.         * @param text A String on the above format.
 102.         * @throws ParseException
 103.         *             when it detects an error in the role string. See also
 104.         *             ParseError.getErrorOffset().
 105.         */
 106.        protected void parseRole(Object role, String text)
 107.            throws ParseException {
 108.            String token;
 109.            boolean hasColon = false;
 110.            boolean hasSlash = false;
 111.            String rolestr = null;
 112.            String basestr = null;
 113.            
 114.            MyTokenizer st = new MyTokenizer(text, " ,\t,/,:");
 115.    
 116.            while (st.hasMoreTokens()) {
 117.                token = st.nextToken();
 118.                if (" ".equals(token) || "\t".equals(token)) {
 119.                    /* Do nothing. */
 120.                } else if ("/".equals(token)) {
 121.                    hasSlash = true;
 122.                    hasColon = false;
 123.    
 124.                } else if (":".equals(token)) {
 125.                    hasColon = true;
 126.                    hasSlash = false;
 127.                    
 128.                } else if (hasColon) {
 129.                    if (basestr != null) {
 130.                        String msg = 
 131.                            "parsing.error.association-role.association-extra-text";
 132.                        throw new ParseException(Translator.localize(msg), st
 133.                                        .getTokenIndex());
 134.                    }
 135.                    basestr = token;
 136.                } else if (hasSlash) {
 137.                    if (rolestr != null) {
 138.                        String msg = 
 139.                            "parsing.error.association-role.association-extra-text";
 140.                        throw new ParseException(Translator.localize(msg), st
 141.                                        .getTokenIndex());
 142.                    }
 143.                    rolestr = token;
 144.                } else {
 145.                    String msg = 
 146.                        "parsing.error.association-role.association-extra-text";
 147.                    throw new ParseException(Translator.localize(msg), 
 148.                            st.getTokenIndex());
 149.                }
 150.            }
 151.            
 152.            if (basestr == null) {
 153.                /* If no base was typed, then only set the name: */
 154.                if (rolestr != null) {
 155.                    Model.getCoreHelper().setName(role, rolestr.trim());
 156.                }
 157.                return;
 158.            }
 159.            /* If the base was not changed, then only set the name: */
 160.            Object currentBase = Model.getFacade().getBase(role);
 161.            if (currentBase != null) {
 162.                String currentBaseStr = Model.getFacade().getName(currentBase);
 163.                if (currentBaseStr == null) {
 164.                    /* TODO: Is this needed? */
 165.                    currentBaseStr = "";
 166.                }
 167.                if (currentBaseStr.equals(basestr)) {
 168.                    if (rolestr != null) {
 169.                        Model.getCoreHelper().setName(role, rolestr.trim());
 170.                    }
 171.                    return;
 172.                }
 173.            }
 174.            Collection c = 
 175.                Model.getCollaborationsHelper().getAllPossibleBases(role);
 176.            Iterator i = c.iterator();
 177.            while (i.hasNext()) {
 178.                Object candidate = i.next();
 179.                if (basestr.equals(Model.getFacade().getName(candidate))) {
 180.                    if (Model.getFacade().getBase(role) != candidate) {
 181.                        /* If the base is already set to this assoc, 
 182.                         * then do not set it again.
 183.                         * This check is needed, otherwise the setbase()
 184.                         *  below gives an exception.*/
 185.                        Model.getCollaborationsHelper().setBase(role, candidate);
 186.                    }
 187.                    /* Only set the name if the base was found: */
 188.                    if (rolestr != null) {
 189.                        Model.getCoreHelper().setName(role, rolestr.trim());
 190.                    }
 191.                    return;
 192.                }
 193.            }
 194.            String msg = "parsing.error.association-role.base-not-found";
 195.            throw new ParseException(Translator.localize(msg), 0);        
 196.        }
 197.    
 198.        private String toString(final Object modelElement) {
 199.            //get the associationRole name
 200.            String name = Model.getFacade().getName(modelElement);
 201.            if (name == null) {
 202.                name = "";
 203.            }
 204.            if (name.length() > 0) {
 205.                name = "/" + name;
 206.            }
 207.            //get the base association name
 208.            Object assoc = Model.getFacade().getBase(modelElement);
 209.            if (assoc != null) {
 210.                String baseName = Model.getFacade().getName(assoc);
 211.                if (baseName != null && baseName.length() > 0) {
 212.                    name = name + ":" + baseName;
 213.                }
 214.            }
 215.            return name;
 216.        }
 217.    
 218.        /*
 219.         * Generate the name of an association role of the form:
 220.         *  ["/" name] [":" name_of_the_base_association]
 221.         * <p>
 222.         * Remark: 
 223.         * So, if both names are empty, then nothing is shown! 
 224.         * See issue 2712.
 225.         */
 226.        @Override
 227.        public String toString(final Object modelElement, 
 228.                final NotationSettings settings) {
 229.            return toString(modelElement);
 230.        }
