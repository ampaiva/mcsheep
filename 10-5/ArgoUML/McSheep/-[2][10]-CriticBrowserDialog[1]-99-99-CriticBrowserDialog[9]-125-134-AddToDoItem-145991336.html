 323.                enableFieldsAndButtons();
 324.                className.setText("");
 325.                headline.setText("");
 326.                priority.setSelectedItem(null);
 327.                priority.repaint();
 328.                moreInfo.setText("");
 329.                desc.setText("");
 330.                return;
 331.            }
 332.            updateButtonsEnabled();
 333.        className.setText(cr.getClass().getName());
 334.        headline.setText(cr.getHeadline());
 335.    
 336.        int p = cr.getPriority();
 337.        if (p == ToDoItem.HIGH_PRIORITY) {
 338.            priority.setSelectedItem(HIGH);
 339.        } else if (p == ToDoItem.MED_PRIORITY) {
 340.            priority.setSelectedItem(MEDIUM);
 341.        } else {
 342.            priority.setSelectedItem(LOW);
 343.        }
 344.        priority.repaint();
 345.    
 346.        moreInfo.setText(cr.getMoreInfoURL());
 347.        desc.setText(cr.getDescriptionTemplate());
 348.        desc.setCaretPosition(0);
 349.        useClar.setSelectedItem(ALWAYS);
 350.        useClar.repaint();
 351.        }

 358.            this.goButton.setEnabled(this.target != null 
 359.                    && this.target.getMoreInfoURL() != null 
 360.                    && this.target.getMoreInfoURL().length() > 0);
 361.            this.networkButton.setEnabled(false);
 362.            this.wakeButton.setEnabled(this.target != null
 363.                                   && (this.target.isSnoozed() 
 364.                                           || !this.target.isEnabled()));
 365.        }

 373.        }

 375.        private void setTargetPriority() {
 376.        if (target == null) {
 377.            return;
 378.        }
 379.        String p = (String) priority.getSelectedItem();
 380.        if (p == null) {
 381.            return;
 382.        }
 383.        if (p.equals(PRIORITIES[0])) {
 384.            target.setPriority(ToDoItem.HIGH_PRIORITY);
 385.        }
 386.        if (p.equals(PRIORITIES[1])) {
 387.            target.setPriority(ToDoItem.MED_PRIORITY);
 388.        }
 389.        if (p.equals(PRIORITIES[2])) {
 390.            target.setPriority(ToDoItem.LOW_PRIORITY);
 391.        }
 392.        }

 451.        public void valueChanged(ListSelectionEvent lse) {
 452.        if (lse.getValueIsAdjusting()) {
 453.            return;
 454.        }
 455.        Object src = lse.getSource();
 456.        if (src != table.getSelectionModel()) {
 457.            LOG.debug("src = " + src);
 458.            return;
 459.        }
 460.        LOG.debug("got valueChanged from " + src);
 461.        int row = table.getSelectedRow();
 462.            if (this.target != null) {
 463.                this.target.deleteObserver(this);
 464.            }
 465.        setTarget((row == -1) ? null : table.getCriticAtRow(row));
 466.            if (this.target != null) {
 467.                this.target.addObserver(this);
 468.            }
 469.        }

 173.        private void doAdd() {
 174.            Designer designer = Designer.theDesigner();
 175.            String headline = headLineTextField.getText();
 176.            int priority = ToDoItem.HIGH_PRIORITY;
 177.            switch (priorityComboBox.getSelectedIndex()) {
 178.        case 0:
 179.            priority = ToDoItem.HIGH_PRIORITY;
 180.            break;
 181.        case 1:
 182.            priority = ToDoItem.MED_PRIORITY;
 183.            break;
 184.        case 2:
 185.            priority = ToDoItem.LOW_PRIORITY;
 186.            break;
 187.            }
 188.            String desc = descriptionTextArea.getText();
 189.            String moreInfoURL = moreinfoTextField.getText();
 190.            ListSet newOffenders = new ListSet();
 191.            for (int i = 0; i < offenderList.getModel().getSize(); i++) {
 192.                newOffenders.add(offenderList.getModel().getElementAt(i));
 193.            }
 194.            ToDoItem item =
 195.                new UMLToDoItem(designer, headline, priority, 
 196.                        desc, moreInfoURL, newOffenders);
 197.            designer.getToDoList().addElement(item); //? inform()
 198.            Designer.firePropertyChange(Designer.MODEL_TODOITEM_ADDED, null, item);
 199.        }
