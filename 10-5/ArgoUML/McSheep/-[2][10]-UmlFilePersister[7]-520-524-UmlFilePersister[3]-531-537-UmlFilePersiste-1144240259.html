 647.        /**
 648.         * Class to filter XML declaration and DOCTYPE declaration from an output
 649.         * stream to allow use as nested XML files.
 650.         * 
 651.         * @author Tom Morris
 652.         */
 653.        class XmlFilterOutputStream extends FilterOutputStream {
 654.    
 655.            private CharsetDecoder decoder;
 656.    
 657.            private boolean headerProcessed = false;
 658.    
 659.            private static final int BUFFER_SIZE = 120;
 660.    
 661.            /**
 662.             * The following three fields make up a doubly mapped buffer with two
 663.             * sets of pointers (the ByteBuffer objects), one for input and one for
 664.             * output. Bytes are written into the buffer using outBB which advances
 665.             * the current position. The following ASCII art shows what the buffer
 666.             * looks like.<code>
 667.             *           vp       vl
 668.             * xxxxxx++++..........
 669.             *       ^p  ^l
 670.             * </code> vp & vl are the output (buffer writing) position and limit,
 671.             * respectively and show the free space that can receive bytes. The ^p
 672.             * and ^l pointers show the input (buffer reading) position and limit.
 673.             * They point to valid bytes which have not yet been converted to
 674.             * characters. The xxx bytes have been both written in and read back
 675.             * out. The +++ bytes have been written into the buffer, but not read
 676.             * out yet.
 677.             */
 678.            private byte[] bytes = new byte[BUFFER_SIZE * 2];
 679.    
 680.            private ByteBuffer outBB = ByteBuffer.wrap(bytes);
 681.    
 682.            private ByteBuffer inBB = ByteBuffer.wrap(bytes);
 683.    
 684.            // Buffer containing characters which have been decoded from the bytes
 685.            // in inBB.
 686.            private CharBuffer outCB = CharBuffer.allocate(BUFFER_SIZE);
 687.    
 688.            // RegEx pattern for XML declaration and, optionally, DOCTYPE
 689.            // Backslashes are doubled up - one for Java, one for Regex
 690.            private final Pattern xmlDeclarationPattern = Pattern
 691.                    .compile("\\s*<\\?xml.*\\?>\\s*(<!DOCTYPE.*>\\s*)?");
 692.    
 693.            /**
 694.             * Construct a filtered output stream using the given character set
 695.             * name.
 696.             * 
 697.             * @param outputStream source output stream to filter
 698.             * @param charsetName name of character set to use for encoding
 699.             */
 700.            public XmlFilterOutputStream(OutputStream outputStream,
 701.                    String charsetName) {
 702.                this(outputStream, Charset.forName(charsetName));
 703.            }
 704.    
 705.            /**
 706.             * Construct a filtered output stream using the given character set.
 707.             * 
 708.             * @param outputStream source output stream to filter
 709.             * @param charset character set to use for encoding
 710.             */
 711.            public XmlFilterOutputStream(OutputStream outputStream, Charset charset) {
 712.                super(outputStream);
 713.                decoder = charset.newDecoder();
 714.                decoder.onMalformedInput(CodingErrorAction.REPORT);
 715.                decoder.onUnmappableCharacter(CodingErrorAction.REPORT);
 716.                startEntry();
 717.            }
 718.    
 719.            /**
 720.             * Reset processing for the beginning of an entry.
 721.             */
 722.            public void startEntry() {
 723.                headerProcessed = false;
 724.                resetBuffers();
 725.            }
 726.    
 727.            private void resetBuffers() {
 728.                inBB.limit(0);
 729.                outBB.position(0);
 730.                outCB.position(0);
 731.            }
 732.    
 733.            @Override
 734.            public void write(byte[] b, int off, int len) throws IOException {
 735.                if ((off | len | (b.length - (len + off)) | (off + len)) < 0) {
 736.                    throw new IndexOutOfBoundsException();
 737.                }
 738.    
 739.                if (headerProcessed) {
 740.                    out.write(b, off, len);
 741.                } else {
 742.                    // TODO: Make this more efficient for large I/Os
 743.                    for (int i = 0; i < len; i++) {
 744.                        write(b[off + i]);
 745.                    }
 746.                }
 747.    
 748.            }
 749.    
 750.            @Override
 751.            public void write(int b) throws IOException {
 752.    
 753.                if (headerProcessed) {
 754.                    out.write(b);
 755.                } else {
 756.                    outBB.put((byte) b);
 757.                    inBB.limit(outBB.position());
 758.                    // Convert from bytes back to characters
 759.                    CoderResult result = decoder.decode(inBB, outCB, false);
 760.                    if (result.isError()) {
 761.                        throw new RuntimeException(
 762.                                "Unknown character decoding error");
 763.                    }
 764.    
 765.                    // This will have problems if the smallest possible
 766.                    // data segment is smaller than the size of the buffer
 767.                    // needed for regex matching
 768.                    if (outCB.position() == outCB.limit()) {
 769.                        processHeader();
 770.                    }
 771.    
 772.                }
 773.            }
 774.    
 775.            private void processHeader() throws IOException {
 776.                headerProcessed = true;
 777.                outCB.position(0); // rewind our character buffer
 778.    
 779.                Matcher matcher = xmlDeclarationPattern.matcher(outCB);
 780.                // Remove anything that matches our pattern
 781.                String headerRemainder = matcher.replaceAll("");
 782.                int index = headerRemainder.length() - 1;
 783.                if (headerRemainder.charAt(index) == '\0') {
 784.                    // Remove null characters at the end
 785.                    do {
 786.                        index--;
 787.                    } while (index >= 0 && headerRemainder.charAt(index) == '\0');
 788.                    headerRemainder = headerRemainder.substring(0, index + 1);
 789.                }
 790.    
 791.                // Reencode the remaining characters as bytes again
 792.                ByteBuffer bb = decoder.charset().encode(headerRemainder);
 793.    
 794.                // and write them to our output stream
 795.                byte[] outBytes = new byte[bb.limit()];
 796.                bb.get(outBytes);
 797.                out.write(outBytes, 0, outBytes.length);
 798.    
 799.                // Write any left over bytes in the input buffer
 800.                // (perhaps from a partially decoded character)
 801.                if (inBB.remaining() > 0) {
 802.                    out.write(inBB.array(), inBB.position(), inBB.remaining());
 803.                    inBB.position(0);
 804.                    inBB.limit(0);
 805.                }
 806.            }
 807.    
 808.            /**
 809.             * This method has no effect to keep sub-writers from closing it
 810.             * accidently. The master can use the method {@link #reallyClose()} to
 811.             * actually close the underlying stream.
 812.             */
 813.            @Override
 814.            public void close() throws IOException {
 815.                flush();
 816.            }
 817.    
 818.            /**
 819.             * Close the stream.
 820.             * 
 821.             * @throws IOException
 822.             */
 823.            public void reallyClose() throws IOException {
 824.                out.close();
 825.            }
 826.    
 827.            /**
 828.             * Flush the stream. If the stream is flushed before the header is
 829.             * completely processed, whatever has been written so far will get
 830.             * processed before the flush.
 831.             */
 832.            @Override
 833.            public void flush() throws IOException {
 834.                if (!headerProcessed) {
 835.                    processHeader();
 836.                }
 837.                out.flush();
 838.            }
 839.    
 840.        }

 140.         * It is being considered to save out individual xmi's from individuals
 141.         * diagrams to make it easier to modularize the output of Argo.
 142.         * 
 143.         * @param file The file to write.
 144.         * @param project the project to save
 145.         * @throws SaveException when anything goes wrong
 146.         * @throws InterruptedException if the thread is interrupted
 147.         * 
 148.         * @see org.argouml.persistence.ProjectFilePersister#save(org.argouml.kernel.Project,
 149.         *      java.io.File)
 150.         */
 151.        public void doSave(Project project, File file) throws SaveException,
 152.            InterruptedException {
 153.    
 154.            ProgressMgr progressMgr = new ProgressMgr();
 155.            progressMgr.setNumberOfPhases(4);
 156.            progressMgr.nextPhase();
 157.    
 158.            File lastArchiveFile = new File(file.getAbsolutePath() + "~");
 159.            File tempFile = null;
 160.    
 161.            try {
 162.                tempFile = createTempFile(file);
 163.            } catch (FileNotFoundException e) {
 164.                throw new SaveException(
 165.                        "Failed to archive the previous file version", e);
 166.            } catch (IOException e) {
 167.                throw new SaveException(
 168.                        "Failed to archive the previous file version", e);
 169.            }
 170.    
 171.            try {
 172.                project.setFile(file);
 173.                project.setVersion(ApplicationVersion.getVersion());
 174.                project.setPersistenceVersion(PERSISTENCE_VERSION);
 175.    
 176.                OutputStream stream = new FileOutputStream(file);
 177.    
 178.                writeProject(project, stream, progressMgr);
 179.    
 180.                stream.close();
 181.    
 182.                progressMgr.nextPhase();
 183.    
 184.                String path = file.getParent();
 185.                if (LOG.isInfoEnabled()) {
 186.                    LOG.info("Dir ==" + path);
 187.                }
 188.    
 189.                // if save did not raise an exception
 190.                // and name+"#" exists move name+"#" to name+"~"
 191.                // this is the correct backup file
 192.                if (lastArchiveFile.exists()) {
 193.                    lastArchiveFile.delete();
 194.                }
 195.                if (tempFile.exists() && !lastArchiveFile.exists()) {
 196.                    tempFile.renameTo(lastArchiveFile);
 197.                }
 198.                if (tempFile.exists()) {
 199.                    tempFile.delete();
 200.                }
 201.    
 202.                progressMgr.nextPhase();
 203.    
 204.            } catch (Exception e) {
 205.                LOG.error("Exception occured during save attempt", e);
 206.    
 207.                // frank: in case of exception
 208.                // delete name and mv name+"#" back to name if name+"#" exists
 209.                // this is the "rollback" to old file
 210.                file.delete();
 211.                tempFile.renameTo(file);
 212.                if (e instanceof InterruptedException) {
 213.                    throw (InterruptedException) e;
 214.                } else {
 215.                    // we have to give a message to user and set the system
 216.                    // to unsaved!
 217.                    throw new SaveException(e);
 218.                }
 219.            }
 220.        }

 232.        /**
 233.         * Write the output for a project on the given stream.
 234.         * 
 235.         * @param project The project to output.
 236.         * @param stream The stream to write to.
 237.         * @throws SaveException If something goes wrong.
 238.         * @throws InterruptedException if the thread is interrupted
 239.         */
 240.        void writeProject(Project project, OutputStream oStream,
 241.                ProgressMgr progressMgr) throws SaveException, InterruptedException {
 242.            OutputStreamWriter outputStreamWriter;
 243.            try {
 244.                outputStreamWriter = new OutputStreamWriter(oStream, Argo
 245.                        .getEncoding());
 246.            } catch (UnsupportedEncodingException e) {
 247.                throw new SaveException(e);
 248.            }
 249.            PrintWriter writer = new PrintWriter(new BufferedWriter(
 250.                    outputStreamWriter));
 251.    
 252.            XmlFilterOutputStream filteredStream = new XmlFilterOutputStream(
 253.                    oStream, Argo.getEncoding());
 254.            try {
 255.                writer.println("<?xml version = \"1.0\" " + "encoding = \""
 256.                        + Argo.getEncoding() + "\" ?>");
 257.                writer.println("<uml version=\"" + PERSISTENCE_VERSION + "\">");
 258.                // Write out header section
 259.                try {
 260.                    Hashtable templates = TemplateReader.getInstance().read(
 261.                            ARGO_TEE);
 262.                    OCLExpander expander = new OCLExpander(templates);
 263.                    expander.expand(writer, project, "  ");
 264.                } catch (ExpansionException e) {
 265.                    throw new SaveException(e);
 266.                }
 267.                writer.flush();
 268.    
 269.                if (progressMgr != null) {
 270.                    progressMgr.nextPhase();
 271.                }
 272.    
 273.                // Note we assume members are ordered correctly already
 274.                for (ProjectMember projectMember : project.getMembers()) {
 275.                    if (LOG.isInfoEnabled()) {
 276.                        LOG.info("Saving member : " + projectMember);
 277.                    }
 278.                    MemberFilePersister persister = getMemberFilePersister(projectMember);
 279.                    filteredStream.startEntry();
 280.                    persister.save(projectMember, filteredStream);
 281.                    try {
 282.                        filteredStream.flush();
 283.                    } catch (IOException e) {
 284.                        throw new SaveException(e);
 285.                    }
 286.                }
 287.    
 288.                writer.println("</uml>");
 289.    
 290.                writer.flush();
 291.            } finally {
 292.                writer.close();
 293.                try {
 294.                    filteredStream.reallyClose();
 295.                } catch (IOException e) {
 296.                    throw new SaveException(e);
 297.                }
 298.            }
 299.        }

 313.        protected Project doLoad(File originalFile, File file,
 314.                ProgressMgr progressMgr) throws OpenException, InterruptedException {
 315.    
 316.            XmlInputStream inputStream = null;
 317.            try {
 318.                Project p = ProjectFactory.getInstance()
 319.                        .createProject(file.toURI());
 320.    
 321.                // Run through any stylesheet upgrades
 322.                int fileVersion = getPersistenceVersionFromFile(file);
 323.    
 324.                LOG.info("Loading uml file of version " + fileVersion);
 325.                if (!checkVersion(fileVersion, getReleaseVersionFromFile(file))) {
 326.                    // If we're about to upgrade the file lets take an archive
 327.                    // of it first.
 328.                    String release = getReleaseVersionFromFile(file);
 329.                    copyFile(originalFile, new File(originalFile.getAbsolutePath()
 330.                            + '~' + release));
 331.    
 332.                    progressMgr.setNumberOfPhases(progressMgr.getNumberOfPhases()
 333.                            + (PERSISTENCE_VERSION - fileVersion));
 334.    
 335.                    while (fileVersion < PERSISTENCE_VERSION) {
 336.                        ++fileVersion;
 337.                        LOG.info("Upgrading to version " + fileVersion);
 338.                        long startTime = System.currentTimeMillis();
 339.                        file = transform(file, fileVersion);
 340.                        long endTime = System.currentTimeMillis();
 341.                        LOG.info("Upgrading took " + ((endTime - startTime) / 1000)
 342.                                + " seconds");
 343.                        progressMgr.nextPhase();
 344.                    }
 345.                }
 346.    
 347.                progressMgr.nextPhase();
 348.    
 349.                inputStream = new XmlInputStream(file.toURI().toURL().openStream(),
 350.                        "argo", file.length(), 100000);
 351.    
 352.                ArgoParser parser = new ArgoParser();
 353.                Reader reader = new InputStreamReader(inputStream, Argo
 354.                        .getEncoding());
 355.                parser.readProject(p, reader);
 356.    
 357.                List memberList = parser.getMemberList();
 358.    
 359.                LOG.info(memberList.size() + " members");
 360.    
 361.                for (int i = 0; i < memberList.size(); ++i) {
 362.                    MemberFilePersister persister = getMemberFilePersister((String) memberList
 363.                            .get(i));
 364.                    LOG.info("Loading member with "
 365.                            + persister.getClass().getName());
 366.                    inputStream.reopen(persister.getMainTag());
 367.                    // TODO: Do we need to set the input encoding here? It was
 368.                    // done for ToDo parsing, but none of the other member types
 369.                    // InputSource inputSource = new InputSource(
 370.                    // new InputStreamReader(inputStream, Argo
 371.                    // .getEncoding()));
 372.                    InputSource inputSource = new InputSource(inputStream);
 373.                    // Don't use systemId here or it will get opened in preference
 374.                    // to inputStream.
 375.                    inputSource.setPublicId(originalFile.toURI().toURL()
 376.                            .toExternalForm());
 377.                    try {
 378.                        persister.load(p, inputSource);
 379.                    } catch (OpenException e) {
 380.                        // UML 2.x files could also contain a profile model.
 381.                        // Try again with uml:Profile as main tag.
 382.                        if ("uml:Model".equals(persister.getMainTag())
 383.                                && e.getCause() instanceof UmlException
 384.                                && e.getCause().getCause() instanceof IOException) {
 385.                            inputStream.reopen("uml:Profile");
 386.                            persister.load(p, inputSource);
 387.                            p.setProjectType(Project.PROFILE_PROJECT);
 388.                        } else {
 389.                            throw e;
 390.                        }
 391.                    }
 392.                }
 393.    
 394.                // let's update the progress
 395.                progressMgr.nextPhase();
 396.                ThreadUtils.checkIfInterrupted();
 397.                inputStream.realClose();
 398.                p.postLoad();
 399.                return p;
 400.            } catch (InterruptedException e) {
 401.                throw e;
 402.            } catch (OpenException e) {
 403.                throw e;
 404.            } catch (IOException e) {
 405.                throw new OpenException(e);
 406.            } catch (SAXException e) {
 407.                throw new OpenException(e);
 408.            }
 409.        }

 411.        protected boolean checkVersion(int fileVersion, String releaseVersion)
 412.            throws OpenException, VersionException {
 413.            // If we're trying to load a file from a future version
 414.            // complain and refuse.
 415.            if (fileVersion > PERSISTENCE_VERSION) {
 416.                throw new VersionException(
 417.                        "The file selected is from a more up to date version of "
 418.                                + "ArgoUML. It has been saved with ArgoUML version "
 419.                                + releaseVersion
 420.                                + ". Please load with that or a more up to date"
 421.                                + "release of ArgoUML");
 422.            }
 423.            return fileVersion >= PERSISTENCE_VERSION;
 424.        }

 426.        /**
 427.         * Transform a string of XML data according to the service required.
 428.         * 
 429.         * @param file The XML file to be transformed
 430.         * @param version the version of the persistence format the XML is to be
 431.         *            transformed to.
 432.         * @return the transformed XML file
 433.         * @throws OpenException on XSLT transformation error or file read
 434.         */
 435.        public final File transform(File file, int version) throws OpenException {
 436.    
 437.            try {
 438.                String upgradeFilesPath = "/org/argouml/persistence/upgrades/";
 439.                String upgradeFile = "upgrade" + version + ".xsl";
 440.    
 441.                String xsltFileName = upgradeFilesPath + upgradeFile;
 442.                URL xsltUrl = UmlFilePersister.class.getResource(xsltFileName);
 443.                LOG.info("Resource is " + xsltUrl);
 444.    
 445.                // Read xsltStream into a temporary file
 446.                // Get url for temp file.
 447.                // openStream from url and wrap in StreamSource
 448.                StreamSource xsltStreamSource = new StreamSource(xsltUrl
 449.                        .openStream());
 450.                xsltStreamSource.setSystemId(xsltUrl.toExternalForm());
 451.    
 452.                TransformerFactory factory = TransformerFactory.newInstance();
 453.                Transformer transformer = factory.newTransformer(xsltStreamSource);
 454.

 647.        /**
 648.         * Class to filter XML declaration and DOCTYPE declaration from an output
 649.         * stream to allow use as nested XML files.
 650.         * 
 651.         * @author Tom Morris
 652.         */
 653.        class XmlFilterOutputStream extends FilterOutputStream {
 654.    
 655.            private CharsetDecoder decoder;
 656.    
 657.            private boolean headerProcessed = false;
 658.    
 659.            private static final int BUFFER_SIZE = 120;
 660.    
 661.            /**
 662.             * The following three fields make up a doubly mapped buffer with two
 663.             * sets of pointers (the ByteBuffer objects), one for input and one for
 664.             * output. Bytes are written into the buffer using outBB which advances
 665.             * the current position. The following ASCII art shows what the buffer
 666.             * looks like.<code>
 667.             *           vp       vl
 668.             * xxxxxx++++..........
 669.             *       ^p  ^l
 670.             * </code> vp & vl are the output (buffer writing) position and limit,
 671.             * respectively and show the free space that can receive bytes. The ^p
 672.             * and ^l pointers show the input (buffer reading) position and limit.
 673.             * They point to valid bytes which have not yet been converted to
 674.             * characters. The xxx bytes have been both written in and read back
 675.             * out. The +++ bytes have been written into the buffer, but not read
 676.             * out yet.
 677.             */
 678.            private byte[] bytes = new byte[BUFFER_SIZE * 2];
 679.    
 680.            private ByteBuffer outBB = ByteBuffer.wrap(bytes);
 681.    
 682.            private ByteBuffer inBB = ByteBuffer.wrap(bytes);
 683.    
 684.            // Buffer containing characters which have been decoded from the bytes
 685.            // in inBB.
 686.            private CharBuffer outCB = CharBuffer.allocate(BUFFER_SIZE);
 687.    
 688.            // RegEx pattern for XML declaration and, optionally, DOCTYPE
 689.            // Backslashes are doubled up - one for Java, one for Regex
 690.            private final Pattern xmlDeclarationPattern = Pattern
 691.                    .compile("\\s*<\\?xml.*\\?>\\s*(<!DOCTYPE.*>\\s*)?");
 692.    
 693.            /**
 694.             * Construct a filtered output stream using the given character set
 695.             * name.
 696.             * 
 697.             * @param outputStream source output stream to filter
 698.             * @param charsetName name of character set to use for encoding
 699.             */
 700.            public XmlFilterOutputStream(OutputStream outputStream,
 701.                    String charsetName) {
 702.                this(outputStream, Charset.forName(charsetName));
 703.            }
 704.    
 705.            /**
 706.             * Construct a filtered output stream using the given character set.
 707.             * 
 708.             * @param outputStream source output stream to filter
 709.             * @param charset character set to use for encoding
 710.             */
 711.            public XmlFilterOutputStream(OutputStream outputStream, Charset charset) {
 712.                super(outputStream);
 713.                decoder = charset.newDecoder();
 714.                decoder.onMalformedInput(CodingErrorAction.REPORT);
 715.                decoder.onUnmappableCharacter(CodingErrorAction.REPORT);
 716.                startEntry();
 717.            }
 718.    
 719.            /**
 720.             * Reset processing for the beginning of an entry.
 721.             */
 722.            public void startEntry() {
 723.                headerProcessed = false;
 724.                resetBuffers();
 725.            }
 726.    
 727.            private void resetBuffers() {
 728.                inBB.limit(0);
 729.                outBB.position(0);
 730.                outCB.position(0);
 731.            }
 732.    
 733.            @Override
 734.            public void write(byte[] b, int off, int len) throws IOException {
 735.                if ((off | len | (b.length - (len + off)) | (off + len)) < 0) {
 736.                    throw new IndexOutOfBoundsException();
 737.                }
 738.    
 739.                if (headerProcessed) {
 740.                    out.write(b, off, len);
 741.                } else {
 742.                    // TODO: Make this more efficient for large I/Os
 743.                    for (int i = 0; i < len; i++) {
 744.                        write(b[off + i]);
 745.                    }
 746.                }
 747.    
 748.            }
 749.    
 750.            @Override
 751.            public void write(int b) throws IOException {
 752.    
 753.                if (headerProcessed) {
 754.                    out.write(b);
 755.                } else {
 756.                    outBB.put((byte) b);
 757.                    inBB.limit(outBB.position());
 758.                    // Convert from bytes back to characters
 759.                    CoderResult result = decoder.decode(inBB, outCB, false);
 760.                    if (result.isError()) {
 761.                        throw new RuntimeException(
 762.                                "Unknown character decoding error");
 763.                    }
 764.    
 765.                    // This will have problems if the smallest possible
 766.                    // data segment is smaller than the size of the buffer
 767.                    // needed for regex matching
 768.                    if (outCB.position() == outCB.limit()) {
 769.                        processHeader();
 770.                    }
 771.    
 772.                }
 773.            }
 774.    
 775.            private void processHeader() throws IOException {
 776.                headerProcessed = true;
 777.                outCB.position(0); // rewind our character buffer
 778.    
 779.                Matcher matcher = xmlDeclarationPattern.matcher(outCB);
 780.                // Remove anything that matches our pattern
 781.                String headerRemainder = matcher.replaceAll("");
 782.                int index = headerRemainder.length() - 1;
 783.                if (headerRemainder.charAt(index) == '\0') {
 784.                    // Remove null characters at the end
 785.                    do {
 786.                        index--;
 787.                    } while (index >= 0 && headerRemainder.charAt(index) == '\0');
 788.                    headerRemainder = headerRemainder.substring(0, index + 1);
 789.                }
 790.    
 791.                // Reencode the remaining characters as bytes again
 792.                ByteBuffer bb = decoder.charset().encode(headerRemainder);
 793.    
 794.                // and write them to our output stream
 795.                byte[] outBytes = new byte[bb.limit()];
 796.                bb.get(outBytes);
 797.                out.write(outBytes, 0, outBytes.length);
 798.    
 799.                // Write any left over bytes in the input buffer
 800.                // (perhaps from a partially decoded character)
 801.                if (inBB.remaining() > 0) {
 802.                    out.write(inBB.array(), inBB.position(), inBB.remaining());
 803.                    inBB.position(0);
 804.                    inBB.limit(0);
 805.                }
 806.            }
 807.    
 808.            /**
 809.             * This method has no effect to keep sub-writers from closing it
 810.             * accidently. The master can use the method {@link #reallyClose()} to
 811.             * actually close the underlying stream.
 812.             */
 813.            @Override
 814.            public void close() throws IOException {
 815.                flush();
 816.            }
 817.    
 818.            /**
 819.             * Close the stream.
 820.             * 
 821.             * @throws IOException
 822.             */
 823.            public void reallyClose() throws IOException {
 824.                out.close();
 825.            }
 826.    
 827.            /**
 828.             * Flush the stream. If the stream is flushed before the header is
 829.             * completely processed, whatever has been written so far will get
 830.             * processed before the flush.
 831.             */
 832.            @Override
 833.            public void flush() throws IOException {
 834.                if (!headerProcessed) {
 835.                    processHeader();
 836.                }
 837.                out.flush();
 838.            }
 839.    
 840.        }
