 117.         *            directory to save to. If null, user will be prompted to

 121.        public boolean trySave(boolean canOverwrite, File directory) {
 122.            overwrite = canOverwrite;
 123.            Project p =  ProjectManager.getManager().getCurrentProject();
 124.            TargetManager tm = TargetManager.getInstance();
 125.            File saveDir = (directory != null) ? directory : getSaveDir(p);
 126.            if (saveDir == null) {
 127.                /* The user cancelled! */
 128.                return false;
 129.            }
 130.            boolean okSoFar = true;
 131.            ArgoDiagram activeDiagram = DiagramUtils.getActiveDiagram();
 132.            for (ArgoDiagram d : p.getDiagramList()) {
 133.                tm.setTarget(d);
 134.                okSoFar = trySaveDiagram(d, saveDir);
 135.                if (!okSoFar) {
 136.                    break;
 137.                }
 138.            }
 139.            tm.setTarget(activeDiagram);
 140.            return okSoFar;
 141.        }

 143.        /**
 144.         * @param target the diagram
 145.         * @param saveDir the directory to save to
 146.         * @return continue exporting diagrams if true
 147.         */
 148.        protected boolean trySaveDiagram(Object target,
 149.                File saveDir) {
 150.            if ( target instanceof Diagram ) {
 151.                String defaultName = ((Diagram) target).getName();
 152.                defaultName = Util.stripJunk(defaultName);
 153.                // TODO: It's probably worthwhile to abstract and factor
 154.                // this chooser and directory stuff. More file handling is
 155.                // coming, I'm sure.
 156.                try {
 157.                    File theFile = new File(saveDir, defaultName + "."
 158.                        + SaveGraphicsManager.getInstance().getDefaultSuffix());
 159.                    String name = theFile.getName();
 160.                    String path = theFile.getParent();
 161.                    SaveGraphicsAction cmd = SaveGraphicsManager.getInstance()
 162.                        .getSaveActionBySuffix(
 163.                            SaveGraphicsManager.getInstance().getDefaultSuffix());
 164.                    if (cmd == null) {
 165.                        showStatus("Unknown graphics file type with extension "
 166.                                + SaveGraphicsManager.getInstance()
 167.                                        .getDefaultSuffix());
 168.                        return false;
 169.                    }
 170.                    showStatus( "Writing " + path + name + "..." );
 171.                    boolean result = saveGraphicsToFile(theFile, cmd);
 172.                    showStatus( "Wrote " + path + name );
 173.                    return result;
 174.                }
 175.                catch ( FileNotFoundException ignore ) {
 176.                    LOG.error("got a FileNotFoundException", ignore);
 177.                }
 178.                catch ( IOException ignore ) {
 179.                    LOG.error("got an IOException", ignore);
 180.                }
 181.            }
 182.            return false;
 183.        }

 212.        /**
 213.         * @param theFile the file to write
 214.         * @param cmd the action to execute to save the graphics
 215.         * @return continue exporting diagrams if true
 216.         * @throws IOException
 217.         */
 218.        private boolean saveGraphicsToFile(File theFile, SaveGraphicsAction cmd) 
 219.            throws IOException {
 220.            if ( theFile.exists() && !overwrite ) {
 221.                String message = Translator.messageFormat(
 222.                        "optionpane.confirm-overwrite",
 223.                        new Object[] {theFile});
 224.                String title = Translator.localize(
 225.                        "optionpane.confirm-overwrite-title"); 
 226.                //Custom button text:
 227.                Object[] options = 
 228.                {Translator.localize(
 229.                        "optionpane.confirm-overwrite.overwrite"), // 0
 230.                 Translator.localize(
 231.                        "optionpane.confirm-overwrite.overwrite-all"), // 1
 232.                 Translator.localize(
 233.                        "optionpane.confirm-overwrite.skip-this-one"), // 2
 234.                 Translator.localize(
 235.                        "optionpane.confirm-overwrite.cancel")}; // 3
 236.    
 237.                int response = 
 238.            JOptionPane.showOptionDialog(ArgoFrame.getFrame(),
 239.                        message,
 240.                        title,
 241.                        JOptionPane.YES_NO_CANCEL_OPTION,
 242.                        JOptionPane.QUESTION_MESSAGE,
 243.                        null,     //do not use a custom Icon
 244.                        options,  //the titles of buttons
 245.                        options[0]); //default button title
 246.    
 247.                if (response == 1) {
 248.                    overwrite = true;
 249.                }
 250.                if (response == 2) {
 251.                    return true;
 252.                }
 253.                if (response == 3) {
 254.                    return false;
 255.                }
 256.                if (response == JOptionPane.CLOSED_OPTION) {
 257.                    return false;
 258.                }
 259.            }
 260.            FileOutputStream fo = null;
 261.            try {
 262.                fo = new FileOutputStream( theFile );
 263.                cmd.setStream(fo);
 264.                cmd.setScale(Configuration.getInteger(
 265.                        SaveGraphicsManager.KEY_GRAPHICS_RESOLUTION, 1));
 266.                cmd.actionPerformed(null);
 267.            } finally {
 268.                if (fo != null) {
 269.                    fo.close();
 270.                }
 271.            }
 272.            return true;
 273.        }

 275.        private JFileChooser getFileChooser(Project p) {
 276.            JFileChooser chooser = null;
 277.            try {
 278.            if ( p != null 
 279.                    && p.getURI() != null
 280.                    && p.getURI().toURL().getFile().length() > 0 ) {
 281.                chooser = new JFileChooser(p.getURI().toURL().getFile());
 282.                }
 283.            }
 284.            catch ( MalformedURLException ex ) {
 285.                LOG.error("exception in opening JFileChooser", ex);
 286.            }
 287.    
 288.            if ( chooser == null ) {
 289.                chooser = new JFileChooser();
 290.            }
 291.            chooser.setDialogTitle(
 292.                    Translator.localize("filechooser.save-all-graphics"));
 293.            chooser.setDialogType(JFileChooser.OPEN_DIALOG);
 294.            chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
 295.            chooser.setMultiSelectionEnabled(false);
 296.            return chooser;
 297.        }

 304.        /**
 305.         * Execute this action from the command line.
 306.         *
 307.         * @see org.argouml.application.api.CommandLineInterface#doCommand(String)

 311.        public boolean doCommand(String argument) {
 312.            File dir = new File(argument);
 313.            if (!dir.exists() || !dir.isDirectory()) {
 314.                LOG.error("The argument must be a path to an existing directory.");
 315.                return false;
 316.            }
 317.            boolean result = true;
 318.            for (Project p : ProjectManager.getManager().getOpenProjects()) {
 319.                TargetManager tm = TargetManager.getInstance();
 320.                for (ArgoDiagram d : p.getDiagramList()) {
 321.                    tm.setTarget(d);
 322.                    if (!trySaveDiagram(d, dir)) {
 323.                        result = false;
 324.                    }
 325.                }
 326.            }
 327.            return result;
 328.        }

  70.    /**
  71.     * Wraps a CmdSaveGIF or CmdSave(E)PS to allow selection of an output file.
  72.     */
  73.    public class ActionSaveGraphics extends AbstractAction
  74.            implements CommandLineInterface {
  75.    
  76.        private static final long serialVersionUID = 3062674953320109889L;
  77.        
  78.        private static final Logger LOG =
  79.            Logger.getLogger(ActionSaveGraphics.class);
  80.    
  81.        /**
  82.         * Constructor for this action.
  83.         */
  84.        public ActionSaveGraphics() {
  85.            super(Translator.localize("action.save-graphics"),
  86.                    ResourceLoaderWrapper.lookupIcon("action.save-graphics"));
  87.        }
  88.    
  89.        /*
  90.         * @see AbstractAction#actionPerformed(ActionEvent)
  91.         */
  92.        public void actionPerformed(ActionEvent ae) {
  93.            trySave(); //TODO: what to do with the return value?
  94.        }
  95.    
  96.        /**
  97.         * Method that does almost everything in this class.<p>
  98.         *
  99.         * @return true if all went well.
 100.         */
 101.        private boolean trySave() {
 102.            Object target = DiagramUtils.getActiveDiagram();
 103.    
 104.            if (!(target instanceof Diagram)) {
 105.                return false;
 106.            }
 107.    
 108.            String defaultName = ((Diagram) target).getName();
 109.            defaultName = Util.stripJunk(defaultName);
 110.    
 111.            Project p =  ProjectManager.getManager().getCurrentProject();
 112.            SaveGraphicsManager sgm = SaveGraphicsManager.getInstance();
 113.            try {
 114.                JFileChooser chooser = null;
 115.    
 116.                if (p != null
 117.                    && p.getURI() != null
 118.                    && p.getURI().toURL().getFile().length() > 0) {
 119.    
 120.                    chooser = new JFileChooser(p.getURI().toURL().getFile());
 121.                }
 122.    
 123.                if (chooser == null) {
 124.                    chooser = new JFileChooser();
 125.                }
 126.    
 127.                Object[] s = {defaultName };
 128.                chooser.setDialogTitle(
 129.                        Translator.messageFormat("filechooser.save-graphics", s));
 130.                // Only specified format are allowed.
 131.                chooser.setAcceptAllFileFilterUsed(false);
 132.                sgm.setFileChooserFilters(chooser, defaultName);
 133.    
 134.                String fn = Configuration.getString(
 135.                        SaveGraphicsManager.KEY_SAVE_GRAPHICS_PATH);
 136.                if (fn.length() > 0) {
 137.                    chooser.setSelectedFile(new File(fn));
 138.                }
 139.    
 140.                int retval = chooser.showSaveDialog(ArgoFrame.getFrame());
 141.                if (retval == JFileChooser.APPROVE_OPTION) {
 142.                    File theFile = chooser.getSelectedFile();
 143.                    if (theFile != null) {
 144.                        String path = theFile.getPath();
 145.                        Configuration.setString(
 146.                                SaveGraphicsManager.KEY_SAVE_GRAPHICS_PATH,
 147.                                path);
 148.    
 149.                        theFile = new File(theFile.getParentFile(),
 150.                                sgm.fixExtension(theFile.getName()));
 151.                        String suffix = sgm.getFilterFromFileName(theFile.getName())
 152.                            .getSuffix();
 153.                        return doSave(theFile, suffix, true);
 154.                    }
 155.                }
 156.            } catch (OutOfMemoryError e) {
 157.                ExceptionDialog ed = new ExceptionDialog(ArgoFrame.getFrame(),
 158.                    "You have run out of memory. "
 159.                    + "Close down ArgoUML and restart with a larger heap size.", e);
 160.                ed.setModal(true);
 161.                ed.setVisible(true);
 162.            } catch (Exception e) {
 163.                ExceptionDialog ed = 
 164.                    new ExceptionDialog(ArgoFrame.getFrame(), e);
 165.                ed.setModal(true);
 166.                ed.setVisible(true);
 167.                LOG.error("Got some exception", e);
 168.            }
 169.    
 170.            return false;
 171.        }
 172.    
 173.        /**
 174.         * Actually do the saving of the graphics file.
 175.         *
 176.         * @return true if it was successful.
 177.         * @param theFile is the file that we are writing to
 178.         * @param suffix is the suffix. Used for deciding what format the file
 179.         * shall have.
 180.         * @param useUI is true if we are supposed to use the UI e.g. to warn
 181.         *              the user that we are replacing an old file.
 182.         */
 183.        private boolean doSave(File theFile,
 184.                   String suffix, boolean useUI)
 185.        throws FileNotFoundException, IOException {
 186.    
 187.            SaveGraphicsManager sgm = SaveGraphicsManager.getInstance();
 188.            SaveGraphicsAction cmd = null;
 189.    
 190.            cmd = sgm.getSaveActionBySuffix(suffix);
 191.            if (cmd == null) {
 192.                return false;
 193.            }
 194.    
 195.            if (useUI) {
 196.                updateStatus(Translator.localize(
 197.                        "statusmsg.bar.save-graphics-status-writing",
 198.                        new Object[] {theFile}));
 199.            }
 200.        if (theFile.exists() && useUI) {
 201.            int response = JOptionPane.showConfirmDialog(
 202.                    ArgoFrame.getFrame(),
 203.                    Translator.messageFormat("optionpane.confirm-overwrite",
 204.                            new Object[] {theFile}),
 205.                    Translator.localize("optionpane.confirm-overwrite-title"),
 206.                    JOptionPane.YES_NO_OPTION);
 207.            if (response != JOptionPane.YES_OPTION) {
 208.            return false;
 209.            }
 210.        }
 211.        FileOutputStream fo = new FileOutputStream(theFile);
 212.        cmd.setStream(fo);
 213.            cmd.setScale(Configuration.getInteger(
 214.                    SaveGraphicsManager.KEY_GRAPHICS_RESOLUTION, 1));
 215.            try {
 216.                cmd.actionPerformed(null);
 217.            } finally {
 218.                fo.close();
 219.            }
 220.            if (useUI) {
 221.                updateStatus(Translator.localize(
 222.                        "statusmsg.bar.save-graphics-status-wrote", 
 223.                        new Object[] {theFile}));
 224.            }
 225.        return true;
 226.        }
 227.        
 228.        private void updateStatus(String status) {
 229.            ArgoEventPump.fireEvent(
 230.                    new ArgoStatusEvent(ArgoEventTypes.STATUS_TEXT,
 231.                    this, status));
 232.        }
 233.    
 234.    
 235.        /**
 236.         * Execute this action from the command line.
 237.         *
 238.         * TODO: The underlying GEF library relies on Acme that doesn't allow
 239.         * us to create these files unless there is a window showing. For this
 240.         * reason I have had to split the performing of commands in
 241.         * {@link org.argouml.application.Main#main(String[])} so that we can,
 242.         * by not supplying the -batch option, run these commands
 243.         * with the window showing. Hopefully this can eventually be fixed.
 244.         *
 245.         * @see org.argouml.application.api.CommandLineInterface#doCommand(String)
 246.         * @param argument is the file name that we save to.
 247.         * @return true if it is OK.
 248.         */
 249.        public boolean doCommand(String argument) {
 250.        File file = new File(argument);
 251.        String suffix = SuffixFilter.getExtension(file);
 252.        if (suffix == null) {
 253.            return false;
 254.        }
 255.    
 256.        try {
 257.            return doSave(file, suffix, false);
 258.        } catch (FileNotFoundException e) {
 259.            LOG.error("File not found error when writing.", e);
 260.        } catch (IOException e) {
 261.            LOG.error("IO error when writing.", e);
 262.        }
 263.        return false;
 264.        }
