 473.        /**
 474.         * Do the search.
 475.         */
 476.        private void doSearch() {
 477.            numFinds++;
 478.            String eName = "";
 479.            if (elementName.getSelectedItem() != null) {
 480.                eName += elementName.getSelectedItem();
 481.                elementName.removeItem(eName);
 482.                elementName.insertItemAt(eName, 0);
 483.                elementName.setSelectedItem(eName);
 484.            }
 485.            String dName = "";
 486.            if (diagramName.getSelectedItem() != null) {
 487.                dName += diagramName.getSelectedItem();
 488.                diagramName.removeItem(dName);
 489.                diagramName.insertItemAt(dName, 0);
 490.                diagramName.setSelectedItem(dName);
 491.            }
 492.            String name = eName;
 493.            if (dName.length() > 0) {
 494.                Object[] msgArgs = {name, dName };
 495.                name =
 496.            Translator.messageFormat(
 497.                        "dialog.find.comboboxitem.element-in-diagram", msgArgs);
 498.                //name += " in " + dName;
 499.            }
 500.            String typeName = type.getSelectedItem().toString();
 501.            if (!typeName.equals("Any Type")) {
 502.                name += " " + typeName;
 503.            }
 504.            if (name.length() == 0) {
 505.                name =
 506.                    Translator.localize("dialog.find.tabname") + (nextResultNum++);
 507.            }
 508.            if (name.length() > 15) {
 509.                // TODO: Localize
 510.                name = name.substring(0, 12) + "...";
 511.            }
 512.    
 513.            String pName = "";
 514.    
 515.            Predicate eNamePred = PredicateStringMatch.create(eName);
 516.            Predicate pNamePred = PredicateStringMatch.create(pName);
 517.            Predicate dNamePred = PredicateStringMatch.create(dName);
 518.            Predicate typePred = (Predicate) type.getSelectedItem();
 519.            PredicateSearch pred =
 520.                new PredicateSearch(eNamePred, pNamePred, dNamePred, typePred);
 521.    
 522.            ChildGenSearch gen = new ChildGenSearch();
 523.            Object root = ProjectManager.getManager().getCurrentProject();
 524.    
 525.            TabResults newResults = new TabResults();
 526.            newResults.setTitle(name);
 527.            newResults.setPredicate(pred);
 528.            newResults.setRoot(root);
 529.            newResults.setGenerator(gen);
 530.            resultTabs.add(newResults);
 531.            results.addTab(name, newResults);
 532.            clearTabs.setEnabled(true);
 533.            getOkButton().setEnabled(true);
 534.            results.setSelectedComponent(newResults);
 535.            Object[] msgArgs = {name };
 536.            location.addItem(Translator.messageFormat(
 537.                    "dialog.find.comboboxitem.in-tab", msgArgs));
 538.            invalidate();
 539.            results.invalidate();
 540.            validate();
 541.            newResults.run();
 542.            newResults.requestFocus();
 543.            newResults.selectResult(0);
 544.        }

 631.        /*
 632.         * @see java.awt.event.MouseListener#mouseClicked(java.awt.event.MouseEvent)
 633.         */
 634.        public void mouseClicked(MouseEvent me) {
 635.            int tab = results.getSelectedIndex();
 636.            if (tab != -1) {
 637.                Rectangle tabBounds = results.getBoundsAt(tab);
 638.                if (!tabBounds.contains(me.getX(), me.getY())) {
 639.            return;
 640.            }
 641.                if (tab >= 1 && me.getClickCount() >= 2) {
 642.                    myDoubleClick(tab - 1); //help tab is 0
 643.            }
 644.            }
 645.        }
 646.    
 647.        /**
 648.         * Double click on tab detaches it in a free floating window.
 649.         *
 650.         * MVW: This is the only place where spawning is still enabled.
 651.         *
 652.         * @param tab the given tab
 653.         */
 654.        private void myDoubleClick(int tab) {
 655.            JPanel t = resultTabs.get(tab);
 656.            if (t instanceof AbstractArgoJPanel) {
 657.                if (((AbstractArgoJPanel) t).spawn() != null) {
 658.                    resultTabs.remove(tab);
 659.                    location.removeItem("In Tab: "
 660.                                    + ((AbstractArgoJPanel) t).getTitle());
 661.                }
 662.            }
 663.        }

 673.     * PredicateMType is a small helper class which converts the
 674.     * string representation of the Type from internal form.
 675.     * Now that Type names aren't prefixed with NSUML's "M" the

 688.        /**
