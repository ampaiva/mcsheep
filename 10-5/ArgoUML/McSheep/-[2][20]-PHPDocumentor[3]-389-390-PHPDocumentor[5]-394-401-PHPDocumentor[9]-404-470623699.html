 667.         */
 668.        private final class DocBlock {
 669.            public static final int TAG_TYPE_ACCESS = 1;
 670.            public static final int TAG_TYPE_AUTHOR = 2;
 671.            public static final int TAG_TYPE_DEPRECATED = 3;
 672.            public static final int TAG_TYPE_PACKAGE = 4;
 673.            public static final int TAG_TYPE_PARAM = 5;
 674.            public static final int TAG_TYPE_RETURN = 6;
 675.            public static final int TAG_TYPE_SEE = 7;

 696.            private TreeMap<String, ITag> tmTags = new TreeMap<String, ITag>();

 719.            /**
 720.             * Formats the DocBlock
 721.             *
 722.             * @param sIndent line indentation string
 723.             *
 724.             * @return The formatted DocBlock.
 725.             */
 726.            public final String toString(String sIndent) {
 727.                String s = "";
 728.    
 729.                if (sDescription == null || sDescription == "") {
 730.                    setDescription(sDefaultDescription);
 731.                }
 732.    
 733.                s += sIndent + "/**\n";
 734.    
 735.                s += wrapDescription(sIndent, 80);
 736.    
 737.                for (Map.Entry<String, ITag> entryMap : tmTags.entrySet()) {
 738.                    ITag entryValue = entryMap.getValue();
 739.                    if (entryValue instanceof ParamList) {
 740.                        ParamList paramList = (ParamList) entryValue;
 741.                        for (ParamTag tag : paramList ) {
 742.                            if (tag.getTag() != null) {
 743.                                s += sIndent + " * " + tag.getTag() + "\n";
 744.                            }
 745.                        }
 746.                    } else {
 747.                        Tag tag = (Tag) entryValue;
 748.                        if (tag.getContent() != null) {
 749.                            if (tag.getTag() != null) {
 750.                                s += sIndent + " * " + tag.getTag() + "\n";
 751.                            }
 752.                        }
 753.                    }
 754.                }
 755.    
 756.                s += sIndent + " */\n";
 757.    
 758.                return s;
 759.            }
 760.    
 761.            /**
 762.             * Wraps long description lines
 763.             *
 764.             * @param sIndent   The indention string.
 765.             * @param iMaxWidth Maximal line length
 766.             *
 767.             * @return The formated description part of DocBlock
 768.             */
 769.            private final String wrapDescription(String sIndent, int iMaxWidth) {
 770.                String sWrapped = "";
 771.    
 772.                if (sDescription != null && sDescription.trim() != "") {
 773.                    StringTokenizer stDescription =
 774.                            new StringTokenizer(sDescription, "\n\r", true);
 775.                    while (stDescription.hasMoreElements()) {
 776.                        String sToken = stDescription.nextToken();
 777.                        if (!sToken.equals("\n")
 778.                                && stDescription.hasMoreElements()) {
 779.                            stDescription.nextToken();
 780.                        } else if (sToken.equals("\n")) {
 781.                            sToken = "";
 782.                        }
 783.    
 784.                        if (sToken.length() > (iMaxWidth - sIndent.length() - 3)) {
 785.                            String sLine = "";
 786.    
 787.                            StringTokenizer stLine =
 788.                                    new StringTokenizer(sToken, " ");
 789.                            while (stLine.hasMoreElements()) {
 790.                                String sLineToken = stLine.nextToken();
 791.    
 792.                                int iMaxLen = iMaxWidth - 4 - sIndent.length();
 793.                                if ((sLine.length() + sLineToken.length())
 794.                                        <= iMaxLen) {
 795.                                    sLine += sLine.trim() != "" ? " " : "";
 796.                                    sLine += sLineToken;
 797.                                } else {
 798.                                    if (sLine.trim() != "") {
 799.                                        sWrapped += sIndent + " * " + sLine + "\n";
 800.                                    }
 801.                                    sLine = "";
 802.                                }
 803.                            }
 804.    
 805.                            if (sLine.trim() != "") {
 806.                                sWrapped += sIndent + " * " + sLine + "\n";
 807.                            }
 808.                        } else {
 809.                            if (sToken.trim() != "") {
 810.                                sWrapped += sIndent + " * " + sToken + "\n";
 811.                            } else if (stDescription.hasMoreElements()) {
 812.                                sWrapped += sIndent + " *\n";
 813.                            }
 814.                        }
 815.                    }
 816.

 200.         * @throws Exception
 201.         */
 202.        private void create(Object modelElement, int iBlockType)
 203.            throws Exception {
 204.            switch (iBlockType) {
 205.            // TODO: move block 4 spaces right if checkstyle is fixed
 206.            case BLOCK_TYPE_UNKNOWN:
 207.                createUnknown(modelElement);
 208.                break;
 209.            case BLOCK_TYPE_FILE:
 210.                createFile(modelElement);
 211.                break;
 212.            case BLOCK_TYPE_INCLUDE:
 213.                createInclude(modelElement);
 214.                break;
 215.            case BLOCK_TYPE_CLASSIFIER:
 216.                createClassifier(modelElement);
 217.                break;
 218.            case BLOCK_TYPE_ATTRIBUTE:
 219.                createAttribute(modelElement);
 220.                break;
 221.            case BLOCK_TYPE_OPERATION:
 222.                createOperation(modelElement);
 223.                break;
 224.            case BLOCK_TYPE_PACKAGE:
 225.                createPackage(modelElement);
 226.                break;
 227.            default:
 228.                throw new IllegalArgumentException(iBlockType
 229.                        + " is not a valid DocBlock type.");
 230.            }
 231.        }

 240.        private void createUnknown(Object modelElement) throws Exception {
 241.            if (modelElement != null) {
 242.                if (Model.getFacade().isAClassifier(modelElement)) {
 243.                    createClassifier(modelElement);
 244.                } else if (Model.getFacade().isAAttribute(modelElement)) {
 245.                    createAttribute(modelElement);
 246.                } else if (Model.getFacade().isAOperation(modelElement)) {
 247.                    createOperation(modelElement);
 248.                } else if (Model.getFacade().isAPackage(modelElement)) {
 249.                    createPackage(modelElement);
 250.                } else {
 251.                    throw new IllegalArgumentException(modelElement.getClass()
 252.                            + " is not a Classifier, Attribute or Operation.");
 253.                }
 254.            } else {
 255.                objDocBlock = new DocBlock();
 256.                iDocBlockType = BLOCK_TYPE_UNKNOWN;
 257.    
 258.                objDocBlock.setDescription("UNKNOWN");
 259.            }
 260.        }

 266.         *
 267.         * @throws Exception
 268.         */
 269.        private void createFile(Object modelElement) throws Exception {
 270.            if (!Model.getFacade().isAClassifier(modelElement)) {
 271.                throw new ClassCastException(modelElement.getClass()
 272.                    + " has wrong object type, Classifier required");
 273.            }
 274.    
 275.            objDocBlock = new DocBlock();
 276.            objModelElement = modelElement;
 277.            iDocBlockType = BLOCK_TYPE_FILE;
 278.    
 279.            objDocBlock.enableTag(DocBlock.TAG_TYPE_PACKAGE);
 280.            objDocBlock.enableTag(DocBlock.TAG_TYPE_SUBPACKAGE);
 281.    
 282.            objDocBlock.disableTag(DocBlock.TAG_TYPE_ACCESS);
 283.            objDocBlock.disableTag(DocBlock.TAG_TYPE_STATIC);
 284.    
 285.            String sPackageName = NameGenerator.generatePackageName(modelElement);
 286.    
 287.            if (sPackageName != null && sPackageName != "") {
 288.                int iFirstUnderscore = sPackageName.indexOf("_");
 289.                if (iFirstUnderscore == -1) {
 290.                    objDocBlock.setTag(DocBlock.TAG_TYPE_PACKAGE, sPackageName);
 291.                } else {
 292.                    objDocBlock.setTag(DocBlock.TAG_TYPE_PACKAGE,
 293.                            sPackageName.substring(0, iFirstUnderscore));
 294.                    objDocBlock.setTag(DocBlock.TAG_TYPE_SUBPACKAGE,
 295.                            sPackageName.substring(iFirstUnderscore + 1));
 296.                }

 308.            // TODO: add license
 309.            // TODO: add copyright
 310.    
 311.            GregorianCalendar calNow = new GregorianCalendar();
 312.            int iDay  = calNow.get(Calendar.DATE);
 313.            int iMon  = (calNow.get(Calendar.MONTH) + 1);
 314.            int iYear = calNow.get(Calendar.YEAR);
 315.            int iHour = calNow.get(Calendar.HOUR_OF_DAY);
 316.            int iMin  = calNow.get(Calendar.MINUTE);

 323.            }
 324.    
 325.            String sDateTime = "";
 326.            sDateTime += iDay  < 10 ? "0" + iDay  + "." : iDay + ".";
 327.            sDateTime += iMon  < 10 ? "0" + iMon  + "." : iMon + ".";
 328.            sDateTime += iYear + ", ";

 334.    
 335.            // TODO: replace "\" in sFilename with "/"
 336.            if (sFilename != null && sProjectName != null) {
 337.                sDescription += sProjectName + " - " + sFilename + "\n\n";
 338.            }
 339.    
 340.            sDescription += "$" + "Id" + "$" + "\n\n";
 341.    
 342.            if (sProjectName != null) {
 343.                sDescription += "This file is part of " + sProjectName + ".\n\n";
 344.            }
 345.    
 346.            sDescription +=
 347.                    "Automatically generated on " + sDateTime
 348.                    + " with ArgoUML PHP module \n(last revised " 
 349.                    + "$Date: 2010-01-12 20:14:42 +0100 (Tue, 12 Jan 2010) $)";
 350.    
 351.            objDocBlock.setDescription(sDescription);
 352.    
 353.            objDocBlock.setTags(Model.getFacade().getTaggedValues(modelElement));

 504.                objDocBlock.setTag(DocBlock.TAG_TYPE_ACCESS, "public");
 505.            } else if (Model.getFacade().isProtected(modelElement)) {
 506.                objDocBlock.setTag(DocBlock.TAG_TYPE_ACCESS, "protected");
 507.            } else if (Model.getFacade().isPrivate(modelElement)) {
 508.                objDocBlock.setTag(DocBlock.TAG_TYPE_ACCESS, "private");
 509.            }
 510.    
 511.            objDocBlock.setTags(Model.getFacade().getTaggedValues(modelElement));
 512.    
 513.            if (Model.getFacade().isAbstract(modelElement)) {
 514.                objDocBlock.setTag(DocBlock.TAG_TYPE_ABSTRACT, "true");
 515.            }
 516.    
 517.            for (Object parameter : Model.getFacade().getParameters(modelElement)) {
 518.                if (Model.getFacade().isReturn(parameter)) {
 519.                    Object returnType = Model.getFacade().getType(parameter);
 520.                    if (returnType != null) {
 521.                        String packageName = NameGenerator
 522.                                .generatePackageName(returnType);
 523.                        if (packageName != null && packageName != "") {
 524.                            objDocBlock.setTag(DocBlock.TAG_TYPE_RETURN, 
 525.                                    packageName
 526.                                    + "_"
 527.                                    + Model.getFacade().getName(returnType));
 528.                        } else {
 529.                            objDocBlock.setTag(DocBlock.TAG_TYPE_RETURN, Model
 530.                                    .getFacade().getName(returnType));
 531.                        }
 532.                    } else {
 533.                        objDocBlock.setTag(DocBlock.TAG_TYPE_RETURN, "mixed");
 534.                    }
 535.                } else {
 536.                    StringBuffer description = new StringBuffer(" ");
 537.                    Object type = Model.getFacade().getType(parameter);
 538.                    if (type != null) {
 539.                        description.append(Model.getFacade().getName(type));
 540.                    }
 541.                    description.append(" ");
 542.                    String name = Model.getFacade().getName(parameter);
 543.                    if (name != null) {
 544.                        description.append(name);
 545.                    }
 546.                    String doc = Model.getFacade().getTaggedValueValue(parameter,
 547.                            "documentation");
 548.                    if (doc != null) {
 549.                        description.append(" ").append(doc);
 550.                    }
 551.                    objDocBlock.addTag(DocBlock.TAG_TYPE_PARAM, description
 552.                            .toString());
 553.                }
 554.            }
 555.        }

 721.             *

 723.             *
 724.             * @return The formatted DocBlock.
 725.             */

 727.                String s = "";
 728.    
 729.                if (sDescription == null || sDescription == "") {
 730.                    setDescription(sDefaultDescription);

 734.

 743.                                s += sIndent + " * " + tag.getTag() + "\n";
 744.                            }
 745.                        }
 746.                    } else {
 747.                        Tag tag = (Tag) entryValue;
 748.                        if (tag.getContent() != null) {
 749.                            if (tag.getTag() != null) {
 750.                                s += sIndent + " * " + tag.getTag() + "\n";
 751.                            }
 752.                        }
 753.                    }
 754.                }
 755.    
 756.                s += sIndent + " */\n";
 757.    
 758.                return s;
 759.            }

 761.            /**
 762.             * Wraps long description lines
 763.             *
 764.             * @param sIndent   The indention string.
 765.             * @param iMaxWidth Maximal line length
 766.             *
 767.             * @return The formated description part of DocBlock
 768.             */
 769.            private final String wrapDescription(String sIndent, int iMaxWidth) {
 770.                String sWrapped = "";
 771.    
 772.                if (sDescription != null && sDescription.trim() != "") {
 773.                    StringTokenizer stDescription =
 774.                            new StringTokenizer(sDescription, "\n\r", true);
 775.                    while (stDescription.hasMoreElements()) {
 776.                        String sToken = stDescription.nextToken();
 777.                        if (!sToken.equals("\n")
 778.                                && stDescription.hasMoreElements()) {
 779.                            stDescription.nextToken();
 780.                        } else if (sToken.equals("\n")) {
 781.                            sToken = "";
 782.                        }
 783.    
 784.                        if (sToken.length() > (iMaxWidth - sIndent.length() - 3)) {
 785.                            String sLine = "";
 786.    
 787.                            StringTokenizer stLine =
 788.                                    new StringTokenizer(sToken, " ");
 789.                            while (stLine.hasMoreElements()) {
 790.                                String sLineToken = stLine.nextToken();
 791.    
 792.                                int iMaxLen = iMaxWidth - 4 - sIndent.length();
 793.                                if ((sLine.length() + sLineToken.length())
 794.                                        <= iMaxLen) {
 795.                                    sLine += sLine.trim() != "" ? " " : "";
 796.                                    sLine += sLineToken;
 797.                                } else {
 798.                                    if (sLine.trim() != "") {
 799.                                        sWrapped += sIndent + " * " + sLine + "\n";
 800.                                    }
 801.                                    sLine = "";
 802.                                }
 803.                            }
 804.    
 805.                            if (sLine.trim() != "") {
 806.                                sWrapped += sIndent + " * " + sLine + "\n";
 807.                            }
 808.                        } else {
 809.                            if (sToken.trim() != "") {
 810.                                sWrapped += sIndent + " * " + sToken + "\n";
 811.                            } else if (stDescription.hasMoreElements()) {
 812.                                sWrapped += sIndent + " *\n";
 813.                            }
 814.                        }
 815.                    }
 816.    
 817.                    sWrapped += sIndent + " *\n";
 818.    
 819.                    return sWrapped;
 820.                }
 821.                return "";
 822.            }
