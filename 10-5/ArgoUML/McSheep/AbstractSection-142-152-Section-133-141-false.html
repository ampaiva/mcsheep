 160.                FileReader f = new FileReader(filename);
 161.                BufferedReader fr = new BufferedReader(f);
 162.    
 163.                String line = "";
 164.                String content = "";
 165.                boolean inSection = false;
 166.                while (line != null) {
 167.                    line = fr.readLine();
 168.                    if (line != null) {
 169.                        if (inSection) {
 170.                            String sectionId = getSectionId(line);
 171.                            if (sectionId != null) {
 172.                                inSection = false;
 173.                                mAry.put(sectionId, content);
 174.                                content = "";
 175.                            } else {
 176.                                content += line + "\n";
 177.                            }
 178.                        } else {
 179.                            String sectionId = getSectionId(line);
 180.                            if (sectionId != null) {
 181.                                inSection = true;
 182.                            }
 183.                        }
 184.                    }
 185.                }
 186.                fr.close();
 187.    
 188.            } catch (IOException e) {
 189.                LOG.debug("Error: " + e.toString());
 190.            }
 191.        }

 193.        private static String getSectionId(String line) {
 194.            int first = line.indexOf(BEGIN);
 195.            int second = line.indexOf(END1);
 196.            if (second < 0) {
 197.                second = line.indexOf(END2);
 198.            }
 199.            String s = null;
 200.            if ((first > 0) && (second > 0)) {
 201.                first = first + BEGIN.length();
 202.                s = line.substring(first, second);
 203.            }
 204.            return s;
