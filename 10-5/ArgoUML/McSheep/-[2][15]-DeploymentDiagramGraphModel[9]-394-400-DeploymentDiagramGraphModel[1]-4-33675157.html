 370.                    if (canAddEdge(specialization)) {

 372.                    }
 373.                    return;
 374.                }
 375.            }
 376.            if (Model.getFacade().isAModelElement(node)) {
 377.                List dependencies =
 378.                    new ArrayList(Model.getFacade().getClientDependencies(node));
 379.                dependencies.addAll(Model.getFacade().getSupplierDependencies(node));
 380.                for (Object dependency : dependencies) {
 381.                    if (canAddEdge(dependency)) {
 382.                        addEdge(dependency);
 383.                    }
 384.                    return;
 385.                }
 386.            }
 387.        }
 388.    
 389.    
 390.        /*
 391.         * @see java.beans.VetoableChangeListener#vetoableChange(java.beans.PropertyChangeEvent)
 392.         */
 393.        public void vetoableChange(PropertyChangeEvent pce) {
 394.            if ("ownedElement".equals(pce.getPropertyName())) {
 395.                List oldOwned = (List) pce.getOldValue();
 396.                Object eo = pce.getNewValue();
 397.                Object me = Model.getFacade().getModelElement(eo);
 398.                if (oldOwned.contains(eo)) {
 399.                    LOG.debug("model removed " + me);
 400.                    if (Model.getFacade().isANode(me)) {
 401.                        removeNode(me);
 402.                    }
 403.                    if (Model.getFacade().isANodeInstance(me)) {
 404.                        removeNode(me);
 405.                    }
 406.                    if (Model.getFacade().isAComponent(me)) {
 407.                        removeNode(me);
 408.                    }
 409.                    if (Model.getFacade().isAComponentInstance(me)) {
 410.                        removeNode(me);
 411.                    }
 412.                    if (Model.getFacade().isAClass(me)) {
 413.                        removeNode(me);
 414.                    }
 415.                    if (Model.getFacade().isAInterface(me)) {
 416.                        removeNode(me);
 417.                    }
 418.                    if (Model.getFacade().isAObject(me)) {
 419.                        removeNode(me);
 420.                    }
 421.                    if (Model.getFacade().isAAssociation(me)) {
 422.                        removeEdge(me);
 423.                    }
 424.                    if (Model.getFacade().isADependency(me)) {
 425.                        removeEdge(me);
 426.                    }
 427.                    if (Model.getFacade().isALink(me)) {
 428.                        removeEdge(me);
 429.                    }
 430.                } else {
 431.                    LOG.debug("model added " + me);
 432.                }
 433.            }
 434.        }

 167.    
 168.        /**
 169.         * Determine if the given node can validly be placed on this
 170.         * graph.<p>
 171.         *
 172.         * This is simply a matter of determining if the node is an actor
 173.         * or use case.<p>
 174.         *
 175.         * <em>Note</em>. This is inconsistent with {@link #addNode},
 176.         * which will not allow a node to be added to the graph if it is
 177.         * already there.<p>
 178.         *
 179.         * @param node  The node to be considered
 180.         *
 181.         * @return      <code>true</code> if the given object is a valid node in
 182.         *              this graph, <code>false</code> otherwise.
 183.         */
 184.        @Override
 185.        public boolean canAddNode(Object node) {
 186.            if (Model.getFacade().isAAssociation(node)
 187.                    && !Model.getFacade().isANaryAssociation(node)) {
 188.                // A binary association is not a node so reject.
 189.                return false;
 190.            }
 191.            if (super.canAddNode(node)) {
 192.                return true;
 193.            }
 194.            if (containsNode(node)) {
 195.            return false;
 196.        }
 197.            return Model.getFacade().isAActor(node)
 198.                || Model.getFacade().isAUseCase(node)
 199.                || Model.getFacade().isAPackage(node);
 200.        }
 201.    
 202.    
 203.        /**
 204.         * Determine if the given edge can validly be placed on this graph.<p>
 205.         *
 206.         * We cannot do so if the edge is already on the graph (unlike
 207.         * nodes they may not appear more than once).<p>
 208.         *
 209.         * Otherwise, for all valid types of edge (binary association,
 210.         * generalization, extend, include, dependency) we get the two
 211.         * ends. If they are both nodes already on the graph we are OK,
 212.         * otherwise we cannot place the edge on the graph.<p>
 213.         *
 214.         * @param edge  The edge to be considered
 215.         *
 216.         * @return      <code>true</code> if the given object is a valid edge in
 217.         *              this graph, <code>false</code> otherwise.
 218.         */
 219.        @Override
 220.        public boolean canAddEdge(Object edge)  {
 221.            if (edge == null) {
 222.                return false;
 223.            }
 224.            if (containsEdge(edge)) {
 225.                return false;
 226.            }
 227.    
 228.            // Get the two ends of any valid edge
 229.            Object sourceModelElement = null;
 230.            Object destModelElement = null;
 231.            if (Model.getFacade().isAAssociation(edge)) {
 232.    
 233.                // Only allow binary associations
 234.    
 235.                Collection conns = Model.getFacade().getConnections(edge);
 236.                Iterator iter = conns.iterator();
 237.    
 238.                if (conns.size() < 2) {
 239.                    return false;
 240.                }
 241.    
 242.                Object associationEnd0 = iter.next();
 243.                Object associationEnd1 = iter.next();
 244.    
 245.                // Give up if the assocation ends don't have a type defined
 246.    
 247.                if ((associationEnd0 == null) || (associationEnd1 == null)) {
 248.                    return false;
 249.                }
 250.    
 251.                sourceModelElement = Model.getFacade().getType(associationEnd0);
 252.                destModelElement = Model.getFacade().getType(associationEnd1);
 253.            } else if (Model.getFacade().isAGeneralization(edge)) {
 254.                sourceModelElement = Model.getFacade().getSpecific(edge);
 255.                destModelElement = Model.getFacade().getGeneral(edge);
 256.            } else if (Model.getFacade().isAExtend(edge)) {
 257.                sourceModelElement = Model.getFacade().getBase(edge);
 258.                destModelElement = Model.getFacade().getExtension(edge);
 259.            } else if (Model.getFacade().isAInclude(edge)) {
 260.    
 261.                sourceModelElement = Model.getFacade().getBase(edge);
 262.                destModelElement = Model.getFacade().getAddition(edge);
 263.            } else if (Model.getFacade().isADependency(edge)) {
 264.    
 265.                // A dependency potentially has many clients and suppliers. We only
 266.                // consider the first of each (not clear that we should really
 267.                // accept the case where there is more than one of either)
 268.    
 269.                Collection clients   = Model.getFacade().getClients(edge);
 270.                Collection suppliers = Model.getFacade().getSuppliers(edge);
 271.    
 272.                if (clients == null || clients.isEmpty() 
 273.                        || suppliers == null || suppliers.isEmpty()) {
 274.                    return false;
 275.                }
 276.                sourceModelElement = clients.iterator().next();
 277.                destModelElement = suppliers.iterator().next();
 278.    
 279.            } else if (edge instanceof CommentEdge) {
 280.                sourceModelElement = ((CommentEdge) edge).getSource();
 281.                destModelElement = ((CommentEdge) edge).getDestination();
 282.            } else {
 283.                return false;
 284.            }
 285.    
 286.            // Both ends must be defined and nodes that are on the graph already.
 287.            if (sourceModelElement == null || destModelElement == null) {
 288.                LOG.error("Edge rejected. Its ends are not attached to anything");
 289.                return false;
 290.            }
 291.    
 292.            if (!containsNode(sourceModelElement)
 293.                    && !containsEdge(sourceModelElement)) {
 294.                LOG.error("Edge rejected. Its source end is attached to "
 295.                        + sourceModelElement
 296.                        + " but this is not in the graph model");
 297.                return false;
 298.            }
 299.            if (!containsNode(destModelElement)
 300.                    && !containsEdge(destModelElement)) {
 301.                LOG.error("Edge rejected. Its destination end is attached to "
 302.                        + destModelElement
 303.                        + " but this is not in the graph model");
 304.                return false;
 305.            }
 306.    
 307.            return true;
 308.        }
 309.    
 310.    
 311.        /**
 312.         * Add the given node to the graph, if valid.<p>
 313.         *
 314.         * We add the node if it is not already on the graph, and
 315.         * (assuming it to be an actor or use case) add it to the owned
 316.         * elements for the model.<p>
 317.         *
 318.         * <em>Needs more work</em>. In adding the node to the owned
 319.         * elements of the model namespace, we are implicitly making it
 320.         * public visibility (it could be private to this namespace).<p>
 321.         *
 322.         * <em>Note</em>.  This method is inconsistent with
 323.         * {@link #canAddNode}, which will allow a node to be added to the
 324.         * graph if it is already there.<p>
 325.         *
 326.         * @param node  The node to be added to the graph.
 327.         */
 328.        @Override
 329.        public void addNode(Object node) {
 330.    
 331.            LOG.debug("adding usecase node");
 332.    
 333.            // Give up if we are already on the graph. This is a bit inconistent
 334.            // with canAddNode above.
 335.    
 336.            if (!canAddNode(node)) {
 337.                return;
 338.            }
 339.    
 340.            // Add the node, check that it is an actor or use case and add it to
 341.            // the model namespace.
 342.    
 343.            getNodes().add(node);
 344.    
 345.            if (Model.getFacade().isAUMLElement(node)
 346.                    && Model.getFacade().getNamespace(node) == null) {
 347.                Model.getCoreHelper().addOwnedElement(getHomeModel(), node);
 348.            }
 349.    
 350.            // Tell GEF its changed
 351.    
 352.            fireNodeAdded(node);
 353.        }
 354.    
 355.    
 356.        /**
 357.         * Add the given edge to the graph, if valid.<p>
 358.         *
 359.         * We add the edge if it is not already on the graph, and
 360.         * (assuming it to be an association, generalization, extend,
 361.         * include or dependency) add it to the owned elements for the
 362.         * model.<p>
 363.         *
 364.         * <em>Needs more work</em>. In adding the edge to the owned
 365.         * elements of the model namespace, we are implicitly making it
 366.         * public visibility (it could be private to this namespace).<p>
 367.         *
 368.         * @param edge  The edge to be added to the graph.
 369.         */
 370.        @Override
 371.        public void addEdge(Object edge) {
 372.            if (edge == null) {
 373.                throw new IllegalArgumentException("Cannot add a null edge");
 374.            }
 375.    
 376.            if (getDestPort(edge) == null || getSourcePort(edge) == null) {
 377.                throw new IllegalArgumentException(
 378.                        "The source and dest port should be provided on an edge");
 379.            }
 380.    
 381.            if (LOG.isInfoEnabled()) {
 382.                LOG.info("Adding an edge of type "
 383.                       + edge.getClass().getName()
 384.                       + " to use case diagram.");
 385.            }
 386.    
 387.            if (!canAddEdge(edge)) {
 388.                LOG.info("Attempt to add edge rejected");
 389.                return;
 390.            }
 391.    
 392.            // Add the element and place it in the namespace of the model
 393.            getEdges().add(edge);
 394.    
 395.            // TODO: assumes public
 396.            if (Model.getFacade().isAUMLElement(edge)
 397.                    && Model.getFacade().getNamespace(edge) == null) {
 398.                Model.getCoreHelper().addOwnedElement(getHomeModel(), edge);
 399.            }
 400.    
 401.            // Tell GEF
 402.    
 403.            fireEdgeAdded(edge);
 404.        }
 405.    
 406.        /**
 407.         * Add the various types of edge that may be connected with the
 408.         * given node.<p>
 409.         *
 410.         * For use cases we may find extend and include relationships. For
 411.         * classifiers (effectively actors and use cases) we may find
 412.         * associations. For generalizable elements (effectively actors
 413.         * and use cases again) we may find generalizations and
 414.         * specializations. For ModelElements (effectively actors and use
 415.         * cases again) we may find dependencies.<p>
 416.         *
 417.         * @param node  The node whose edges are to be added.
 418.         */
 419.        @Override
 420.        public void addNodeRelatedEdges(Object node) {
 421.            super.addNodeRelatedEdges(node);
 422.    
 423.            if (Model.getFacade().isAUseCase(node)) {
 424.                List relations = new ArrayList();
 425.    
 426.                relations.addAll(Model.getFacade().getIncludes(node));
 427.                relations.addAll(Model.getFacade().getIncluders(node));
 428.                relations.addAll(Model.getFacade().getExtends(node));
 429.                relations.addAll(Model.getFacade().getExtenders(node));
 430.    
 431.                for (Object relation : relations) {
 432.                    if (canAddEdge(relation)) {
 433.                        addEdge(relation);
 434.                    }
 435.                }
 436.            }
 437.    
 438.            if (Model.getFacade().isAClassifier(node)) {
 439.                Collection ends = Model.getFacade().getAssociationEnds(node);
 440.                for (Object ae : ends) {
 441.                    if (canAddEdge(Model.getFacade().getAssociation(ae))) {
 442.                        addEdge(Model.getFacade().getAssociation(ae));
 443.                    }
 444.                }
 445.            }
 446.    
 447.            if (Model.getFacade().isAGeneralizableElement(node)) {
 448.                Collection gn = Model.getFacade().getGeneralizations(node);
 449.                for (Object g : gn) {
 450.                    if (canAddEdge(g)) {
 451.                        addEdge(g);
 452.                    }
 453.                }
 454.                Collection sp = Model.getFacade().getSpecializations(node);
 455.                for (Object s : sp) {
 456.                    if (canAddEdge(s)) {
 457.                        addEdge(s);
 458.                    }
 459.                }
 460.            }
 461.    
 462.            if (Model.getFacade().isAUMLElement(node)) {
 463.                Collection dependencies =
 464.                    new ArrayList(Model.getFacade().getClientDependencies(node));
 465.    
 466.                dependencies.addAll(Model.getFacade().getSupplierDependencies(node));
 467.    
 468.                for (Object dependency : dependencies) {
 469.                    if (canAddEdge(dependency)) {
 470.                        addEdge(dependency);
 471.                    }
 472.                }
 473.            }
 474.        }
 475.    
 476.    
 477.    
 478.        /**
 479.         * Determine if the two given ports can be connected by a kind of
 480.         * edge to be determined by the ports.<p>
 481.         *
 482.         * <em>Note</em>. There appears to be a problem with the
 483.         * implementation, since it suggests actors cannot connect. In
 484.         * fact generalization is permitted, and this works, suggesting
 485.         * this method is not actually invoked in the current
 486.         * implementation of ArgoUML.<p>
 487.         *
 488.         * @param fromP  The source port of the connection
 489.         *
 490.         * @param toP    The destination port of the connection.
 491.         *
 492.         * @return       <code>true</code> if the two given ports can be connected
 493.         *               by a kind of edge to be determined by the
 494.         *               ports. <code>false</code> otherwise.
 495.         */
 496.        @Override
 497.        public boolean canConnect(Object fromP, Object toP) {
 498.    
 499.            // Suggest that actors may not connect (see JavaDoc comment about
 500.            // this).
 501.    
 502.            if (Model.getFacade().isAActor(fromP)
 503.                    && Model.getFacade().isAActor(toP)) {
 504.                return false;
 505.            }
 506.    
 507.            // Everything else is OK
 508.    
 509.            return true;
 510.        }

 513.        ///////////////////////////////////////////////////////////////////////////
 514.        //
 515.        // Methods that implement the VetoableChangeListener interface
 516.        //
 517.        ///////////////////////////////////////////////////////////////////////////
 518.    
 519.        /**
 520.         * Called when a property of interest has been changed - in this
 521.         * case the owned elements of the model. Provided to implement the
 522.         * {@link VetoableChangeListener} interface.<p>
 523.         *
 524.         * We could throw a PropertyVetoException if we wished to allow
 525.         * the change to be rolled back, but we don't.<p>
 526.         *
 527.         * @param pce  The event that triggered us, and from which we can extract
 528.         *             the name of the property that triggered us.
 529.         */
 530.        public void vetoableChange(PropertyChangeEvent pce) {
 531.    
 532.            // Only interested in the "ownedElement" property. Either something has
 533.            // been added to the namespace for this model, or removed. In the
 534.            // latter case the "something" will be in the old value of the
 535.            // property, which is the collection of owned elements, and the new value
 536.            // will be the element import describing the model element and the
 537.            // model from which it was removed
 538.    
 539.            if ("ownedElement".equals(pce.getPropertyName())) {
 540.                List oldOwned = (List) pce.getOldValue();
 541.    
 542.                Object eo = /*(MElementImport)*/ pce.getNewValue();
 543.                Object  me = Model.getFacade().getModelElement(eo);
 544.    
 545.                // If the element import is in the old owned, it means it must have
 546.                // been removed. Make sure the associated model element is removed.
 547.    
 548.                if (oldOwned.contains(eo)) {
 549.    
 550.                    LOG.debug("model removed " + me);
 551.    
 552.                    // Remove a node
 553.    
 554.                    if ((Model.getFacade().isAActor(me))
 555.                || (Model.getFacade().isAUseCase(me))) {
 556.    
 557.                        removeNode(me);
 558.                    } else if ((Model.getFacade().isAAssociation(me))
 559.                 || (Model.getFacade().isAGeneralization(me))
 560.                 || (Model.getFacade().isAExtend(me))
 561.                 || (Model.getFacade().isAInclude(me))
 562.                 || (Model.getFacade().isADependency(me))) {
 563.                        // Remove an edge
 564.                        removeEdge(me);
 565.                    }
 566.                } else {
 567.                    // Something was added - nothing for us to worry about
 568.                    LOG.debug("model added " + me);
 569.                }
 570.            }
 571.        }
