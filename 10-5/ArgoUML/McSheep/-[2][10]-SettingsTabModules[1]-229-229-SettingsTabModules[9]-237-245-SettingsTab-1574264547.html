 175.            fieldGraphicsFormat.removeAllItems();

 177.            fieldGraphicsFormat.setModel(new DefaultComboBoxModel(c.toArray()));
 178.    
 179.            fieldGraphicsResolution.removeAllItems();
 180.            fieldGraphicsResolution.setModel(new DefaultComboBoxModel(
 181.                    theResolutions.toArray()));

 183.                Configuration.getInteger(
 184.                    SaveGraphicsManager.KEY_GRAPHICS_RESOLUTION, 1);

 187.                    fieldGraphicsResolution.setSelectedItem(gr);
 188.                    break;

 191.        }
