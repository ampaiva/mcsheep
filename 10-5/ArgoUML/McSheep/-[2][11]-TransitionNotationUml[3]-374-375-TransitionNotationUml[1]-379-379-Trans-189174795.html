 439.        /**
 440.         * Handle the Guard of a Transition.<p>
 441.         *
 442.         * We can distinct between 4 cases:<ol>
 443.         * <li>A guard is given. None exists yet.
 444.         * <li>The expression of the guard was present, but is altered.
 445.         * <li>A guard is not given. None exists yet.
 446.         * <li>The expression of the guard was present, but is removed.
 447.         * </ol>
 448.         *
 449.         * The reaction in these cases should be:<ol>
 450.         * <li>Create a new guard, set its name, language & expression,
 451.         *     and hook it to the transition.
 452.         * <li>Change the guard's expression. Leave the name & language
 453.         *     untouched. See also issue 2742.
 454.         * <li>Nop.
 455.         * <li>Unhook and erase the existing guard.
 456.         * </ol>
 457.         *
 458.         * @param trans the UML element transition
 459.         * @param guard the string that represents the guard expression
 460.         */
 461.        private void parseGuard(Object trans, String guard) {
 462.            Object g = Model.getFacade().getGuard(trans);
 463.            if (guard.length() > 0) {
 464.                if (g == null) {
 465.                    // case 1
 466.                    /*TODO: In the next line, I should use buildGuard(),
 467.                     * but it doesn't show the guard on the diagram...
 468.                     * Why? (MVW)
 469.                     */
 470.                    g = Model.getStateMachinesFactory().createGuard();
 471.                    if (g != null) {
 472.                        Model.getStateMachinesHelper().setExpression(g,
 473.                                Model.getDataTypesFactory()
 474.                                    .createBooleanExpression("", guard));
 475.                        Model.getCoreHelper().setName(g, "anon");
 476.                        Model.getCommonBehaviorHelper().setTransition(g, trans);
 477.    
 478.                        // NSUML does this (?)
 479.                        // Model.getFacade().setGuard(trans, g);
 480.                    }
 481.                } else {
 482.                    // case 2
 483.                    Object expr = Model.getFacade().getExpression(g);
 484.                    String language = "";
 485.    
 486.                    /* TODO: This does not work! (MVW)
 487.                     Model.getFacade().setBody(expr,guard);
 488.                     Model.getFacade().setExpression(g,expr); */
 489.    
 490.                    //hence a less elegant workaround that works:
 491.                    if (expr != null) {
 492.                        language = Model.getDataTypesHelper().getLanguage(expr);
 493.                    }
 494.                    Model.getStateMachinesHelper().setExpression(g,
 495.                            Model.getDataTypesFactory()
 496.                                    .createBooleanExpression(language, guard));
 497.                    /* TODO: In this case, the properties panel
 498.                     is not updated with the changed expression! */
 499.                }
 500.            } else {
 501.                if (g == null) {
 502.                    /* case 3 */
 503.                } else {
 504.                    // case 4
 505.                    delete(g); // erase it
 506.                }
 507.            }
 508.        }

 510.        /**
 511.         * Handle the Effect (Action) of a Transition.<p>
 512.         *
 513.         * We can distinct between 4 cases:<ul>
 514.         * <li>1. An effect is given. None exists yet.
 515.         * <li>2. The expression of the effect was present, but is altered.
 516.         * <li>3. An effect is not given. None exists yet.
 517.         * <li>4. The expression of the effect was present, but is removed.
 518.         * </ul>
 519.         *
 520.         * The reaction in these cases should be:<ul>
 521.         * <li>1. Create a new CallAction, set its name, language &
 522.         * expression, and hook it to the transition.
 523.         * <li>2. Change the effect's expression. Leave the actiontype, name
 524.         * & language untouched.
 525.         * <li>3. Nop.
 526.         * <li>4. Unhook and erase the existing effect.
 527.         * </ul>
 528.         *
 529.         * @param actions the string to be parsed
 530.         * @param trans the transition that causes the effect (actions)
 531.         */
 532.        private void parseEffect(Object trans, String actions) {
 533.            Object effect = Model.getFacade().getEffect(trans);
 534.            if (actions.length() > 0) {
 535.                if (effect == null) { // case 1
 536.                    effect =
 537.                        Model.getCommonBehaviorFactory()
 538.                            .createCallAction();
 539.                    /* And hook it to the transition immediately,
 540.                     * so that an exception can not cause it to remain dangling: */
 541.                    Model.getStateMachinesHelper().setEffect(trans, effect);
 542.                    Model.getCommonBehaviorHelper().setScript(effect,
 543.                            Model.getDataTypesFactory()
 544.                                    .createActionExpression(""/*language*/,
 545.                                                            actions));
 546.                    Model.getCoreHelper().setName(effect, "anon");
 547.                } else { // case 2
 548.                    Object script = Model.getFacade().getScript(effect);
 549.                    String language = (script == null) ? null
 550.                            : Model.getDataTypesHelper().getLanguage(script);
 551.                    Model.getCommonBehaviorHelper().setScript(effect,
 552.                            Model.getDataTypesFactory()
 553.                                    .createActionExpression(language, actions));
 554.                }
 555.            } else { // case 3 & 4
 556.                if (effect == null) {
 557.                    // case 3
 558.                } else {
 559.                    // case 4
 560.                    delete(effect); // erase it
 561.                }
 562.            }
 563.        }

 593.            Object effect = Model.getFacade().getEffect(modelElement);

 598.                t += " [" + g + "]";

 606.        /**
 607.         * Generates the text for a (trigger) event.
 608.         *
 609.         * @param m Object of any MEvent kind
 610.         * @return the string representing the event
 611.         */
 612.        private String generateEvent(Object m) {
 613.            if (m == null) {
 614.                return "";
 615.            }
 616.            StringBuffer event = new StringBuffer();
 617.            if (Model.getFacade().isAChangeEvent(m)) {
 618.                event.append("when(");
 619.                event.append(
 620.                        generateExpression(Model.getFacade().getExpression(m)));
 621.                event.append(")");
 622.            } else if (Model.getFacade().isATimeEvent(m)) {
 623.                event.append("after(");
 624.                event.append(
 625.                        generateExpression(Model.getFacade().getExpression(m)));
 626.                event.append(")");
 627.            } else if (Model.getFacade().isASignalEvent(m)) {
 628.                event.append(Model.getFacade().getName(m));
 629.            } else if (Model.getFacade().isACallEvent(m)) {
 630.                event.append(Model.getFacade().getName(m));
 631.                event.append(generateParameterList(m));
 632.            }
 633.            return event.toString();
 634.        }

 636.        /**
 637.         * Generates a string representing the given Guard. <p>
 638.         *
 639.         * If there is an expression, then its body text is returned.
 640.         * Else, a 0 length string is returned. <p>
 641.         *
 642.         * Apparently, the AndroMDA people are convinced that the
 643.         * name of the guard should be shown on the diagram, while this
 644.         * is not correct according the UML standard.
 645.         * ArgoUML does not support this feature because of its down-side:
 646.         * The name would end up in the expression
 647.         * if you edit the transition text on the diagram,
 648.         * while the name remains containing the old value.
 649.         *
 650.         * @param m the UML Guard object
 651.         * @return a string
 652.         */
 653.        private String generateGuard(Object m) {
 654.            if (m != null) {
 655.                if (Model.getFacade().getExpression(m) != null) {
 656.                    return generateExpression(Model.getFacade().getExpression(m));
 657.                }
 658.            }
 659.            return "";
 660.        }

 662.        /**
 663.         * Generates a list of parameters. The parameters belong to the
 664.         * given object.  The returned string will have the following
 665.         * syntax:<p>
 666.         *
 667.         * (param1, param2, param3, ..., paramN)<p>
 668.         *
 669.         * If there are no parameters, then "()" is returned.
 670.         *
 671.         * @param parameterListOwner the 'owner' of the parameters
 672.         * @return the generated parameter list
 673.         */
 674.        private String generateParameterList(Object parameterListOwner) {
 675.            Iterator it =
 676.                Model.getFacade().getParameters(parameterListOwner).iterator();
 677.            StringBuffer list = new StringBuffer();
 678.            list.append("(");
 679.            if (it.hasNext()) {
 680.                while (it.hasNext()) {
 681.                    Object param = it.next();
 682.                    list.append(generateParameter(param));
 683.                    if (it.hasNext()) {
 684.                        list.append(", ");
 685.                    }
 686.                }
 687.            }
 688.            list.append(")");
 689.            return list.toString();
 690.        }

 697.                }
 698.            }

 700.        }

 702.        /**
 703.         * Generates the representation of a parameter on the display
 704.         * (diagram). The string to be returned will have the following
 705.         * syntax:<p>
 706.         *
 707.         * kind name : type-expression = default-value
 708.         *
 709.         * @param parameter the parameter
 710.         * @return the generated text
 711.         */
 712.        public String generateParameter(Object parameter) {
 713.            StringBuffer s = new StringBuffer();
 714.            s.append(generateKind(Model.getFacade().getKind(parameter)));
 715.            if (s.length() > 0) {
 716.                s.append(" ");
 717.            }
 718.            s.append(Model.getFacade().getName(parameter));
 719.            String classRef =
 720.                generateClassifierRef(Model.getFacade().getType(parameter));
 721.            if (classRef.length() > 0) {
 722.                s.append(" : ");
 723.                s.append(classRef);
 724.            }
 725.            String defaultValue =
 726.                generateExpression(Model.getFacade().getDefaultValue(parameter));
 727.            if (defaultValue.length() > 0) {
 728.                s.append(" = ");
 729.                s.append(defaultValue);
 730.            }
 731.            return s.toString();
 732.        }

 740.                s.append("inout");
 741.            } else if (kind == Model.getDirectionKind().getReturnParameter()) {
 742.                // return nothing
 743.            } else if (kind == Model.getDirectionKind().getOutParameter()) {

 745.            }

 747.        }

  89.        /**
  90.         * Parse a transition description line of the form:<pre>
  91.         *    "event-signature [guard-condition] / action-expression".
  92.         * </pre>
  93.         *
  94.         * A ";" is not interpreted as having any special meaning. <p>
  95.         *
  96.         * The "event-signature" may be one of the 4
  97.         * formats:<ul>
  98.         * <li> ChangeEvent: "when(condition)"
  99.         * <li> TimeEvent: "after(duration)"
 100.         * <li> CallEvent: "a(parameter-list)".
 101.         * <li> SignalEvent: any string without ().
 102.         * </ul>
 103.         *
 104.         * Remark: The UML standard does not make a distinction between
 105.         * the syntax of a CallEvent and SignalEvent:
 106.         * both may have parameters between ().
 107.         * For simplicity and user-friendliness, we chose for this distinction.
 108.         * If a user wants parameters for a SignalEvent,
 109.         * then he may add them in the properties panels, but not on the diagram.
 110.         * <p>
 111.         *
 112.         * An alternative solution would be to create a CallEvent by default,
 113.         * and when editing an existing event, do not change the type.<p>
 114.         *
 115.         * TODO: This function fails when the event-signature contains a "["
 116.         * or a "/". See issue 5983 for other cases that were 
 117.         * a problem in the past.
 118.         *
 119.         * @param trans the transition object to which this string applies
 120.         * @param s     the string to be parsed
 121.         * @return      the transition object
 122.         * @throws ParseException when no matching [] are found
 123.         */
 124.        protected Object parseTransition(Object trans, String s)
 125.            throws ParseException {
 126.            s = s.trim();
 127.    
 128.            int a = s.indexOf("[");
 129.            int b = s.indexOf("]");
 130.            int c = s.indexOf("/");
 131.            if (((a < 0) && (b >= 0)) || ((b < 0) && (a >= 0)) || (b < a)) {
 132.                String msg = "parsing.error.transition.no-matching-square-brackets";
 133.                throw new ParseException(Translator.localize(msg), 0);
 134.            }
 135.            if ((c >= 0) && (c < b) && (c > a) && (a > 0)) {
 136.                String msg = "parsing.error.transition.found-bracket-instead-slash";
 137.                throw new ParseException(Translator.localize(msg), 0);
 138.            }
 139.    
 140.            String[] s1 = s.trim().split("/", 2);
 141.            String eg = s1[0].trim();
 142.            String[] s2 = eg.split("\\[", 2);
 143.            
 144.            if (s2[0].trim().length() > 0) {
 145.                parseTrigger(trans, s2[0].trim());
 146.            }
 147.            if (s2.length > 1)  {
 148.                if (s2[1].trim().endsWith("]")) {
 149.                    String g = s2[1].trim();
 150.                    g = g.substring(0, g.length() - 1).trim();
 151.                    if (g.length() > 0) {
 152.                        parseGuard(trans, g);
 153.                    }
 154.                }
 155.            }
 156.            if (s1.length > 1)  {
 157.                if (s1[1].trim().length() > 0) {
 158.                    parseEffect(trans, s1[1].trim()); 
 159.                }
 160.            }
 161.            return trans;
 162.        }

 164.        /**
 165.         * Parse the Event that is the trigger of the given transition.
 166.         *
 167.         * @param trans the transition which is triggered by the given event
 168.         * @param trigger the given trigger
 169.         * @throws ParseException
 170.         */
 171.        private void parseTrigger(Object trans, String trigger)
 172.            throws ParseException {
 173.            // let's look for a TimeEvent, ChangeEvent, CallEvent or SignalEvent
 174.            String s = "";
 175.            boolean timeEvent = false;
 176.            boolean changeEvent = false;
 177.            boolean callEvent = false;
 178.            boolean signalEvent = false;
 179.            trigger = trigger.trim();
 180.    
 181.            StringTokenizer tokenizer = new StringTokenizer(trigger, "()");
 182.            String name = tokenizer.nextToken().trim();
 183.            if (name.equalsIgnoreCase("after")) {
 184.                timeEvent = true;
 185.            } else if (name.equalsIgnoreCase("when")) {
 186.                changeEvent = true;
 187.            } else {
 188.                // the part after the || is for when there's nothing between the ()
 189.                if (tokenizer.hasMoreTokens()
 190.                        || (trigger.indexOf("(") > 0)
 191.                        || (trigger.indexOf(")") > 1)) {
 192.                    callEvent = true;
 193.                    if (!trigger.endsWith(")") || !(trigger.indexOf("(") > 0)) {
 194.                        String msg =
 195.                            "parsing.error.transition.no-matching-brackets";
 196.                        throw new ParseException(
 197.                                Translator.localize(msg), 0);
 198.                    }
 199.                } else {
 200.                    signalEvent = true;
 201.                }
 202.            }
 203.            if (timeEvent || changeEvent || callEvent) {
 204.                if (tokenizer.hasMoreTokens()) {
 205.                    s = tokenizer.nextToken().trim();
 206.                } // else the empty s will do
 207.            }
 208.    
 209.            /*
 210.             * We can distinguish between 4 cases:
 211.             * 1. A trigger is given. None exists yet.
 212.             * 2. The trigger was present, and it is the same type,
 213.             * or a different type, and its text is changed, or the same.
 214.             * 3. A trigger is not given. None exists yet.
 215.             * 4. The name of the trigger was present, but is removed.
 216.             * The reaction in these cases should be:
 217.             * 1. Find the referred trigger (issue 5988) or create a new one, and hook it to the transit...
 218.             * 2. Rename the trigger.
 219.             * 3. Nop.
 220.             * 4. Unhook and erase the existing trigger.
 221.             */
 222.            Object evt = Model.getFacade().getTrigger(trans);
 223.            /* It is safe to give a null to the next function,
 224.             * since a statemachine is always composed by a model anyhow. */
 225.            Object ns =
 226.                Model.getStateMachinesHelper()
 227.                    .findNamespaceForEvent(trans, null);
 228.            StateMachinesFactory sMFactory =
 229.                    Model.getStateMachinesFactory();
 230.            boolean weHaveAnEvent = false;
 231.            if (trigger.length() > 0) {
 232.                // case 1 and 2
 233.                if (evt == null) {
 234.                    // case 1
 235.                    if (timeEvent) { // after(...)
 236.                        evt = findOrBuildTimeEvent(s, ns);
 237.                        /* Do not set the name. */
 238.                    }
 239.                    if (changeEvent) { // when(...)
 240.                        evt = findOrBuildChangeEvent(s, ns);
 241.                        /* Do not set the name. */
 242.                    }
 243.                    if (callEvent) { // operation(paramlist)
 244.                        String triggerName =
 245.                            trigger.indexOf("(") > 0
 246.                            ? trigger.substring(0, trigger.indexOf("(")).trim()
 247.                            : trigger;
 248.                        /* This case is a bit different, because of the parameters. 
 249.                         * If the event already exists, the parameters are ignored. */
 250.                        evt = findCallEvent(triggerName, ns);
 251.                        if (evt == null) {
 252.                            evt = sMFactory.buildCallEvent(trans, triggerName, ns);
 253.                            // and parse the parameter list
 254.                            NotationUtilityUml.parseParamList(evt, s, 0);
 255.                        }
 256.                    }
 257.                    if (signalEvent) { // signalname
 258.                        evt = findOrBuildSignalEvent(trigger, ns);
 259.                    }
 260.                    weHaveAnEvent = true;
 261.                } else {
 262.                    // case 2
 263.                    if (timeEvent) {
 264.                        if (Model.getFacade().isATimeEvent(evt)) {
 265.                            /* Just change the time expression */
 266.                            Object timeExpr = Model.getFacade().getWhen(evt);
 267.                            if (timeExpr == null) {
 268.                                // we have an event without expression
 269.                                timeExpr = Model.getDataTypesFactory().createTimeExpression("", s);
 270.                                Model.getStateMachinesHelper().setWhen(evt, timeExpr);
 271.                            } else {
 272.                                Model.getDataTypesHelper().setBody(timeExpr, s);
 273.                            }
 274.                        } else {
 275.                            /* It's a time-event now,
 276.                             * but was of another type before! */
 277.                            delete(evt); /* TODO: What if used elsewhere? */
 278.                            evt = sMFactory.buildTimeEvent(s, ns);
 279.                            weHaveAnEvent = true;
 280.                        }
 281.                    }
 282.                    if (changeEvent) {
 283.                        if (Model.getFacade().isAChangeEvent(evt)) {
 284.                            /* Just change the ChangeExpression */
 285.                            Object changeExpr =
 286.                                Model.getFacade().getChangeExpression(evt);
 287.                            if (changeExpr == null) {
 288.                                /* Create a new expression: */
 289.                                changeExpr = Model.getDataTypesFactory()
 290.                                    .createBooleanExpression("", s);
 291.                                Model.getStateMachinesHelper().setExpression(evt,
 292.                                        changeExpr);
 293.                            } else {
 294.                                Model.getDataTypesHelper().setBody(changeExpr, s);
 295.                            }
 296.                        } else {
 297.                            /* The parsed text describes a change-event,
 298.                             * but the model contains another type! */
 299.                            delete(evt); /* TODO: What if used elsewhere? */
 300.                            evt = sMFactory.buildChangeEvent(s, ns);
 301.                            weHaveAnEvent = true;
 302.                        }
 303.                    }
 304.                    if (callEvent) {
 305.                        if (Model.getFacade().isACallEvent(evt)) {
 306.                            /* Just change the Name and linked operation */
 307.                            String triggerName =
 308.                                trigger.indexOf("(") > 0
 309.                                ? trigger.substring(0, trigger.indexOf("(")).trim()
 310.                                        : trigger;
 311.                            if (!Model.getFacade().getName(evt)
 312.                                    .equals(triggerName)) {
 313.                                Model.getCoreHelper().setName(evt, triggerName);
 314.                            }
 315.                                /* TODO: Change the linked operation. */
 316.                        } else {
 317.                            delete(evt); /* TODO: What if used elsewhere? */
 318.                            evt = sMFactory.buildCallEvent(trans, trigger, ns);
 319.                            // and parse the parameter list
 320.                            NotationUtilityUml.parseParamList(evt, s, 0);
 321.                            weHaveAnEvent = true;
 322.                        }
 323.                    }
 324.                    if (signalEvent) {
 325.                        if (Model.getFacade().isASignalEvent(evt)) {
 326.                            /* Just change the Name and linked signal */
 327.                            if (!Model.getFacade().getName(evt).equals(trigger)) {
 328.                                Model.getCoreHelper().setName(evt, trigger);
 329.                            }
 330.                            /* TODO: link to the Signal. */
 331.                        } else {
 332.                            delete(evt); /* TODO: What if used elsewhere? */
 333.                            evt = sMFactory.buildSignalEvent(trigger, ns);
 334.                            weHaveAnEvent = true;
 335.                        }
 336.                    }
 337.                }
 338.                if (weHaveAnEvent && (evt != null)) {
 339.                    Model.getStateMachinesHelper().setEventAsTrigger(trans, evt);
 340.                }
 341.            } else {
 342.                // case 3 and 4
 343.                if (evt == null) {
 344.                    /* case 3 */
 345.                } else {
 346.                    // case 4
 347.                    delete(evt); // erase it
 348.                }
 349.            }
 350.        }

 439.        /**
 440.         * Handle the Guard of a Transition.<p>
 441.         *
 442.         * We can distinct between 4 cases:<ol>
 443.         * <li>A guard is given. None exists yet.
 444.         * <li>The expression of the guard was present, but is altered.
 445.         * <li>A guard is not given. None exists yet.
 446.         * <li>The expression of the guard was present, but is removed.
 447.         * </ol>
 448.         *
 449.         * The reaction in these cases should be:<ol>
 450.         * <li>Create a new guard, set its name, language & expression,
 451.         *     and hook it to the transition.
 452.         * <li>Change the guard's expression. Leave the name & language
 453.         *     untouched. See also issue 2742.
 454.         * <li>Nop.
 455.         * <li>Unhook and erase the existing guard.
 456.         * </ol>
 457.         *
 458.         * @param trans the UML element transition
 459.         * @param guard the string that represents the guard expression
 460.         */
 461.        private void parseGuard(Object trans, String guard) {
 462.            Object g = Model.getFacade().getGuard(trans);
 463.            if (guard.length() > 0) {
 464.                if (g == null) {
 465.                    // case 1
 466.                    /*TODO: In the next line, I should use buildGuard(),
 467.                     * but it doesn't show the guard on the diagram...
 468.                     * Why? (MVW)
 469.                     */
 470.                    g = Model.getStateMachinesFactory().createGuard();
 471.                    if (g != null) {
 472.                        Model.getStateMachinesHelper().setExpression(g,
 473.                                Model.getDataTypesFactory()
 474.                                    .createBooleanExpression("", guard));
 475.                        Model.getCoreHelper().setName(g, "anon");
 476.                        Model.getCommonBehaviorHelper().setTransition(g, trans);
 477.    
 478.                        // NSUML does this (?)
 479.                        // Model.getFacade().setGuard(trans, g);
 480.                    }
 481.                } else {
 482.                    // case 2
 483.                    Object expr = Model.getFacade().getExpression(g);
 484.                    String language = "";
 485.    
 486.                    /* TODO: This does not work! (MVW)
 487.                     Model.getFacade().setBody(expr,guard);
 488.                     Model.getFacade().setExpression(g,expr); */
 489.    
 490.                    //hence a less elegant workaround that works:
 491.                    if (expr != null) {
 492.                        language = Model.getDataTypesHelper().getLanguage(expr);
 493.                    }
 494.                    Model.getStateMachinesHelper().setExpression(g,
 495.                            Model.getDataTypesFactory()
 496.                                    .createBooleanExpression(language, guard));
 497.                    /* TODO: In this case, the properties panel
 498.                     is not updated with the changed expression! */
 499.                }
 500.            } else {
 501.                if (g == null) {
 502.                    /* case 3 */
 503.                } else {
 504.                    // case 4
 505.                    delete(g); // erase it
 506.                }
 507.            }
 508.        }

 510.        /**
 511.         * Handle the Effect (Action) of a Transition.<p>
 512.         *
 513.         * We can distinct between 4 cases:<ul>
 514.         * <li>1. An effect is given. None exists yet.
 515.         * <li>2. The expression of the effect was present, but is altered.
 516.         * <li>3. An effect is not given. None exists yet.
 517.         * <li>4. The expression of the effect was present, but is removed.
 518.         * </ul>
 519.         *
 520.         * The reaction in these cases should be:<ul>
 521.         * <li>1. Create a new CallAction, set its name, language &
 522.         * expression, and hook it to the transition.
 523.         * <li>2. Change the effect's expression. Leave the actiontype, name
 524.         * & language untouched.
 525.         * <li>3. Nop.
 526.         * <li>4. Unhook and erase the existing effect.
 527.         * </ul>
 528.         *
 529.         * @param actions the string to be parsed
 530.         * @param trans the transition that causes the effect (actions)
 531.         */
 532.        private void parseEffect(Object trans, String actions) {
 533.            Object effect = Model.getFacade().getEffect(trans);
 534.            if (actions.length() > 0) {
 535.                if (effect == null) { // case 1
 536.                    effect =
 537.                        Model.getCommonBehaviorFactory()
 538.                            .createCallAction();
 539.                    /* And hook it to the transition immediately,
 540.                     * so that an exception can not cause it to remain dangling: */
 541.                    Model.getStateMachinesHelper().setEffect(trans, effect);
 542.                    Model.getCommonBehaviorHelper().setScript(effect,
 543.                            Model.getDataTypesFactory()
 544.                                    .createActionExpression(""/*language*/,
 545.                                                            actions));
 546.                    Model.getCoreHelper().setName(effect, "anon");
 547.                } else { // case 2
 548.                    Object script = Model.getFacade().getScript(effect);
 549.                    String language = (script == null) ? null
 550.                            : Model.getDataTypesHelper().getLanguage(script);
 551.                    Model.getCommonBehaviorHelper().setScript(effect,
 552.                            Model.getDataTypesFactory()
 553.                                    .createActionExpression(language, actions));
 554.                }
 555.            } else { // case 3 & 4
 556.                if (effect == null) {
 557.                    // case 3
 558.                } else {
 559.                    // case 4
 560.                    delete(effect); // erase it
 561.                }
 562.            }
 563.        }

 590.        private String toString(Object modelElement) {
 591.            Object trigger = Model.getFacade().getTrigger(modelElement);
 592.            Object guard = Model.getFacade().getGuard(modelElement);
 593.            Object effect = Model.getFacade().getEffect(modelElement);

 597.            if (g.length() > 0) {

 600.            if (e.length() > 0) {

 606.        /**
 607.         * Generates the text for a (trigger) event.
 608.         *
 609.         * @param m Object of any MEvent kind
 610.         * @return the string representing the event
 611.         */
 612.        private String generateEvent(Object m) {
 613.            if (m == null) {
 614.                return "";
 615.            }
 616.            StringBuffer event = new StringBuffer();
 617.            if (Model.getFacade().isAChangeEvent(m)) {
 618.                event.append("when(");
 619.                event.append(
 620.                        generateExpression(Model.getFacade().getExpression(m)));
 621.                event.append(")");
 622.            } else if (Model.getFacade().isATimeEvent(m)) {
 623.                event.append("after(");
 624.                event.append(
 625.                        generateExpression(Model.getFacade().getExpression(m)));
 626.                event.append(")");
 627.            } else if (Model.getFacade().isASignalEvent(m)) {
 628.                event.append(Model.getFacade().getName(m));
 629.            } else if (Model.getFacade().isACallEvent(m)) {
 630.                event.append(Model.getFacade().getName(m));
 631.                event.append(generateParameterList(m));
 632.            }
 633.            return event.toString();
 634.        }

 636.        /**
 637.         * Generates a string representing the given Guard. <p>
 638.         *
 639.         * If there is an expression, then its body text is returned.
 640.         * Else, a 0 length string is returned. <p>
 641.         *
 642.         * Apparently, the AndroMDA people are convinced that the
 643.         * name of the guard should be shown on the diagram, while this
 644.         * is not correct according the UML standard.
 645.         * ArgoUML does not support this feature because of its down-side:
 646.         * The name would end up in the expression
 647.         * if you edit the transition text on the diagram,
 648.         * while the name remains containing the old value.
 649.         *
 650.         * @param m the UML Guard object
 651.         * @return a string
 652.         */
 653.        private String generateGuard(Object m) {
 654.            if (m != null) {
 655.                if (Model.getFacade().getExpression(m) != null) {
 656.                    return generateExpression(Model.getFacade().getExpression(m));
 657.                }
 658.            }
 659.            return "";
 660.        }

 662.        /**
 663.         * Generates a list of parameters. The parameters belong to the
 664.         * given object.  The returned string will have the following
 665.         * syntax:<p>
 666.         *
 667.         * (param1, param2, param3, ..., paramN)<p>
 668.         *
 669.         * If there are no parameters, then "()" is returned.
 670.         *
 671.         * @param parameterListOwner the 'owner' of the parameters
 672.         * @return the generated parameter list
 673.         */
 674.        private String generateParameterList(Object parameterListOwner) {
 675.            Iterator it =
 676.                Model.getFacade().getParameters(parameterListOwner).iterator();
 677.            StringBuffer list = new StringBuffer();
 678.            list.append("(");
 679.            if (it.hasNext()) {
 680.                while (it.hasNext()) {
 681.                    Object param = it.next();
 682.                    list.append(generateParameter(param));
 683.                    if (it.hasNext()) {
 684.                        list.append(", ");
 685.                    }
 686.                }
 687.            }
 688.            list.append(")");
 689.            return list.toString();
 690.        }

 695.                if (body != null) {
 696.                    return (String) body;
 697.                }
 698.            }
 699.            return "";
 700.        }

 702.        /**
 703.         * Generates the representation of a parameter on the display
 704.         * (diagram). The string to be returned will have the following
 705.         * syntax:<p>
 706.         *
 707.         * kind name : type-expression = default-value
 708.         *
 709.         * @param parameter the parameter
 710.         * @return the generated text
 711.         */
 712.        public String generateParameter(Object parameter) {
 713.            StringBuffer s = new StringBuffer();
 714.            s.append(generateKind(Model.getFacade().getKind(parameter)));
 715.            if (s.length() > 0) {
 716.                s.append(" ");
 717.            }
 718.            s.append(Model.getFacade().getName(parameter));
 719.            String classRef =
 720.                generateClassifierRef(Model.getFacade().getType(parameter));
 721.            if (classRef.length() > 0) {
 722.                s.append(" : ");
 723.                s.append(classRef);
 724.            }
 725.            String defaultValue =
 726.                generateExpression(Model.getFacade().getDefaultValue(parameter));
 727.            if (defaultValue.length() > 0) {
 728.                s.append(" = ");
 729.                s.append(defaultValue);
 730.            }
 731.            return s.toString();
 732.        }

 734.        private String generateKind(Object /*Parameter etc.*/ kind) {
 735.            StringBuffer s = new StringBuffer();
 736.            if (kind == null /* "in" is the default */
 737.                    || kind == Model.getDirectionKind().getInParameter()) {

 740.                s.append("inout");
 741.            } else if (kind == Model.getDirectionKind().getReturnParameter()) {
 742.                // return nothing
 743.            } else if (kind == Model.getDirectionKind().getOutParameter()) {
 744.                s.append("out");
 745.            }
 746.            return s.toString();
 747.        }
