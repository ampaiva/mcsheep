  70.    /**
  71.     * Class providing access to domain mappings. Domain mappings are required to
  72.     * map uml datatypes (domains in db-language) to datatypes of a target database.
  73.     * This way there can be modelled <code>String</code>s, <code>int</code>s
  74.     * or <code>double</code>s for a database model. The DomainMapper maps
  75.     * <code>String</code> to a <code>VARCHAR(100)</code>, <code>int</code>
  76.     * to <code>INTEGER</code> or <code>double</code> to
  77.     * <code>DOUBLE PRECISION</code>, according its configuration. This
  78.     * configuration can be changed in the settings dialog.
  79.     * 
  80.     * @author drahmann
  81.     */

 167.        private InputStream getDomainMap() {
 168.            String domainMap = Configuration.getString(MAPPING_KEY);

 173.                    is = url.openStream();

 191.        public void load() {
 192.    
 193.            DocumentBuilderFactory docFactory = DocumentBuilderFactory
 194.                    .newInstance();
 195.            try {
 196.                DocumentBuilder docBuilder = docFactory.newDocumentBuilder();
 197.                Document document = docBuilder.parse(getDomainMap());
 198.                Element root = document.getDocumentElement();
 199.                NodeList childs = root.getChildNodes();
 200.    
 201.                for (int i = 0; i < childs.getLength(); i++) {
 202.                    Node child = childs.item(i);
 203.                    if (child.getNodeType() != Node.ELEMENT_NODE) {
 204.                        continue;
 205.                    }
 206.    
 207.                    NamedNodeMap attributes = child.getAttributes();
 208.                    String name = attributes.getNamedItem("name").getTextContent();
 209.    
 210.                    Map<String, String> mappings = getMappingsFor(name);
 211.                    readMappings(mappings, child.getChildNodes());
 212.                }
 213.            } catch (ParserConfigurationException e) {
 214.                LOG.error("Exception", e);
 215.            } catch (SAXException e) {
 216.                LOG.error("Exception", e);
 217.            } catch (IOException e) {
 218.                LOG.error("Exception", e);
 219.            }
 220.        }

 223.         * Save all mappings to the file domainmapping.xml located in the same

 228.        public void save() {
 229.            Writer sw = new StringWriter(1024);
 230.            try {

 232.                sw.write(GeneratorSql.LINE_SEPARATOR);
 233.                sw.write(ROOT_TAG);
 234.                sw.write(GeneratorSql.LINE_SEPARATOR);
 235.    
 236.                indent = "\t";
 237.                Set<Entry<String,Map<String,String>>> dbEntries = databases.entrySet();
 238.                for (Iterator<Entry<String, Map<String,String>>> it = dbEntries.iterator(); it.hasNext()...
 239.                    Entry<String, Map<String, String>> entry = it.next();
 240.                    String className = (String) entry.getKey();
 241.                    Map<String, String> mappings = entry.getValue();
 242.    
 243.                    StringBuffer sb = new StringBuffer();
 244.                    sb.append(indent);
 245.                    sb.append("<tns:database name=\"");
 246.                    sb.append(className);
 247.                    sb.append("\">").append(GeneratorSql.LINE_SEPARATOR);
 248.                    sw.write(sb.toString());
 249.    
 250.                    writeMappings(sw, mappings);
 251.    
 252.                    sw.write("</tns:database>");
 253.                }
 254.    
 255.                sw.write("</tns:mappings>");
 256.                sw.close();
 257.            } catch (IOException e) {
 258.                LOG.error("Exception", e);
 259.            }
 260.            
 261.            Configuration.setString(MAPPING_KEY, sw.toString());
 262.        }

 268.         *            The class of the code creator for which this mapping should be

 280.    
 281.        private void readMappings(Map<String, String> mappings, NodeList nodes) {
 282.            for (int i = 0; i < nodes.getLength(); i++) {
 283.                Node mapping = nodes.item(i);
 284.                if (mapping.getNodeType() != Node.ELEMENT_NODE) {
 285.                    continue;
 286.                }
 287.    
 288.                NamedNodeMap attributes = mapping.getAttributes();
 289.                Node src = attributes.getNamedItem("umltype");
 290.                Node dst = attributes.getNamedItem("dbtype");
 291.                String srcText = src.getTextContent();
 292.                String dstText = dst.getTextContent();
 293.    
 294.                mappings.put(srcText, dstText);
 295.            }
 296.        }

 298.        private void writeMappings(Writer fw, Map<String, String> mappings) throws IOException {
 299.            String oldIndent = indent;
 300.            indent += "\t";
 301.            Set<Entry<String, String>> entries = mappings.entrySet();
 302.            for (Iterator<Entry<String, String>> it = entries.iterator(); it.hasNext();) {
 303.                Entry<String, String> entry = it.next();
 304.                String domain = entry.getKey();
 305.                String datatype = entry.getValue();
 306.    
 307.                StringBuffer sb = new StringBuffer();
 308.                sb.append(indent);
 309.                sb.append("<tns:mapping umltype=\"");
 310.                sb.append(domain);
 311.                sb.append("\" dbtype=\"");
 312.                sb.append(datatype);
 313.                sb.append("\" />").append(GeneratorSql.LINE_SEPARATOR);
 314.    
 315.                fw.write(sb.toString());
 316.            }
 317.            indent = oldIndent;
 318.        }

  53.    import java.util.Set;

  55.

  70.    /**
  71.     * Class providing access to domain mappings. Domain mappings are required to
  72.     * map uml datatypes (domains in db-language) to datatypes of a target database.
  73.     * This way there can be modelled <code>String</code>s, <code>int</code>s
  74.     * or <code>double</code>s for a database model. The DomainMapper maps
  75.     * <code>String</code> to a <code>VARCHAR(100)</code>, <code>int</code>
  76.     * to <code>INTEGER</code> or <code>double</code> to
  77.     * <code>DOUBLE PRECISION</code>, according its configuration. This
  78.     * configuration can be changed in the settings dialog.
  79.     * 
  80.     * @author drahmann
  81.     */
  82.    public class DomainMapper {
  83.        private static final ConfigurationKey MAPPING_KEY = Configuration.makeKey("sql","domainmapping");
  84.        
  85.        private static final String ROOT_TAG = "<tns:mappings "
  86.                + "xmlns:tns=\"http://www.argouml.org/Namespace/argouml-sql\" "
  87.                + "xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" "
  88.                + "xsi:schemaLocation=\""
  89.                + "http://www.argouml.org/Namespace/argouml-sql domainmapping.xsd \">";
  90.    
  91.        private static final String XML_FILE_NAME = "domainmapping.xml";
  92.    
  93.        private static final String XML_TAG = "<?xml version=\"1.0\" "
  94.                + "encoding=\"UTF-8\"?>";
  95.    
  96.        private Map<String, Map<String,String>> databases;
  97.    
  98.        private String indent;
  99.    
 100.        private static final Logger LOG = Logger.getLogger(DomainMapper.class);
 101.    
 102.        /**
 103.         * Creates a new DomainMapper.
 104.         * 
 105.         */
 106.        public DomainMapper() {
 107.            databases = new HashMap<String, Map<String,String>>();
 108.            // TODO: lazy load mappings
 109.            load();
 110.        }
 111.    
 112.        /**
 113.         * Clears all mappings for the specified database code creator.
 114.         * 
 115.         * @param codeCreatorClass The class object of the code creator.
 116.         */
 117.        public void clear(Class codeCreatorClass) {
 118.            getMappingsFor(codeCreatorClass).clear();
 119.        }
 120.    
 121.        /**
 122.         * Returns the datatype for a given domain and code creator. The domain
 123.         * itself is returned if
 124.         * <ul>
 125.         * <li>there does not exist a mapping for the given code creator or</li>
 126.         * <li>there is no defined mapping for the given domain</li>
 127.         * </ul>
 128.         * 
 129.         * @param codeCreatorClass
 130.         *            The class of the code creator
 131.         * @param domain
 132.         *            The domain
 133.         * @return The database-specific datatype for the given domain
 134.         */
 135.        public String getDatatype(Class codeCreatorClass, String domain) {
 136.            Map<String, String> mappings = getMappingsFor(codeCreatorClass);
 137.            String datatype = domain;
 138.            if (mappings != null) {
 139.                String dt = mappings.get(domain);
 140.                if (dt != null) {
 141.                    datatype = dt;
 142.                }
 143.            }
 144.    
 145.            return datatype;
 146.        }
 147.    
 148.        private Map<String, String> getMappingsFor(String codeCreatorClassName) {
 149.            Map<String, String> mappings = databases.get(codeCreatorClassName);
 150.            if (mappings == null) {
 151.                mappings = new TreeMap<String, String>(String.CASE_INSENSITIVE_ORDER);
 152.                databases.put(codeCreatorClassName, mappings);
 153.            }
 154.            return mappings;
 155.        }
 156.    
 157.        /**
 158.         * 
 159.         * @param codeCreatorClass
 160.         * @return All mappings for the given database code creator class.
 161.         */
 162.        public Map<String, String> getMappingsFor(Class codeCreatorClass) {
 163.            return getMappingsFor(codeCreatorClass.getName());
 164.        }
 165.    
 166.    
 167.        private InputStream getDomainMap() {
 168.            String domainMap = Configuration.getString(MAPPING_KEY);
 169.            InputStream is = new StringBufferInputStream(domainMap);
 170.            if ("".equals(domainMap)) {
 171.                try {
 172.                    URL url = getClass().getResource(XML_FILE_NAME).toURI().toURL();
 173.                    is = url.openStream();
 174.                } catch (URISyntaxException e) {
 175.                    LOG.warn("Could not find domainmapping file", e);
 176.                } catch (MalformedURLException e) {
 177.                    LOG.warn("Could not find domainmapping file", e);
 178.                } catch (IOException e) {
 179.                    LOG.warn("Error reading/fetching domain map", e);
 180.                }
 181.            }
 182.            return is;
 183.        }
 184.    
 185.        /**
 186.         * Load all mappings from the file domainmapping.xml located in the same
 187.         * directory than the module.
 188.         * 
 189.         * TODO: This should first try to load from user preferences, then a URI.
 190.         */
 191.        public void load() {
 192.    
 193.            DocumentBuilderFactory docFactory = DocumentBuilderFactory
 194.                    .newInstance();
 195.            try {
 196.                DocumentBuilder docBuilder = docFactory.newDocumentBuilder();
 197.                Document document = docBuilder.parse(getDomainMap());
 198.                Element root = document.getDocumentElement();
 199.                NodeList childs = root.getChildNodes();
 200.    
 201.                for (int i = 0; i < childs.getLength(); i++) {
 202.                    Node child = childs.item(i);
 203.                    if (child.getNodeType() != Node.ELEMENT_NODE) {
 204.                        continue;
 205.                    }
 206.    
 207.                    NamedNodeMap attributes = child.getAttributes();
 208.                    String name = attributes.getNamedItem("name").getTextContent();
 209.    
 210.                    Map<String, String> mappings = getMappingsFor(name);
 211.                    readMappings(mappings, child.getChildNodes());
 212.                }
 213.            } catch (ParserConfigurationException e) {
 214.                LOG.error("Exception", e);
 215.            } catch (SAXException e) {
 216.                LOG.error("Exception", e);
 217.            } catch (IOException e) {
 218.                LOG.error("Exception", e);
 219.            }
 220.        }
 221.    
 222.        /**
 223.         * Save all mappings to the file domainmapping.xml located in the same
 224.         * directory than the module.
 225.         * 
 226.         * TODO: This needs to be modified to save use the user preference store
 227.         */
 228.        public void save() {
 229.            Writer sw = new StringWriter(1024);
 230.            try {
 231.                sw.write(XML_TAG);
 232.                sw.write(GeneratorSql.LINE_SEPARATOR);
 233.                sw.write(ROOT_TAG);
 234.                sw.write(GeneratorSql.LINE_SEPARATOR);
 235.    
 236.                indent = "\t";
 237.                Set<Entry<String,Map<String,String>>> dbEntries = databases.entrySet();
 238.                for (Iterator<Entry<String, Map<String,String>>> it = dbEntries.iterator(); it.hasNext()...
 239.                    Entry<String, Map<String, String>> entry = it.next();
 240.                    String className = (String) entry.getKey();
 241.                    Map<String, String> mappings = entry.getValue();
 242.    
 243.                    StringBuffer sb = new StringBuffer();
 244.                    sb.append(indent);
 245.                    sb.append("<tns:database name=\"");
 246.                    sb.append(className);
 247.                    sb.append("\">").append(GeneratorSql.LINE_SEPARATOR);
 248.                    sw.write(sb.toString());
 249.    
 250.                    writeMappings(sw, mappings);
 251.    
 252.                    sw.write("</tns:database>");
 253.                }
 254.    
 255.                sw.write("</tns:mappings>");
 256.                sw.close();
 257.            } catch (IOException e) {
 258.                LOG.error("Exception", e);
 259.            }
 260.            
 261.            Configuration.setString(MAPPING_KEY, sw.toString());
 262.        }
 263.    
 264.        /**
 265.         * Set specified mapping for the given database code creator class.
 266.         * 
 267.         * @param codeCreatorClass
 268.         *            The class of the code creator for which this mapping should be
 269.         *            set.
 270.         * @param domain
 271.         *            The domain (uml datatype).
 272.         * @param datatype
 273.         *            The datatype (database-specific).
 274.         */
 275.        public void setDatatype(Class codeCreatorClass, String domain,
 276.                String datatype) {
 277.            Map<String, String> mappings = getMappingsFor(codeCreatorClass);
 278.            mappings.put(domain, datatype);
 279.        }
 280.    
 281.        private void readMappings(Map<String, String> mappings, NodeList nodes) {
 282.            for (int i = 0; i < nodes.getLength(); i++) {
 283.                Node mapping = nodes.item(i);
 284.                if (mapping.getNodeType() != Node.ELEMENT_NODE) {
 285.                    continue;
 286.                }
 287.    
 288.                NamedNodeMap attributes = mapping.getAttributes();
 289.                Node src = attributes.getNamedItem("umltype");
 290.                Node dst = attributes.getNamedItem("dbtype");
 291.                String srcText = src.getTextContent();
 292.                String dstText = dst.getTextContent();
 293.    
 294.                mappings.put(srcText, dstText);
 295.            }
 296.        }
 297.    
 298.        private void writeMappings(Writer fw, Map<String, String> mappings) throws IOException {
 299.            String oldIndent = indent;
 300.            indent += "\t";
 301.            Set<Entry<String, String>> entries = mappings.entrySet();
 302.            for (Iterator<Entry<String, String>> it = entries.iterator(); it.hasNext();) {
 303.                Entry<String, String> entry = it.next();
 304.                String domain = entry.getKey();
 305.                String datatype = entry.getValue();
 306.    
 307.                StringBuffer sb = new StringBuffer();
 308.                sb.append(indent);
 309.                sb.append("<tns:mapping umltype=\"");
 310.                sb.append(domain);
 311.                sb.append("\" dbtype=\"");
 312.                sb.append(datatype);
 313.                sb.append("\" />").append(GeneratorSql.LINE_SEPARATOR);
 314.    
 315.                fw.write(sb.toString());
 316.            }
 317.            indent = oldIndent;
 318.        }
 319.    }
