  72.         *     UML entity that is being checked.

  79.            if (!(Model.getFacade().isAStateVertex(dm))) {
  80.                return NO_PROBLEM;
  81.            }
  82.            /* Now we are sure dm is a StateVertex. */
  83.            if (Model.getFacade().isAPseudostate(dm)) {
  84.                Object k = Model.getFacade().getKind(dm);

  90.                }

  92.                    return NO_PROBLEM;
  93.                }
  94.            }
  95.            if (!Model.getFacade().isAState(dm)) {

 100.            if (sm != null && Model.getFacade().getTop(sm) == dm) {
 101.                /* If dm is the top state of the statemachine, then it is 
 102.                 * not supposed to have incoming transitions. */

 107.            if (incoming.size() > 0) {
 108.                return NO_PROBLEM;
 109.            }

 111.            if (!Model.getFacade().isACompositeState(dm)) {
 112.                return PROBLEM_FOUND;
 113.            }
 114.            /* Now we are sure dm is a Composite State. */
 115.

 117.             * at a sub-state of the composite state: */
 118.            Collection transitions = Model.getFacade().getTransitions(sm);
 119.            for (Object t : transitions) {
 120.                Object sourceState = Model.getFacade().getSource(t);
 121.                Object targetState = Model.getFacade().getTarget(t);
 122.                if (!isSomeSubvertexOf(sourceState, dm) && isSomeSubvertexOf(targetState, dm)) {
 123.                    return NO_PROBLEM;
 124.                }
 125.            }
 126.    
 127.            return PROBLEM_FOUND;
 128.        }
 129.        
 130.        /**
 131.         * Test if a state is contained within a composite state recursively. 
 132.         * This is done by checking if the parent of the subject 
 133.         * equals the composite, or the parent of the parent, etc.
 134.         * 
 135.         * @param subject the StateVertex that is investigated
 136.         * @param composite the Composite state that may or may not contain the subject
 137.         * @return true if and only if the given composite contains recursively the given subject
 138.         */
 139.        private boolean isSomeSubvertexOf(Object subject, Object composite) {
 140.            Object c = subject;
 141.            while (c != null) {
 142.                if (c == composite) {
 143.                    return true;
 144.                }
 145.                c = Model.getFacade().getContainer(c);
 146.            }
 147.            return false;
 148.        }

 151.         * @see org.argouml.uml.cognitive.critics.CrUML#getCriticizedDesignMaterials()
 152.         */
 153.        @Override
 154.        public Set<Object> getCriticizedDesignMaterials() {
 155.            Set<Object> ret = new HashSet<Object>();
 156.            ret.add(Model.getMetaTypes().getStateVertex());
 157.            return ret;
 158.        }

  49.    /**
  50.     * A critic to detect when a state has no outgoing transitions.
  51.     * <p>
  52.     * Also a transition from a state contained in the dm going out of the dm
  53.     * counts as an "outgoing" transition thanks to issue 689.
  54.     *
  55.     * @author jrobbins
  56.     */
  57.    public class CrNoOutgoingTransitions extends CrUML {
  58.    
  59.        /**
  60.         * Constructor.
  61.         */
  62.        public CrNoOutgoingTransitions() {
  63.            setupHeadAndDesc();
  64.        addSupportedDecision(UMLDecision.STATE_MACHINES);
  65.        addTrigger("outgoing");
  66.        }
  67.    
  68.        /*
  69.         * @see org.argouml.uml.cognitive.critics.CrUML#predicate2(java.lang.Object, org.argouml.cogniti...
  70.         */
  71.        @Override
  72.        public boolean predicate2(Object dm, Designer dsgr) {
  73.            if (!(Model.getFacade().isAStateVertex(dm))) {
  74.                return NO_PROBLEM;
  75.            }
  76.            /* Now we are sure dm is a StateVertex. */
  77.    
  78.            if (Model.getFacade().isAPseudostate(dm)) {
  79.                Object k = Model.getFacade().getKind(dm);
  80.                if (k.equals(Model.getPseudostateKind().getChoice())) {
  81.                    return NO_PROBLEM;
  82.                }
  83.                if (k.equals(Model.getPseudostateKind().getJunction())) {
  84.                    return NO_PROBLEM;
  85.                }
  86.            }
  87.            if (!Model.getFacade().isAState(dm)) {
  88.                return NO_PROBLEM;
  89.            }
  90.            if (Model.getFacade().isAFinalState(dm)) {
  91.                return NO_PROBLEM;
  92.            }
  93.            /* Now we are sure dm is a State. */
  94.            Object stateMachine = Model.getFacade().getStateMachine(dm);
  95.            if (stateMachine == null) {
  96.                return NO_PROBLEM;
  97.            }
  98.            
  99.            if (stateMachine != null && Model.getFacade().getTop(stateMachine) == dm) {
 100.                /* If dm is the top state of the statemachine, then it is 
 101.                 * not supposed to have outgoing transitions. */
 102.                return NO_PROBLEM;
 103.            }
 104.    
 105.            Collection outgoing = Model.getFacade().getOutgoings(dm);
 106.            if (outgoing == null || outgoing.size() > 0) {
 107.                return NO_PROBLEM;
 108.            }
 109.    
 110.            if (!Model.getFacade().isACompositeState(dm)) {
 111.                return PROBLEM_FOUND;
 112.            }
 113.            /* Now we are sure dm is a Composite State. */
 114.    
 115.            /* Issue 689: Look for a transition that starts 
 116.             * at a sub-state and goes out of the composite state: */
 117.            Collection transitions = Model.getFacade().getTransitions(stateMachine);
 118.            for (Object t : transitions) {
 119.                Object sourceState = Model.getFacade().getSource(t);
 120.                Object targetState = Model.getFacade().getTarget(t);
 121.                if (isSomeSubvertexOf(sourceState, dm) && !isSomeSubvertexOf(targetState, dm)) {
 122.                    return NO_PROBLEM;
 123.                }
 124.            }
 125.    
 126.            return PROBLEM_FOUND;
 127.        }
 128.    
 129.        /**
 130.         * Test if a state is contained within a composite state recursively. 
 131.         * This is done by checking if the parent of the subject 
 132.         * equals the composite, or the parent of the parent, etc.
 133.         * 
 134.         * @param subject the StateVertex that is investigated
 135.         * @param composite the Composite state that may or may not contain the subject
 136.         * @return true if and only if the given composite contains recursively the given subject
 137.         */
 138.        private boolean isSomeSubvertexOf(Object subject, Object composite) {
 139.            Object c = subject;
 140.            while (c != null) {
 141.                if (c == composite) {
 142.                    return true;
 143.                }
 144.                c = Model.getFacade().getContainer(c);
 145.            }
 146.            return false;
 147.        }
