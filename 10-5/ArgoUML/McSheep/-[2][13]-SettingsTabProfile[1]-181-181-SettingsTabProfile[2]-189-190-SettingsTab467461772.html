 504.        public void handleSettingsTabSave() {
 505.            List<Profile> toRemove = new ArrayList<Profile>();
 506.            List<Profile> usedItens = new ArrayList<Profile>();
 507.    
 508.            MutableComboBoxModel modelUsd = ((MutableComboBoxModel) defaultList
 509.                    .getModel());
 510.            MutableComboBoxModel modelDir = ((MutableComboBoxModel) directoryList
 511.                    .getModel());
 512.    
 513.            for (int i = 0; i < modelUsd.getSize(); ++i) {
 514.                usedItens.add((Profile) modelUsd.getElementAt(i));
 515.            }
 516.    
 517.            for (Profile profile 
 518.                    : ProfileFacade.getManager().getDefaultProfiles()) {
 519.                if (!usedItens.contains(profile)) {
 520.                    toRemove.add(profile);
 521.                }
 522.            }
 523.    
 524.            for (Profile profile : toRemove) {
 525.                ProfileFacade.getManager().removeFromDefaultProfiles(profile);
 526.            }
 527.    
 528.            for (Profile profile : usedItens) {
 529.                if (!ProfileFacade.getManager().getDefaultProfiles().contains(
 530.                        profile)) {
 531.                    ProfileFacade.getManager().addToDefaultProfiles(profile);
 532.                }
 533.            }
 534.    
 535.            List<String> toRemoveDir = new ArrayList<String>();
 536.            List<String> usedItensDir = new ArrayList<String>();
 537.    
 538.            for (int i = 0; i < modelDir.getSize(); ++i) {
 539.                usedItensDir.add((String) modelDir.getElementAt(i));
 540.            }
 541.    
 542.            for (String dirEntry : ProfileFacade.getManager()
 543.                    .getSearchPathDirectories()) {
 544.                if (!usedItensDir.contains(dirEntry)) {
 545.                    toRemoveDir.add(dirEntry);
 546.                }
 547.            }
 548.    
 549.            for (String dirEntry : toRemoveDir) {
 550.                ProfileFacade.getManager().removeSearchPathDirectory(dirEntry);
 551.            }
 552.    
 553.            for (String dirEntry : usedItensDir) {
 554.                if (!ProfileFacade.getManager().getSearchPathDirectories()
 555.                        .contains(dirEntry)) {
 556.                    ProfileFacade.getManager().addSearchPathDirectory(dirEntry);
 557.                }
 558.            }
 559.    
 560.        }

 505.                        SOURCE_LANGUAGE_TAG);

 507.                    return false;
 508.                }
 509.                String savedLang = Model.getFacade().getValueOfTag(taggedValue);
 510.                return (lang.getName().equals(savedLang));

 613.            public Object getValueAt(int row, int col) {

 635.            @Override
 636.            public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
 637.                if (columnIndex == getLanguagesCount()) {
 638.                    return;
 639.                }
 640.                if (columnIndex >= getColumnCount()) {
 641.                    return;
 642.                }
 643.                if (!(aValue instanceof Boolean)) {
 644.                    return;
 645.                }
 646.                boolean val = ((Boolean) aValue).booleanValue();
 647.                Object cls = classes.get(rowIndex);
 648.    
 649.                if (columnIndex >= 0 && columnIndex < getLanguagesCount()) {
 650.                    if (val) {
 651.                        checked[columnIndex].add(cls);
 652.                    } else {
 653.                        checked[columnIndex].remove(cls);
 654.                    }
 655.                }
 656.    
 657.                if (val && !getOkButton().isEnabled()) {
 658.                    getOkButton().setEnabled(true);
 659.                } else if (!val && getOkButton().isEnabled()
 660.                        && getChecked().size() == 0) {
 661.                    getOkButton().setEnabled(false);

 665.            /**

 667.             * classes.
 668.             * 
 669.             * @param value If false then all checkmarks are cleared for all
 670.             *            languages. If true then all are cleared, except for one
 671.             *            language column, these are all set.
 672.             */
 673.            public void setAllChecks(boolean value) {
 674.                int rows = getRowCount();
 675.                int checks = getLanguagesCount();
 676.    
 677.                if (rows == 0) {
 678.                    return;
 679.                }
 680.    
 681.                for (int i = 0; i < rows; ++i) {
 682.                    Object cls = classes.get(i);
 683.    
 684.                    for (int j = 0; j < checks; ++j) {
 685.                        if (value && (j == languageHistory)) {
 686.                            checked[j].add(cls);
 687.                        } else {
 688.                            checked[j].remove(cls);
 689.                        }
 690.                    }
 691.                }
 692.                if (value) {
 693.                    if (++languageHistory >= checks) {
 694.                        languageHistory = 0;
 695.                    }
 696.                }
 697.                getOkButton().setEnabled(value);
 698.            }
