  57.     * This class is the GUI front for a mutable linked list. The user can add,
  58.     * delete or create modelelements to the model. He can do that via a popup menu.
  59.     * <p>
  60.     *
  61.     * The developer using this class can turn on and off the actions the user can
  62.     * do via various configuration switches. To turn on/off the add option for
  63.     * example, he can call the method setAddOption. Default the options for delete
  64.     * and add are on. The option for new is off, since this is much less used.
  65.     * <p>
  66.     *

  68.     * ActionClasses. ActionRemoveFromModel for the delete, the other actionclasses
  69.     * need to be provided when constructing this object.
  70.     * <p>
  71.     *
  72.     * Since december 14th, an option is added to configure the popupmenu that
  73.     * should be shown.
  74.     *

  56.    /**
  57.     * This class is the GUI front for a mutable linked list. The user can add,
  58.     * delete or create modelelements to the model. He can do that via a popup menu.
  59.     * <p>
  60.     *
  61.     * The developer using this class can turn on and off the actions the user can
  62.     * do via various configuration switches. To turn on/off the add option for
  63.     * example, he can call the method setAddOption. Default the options for delete
  64.     * and add are on. The option for new is off, since this is much less used.
  65.     * <p>
  66.     *
  67.     * The implementation of the three actions, are delegated to several other
  68.     * ActionClasses. ActionRemoveFromModel for the delete, the other actionclasses
  69.     * need to be provided when constructing this object.
  70.     * <p>
  71.     *
  72.     * Since december 14th, an option is added to configure the popupmenu that
  73.     * should be shown.
  74.     *
  75.     * @since Oct 2, 2002
  76.     * @author jaap.branderhorst@xs4all.nl
  77.     */

  99.        private class PopupMenu extends JPopupMenu {
 100.            public PopupMenu() {
 101.                super();
 102.                if (isAdd()) {
 103.                    addAction.setTarget(getTarget());
 104.                    add(addAction);
 105.                    if (isNew() || isDelete()) {
 106.                        addSeparator();
 107.                    }
 108.                }
 109.                if (isNew()) {
 110.                    newAction.setTarget(getTarget());
 111.                    add(newAction);
 112.                    if (isDelete()) {
 113.                        addSeparator();
 114.                    }
 115.                }
 116.                if (isDelete()) {
 117.                    deleteAction.setObjectToRemove(getSelectedValue());
 118.                    deleteAction.setTarget(getTarget());
 119.                    add(deleteAction);
 120.                }
 121.            }
 122.        }

 127.         * itself).
 128.         *
 129.         * @param dataModel the data model
 130.         * @param theAddAction the action for adding
 131.         * @param theNewAction the action for new

 138.                AbstractActionRemoveElement theDeleteAction) {
 139.            super(dataModel, true);
 140.            setAddAction(theAddAction);
 141.            setNewAction(theNewAction);
 142.            deleteAction = theDeleteAction;
 143.            addMouseListener(this);
 144.        }

 298.        /**
 299.         * Tell the actions what objects they should work on.
 300.         */
 301.        protected void initActions() {
 302.            if (isAdd()) {
 303.                addAction.setTarget(getTarget());
 304.            }
 305.            if (isNew()) {
 306.                newAction.setTarget(getTarget());
 307.            }
 308.            if (isDelete()) {
 309.                deleteAction.setObjectToRemove(getSelectedValue());
 310.                deleteAction.setTarget(getTarget());
 311.            }
 312.        }
 313.    
 314.        /*
 315.         * @see java.awt.event.MouseListener#mouseReleased(
 316.         *      java.awt.event.MouseEvent)
 317.         */
 318.        @Override
 319.        public void mouseReleased(MouseEvent e) {
 320.            if (e.isPopupTrigger()
 321.                    && !Model.getModelManagementHelper().isReadOnly(getTarget())) {
 322.                Point point = e.getPoint();
 323.                int index = locationToIndex(point);
 324.                JPopupMenu popup = getPopupMenu();
 325.                Object model = getModel();
 326.                if (model instanceof UMLModelElementListModel) {
 327.                    ((UMLModelElementListModel) model).buildPopup(popup, index);
 328.                }
 329.                if (popup.getComponentCount() > 0) {
 330.                    initActions();
 331.                    LOG.info("Showing popup at " + e.getX() + "," + e.getY());
 332.                    popup.show(this, e.getX(), e.getY());
 333.                }
 334.                e.consume();
 335.            }
 336.        }
 337.

 342.        public void mousePressed(MouseEvent e) {
 343.            if (e.isPopupTrigger()
 344.                    && !Model.getModelManagementHelper().isReadOnly(getTarget())) {
 345.                JPopupMenu popup = getPopupMenu();
 346.                if (popup.getComponentCount() > 0) {
 347.                    initActions();
 348.                    LOG.debug("Showing popup at " + e.getX() + "," + e.getY());
 349.                    getPopupMenu().show(this, e.getX(), e.getY());
 350.                }
 351.                e.consume();
 352.            }
 353.        }
