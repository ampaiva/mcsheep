 406.        /**
 407.         * Finds a type in a model by name
 408.         * 
 409.         * FIXME: duplicated from the method with the same name in 
 410.         * org.argouml.profile.internal.ModelUtils.
 411.         * 
 412.         * @param s the type name
 413.         * @param model the model
 414.         * @return the type or <code>null</code> if the type has not been found.
 415.         */
 416.        public static Object findTypeInModel(String s, Object model) {
 417.    
 418.            if (!Model.getFacade().isANamespace(model)) {
 419.                throw new IllegalArgumentException(
 420.                        "Looking for the classifier " + s
 421.                        + " in a non-namespace object of " + model
 422.                        + ". A namespace was expected.");
 423.            }
 424.    
 425.            Collection allClassifiers =
 426.                Model.getModelManagementHelper()
 427.                    .getAllModelElementsOfKind(model,
 428.                            Model.getMetaTypes().getClassifier());
 429.    
 430.            Object[] classifiers = allClassifiers.toArray();
 431.            Object classifier = null;
 432.    
 433.            for (int i = 0; i < classifiers.length; i++) {
 434.    
 435.                classifier = classifiers[i];
 436.                if (Model.getFacade().getName(classifier) != null
 437.                            && Model.getFacade().getName(classifier).equals(s)) {
 438.                    return classifier;
 439.                }
 440.            }
 441.    
 442.            return null;
 443.        }
