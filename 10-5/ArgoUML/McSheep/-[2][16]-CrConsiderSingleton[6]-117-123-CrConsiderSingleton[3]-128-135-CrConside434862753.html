 140.            //that inherit all ops but define none of their own.
 141.    
 142.            if (findAssociation(dm, 0)) {
 143.                return NO_PROBLEM;
 144.            }
 145.            return PROBLEM_FOUND;
 146.        }
 147.    
 148.        /**
 149.         * @param dm The classifier to examine.
 150.         * @param depth Number of levels searched.
 151.         * @return true if an association can be found in this classifier
 152.         *        or in any of its generalizations.
 153.         */
 154.        private boolean findAssociation(Object dm, int depth) {
 155.            if (Model.getFacade().getAssociationEnds(dm).iterator().hasNext()) {
 156.                return true;
 157.            }
 158.    
 159.            if (depth > 50) {
 160.                return false;
 161.            }
 162.    
 163.            Iterator iter = Model.getFacade().getGeneralizations(dm).iterator();
 164.    
 165.            while (iter.hasNext()) {
 166.                Object parent = Model.getFacade().getGeneral(iter.next());
 167.    
 168.                if (parent == dm) {
 169.                    continue;
 170.                }
 171.    
 172.                if (Model.getFacade().isAClassifier(parent)) {
 173.                    if (findAssociation(parent, depth + 1)) {
 174.                        return true;
 175.                    }
 176.                }
 177.            }
 178.    
 179.            if (Model.getFacade().isAUseCase(dm)) {
 180.                // for use cases we need to check for extend/includes
 181.                // actors cannot have them, so no need to check
 182.                Iterator iter2 = Model.getFacade().getExtends(dm).iterator();
 183.                while (iter2.hasNext()) {
 184.                    Object parent = Model.getFacade().getExtension(iter2.next());
 185.    
 186.                    if (parent == dm) {
 187.                        continue;
 188.                    }
 189.    
 190.                    if (Model.getFacade().isAClassifier(parent)) {
 191.                        if (findAssociation(parent, depth + 1)) {
 192.                            return true;
 193.                        }
 194.                    }
 195.                }
 196.    
 197.                Iterator iter3 = Model.getFacade().getIncludes(dm).iterator();
 198.                while (iter3.hasNext()) {
 199.                    Object parent = Model.getFacade().getBase(iter3.next());
 200.    
 201.                    if (parent == dm) {
 202.                        continue;
 203.                    }
 204.    
 205.                    if (Model.getFacade().isAClassifier(parent)) {
 206.                        if (findAssociation(parent, depth + 1)) {
 207.                            return true;
 208.                        }
 209.                    }
 210.                }
 211.            }
 212.            return false;
 213.        }

 215.        /*
 216.         * @see org.argouml.uml.cognitive.critics.CrUML#getCriticizedDesignMaterials()
 217.         */
 218.        @Override
 219.        public Set<Object> getCriticizedDesignMaterials() {
 220.            Set<Object> ret = new HashSet<Object>();
 221.            ret.add(Model.getMetaTypes().getUMLClass());
 222.            ret.add(Model.getMetaTypes().getActor());
 223.            ret.add(Model.getMetaTypes().getUseCase());
 224.            return ret;
 225.        }
