 816.            findOrCreateGeneralization(parent, contextStack.peek());

 826.                if (contextAccessSpecifier == null) { // default is private
 827.                    contextAccessSpecifier = Model.getVisibilityKind().getPrivate();
 828.                }

 830.                        generalization, 
 831.                        getFacade().getName(contextAccessSpecifier));

 835.            }

 841.         *
 842.         * @param parent the parent Classifier
 843.         * @param child the child Classifier
 844.         * @return the found Generalization model element if found, otherwise a
 845.         *         newly created

 847.        private Object findOrCreateGeneralization(Object parent,
 848.                Object child) {
 849.            Object generalization =
 850.            Model.getFacade().getGeneralization(child, parent);
 851.            Object stereotype = null;

 853.                try {
 854.                    generalization = getUmlFactory().buildConnection(
 855.                            getMetaTypes().getGeneralization(), child, null,

 861.        } else {
 862.                Collection stereotypes = getFacade().getStereotypes(generalization);
 863.                for (Object aStereotype : stereotypes) {
 864.                    if (STEREO_NAME_GENERALIZATION.equals(
 865.                            getFacade().getName(aStereotype))) {
 866.                        stereotype = aStereotype;
 867.                    }
 868.                }
 869.            }
 870.            assert generalization != null;
 871.            if (stereotype == null) {
 872.                stereotype = profile.getCppGeneralizationStereotype();
 873.                assert stereotype != null;
 874.                getCoreHelper().addStereotype(generalization, stereotype);
