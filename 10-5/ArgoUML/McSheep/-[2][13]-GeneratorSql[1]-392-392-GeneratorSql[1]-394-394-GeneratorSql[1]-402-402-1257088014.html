 371.            Collection<ForeignKeyDefinition> fkDefs = 
 372.                new HashSet<ForeignKeyDefinition>();
 373.    
 374.            for (Object assocEnd : Model.getFacade().getAssociationEnds(relation)) {
 375.    
 376.                Collection otherAssocEnds = Model.getFacade()
 377.                        .getOtherAssociationEnds(assocEnd);
 378.                Object otherAssocEnd = otherAssocEnds.iterator().next();
 379.    
 380.                ForeignKeyDefinition fkDef = getFkDef(relation, assocEnd,
 381.                        otherAssocEnd);
 382.                if (fkDef != null) {
 383.                    fkDefs.add(fkDef);
 384.                }
 385.            }
 386.    
 387.            return fkDefs;
 388.        }
 389.    
 390.        private ForeignKeyDefinition getFkDef(Object relation, Object assocEnd,
 391.                Object otherAssocEnd) {
 392.            Object assoc = Model.getFacade().getAssociation(assocEnd);
 393.            List fkAttributes = Utils.getFkAttributes(relation, assoc);
 394.            int otherUpper = Model.getFacade().getUpper(otherAssocEnd);
 395.            if (otherUpper != 1 || fkAttributes.size() == 0) {
 396.                return null;
 397.            }
 398.            ForeignKeyDefinition fkDef = new ForeignKeyDefinition();
 399.    
 400.            List<Object> srcAttributes = new ArrayList<Object>();
 401.    
 402.            Object srcRelation = Model.getFacade().getClassifier(otherAssocEnd);
 403.            for (Object fkAttr : fkAttributes) {
 404.                Object srcAttr = Utils.getSourceAttribute(fkAttr, srcRelation);
 405.                srcAttributes.add(srcAttr);
 406.            }
 407.    
 408.            // List colNames = new ArrayList();
 409.            TableDefinition tableDef = tableDefinitions
 410.                    .get(relation);
 411.            fkDef.setTable(tableDef);
 412.            for (Object fkAttr : fkAttributes) {
 413.                // colNames.add(Model.getFacade().getName(fkAttr));
 414.    
 415.                ColumnDefinition colDef = tableDef.getColumnDefinition(Model
 416.                        .getFacade().getName(fkAttr));
 417.                fkDef.addColumnDefinition(colDef);
 418.            }
 419.    
 420.            // List refColNames = new ArrayList();
 421.            tableDef = tableDefinitions.get(srcRelation);
 422.            fkDef.setReferencesTable(tableDef);
 423.            for (Object srcAttr : srcAttributes) {
 424.                // refColNames.add(Model.getFacade().getName(srcAttr));
 425.    
 426.                ColumnDefinition colDef = tableDef.getColumnDefinition(Model
 427.                        .getFacade().getName(srcAttr));
 428.                fkDef.addReferencesColumn(colDef);
 429.            }
 430.    
 431.            int lower = Model.getFacade().getLower(assocEnd);
 432.            int upper = Model.getFacade().getUpper(assocEnd);
 433.            int refLower = Model.getFacade().getLower(otherAssocEnd);
 434.            int refUpper = Model.getFacade().getUpper(otherAssocEnd);
 435.    
 436.            fkDef.setForeignKeyName(Model.getFacade().getName(assoc));
 437.    
 438.            // fkDef.setTableName(Model.getFacade().getName(relation));
 439.            // fkDef.setColumnNames(colNames);
 440.    
 441.            // fkDef.setReferencesTableName(Model.getFacade().getName(srcRelation));
 442.            // fkDef.setReferencesColumnNames(refColNames);
 443.    
 444.            fkDef.setLower(lower);
 445.            fkDef.setUpper(upper);
 446.            fkDef.setReferencesLower(refLower);
 447.            fkDef.setReferencesUpper(refUpper);
 448.    
 449.            return fkDef;
 450.        }

  52.    /**
  53.     * A critic to check whether to classes sharing a 1..1 association can or
  54.     * should be combined.
  55.     */
  56.    public class CrMergeClasses extends CrUML {
  57.    
  58.        /**
  59.         * The constructor.
  60.         */
  61.        public CrMergeClasses() {
  62.            setupHeadAndDesc();
  63.        setPriority(ToDoItem.LOW_PRIORITY);
  64.        addSupportedDecision(UMLDecision.CLASS_SELECTION);
  65.        addTrigger("associationEnd");
  66.        }
  67.    
  68.    
  69.        /*
  70.         * @see org.argouml.uml.cognitive.critics.CrUML#predicate2(
  71.         *      java.lang.Object, org.argouml.cognitive.Designer)
  72.         */
  73.        @Override
  74.        public boolean predicate2(Object dm, Designer dsgr) {
  75.        if (!(Model.getFacade().isAClass(dm))) {
  76.            return NO_PROBLEM;
  77.        }
  78.        Object cls = dm;
  79.        Collection ends = Model.getFacade().getAssociationEnds(cls);
  80.        if (ends == null || ends.size() != 1) {
  81.            return NO_PROBLEM;
  82.        }
  83.        Object myEnd = ends.iterator().next();
  84.        Object asc = Model.getFacade().getAssociation(myEnd);
  85.        List conns = new ArrayList(Model.getFacade().getConnections(asc));
  86.            // Do we have 2 connection ends?
  87.            if (conns == null || conns.size() != 2) {
  88.                return NO_PROBLEM;
  89.            }
  90.        Object ae0 = conns.get(0);
  91.        Object ae1 = conns.get(1);
  92.        // both ends must be classes, otherwise there is nothing to merge
  93.        if (!(Model.getFacade().isAClass(Model.getFacade().getType(ae0))
  94.                && Model.getFacade().isAClass(Model.getFacade().getType(ae1)))) {
  95.            return NO_PROBLEM;
  96.        }
  97.        // both ends must be navigable, otherwise there is nothing to merge
  98.        if (!(Model.getFacade().isNavigable(ae0)
  99.                && Model.getFacade().isNavigable(ae1))) {
 100.            return NO_PROBLEM;
 101.        }
 102.        if (Model.getFacade().getLower(ae0) == 1
 103.                    && Model.getFacade().getUpper(ae0) == 1
 104.                    && Model.getFacade().getLower(ae1) == 1
 105.                    && Model.getFacade().getUpper(ae1) == 1) {
 106.            return PROBLEM_FOUND;
 107.        }
 108.        return NO_PROBLEM;
 109.        }
