  91.         * @param s "src" or "dest". Anything else is interpreted as "dest".

  94.            super(m, null);
  95.            // Set the tooltip string:

  97.        str = s;

  99.        }

 102.        /*
 103.         * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
 104.         */
 105.        @Override
 106.        public void actionPerformed(ActionEvent ae) {
 107.            super.actionPerformed(ae);
 108.            List sels = Globals.curEditor().getSelectionManager().selections();
 109.        if (sels.size() == 1) {
 110.            Selection sel = (Selection) sels.get(0);
 111.            Fig f = sel.getContent();
 112.            Object owner = ((FigEdgeModelElement) f).getOwner();
 113.            Collection ascEnds = Model.getFacade().getConnections(owner);
 114.                Iterator iter = ascEnds.iterator();
 115.            Object ascEnd = null;
 116.            if (str.equals("src")) {
 117.            ascEnd = iter.next();
 118.                } else {
 119.                    while (iter.hasNext()) {
 120.                        ascEnd = iter.next();
 121.                    }
 122.                }
 123.    
 124.                if (!mult.equals(Model.getFacade().toString(
 125.                        Model.getFacade().getMultiplicity(ascEnd)))) {
 126.                    Model.getCoreHelper().setMultiplicity(ascEnd, (String) mult);
 127.                }
 128.    
 129.        }
 130.        }
