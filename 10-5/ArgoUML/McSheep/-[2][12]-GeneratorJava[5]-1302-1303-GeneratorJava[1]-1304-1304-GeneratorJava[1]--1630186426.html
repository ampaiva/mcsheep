 137.         * Returns the full path name of the the generated file or
 138.         * null if no file can be generated.
 139.         *
 140.         * @param modelElement the element to be generated
 141.         * @param path the path where the element will be generated
 142.         * @return String full path name of the the generated file
 143.         */
 144.        private String generateFile(Object modelElement, String path) {
 145.            String name = Model.getFacade().getName(modelElement);
 146.            if (name == null || name.length() == 0) {
 147.                return null;
 148.            }
 149.            Object classifier = modelElement;
 150.            String filename = name + ".java";
 151.            StringBuilder sbPath = new StringBuilder(path);
 152.            if (!path.endsWith(FILE_SEPARATOR)) {
 153.                sbPath.append(FILE_SEPARATOR);
 154.            }
 155.    
 156.            String packagePath =
 157.            getPackageName(Model.getFacade().getNamespace(classifier));
 158.    
 159.            int lastIndex = -1;
 160.            do {
 161.                File f = new File(sbPath.toString());
 162.                if (!f.isDirectory()) {
 163.                    if (!f.mkdir()) {
 164.                        LOG.error(" could not make directory " + path);
 165.                        return null;
 166.                    }
 167.                }
 168.                
 169.                if (lastIndex == packagePath.length()) {
 170.                    break;
 171.                }
 172.    
 173.                int index = packagePath.indexOf(".", lastIndex + 1);
 174.                if (index == -1) {
 175.                    index = packagePath.length();
 176.                }
 177.    
 178.                sbPath.append(packagePath.substring(lastIndex + 1, index)
 179.                    + FILE_SEPARATOR);
 180.                lastIndex = index;
 181.            } while (true);
 182.    
 183.            String pathname = sbPath.toString() + filename;
 184.            //cat.info("-----" + pathname + "-----");
 185.    
 186.            //now decide whether file exist and need an update or is to be
 187.            //newly generated
 188.            File f = new File(pathname);
 189.            isFileGeneration = true; // used to produce method javadoc
 190.    
 191.            //String pathname = path + filename;
 192.            // TODO: package, project basepath, tagged values to configure
 193.            LOG.info("Generating " + f.getPath());
 194.            isFileGeneration = true;
 195.            String header =
 196.            generateHeader(classifier, pathname, packagePath);
 197.            String src = generateClassifier(classifier);
 198.            BufferedWriter fos = null;
 199.            try {
 200.            if (Configuration.getString(Argo.KEY_INPUT_SOURCE_ENCODING) == null
 201.            || Configuration.getString(Argo.KEY_INPUT_SOURCE_ENCODING)
 202.                .trim().equals("")) {
 203.                    fos =
 204.                new BufferedWriter(
 205.                        new OutputStreamWriter(new FileOutputStream(f),
 206.                                System.getProperty("file.encoding")));
 207.            } else {
 208.                    fos =
 209.                new BufferedWriter(
 210.                        new OutputStreamWriter(new FileOutputStream(f),
 211.                                Configuration.getString(
 212.                                        Argo.KEY_INPUT_SOURCE_ENCODING)));
 213.            }
 214.                fos.write(header);
 215.                fos.write(src);
 216.            } catch (IOException exp) {
 217.                LOG.error("IO Exception: " + exp + ", for file: " + f.getPath());
 218.            } finally {
 219.                isFileGeneration = false;
 220.                try {
 221.                    if (fos != null) {
 222.                        fos.close();
 223.                    }
 224.                } catch (IOException exp) {
 225.                    LOG.error("FAILED: " + f.getPath());
 226.                }
 227.            }
 228.    
 229.            //cat.info("----- end updating -----");
 230.            return pathname;
 231.        }

 341.        private String generateImports(Object cls, String packagePath) {
 342.            // If the model is built by the import of Java source code, then a
 343.            // component named after the filename was created, which manages the
 344.            // import statements for all included classes/interfaces. This
 345.            // component is now searched for cls in order to extract the imports.
 346.            Object ns = Model.getFacade().getNamespace(cls);
 347.            if (ns != null) {
 348.                for (Object oe : Model.getFacade().getOwnedElements(ns)) {
 349.                    if (Model.getFacade().getUmlVersion().charAt(0) == '1' &&
 350.                            Model.getFacade().isAComponent(oe)) {
 351.                        for (Object re
 352.                            : Model.getFacade().getResidentElements(oe)) {
 353.                            Object r = Model.getFacade().getResident(re);
 354.                            if (r.equals(cls)) {
 355.                                return generateArtifactImports(oe);
 356.                            }
 357.                        }
 358.                    } else if (Model.getFacade().isAArtifact(oe)) {
 359.                        if (Model.getCoreHelper().getUtilizedElements(oe)
 360.                                .contains(cls)) {
 361.                            return generateArtifactImports(oe);
 362.                        }
 363.                    }
 364.                }
 365.            }
 366.            // We now have the situation that no component with package imports
 367.            // was found, so the import statements are guessed from the used model
 368.            // elements inside cls.
 369.            StringBuffer sb = new StringBuffer(80);
 370.            HashSet<String> importSet = new java.util.HashSet<String>();
 371.    
 372.            // now check packages of all feature types
 373.            for (Object mFeature : Model.getFacade().getFeatures(cls)) {
 374.                if (Model.getFacade().isAAttribute(mFeature)) {
 375.                    String ftype = generateImportType(Model.getFacade().getType(
 376.                            mFeature), packagePath);
 377.                    if (ftype != null) {
 378.                        importSet.add(ftype);
 379.                    }
 380.                } else if (Model.getFacade().isAOperation(mFeature)) {
 381.                    // check the parameter types
 382.                    for (Object parameter : Model.getFacade().getParameters(
 383.                            mFeature)) {
 384.                        String ftype = generateImportType(Model.getFacade()
 385.                                .getType(parameter), packagePath);
 386.                        if (ftype != null) {
 387.                            importSet.add(ftype);
 388.                        }
 389.                    }
 390.    
 391.                    // check the return parameter types
 392.                    for (Object parameter 
 393.                            : Model.getCoreHelper().getReturnParameters(mFeature)) {
 394.                        String ftype = generateImportType(Model.getFacade()
 395.                                .getType(parameter), packagePath);
 396.                        if (ftype != null) {
 397.                            importSet.add(ftype);
 398.                        }
 399.                    }
 400.    
 401.                    // check raised signals
 402.                    for (Object signal 
 403.                            : Model.getFacade().getRaisedSignals(mFeature)) {
 404.                        if (!Model.getFacade().isAException(signal)) {
 405.                            continue;
 406.                        }
 407.                        String ftype = generateImportType(Model.getFacade()
 408.                                .getType(signal), packagePath);
 409.                        if (ftype != null) {
 410.                            importSet.add(ftype);
 411.                        }
 412.                    }
 413.                }
 414.            }
 415.    
 416.            // now check generalizations
 417.            for (Object gen : Model.getFacade().getGeneralizations(cls)) {
 418.                Object parent = Model.getFacade().getGeneral(gen);
 419.                if (parent == cls) {
 420.                    continue;
 421.                }
 422.    
 423.                String ftype = generateImportType(parent, packagePath);
 424.                if (ftype != null) {
 425.                    importSet.add(ftype);
 426.                }
 427.            }
 428.    
 429.            // now check packages of the interfaces
 430.            for (Object iface : Model.getFacade().getSpecifications(cls)) {
 431.                String ftype = generateImportType(iface, packagePath);
 432.                if (ftype != null) {
 433.                    importSet.add(ftype);
 434.                }
 435.            }
 436.    
 437.            // check association end types
 438.            for (Object associationEnd : Model.getFacade().getAssociationEnds(cls))
 439.            {
 440.                Object association =
 441.                    Model.getFacade().getAssociation(associationEnd);
 442.                for (Object associationEnd2 
 443.                        : Model.getFacade().getConnections(association)) {
 444.                    if (associationEnd2 != associationEnd
 445.                            && Model.getFacade().isNavigable(associationEnd2)
 446.                            && !Model.getFacade().isAbstract(
 447.                                    Model.getFacade().getAssociation(
 448.                                            associationEnd2))) {
 449.                        // association end found
 450.                        if (Model.getFacade().getUpper(associationEnd2) != 1) {
 451.                            importSet.add("java.util.Vector");
 452.                        } else {
 453.                            String ftype =
 454.                                generateImportType(Model.getFacade().getType(
 455.                                        associationEnd2),
 456.                                        packagePath);
 457.                            if (ftype != null) {
 458.                                importSet.add(ftype);
 459.                            }
 460.                        }
 461.                    }
 462.                }
 463.
