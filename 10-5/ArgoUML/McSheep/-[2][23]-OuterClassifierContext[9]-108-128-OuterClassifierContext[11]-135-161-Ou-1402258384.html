 105.        public Object get(String name, boolean interfacesOnly)
 106.        throws ClassifierNotFoundException {
 107.        // Search in model
 108.        Object mClassifier = Model.getFacade().lookupIn(mPackage, name);
 109.    
 110.        if (mClassifier == null) {
 111.            Class classifier;
 112.            // Try to find it via the classpath
 113.            try {
 114.    
 115.            // Special case for model
 116.            if (Model.getFacade().isAModel(mPackage)) {
 117.                classifier = Class.forName(name);
 118.            }
 119.            else {
 120.                        String clazzName = javaName + "." + name;
 121.                        classifier = Class.forName(clazzName);
 122.            }
 123.            if (classifier.isInterface()) {
 124.                mClassifier =
 125.                                Model.getCoreFactory().buildInterface(
 126.                                        name, mPackage);
 127.            } else {
 128.                        if (!interfacesOnly) {
 129.                            mClassifier =
 130.                                    Model.getCoreFactory().buildClass(
 131.                                            name, mPackage);
 132.                        }
 133.            }
 134.                    if (mClassifier != null) {
 135.                        setGeneratedTag(mClassifier);
 136.                    }
 137.            }
 138.            catch (ClassNotFoundException e) {
 139.            // No class or interface found
 140.                    // try USER classpath
 141.    
 142.                    try {
 143.                        // Special case for model
 144.                        if (Model.getFacade().isAModel(mPackage)) {
 145.                            classifier =
 146.                    ImportClassLoader.getInstance().loadClass(name);
 147.                        }
 148.                        else {
 149.                            String clazzName = javaName + "." + name;
 150.                            classifier =
 151.                    ImportClassLoader.getInstance()
 152.                        .loadClass(clazzName);
 153.                        }
 154.                if (classifier.isInterface()) {
 155.                mClassifier =
 156.                                    Model.getCoreFactory().buildInterface(
 157.                                            name, mPackage);
 158.                } else {
 159.                    if (!interfacesOnly) {
 160.                                mClassifier =
 161.                                        Model.getCoreFactory().buildClass(
 162.                                                name, mPackage);
 163.                            }
 164.                }
 165.                        if (mClassifier != null) {
 166.                            setGeneratedTag(mClassifier);
 167.                        }
 168.                    }
 169.                    catch (ClassNotFoundException e1) {
 170.                        // Ignore - we'll deal with this later by checking to see
 171.                        // if we found anything.
 172.                    } catch (MalformedURLException e1) {
 173.                        LOG.warn("Classpath configuration error", e1);
 174.                    }
 175.                }
 176.        }
 177.        if (mClassifier == null) {
 178.            // Continue the search through the rest of the model
 179.            if (getContext() != null) {
 180.            mClassifier = getContext().get(name, interfacesOnly);
 181.            } else {
 182.            // Check for java data types
 183.                if (!interfacesOnly 
 184.                        && name.equals("int")
 185.                        || name.equals("long")
 186.                        || name.equals("short")
 187.                        || name.equals("byte")
 188.                        || name.equals("char")
 189.                        || name.equals("float")
 190.                        || name.equals("double")
 191.                        || name.equals("boolean")
 192.                        || name.equals("void")
 193.                        // How do I represent arrays in UML?
 194.                        || name.indexOf("[]") != -1) {
 195.                mClassifier =
 196.                Model.getCoreFactory()
 197.                    .buildDataType(name, mPackage);
 198.            }
 199.            }
 200.        }
 201.        if (mClassifier == null) {
 202.            throw new ClassifierNotFoundException(name);
 203.        }
 204.    
 205.        return mClassifier;
 206.        }

 208.        // Historically this used the value "yes", but all existing
 209.        // code only checks for the presence of the tag, not its value
 210.        private static final String GENERATED_TAG_VALUE = "true";
 211.        
 212.        /**
 213.         * Set the tagged value which indicates this element was 
 214.         * generated as a result of reverse engineering.
 215.         * 
 216.         * @param element the ModelElement to set the tag on
 217.         */
 218.        private void setGeneratedTag(Object element) {
 219.            Object tv =
 220.                    Model.getFacade().getTaggedValue(element, Facade.GENERATED_TAG);
 221.            if (tv == null) {
 222.                Model.getExtensionMechanismsHelper().addTaggedValue(
 223.                        element,
 224.                        Model.getExtensionMechanismsFactory().buildTaggedValue(
 225.                                Facade.GENERATED_TAG, GENERATED_TAG_VALUE));
 226.            } else {
 227.                Model.getExtensionMechanismsHelper().setValueOfTag(
 228.                        tv, GENERATED_TAG_VALUE);
 229.            }
 230.        }
