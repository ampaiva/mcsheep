 449.        /**

 451.         * the wellformednessrules as defined in section 2.10.3 of the UML 1.3 spec.
 452.         * <p>
 453.         * Beware: this function does not return the actual base! 
 454.         * Which is by design; there are easier ways to retrieve the actual base.
 455.         * 
 456.         * @param aRole
 457.         *            the given associationrole
 458.         * @return Collection all possible bases
 459.         */
 460.        private Collection getAllPossibleBases(AssociationRole aRole) {
 461.            Set<UmlAssociation> ret = new HashSet<UmlAssociation>();
 462.            if (aRole == null || aRole.getNamespace() == null) {
 463.                return ret;
 464.            }
 465.    
 466.            // find the bases of the connected classifierroles so that we can see
 467.            // what associations are between them. If there are bases then the
 468.            // associations between those bases form the possible bases. Otherwise
 469.            // the bases are formed by all associations in the namespace of the
 470.            // collaboration
 471.            Set<Classifier> bases = new HashSet<Classifier>();
 472.            for (AssociationEnd end : aRole.getConnection()) {
 473.                assert end instanceof AssociationEndRole;
 474.                ClassifierRole type = (ClassifierRole) end.getParticipant();
 475.                if (type != null) {
 476.                    bases.addAll(type.getBase());
 477.                }
 478.            }
 479.            if (bases.isEmpty()) {
 480.                ModelManagementHelper mmh = modelImpl.getModelManagementHelper();
 481.                Namespace ns = 
 482.                    ((Collaboration) aRole.getNamespace()).getNamespace();
 483.                ret.addAll(
 484.                        mmh.getAllModelElementsOfKind(ns, UmlAssociation.class));
 485.                ret.removeAll(mmh.getAllModelElementsOfKind(ns,
 486.                        AssociationRole.class));
 487.            } else {
 488.                CoreHelper ch = modelImpl.getCoreHelper();
 489.                /* This double 'for' loop may be optimised as follows:
 490.                 * - Use arrays in stead of a Set for bases
 491.                 * - Have the second loop start from the element after base1 
 492.                 * ... but I chose not to do this, since the gain is small,
 493.                 * and this is only used for a lazily filled combo.
 494.                 * BTW: This is also used by Notation.*/
 495.                for (Classifier base1 : bases)  {
 496.                    for (Classifier base2 : bases) {
 497.                        // include associations to self - see issue 5602
 498.                        ret.addAll(ch.getAssociations(base1, base2));
 499.                    }
 500.                }
 501.            }
 502.            // An Association can only have a single unnamed ClassifierRole, so
 503.            Collection<UmlAssociation> listToRemove = new ArrayList<UmlAssociation>();
 504.            for (UmlAssociation association : ret) {
 505.                Collection<AssociationRole> associationRoles = 
 506.                    ((org.omg.uml.UmlPackage) (association)
 507.                        .refOutermostPackage()).getCollaborations()
 508.                        .getABaseAssociationRole().getAssociationRole(association);
 509.                if (associationRoles.isEmpty()) {
 510.                    continue;
 511.                }
 512.                // if we are unnamed eliminate all classifiers which are already 
 513.                // the base of some role
 514.                if (aRole.getName() == null || aRole.getName().equals("")) {
 515.                    listToRemove.add(association);
 516.                } else {
 517.                    // eliminate Classifiers which already have an unnamed role
 518.                    for (AssociationRole ar : associationRoles) {
 519.                        if (ar.getName() == null || ar.getName().equals("")) {
 520.                            listToRemove.add(association);
 521.                        }
 522.                    }
 523.                }
 524.            }
 525.            ret.removeAll(listToRemove);
 526.    
 527.            return ret;
 528.        }

 533.         * <p>

 541.                return Collections.EMPTY_SET;

 549.            returnList.removeAll(mmh.getAllModelElementsOfKind(ns,
 550.                    ClassifierRole.class));

 552.            // A Classifier can only have a single unnamed ClassifierRole, so
 553.            // TODO: This probably belongs in a critic instead of here

 593.            if (modelImpl.getFacade().isAPackage(ns)) {

 259.        public void setActivator(Object ames, Object anactivator) {
 260.            if (ames == null) {
 261.                throw new IllegalArgumentException("message is null");
 262.            }
 263.            if (!(ames instanceof Message)) {
 264.                throw new IllegalArgumentException("message");
 265.            }
 266.            if (anactivator != null && !(anactivator instanceof Message)) {
 267.                throw new IllegalArgumentException(
 268.                        "An activator must be a message");
 269.            }
 270.            Message mes = (Message) ames;
 271.            Message activator = (Message) anactivator;
 272.            if (mes == activator) {
 273.                throw new IllegalArgumentException("In setActivator: message may "
 274.                        + "not be equal to activator");
 275.            }
 276.    
 277.            if (activator != null) {
 278.                if (mes.getInteraction() != activator.getInteraction()) {
 279.                    throw new IllegalArgumentException(
 280.                            "In setActivator: interaction "
 281.                                    + "of message should equal "
 282.                                    + "interaction of activator");
 283.                }
 284.                // we must find out if the activator itself does not have
 285.                // message as it's activator
 286.                if (hasAsActivator(activator, mes)) {
 287.                    throw new IllegalArgumentException(
 288.                            "In setActivator: message may "
 289.                                    + "not be the activator for "
 290.                                    + "the original activator");
 291.                }
 292.                // An activator can't also be a predecessor of a message
 293.                if (mes.getPredecessor().contains(activator)) {
 294.                    mes.getPredecessor().remove(activator);
 295.                }
 296.            }
 297.            List<Message> listToChange = new ArrayList<Message>();
 298.            Collection<Message> predecessors = mes.getPredecessor();
 299.            listToChange.addAll(predecessors);
 300.            listToChange.add(mes);
 301.            Interaction inter = mes.getInteraction();
 302.            for (Message mes2 : inter.getMessage()) {
 303.                if (mes2.getPredecessor().contains(mes)) {
 304.                    listToChange.add(mes2);
 305.                }
 306.            }
 307.            // This causes problems. It can make multiple return messages
 308.            // refer to the same activator even if not returning to the same
 309.            // classifier role as the activator emenates from.
 310.            // I'm not sure that changing the activator of one message
 311.            // should amend any other messages but this is certainly changing
 312.            // too many - Bob.
 313.            for (Message mes2 : listToChange) {
 314.                mes2.setActivator(activator);
 315.            }
 316.    
 317.        }

 452.         * <p>
 453.         * Beware: this function does not return the actual base! 
 454.         * Which is by design; there are easier ways to retrieve the actual base.
 455.         *

 459.         */
 460.        private Collection getAllPossibleBases(AssociationRole aRole) {
 461.            Set<UmlAssociation> ret = new HashSet<UmlAssociation>();

 468.            // associations between those bases form the possible bases. Otherwise

 470.            // collaboration

 472.            for (AssociationEnd end : aRole.getConnection()) {
 473.                assert end instanceof AssociationEndRole;
 474.                ClassifierRole type = (ClassifierRole) end.getParticipant();
 475.                if (type != null) {
 476.                    bases.addAll(type.getBase());
 477.                }
 478.            }

 480.                ModelManagementHelper mmh = modelImpl.getModelManagementHelper();
 481.                Namespace ns = 
 482.                    ((Collaboration) aRole.getNamespace()).getNamespace();
 483.                ret.addAll(
 484.                        mmh.getAllModelElementsOfKind(ns, UmlAssociation.class));
 485.                ret.removeAll(mmh.getAllModelElementsOfKind(ns,
 486.                        AssociationRole.class));
 487.            } else {
 488.                CoreHelper ch = modelImpl.getCoreHelper();
 489.                /* This double 'for' loop may be optimised as follows:
 490.                 * - Use arrays in stead of a Set for bases
 491.                 * - Have the second loop start from the element after base1 
 492.                 * ... but I chose not to do this, since the gain is small,

 501.            }

 507.                        .refOutermostPackage()).getCollaborations()

 510.                    continue;
 511.                }

 513.                // the base of some role
 514.                if (aRole.getName() == null || aRole.getName().equals("")) {
 515.                    listToRemove.add(association);
 516.                } else {
 517.                    // eliminate Classifiers which already have an unnamed role
 518.                    for (AssociationRole ar : associationRoles) {
 519.                        if (ar.getName() == null || ar.getName().equals("")) {
 520.                            listToRemove.add(association);
 521.                        }
 522.                    }
 523.                }
 524.            }
 525.            ret.removeAll(listToRemove);
