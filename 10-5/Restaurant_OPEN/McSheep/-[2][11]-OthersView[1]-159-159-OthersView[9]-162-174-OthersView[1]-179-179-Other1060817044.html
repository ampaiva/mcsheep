 158.        private void updateGuestNumber() {

 161.            
 162.            NumberSelectionDialog2 dialog = new NumberSelectionDialog2();

 165.            dialog.pack();

 168.            if(dialog.isCanceled()) {
 169.                return;
 170.            }

 176.            }

 179.            updateView();

 187.        private void updateTableNumber() {
 188.            Ticket thisTicket = getCurrentTicket();
 189.            int tableNumber = thisTicket.getTableNumber();
 190.            
 191.            NumberSelectionDialog2 dialog = new NumberSelectionDialog2();
 192.            dialog.setTitle(com.floreantpos.POSConstants.TABLE_NUMBER);
 193.            dialog.setValue(tableNumber);
 194.            dialog.pack();
 195.            dialog.open();
 196.            
 197.            if(dialog.isCanceled()) {
 198.                return;
 199.            }
 200.            
 201.            tableNumber = (int) dialog.getValue();
 202.            if(tableNumber == 0) {
 203.                POSMessageDialog.showError(Application.getPosWindow(), com.floreantpos.POSConstants.TABLE_NUMBER_...
 204.                return;
 205.            }
 206.            
 207.            TicketDAO dao = TicketDAO.getInstance();
 208.            Ticket otherTicket = dao.findTicketByTableNumber(tableNumber);
 209.            
 210.            if(otherTicket != null && !otherTicket.equals(thisTicket)) {
 211.                POSMessageDialog.showError(com.floreantpos.POSConstants.TABLE_OCCUPIED);
 212.                return;
 213.            }
 214.    
 215.            thisTicket.setTableNumber(tableNumber);
 216.            updateView();
 217.        }

 230.        public void updateView() {
 231.            if(currentTicket != null) {
 232.                if(currentTicket.getTableNumber() == Ticket.TAKE_OUT) {
 233.                    btnCustomerNumber.setText("");
 234.                    btnTableNumber.setText("");
 235.                    
 236.                    btnCustomerNumber.setEnabled(false);
 237.                    btnTableNumber.setEnabled(false);
 238.                }
 239.                else {
 240.                    btnCustomerNumber.setEnabled(true);
 241.                    btnTableNumber.setEnabled(true);
 242.                    
 243.                    btnCustomerNumber.setText(currentTicket.getNumberOfGuests() + " " + POSConstants.GUEST + "s")...
 244.                    btnTableNumber.setText(POSConstants.TABLE_NO + ": " + currentTicket.getTableNumber());
 245.                }
 246.            }
 247.        }
