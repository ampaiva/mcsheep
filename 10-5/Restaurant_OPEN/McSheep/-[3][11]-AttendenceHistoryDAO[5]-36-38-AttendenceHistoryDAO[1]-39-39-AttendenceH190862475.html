 103.        public void voidTicket(Ticket ticket) throws Exception {
 104.            Session session = null;
 105.            Transaction tx = null;
 106.            try {
 107.                ticket.setVoided(true);
 108.                ticket.setPaid(false);
 109.                ticket.setClosed(true);
 110.                ticket.setClosingDate(new Date());
 111.    
 112.                session = createNewSession();
 113.                tx = session.beginTransaction();
 114.    
 115.                session.update(ticket);
 116.    
 117.                Criteria criteria = session.createCriteria(com.floreantpos.model.PosTransaction.class);
 118.                criteria.add(Restrictions.eq(com.floreantpos.model.PosTransaction.PROP_TICKET, ticket));
 119.                List list = criteria.list();
 120.    
 121.                if(list != null) {
 122.                    for (Iterator iter = list.iterator(); iter.hasNext();) {
 123.                        PosTransaction transaction = (PosTransaction) iter.next();
 124.                        Terminal terminal = transaction.getTerminal();
 125.                        terminal.setCurrentBalance(terminal.getCurrentBalance() - transaction.getSubtotalAmount());
 126.                        session.update(terminal);
 127.                        transaction.setTerminal(null);
 128.                        session.delete(transaction);
 129.                    }
 130.                }
 131.                session.flush();
 132.                tx.commit();
 133.            } catch (Exception x) {
 134.                try {
 135.                    tx.rollback();
 136.                } catch (Exception e) {
 137.                }
 138.                throw x;
 139.            }
 140.    
 141.            finally {
 142.                closeSession(session);
 143.            }
 144.        }
