 185.        private void btnNewVoidReasonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_...
 186.            try {
 187.                NotesDialog dialog = new NotesDialog((Frame) getOwner(), true);
 188.                dialog.setTitle(com.floreantpos.POSConstants.ENTER_VOID_REASON);
 189.                dialog.pack();
 190.                dialog.open();
 191.    
 192.                if (!dialog.isCanceled()) {
 193.                    String newVoidReason = dialog.getNote();
 194.                    VoidReason voidReason = new VoidReason();
 195.                    voidReason.setReasonText(newVoidReason);
 196.    
 197.                    VoidReasonDAO dao = new VoidReasonDAO();
 198.                    dao.save(voidReason);
 199.                    
 200.                    if(cbVoidReasons.getModel() instanceof ListComboBoxModel) {
 201.                        ListComboBoxModel model = (ListComboBoxModel) cbVoidReasons.getModel();
 202.                        model.addElement(voidReason);
 203.                    }
 204.                }
 205.            } catch (Throwable e) {
 206.                POSMessageDialog.showError(POSConstants.ERROR_MESSAGE, e);
 207.            }
 208.        }//GEN-LAST:event_btnNewVoidReasonActionPerformed
