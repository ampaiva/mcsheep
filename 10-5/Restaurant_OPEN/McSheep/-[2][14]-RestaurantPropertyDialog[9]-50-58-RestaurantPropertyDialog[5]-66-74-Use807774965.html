 230.        @Override
 231.        public boolean save() {
 232.            try {
 233.                updateModel();
 234.            } catch (IllegalModelStateException e) {
 235.                POSMessageDialog.showError(this, e.getMessage());
 236.                return false;
 237.            }
 238.    
 239.            User user = (User) getBean();
 240.            UserDAO userDAO = UserDAO.getInstance();
 241.    
 242.            if (!editMode) {
 243.                if (userDAO.isUserExist(user.getUserId())) {
 244.                    POSMessageDialog.showError(this, "User with ID: " + user.getUserId() + " already exists.");
 245.                    return false;
 246.                }
 247.            }
 248.    
 249.            try {
 250.                userDAO.saveOrUpdate(user, editMode);
 251.            } catch (PosException x) {
 252.                POSMessageDialog.showError(this, x.getMessage(), x);
 253.                x.printStackTrace();
 254.                return false;
 255.            } catch (Exception x) {
 256.                POSMessageDialog.showError(this, "Could not save user", x);
 257.                x.printStackTrace();
 258.                return false;
 259.            }
 260.    
 261.            return true;
 262.        }

 264.        @Override
 265.        protected boolean updateModel() throws IllegalModelStateException {
 266.            User user = null;
 267.            if (!(getBean() instanceof User)) {
 268.                user = new User();
 269.            }
 270.            else {
 271.                user = (User) getBean();
 272.            }
 273.    
 274.            int id = 1000;
 275.            try {
 276.                id = Integer.parseInt(tfId.getText());
 277.            } catch (Exception x) {
 278.                throw new IllegalModelStateException("ID is not valid");
 279.            }
 280.    
 281.            String ssn = tfSsn.getText();
 282.            String firstName = tfFirstName.getText();
 283.            String lastName = tfLastName.getText();
 284.            String password1 = new String(tfPassword1.getPassword());
 285.            String password2 = new String(tfPassword2.getPassword());
 286.    
 287.            if (POSUtil.isBlankOrNull(ssn)) {
 288.                throw new IllegalModelStateException("SSN cannot be empty");
 289.            }
 290.            if (POSUtil.isBlankOrNull(firstName)) {
 291.                throw new IllegalModelStateException("First name cannot be empty");
 292.            }
 293.            if (POSUtil.isBlankOrNull(lastName)) {
 294.                throw new IllegalModelStateException("Last name cannot be empty");
 295.            }
 296.            if (POSUtil.isBlankOrNull(password1)) {
 297.                throw new IllegalModelStateException("Password1 cannot be empty");
 298.            }
 299.            if (POSUtil.isBlankOrNull(password2)) {
 300.                throw new IllegalModelStateException("Password2 cannot be empty");
 301.            }
 302.            if (!password1.equals(password2)) {
 303.                throw new IllegalModelStateException("Password did not match");
 304.            }
 305.    
 306.            double cost = 0;
 307.    
 308.            try {
 309.                cost = Double.parseDouble(tfCostPerHour.getText());
 310.            } catch (Exception x) {
 311.                throw new IllegalModelStateException("Cost per hour for " + firstName + " " + lastName + " is not...
 312.            }
 313.    
 314.            user.setNewUserType((UserType) cbUserType.getSelectedItem());
 315.            user.setCostPerHour(cost);
 316.    
 317.            user.setSsn(ssn);
 318.            user.setUserId(id);
 319.            user.setFirstName(firstName);
 320.            user.setLastName(lastName);
 321.            user.setPassword(PasswordHasher.hashPassword(password1));
 322.    
 323.            setBean(user);
 324.            return true;
 325.        }
