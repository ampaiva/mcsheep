 323.        public void actionPerformed(ActionEvent e) {
 324.            if ("scrollUP".equals(e.getActionCommand())) {
 325.                if (couponList == null || couponList.size() == 0)
 326.                    return;
 327.    
 328.                int selectedRow = listCoupons.getSelectedIndex();
 329.    
 330.                if (selectedRow <= 0) {
 331.                    selectedRow = 0;
 332.                }
 333.                else {
 334.                    --selectedRow;
 335.                }
 336.    
 337.                listCoupons.setSelectedIndex(selectedRow);
 338.                Rectangle cellRect = listCoupons.getCellBounds(selectedRow, selectedRow);
 339.                listCoupons.scrollRectToVisible(cellRect);
 340.            }
 341.            else if ("scrollDown".equals(e.getActionCommand())) {
 342.                if (couponList == null || couponList.size() == 0)
 343.                    return;
 344.    
 345.                int selectedRow = listCoupons.getSelectedIndex();
 346.    
 347.                if (selectedRow < 0) {
 348.                    selectedRow = 0;
 349.                }
 350.                else if (selectedRow >= couponList.size() - 1) {
 351.                    //return;
 352.                }
 353.                else {
 354.                    ++selectedRow;
 355.                }
 356.    
 357.                listCoupons.setSelectedIndex(selectedRow);
 358.                Rectangle cellRect = listCoupons.getCellBounds(selectedRow, selectedRow);
 359.                cellRect.y += 20;
 360.                listCoupons.scrollRectToVisible(cellRect);
 361.            }
 362.        }

 227.        @Override
 228.        public boolean save() {
 229.            try {
 230.                if(!updateModel()) return false;
 231.                
 232.                MenuModifier modifier = (MenuModifier) getBean();
 233.                ModifierDAO dao = new ModifierDAO();
 234.                dao.saveOrUpdate(modifier);
 235.            } catch (Exception e) {
 236.                MessageDialog.showError(com.floreantpos.POSConstants.SAVE_ERROR, e);
 237.                return false;
 238.            }
 239.            return true;
 240.        }

 246.        @Override
 247.        protected void updateView() {
 248.            MenuModifier modifier = (MenuModifier) getBean();
 249.            
 250.            if(modifier == null) {
 251.                tfName.setText("");
 252.                tfPrice.setText("0");
 253.                tfExtraPrice.setText("0");
 254.                return;
 255.            }
 256.            
 257.            tfName.setText(modifier.getName());
 258.            tfPrice.setValue(new Double(modifier.getPrice()));
 259.            tfExtraPrice.setValue(Double.valueOf(modifier.getExtraPrice()));
 260.            cbModifierGroup.setSelectedItem(modifier.getModifierGroup());
 261.            btnPrintToKitchen.setSelected(modifier.isShouldPrintToKitchen());
 262.            if(modifier.getTax() != null) {
 263.                cbTaxes.setSelectedItem(modifier.getTax());
 264.            }
 265.        }
 266.    
 267.        @Override
 268.        protected boolean updateModel() {
 269.            MenuModifier modifier = (MenuModifier) getBean();
 270.    
 271.            String name = tfName.getText();
 272.            if(POSUtil.isBlankOrNull(name)) {
 273.                MessageDialog.showError("Name is required");
 274.                return false;
 275.            }
 276.            
 277.            modifier.setName(name);
 278.            modifier.setPrice(((Double) tfPrice.getValue()).doubleValue());
 279.            modifier.setExtraPrice(((Double) tfExtraPrice.getValue()).doubleValue());
 280.            modifier.setTax((Tax) cbTaxes.getSelectedItem());
 281.            modifier.setModifierGroup((MenuModifierGroup) cbModifierGroup.getSelectedItem());
 282.            modifier.setShouldPrintToKitchen(Boolean.valueOf(btnPrintToKitchen.isSelected()));
 283.            
 284.            return true;
 285.        }
