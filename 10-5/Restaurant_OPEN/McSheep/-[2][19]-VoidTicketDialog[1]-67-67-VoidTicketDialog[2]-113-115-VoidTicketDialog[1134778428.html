  97.        protected void updateView() {
  98.            MenuCategory foodCategory = (MenuCategory) getBean();
  99.            if(foodCategory == null) {
 100.                tfName.setText("");
 101.                chkVisible.setSelected(false);
 102.                return;
 103.            }
 104.            tfName.setText(foodCategory.getName());
 105.            chkBeverage.setSelected(foodCategory.isBeverage());
 106.            if(foodCategory.getId() == null) {
 107.                chkVisible.setSelected(true);
 108.            }
 109.            else {
 110.                chkVisible.setSelected(foodCategory.isVisible());
 111.            }
 112.        }

 114.        protected boolean updateModel() {
 115.            MenuCategory foodCategory = (MenuCategory) getBean();
 116.            if(foodCategory == null) {
 117.                return false;
 118.            }
 119.            
 120.            
 121.            String categoryName = tfName.getText();
 122.            if(POSUtil.isBlankOrNull(categoryName)) {
 123.                MessageDialog.showError("Name is required");
 124.                return false;
 125.            }
 126.            
 127.            foodCategory.setName(categoryName);
 128.            foodCategory.setBeverage(chkBeverage.isSelected());
 129.            foodCategory.setVisible(chkVisible.isSelected());
 130.            
 131.            return true;
 132.        }
