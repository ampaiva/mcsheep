 120.        private void doRefreshReport(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_doRefreshReport ...
 121.            Date fromDate = dpStartDate.getDate();
 122.            Date toDate = dpEndDate.getDate();
 123.            
 124.            if(fromDate.after(toDate)) {
 125.                POSMessageDialog.showError(Application.getInstance().getBackOfficeWindow(), com.floreantpos.POSCo...
 126.                return;
 127.            }
 128.            
 129.            try {
 130.                reportPanel.removeAll();
 131.                reportPanel.revalidate();
 132.                
 133.                if (report != null) {
 134.                    int reportType = cbReportType.getSelectedIndex();
 135.    
 136.                    report.setReportType(reportType);
 137.                    report.setStartDate(fromDate);
 138.                    report.setEndDate(toDate);
 139.    
 140.                    report.refresh();
 141.                    
 142.                    if (report != null && report.getViewer() != null) {
 143.                        reportPanel.add(report.getViewer());
 144.                        reportPanel.revalidate();
 145.                    }
 146.                }
 147.                
 148.            } catch (Exception e) {
 149.                MessageDialog.showError(com.floreantpos.POSConstants.ERROR_MESSAGE, e);
 150.            }
 151.        }//GEN-LAST:event_doRefreshReport
