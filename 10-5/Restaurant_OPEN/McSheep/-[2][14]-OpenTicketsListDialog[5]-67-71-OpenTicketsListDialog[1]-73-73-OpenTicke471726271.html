 254.        public void updateView() {
 255.            try {
 256.                TicketDAO dao = new TicketDAO();
 257.                List<Ticket> openTickets = dao.findOpenTickets();
 258.    
 259.                TransparentPanel ticketPanel = null;
 260.                int count = 0;
 261.                GridBagConstraints gbc = new GridBagConstraints();
 262.                gbc.weightx = 1;
 263.                gbc.gridx = 0;
 264.                gbc.gridy = GridBagConstraints.RELATIVE;
 265.                gbc.fill = GridBagConstraints.HORIZONTAL;
 266.                gbc.insets = new Insets(3, 3, 3, 3);
 267.    
 268.                for (Iterator iter = openTickets.iterator(); iter.hasNext(); count++) {
 269.                    Ticket ticket = (Ticket) iter.next();
 270.                    if (count % 6 == 0) {
 271.                        ticketPanel = new TransparentPanel(new GridBagLayout());
 272.                        ticketListPanel.add(ticketPanel, String.valueOf(count));
 273.                    }
 274.                    ticketPanel.add(new TicketButton(ticket), gbc);
 275.                }
 276.                openTickets.clear();
 277.            } catch (Exception e) {
 278.                MessageDialog.showError(e);
 279.            }
 280.        }

 284.    
 285.            if (actionCommand.equals(com.floreantpos.POSConstants.GO)) {
 286.                try {
 287.                    TicketDAO dao = new TicketDAO();
 288.                    int ticketId = 0;

 290.                        ticketId = Integer.parseInt(tfTicketNumber.getText());
 291.                    } catch (NullPointerException x) {
 292.                    }
 293.                    Ticket ticket = dao.get(ticketId);

 304.            }

 306.            else {
