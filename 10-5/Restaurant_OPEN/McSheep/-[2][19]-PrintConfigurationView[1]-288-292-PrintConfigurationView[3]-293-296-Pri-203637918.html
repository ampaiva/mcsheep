 127.        @Override
 128.        protected boolean updateModel() throws IllegalModelStateException {
 129.            if(userType == null) {
 130.                userType = new UserType();
 131.            }
 132.            
 133.            String name = tfTypeName.getText();
 134.            if(StringUtils.isEmpty(name)) {
 135.                throw new IllegalModelStateException(com.floreantpos.POSConstants.TYPE_NAME_CANNOT_BE_EMPTY);
 136.            }
 137.            
 138.            userType.setName(name);
 139.            userType.clearPermissions();
 140.            
 141.            Object[] checkedValues = (Object[]) listPermissions.getCheckedValues();
 142.            for (int i = 0; i < checkedValues.length; i++) {
 143.                userType.addTopermissions((UserPermission) checkedValues[i]);
 144.            }
 145.            
 146.            setBean(userType);
 147.            return true;
 148.        }

 150.        @Override
 151.        protected void updateView() {
 152.            if(userType == null) {
 153.                listPermissions.clearSelection();
 154.                return;
 155.            }
 156.            
 157.            tfTypeName.setText(userType.getName());
 158.            
 159.            Set<UserPermission> permissions = userType.getPermissions();
 160.            if(permissions == null) {
 161.                listPermissions.clearSelection();
 162.                return;
 163.            }
 164.            
 165.            CheckBoxListModel model = (CheckBoxListModel) listPermissions.getModel();
 166.            for (UserPermission permission : permissions) {
 167.                for (int i = 0; i < model.getItems().size(); i++) {
 168.                    Entry entry = (Entry) model.getItems().elementAt(i);
 169.                    if(entry.getValue().equals(permission)) {
 170.                        entry.setChecked(true);
 171.                    }
 172.                }
 173.            }
 174.            model.fireTableRowsUpdated(0, model.getRowCount());
 175.        }
