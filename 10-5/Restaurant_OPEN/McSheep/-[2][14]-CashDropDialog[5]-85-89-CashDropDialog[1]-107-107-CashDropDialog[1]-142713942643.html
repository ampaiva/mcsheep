 192.        private void doScrollDown(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_doScrollDown
 193.            if (cashDropList == null)
 194.                return;
 195.    
 196.            int selectedRow = tableCashDrops.getSelectedRow();
 197.            if (selectedRow < 0) {
 198.                selectedRow = 0;
 199.            }
 200.            else if (selectedRow >= cashDropList.size() - 1) {
 201.                selectedRow = 0;

 203.            else {
 204.                ++selectedRow;
 205.            }
 206.    
 207.            selectionModel.setLeadSelectionIndex(selectedRow);

 212.        private void doScrollUp(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_doScrollUp
