 132.        @Override

 136.            
 137.            try {

 139.                maxQuantity = Integer.parseInt(tfMaxQuantity.getText());

 141.            
 142.            MenuModifierGroup group = (MenuModifierGroup) cbModifierGroups.getSelectedItem();

 146.            }

 148.            modifierGroup.setModifierGroup(group);
