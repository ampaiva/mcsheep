 417.        public void transferTicketItem(TicketItem ticketItem, TicketForSplitView toTicketView) {
 418.            TicketItem newTicketItem = new TicketItem();
 419.            newTicketItem.setItemCount(1);
 420.            newTicketItem.setItemId(ticketItem.getItemId());
 421.            newTicketItem.setHasModifiers(ticketItem.isHasModifiers());
 422.            newTicketItem.setTicketItemModifierGroups(new ArrayList(ticketItem.getTicketItemModifierGroups()));
 423.            newTicketItem.setName(ticketItem.getName());
 424.            newTicketItem.setGroupName(ticketItem.getGroupName());
 425.            newTicketItem.setCategoryName(ticketItem.getCategoryName());
 426.            newTicketItem.setUnitPrice(ticketItem.getUnitPrice());
 427.            newTicketItem.setDiscountRate(ticketItem.getDiscountRate());
 428.            newTicketItem.setTaxRate(ticketItem.getTaxRate());
 429.            newTicketItem.setBeverage(ticketItem.isBeverage());
 430.            newTicketItem.setShouldPrintToKitchen(ticketItem.isShouldPrintToKitchen());
 431.            
 432.            int itemCount = ticketItem.getItemCount();
 433.    
 434.            toTicketView.ticketViewerTable.addTicketItem(newTicketItem);
 435.            if (itemCount > 1) {
 436.                ticketItem.setItemCount(--itemCount);
 437.            }
 438.            else {
 439.                ticketViewerTable.delete(ticketItem.getTableRowNum());
 440.            }
 441.            repaint();
 442.            toTicketView.repaint();
 443.            updateView();
 444.            toTicketView.updateView();
 445.        }
