 269.        private void doFinish(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_doFinish
 270.            double faceValue = 0;
 271.            try {
 272.                faceValue = Double.parseDouble(tfFaceValue.getText());
 273.            } catch(Exception e) {
 274.                POSMessageDialog.showError("Gift certificate face value is not valid");
 275.                return;
 276.            }
 277.            double tenderedAmount = 0;
 278.            try {
 279.                tenderedAmount = Double.parseDouble(tfAmountTendered.getText());
 280.            } catch (Exception e) {
 281.                POSMessageDialog.showError("Amount tendered is not valid");
 282.                return;
 283.            }
 284.            
 285.            if(faceValue < tenderedAmount) {
 286.                POSMessageDialog.showError("Face value cannot be less than tendered amount.");
 287.                return;
 288.            }
 289.            
 290.            GiftCertificateTransaction transaction = new GiftCertificateTransaction();
 291.            transaction.setFaceValue(faceValue);
 292.            transaction.setPaidAmount(tenderedAmount);
 293.            transaction.setCashBackAmount(faceValue - tenderedAmount);
 294.            settleTickets(tenderedAmount, 0, transaction, null, null);
 295.        }//GEN-LAST:event_doFinish

 365.            public void actionPerformed(ActionEvent e) {
 366.                JTextField tf = getFocusedTextField();
 367.                PosButton button = (PosButton) e.getSource();
 368.                String s = button.getActionCommand();
 369.                if (s.equals("CLEAR")) {
 370.                    tf.setText("0");
 371.                }
 372.                else if (s.equals(".")) {
 373.                    if (tf.getText().indexOf('.') < 0) {
 374.                        tf.setText(tf.getText() + ".");
 375.                    }
 376.                }
 377.                else {
 378.                    String string = tf.getText();
 379.                    int index = string.indexOf('.');
 380.                    if (index < 0) {
 381.                        double value = 0;
 382.                        try {
 383.                            value = Double.parseDouble(string);
 384.                        } catch (NumberFormatException x) {
 385.                            Toolkit.getDefaultToolkit().beep();
 386.                        }
 387.                        if (value == 0) {
 388.                            tf.setText(s);
 389.                        }
 390.                        else {
 391.                            tf.setText(string + s);
 392.                        }
 393.                    }
 394.                    else {
 395.                        tf.setText(string + s);
 396.                    }
 397.                }
 398.            }
 399.        };
