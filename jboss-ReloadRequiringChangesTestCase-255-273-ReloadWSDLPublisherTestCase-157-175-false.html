 185.                throw new Exception(result.get(FAILURE_DESCRIPTION).toString());

  70.    @RunWith(Arquillian.class)
  71.    @RunAsClient
  72.    public class ReloadWSDLPublisherTestCase {
  73.    
  74.        private static final String DEFAULT_JBOSSAS = "default-jbossas";
  75.        private static final String DEPLOYMENT = "jaxws-manual-pojo";
  76.        private static final String keepAlive = System.getProperty("http.keepAlive") == null ? "true" : System.getProperty("http.keepAlive");
  77.        private static final String maxConnections = System.getProperty("http.maxConnections") == null ? "5" : System.getProperty("http.maxConnections");
  78.    
  79.        @ArquillianResource
  80.        ContainerController containerController;
  81.    
  82.        @ArquillianResource
  83.        Deployer deployer;
  84.    
  85.        @Deployment(name = DEPLOYMENT, testable = false, managed = false)
  86.        public static WebArchive deployment() {
  87.            WebArchive pojoWar = ShrinkWrap.create(WebArchive.class, DEPLOYMENT + ".war").addClasses(
  88.                    EndpointIface.class, PojoEndpoint.class);
  89.            return pojoWar;
  90.        }
  91.    
  92.        @Before
  93.        public void endpointLookup() throws Exception {
  94.            containerController.start(DEFAULT_JBOSSAS);
  95.            if (containerController.isStarted(DEFAULT_JBOSSAS)) {
  96.                deployer.deploy(DEPLOYMENT);
  97.            }
  98.            System.setProperty("http.keepAlive", "false");
  99.            System.setProperty("http.maxConnections", "1");
 100.        }
 101.    
 102.        @Test
 103.        @OperateOnDeployment(DEPLOYMENT)
 104.        public void testHelloStringAfterReload() throws Exception {
 105.            Assert.assertTrue(containerController.isStarted(DEFAULT_JBOSSAS));
 106.            ManagementClient managementClient = new ManagementClient(TestSuiteEnvironment.getModelControllerClient(),
 107.                    TestSuiteEnvironment.getServerAddress(), TestSuiteEnvironment.getServerPort(), "http-remoting");
 108.            QName serviceName = new QName("http://jbossws.org/basic", "POJOService");
 109.            URL wsdlURL = new URL(managementClient.getWebUri().toURL(), '/' + DEPLOYMENT + "/POJOService?wsdl");
 110.            checkWsdl(wsdlURL);
 111.            Service service = Service.create(wsdlURL, serviceName);
 112.            EndpointIface proxy = service.getPort(EndpointIface.class);
 113.            Assert.assertEquals("Hello World!", proxy.helloString("World"));
 114.            reloadServer(managementClient, 100000);
 115.            checkWsdl(wsdlURL);
 116.            serviceName = new QName("http://jbossws.org/basic", "POJOService");
 117.            service = Service.create(wsdlURL, serviceName);
 118.            proxy = service.getPort(EndpointIface.class);
 119.            Assert.assertEquals("Hello World!", proxy.helloString("World"));
 120.            Assert.assertTrue(containerController.isStarted(DEFAULT_JBOSSAS));
 121.        }

 123.        @After
 124.        public void stopContainer() {
 125.            System.setProperty("http.keepAlive", keepAlive);
 126.            System.setProperty("http.maxConnections", maxConnections);
 127.            if (containerController.isStarted(DEFAULT_JBOSSAS)) {
 128.                deployer.undeploy(DEPLOYMENT);
 129.            }
 130.            if (containerController.isStarted(DEFAULT_JBOSSAS)) {
 131.                containerController.stop(DEFAULT_JBOSSAS);
 132.            }
 133.        }

 135.        private void reloadServer(ManagementClient managementClient, int timeout) throws Exception {
 136.            executeReload(managementClient.getControllerClient());
 137.            waitForLiveServerToReload(timeout);
 138.        }
 139.    
 140.        private void executeReload(ModelControllerClient client) throws IOException {
 141.            ModelNode operation = new ModelNode();
 142.            operation.get(OP_ADDR).setEmptyList();
 143.            operation.get(OP).set("reload");
 144.            try {
 145.                Assert.assertTrue(Operations.isSuccessfulOutcome(client.execute(operation)));
 146.            } catch(IOException e) {
 147.                final Throwable cause = e.getCause();
 148.                if (!(cause instanceof ExecutionException) && !(cause instanceof CancellationException)) {
 149.                    throw e;
 150.                } // else ignore, this might happen if the channel gets closed before we got the response
 151.            } finally {
 152.                client.close();
 153.            }
 154.        }
 155.    
 156.        private void waitForLiveServerToReload(int timeout) throws Exception {
 157.            long start = System.currentTimeMillis();
 158.            ModelNode operation = new ModelNode();
 159.            operation.get(OP_ADDR).setEmptyList();
 160.            operation.get(OP).set(READ_ATTRIBUTE_OPERATION);
 161.            operation.get(NAME).set("server-state");
 162.            while (System.currentTimeMillis() - start < timeout) {
 163.                ModelControllerClient liveClient = ModelControllerClient.Factory.create(
 164.                        TestSuiteEnvironment.getServerAddress(), TestSuiteEnvironment.getServerPort());
 165.                try {
 166.                    ModelNode result = liveClient.execute(operation);
 167.                    if ("running".equals(result.get(RESULT).asString())) {
 168.                        return;
 169.                    }
 170.                } catch (IOException e) {
 171.                } finally {
 172.                    IoUtils.safeClose(liveClient);
 173.                }
 174.                try {
 175.                    Thread.sleep(100);
 176.                } catch (InterruptedException e) {
 177.                }
 178.            }
 179.            fail("Live Server did not reload in the imparted time.");
 180.        }
 181.    
 182.        private void checkWsdl(URL wsdlURL) throws IOException {
 183.            StringBuilder proxyUsed = new StringBuilder();
 184.            try {
 185.                List proxies = ProxySelector.getDefault().select(wsdlURL.toURI());
 186.                for(Proxy proxy : proxies) {
 187.                    System.out.println("To connect to " + wsdlURL + " we are using proxy " + proxy);
 188.                    proxyUsed.append("To connect to ").append(wsdlURL).append(" we are using proxy ").append(proxy).append("\r\n");
 189.                }
 190.            } catch (URISyntaxException ex) {
 191.                Logger.getLogger(ReloadWSDLPublisherTestCase.class.getName()).log(Level.SEVERE, null, ex);
 192.            }
 193.            HttpURLConnection connection = (HttpURLConnection) wsdlURL.openConnection();
 194.            try {
 195.                connection.connect();
 196.                Assert.assertEquals(proxyUsed.toString(), HttpServletResponse.SC_OK, connection.getResponseCode());
 197.            } finally {
 198.                connection.disconnect();
 199.            }
 200.        }
