 210.        static URL discoverXsd(final String xsdName) {
 211.            if (OUTDATED_NAMESPACES.containsKey(xsdName)) {
 212.                throw new RuntimeException("Default configs are not in line with most recent schemas " + xsdName + " has been superseded by " + OUTDATED_NAMESPACES.get(xsdName));
 213.            }
 214.            final File file = JBOSS_SCHEMAS_MAP.get(xsdName);
 215.            URL url = null;
 216.            try {
 217.                if (file != null) {
 218.                    url = file.toURI().toURL();
 219.                }
 220.            } catch (IOException e) {
 221.                url = null;
 222.            }
 223.            String fileName = xsdName;
 224.            int index = fileName.lastIndexOf("/");
 225.            if(index == -1) {
 226.                index = fileName.lastIndexOf("\\");
 227.            }
 228.            if(index != -1) {
 229.                fileName = fileName.substring(index + 1);
 230.            }
 231.            // Search
 232.            final ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
 233.            if (url == null)
 234.                url = classLoader.getResource("docs/schema/" + fileName);
 235.            if (url == null)
 236.                url = classLoader.getResource("docs/" + fileName);

 248.        private static class SchemaFilter implements FilenameFilter {
 249.            private static final Pattern PATTERN = Pattern.compile("(jboss|wildfly)-.*\\.xsd$");
 250.            private final String[] exclusions;
 251.    
 252.            SchemaFilter() {
 253.                this.exclusions = new String[0];
 254.            }
 255.    
 256.            SchemaFilter(final String[] exclusions) {
 257.                this.exclusions = exclusions;
 258.            }
 259.    
 260.            @Override
 261.            public boolean accept(final File dir, final String name) {
 262.                final boolean accepted = PATTERN.matcher(name).find();
 263.                if (accepted) {
 264.                    // check for explicit excluded files
 265.                    for (final String excludedFile : exclusions) {
 266.                        if (excludedFile.equals(name)) {
 267.                            // file is in exclude list, so we don't accept this file
 268.                            return false;
 269.                        }
 270.                    }
 271.                }
 272.                return accepted;
 273.            }

  63.    /**
  64.     * Deployment processor responsible for analyzing each attached {@link org.jboss.as.ee.component.ComponentDescription} instance to configure
  65.     * required resource injection configurations.
  66.     *
  67.     * @author John Bailey
  68.     * @author David M. Lloyd
  69.     * @author Richard Opalka
  70.     * @author Eduardo Martins
  71.     */
  72.    public class ResourceInjectionAnnotationParsingProcessor implements DeploymentUnitProcessor {
  73.    
  74.        private static final DotName RESOURCE_ANNOTATION_NAME = DotName.createSimple(Resource.class.getName());
  75.        private static final DotName RESOURCES_ANNOTATION_NAME = DotName.createSimple(Resources.class.getName());
  76.        private static final String JAVAX_NAMING_CONTEXT = "javax.naming.Context";
  77.        public static final Map FIXED_LOCATIONS;
  78.        public static final Set SIMPLE_ENTRIES;
  79.    
  80.        static {
  81.            final Map locations = new HashMap();
  82.            locations.put("javax.transaction.UserTransaction", "java:jboss/UserTransaction");
  83.            locations.put("javax.transaction.TransactionSynchronizationRegistry", "java:jboss/TransactionSynchronizationRegistry");
  84.    
  85.            //we have to be careful with java:comp lookups here
  86.            //as they will not work in entries in application.xml, as there is no comp context available
  87.            //so we can only use it for resources that are not valid to be entries in application.xml
  88.            locations.put("javax.enterprise.inject.spi.BeanManager", "java:comp/BeanManager");
  89.            locations.put("javax.ejb.TimerService", "java:comp/TimerService");
  90.            locations.put("org.omg.CORBA.ORB", "java:comp/ORB");
  91.            FIXED_LOCATIONS = Collections.unmodifiableMap(locations);
  92.

  97.            simpleEntries.add("short");

 106.            simpleEntries.add("java.lang.Integer");
 107.            simpleEntries.add("java.lang.Long");
 108.            simpleEntries.add("java.lang.Double");
 109.            simpleEntries.add("java.lang.Float");
 110.            simpleEntries.add("java.lang.String");
 111.            simpleEntries.add("java.lang.Class");
 112.            SIMPLE_ENTRIES = Collections.unmodifiableSet(simpleEntries);

 116.        public void deploy(final DeploymentPhaseContext phaseContext) throws DeploymentUnitProcessingException {
 117.            final DeploymentUnit deploymentUnit = phaseContext.getDeploymentUnit();
 118.            final EEModuleDescription eeModuleDescription = deploymentUnit.getAttachment(Attachments.EE_MODULE_DESCRIPTION);
 119.            final CompositeIndex index = deploymentUnit.getAttachment(org.jboss.as.server.deployment.Attachments.COMPOSITE_ANNOTATION_INDEX);
 120.            final EEApplicationClasses applicationClasses = deploymentUnit.getAttachment(Attachments.EE_APPLICATION_CLASSES_DESCRIPTION);
 121.            final Module module = deploymentUnit.getAttachment(org.jboss.as.server.deployment.Attachments.MODULE);
 122.            final PropertyReplacer replacer = EJBAnnotationPropertyReplacement.propertyReplacer(deploymentUnit);
 123.            if (module == null) {
 124.                return;
 125.            }
 126.            final List resourceAnnotations = index.getAnnotations(RESOURCE_ANNOTATION_NAME);
 127.            for (AnnotationInstance annotation : resourceAnnotations) {
 128.                final AnnotationTarget annotationTarget = annotation.target();
 129.                final AnnotationValue nameValue = annotation.value("name");
 130.                final String name = (nameValue != null) ? replacer.replaceProperties(nameValue.asString()) : null;
 131.                final AnnotationValue typeValue = annotation.value("type");
 132.                final String type = typeValue != null ? typeValue.asClass().name().toString() : null;
 133.                if (annotationTarget instanceof FieldInfo) {
 134.                    final FieldInfo fieldInfo = (FieldInfo) annotationTarget;
 135.                    final ClassInfo classInfo = fieldInfo.declaringClass();
 136.                    EEModuleClassDescription classDescription = eeModuleDescription.addOrGetLocalClassDescription(classInfo.name().toString());
 137.                    processFieldResource(phaseContext, fieldInfo, name, type, classDescription, annotation, eeModuleDescription, module, applicationClasses, replacer);
 138.                } else if (annotationTarget instanceof MethodInfo) {
 139.                    final MethodInfo methodInfo = (MethodInfo) annotationTarget;
 140.                    ClassInfo classInfo = methodInfo.declaringClass();
 141.                    EEModuleClassDescription classDescription = eeModuleDescription.addOrGetLocalClassDescription(classInfo.name().toString());
 142.                    processMethodResource(phaseContext, methodInfo, name, type, classDescription, annotation, eeModuleDescription, module, applicationClasses, replacer);
 143.                } else if (annotationTarget instanceof ClassInfo) {
 144.                    final ClassInfo classInfo = (ClassInfo) annotationTarget;
 145.                    EEModuleClassDescription classDescription = eeModuleDescription.addOrGetLocalClassDescription(classInfo.name().toString());
 146.                    processClassResource(phaseContext, name, type, classDescription, annotation, eeModuleDescription, module, applicationClasses, replacer);
 147.                }
 148.            }
 149.            final List resourcesAnnotations = index.getAnnotations(RESOURCES_ANNOTATION_NAME);
 150.            for (AnnotationInstance outerAnnotation : resourcesAnnotations) {
 151.                final AnnotationTarget annotationTarget = outerAnnotation.target();
 152.                if (annotationTarget instanceof ClassInfo) {
 153.                    final ClassInfo classInfo = (ClassInfo) annotationTarget;
 154.                    final AnnotationInstance[] values = outerAnnotation.value("value").asNestedArray();
 155.                    for (AnnotationInstance annotation : values) {
 156.                        final AnnotationValue nameValue = annotation.value("name");
 157.                        final String name = (nameValue != null) ? replacer.replaceProperties(nameValue.asString()) : null;
 158.                        final AnnotationValue typeValue = annotation.value("type");
 159.                        final String type = (typeValue != null) ? typeValue.asClass().name().toString() : null;
 160.                        EEModuleClassDescription classDescription = eeModuleDescription.addOrGetLocalClassDescription(classInfo.name().toString());
 161.                        processClassResource(phaseContext, name, type, classDescription, annotation, eeModuleDescription, module, applicationClasses, replacer);
 162.                    }
 163.                }
 164.            }
 165.        }

 168.        }

 170.        protected void processFieldResource(final DeploymentPhaseContext phaseContext, final FieldInfo fieldInfo, final String name, final String type, final EEModuleClassDescription classDescription, final AnnotationInstance annotation, final EEModuleDescription eeModuleDescription, final Module module, final EEApplicationClasses applicationClasses, final PropertyReplacer replacer) throws DeploymentUnitProcessingException {
 171.            final String fieldName = fieldInfo.name();
 172.            final String injectionType = isEmpty(type) || type.equals(Object.class.getName()) ? fieldInfo.type().name().toString() : type;
 173.            final String localContextName = isEmpty(name) ? fieldInfo.declaringClass().name().toString() + "/" + fieldName : name;
 174.            final InjectionTarget targetDescription = new FieldInjectionTarget(fieldInfo.declaringClass().name().toString(), fieldName, fieldInfo.type().name().toString());
 175.            process(phaseContext, classDescription, annotation, injectionType, localContextName, targetDescription, eeModuleDescription, module, applicationClasses, replacer);
 176.        }
 177.    
 178.        protected void processMethodResource(final DeploymentPhaseContext phaseContext, final MethodInfo methodInfo, final String name, final String type, final EEModuleClassDescription classDescription, final AnnotationInstance annotation, final EEModuleDescription eeModuleDescription, final Module module, final EEApplicationClasses applicationClasses, final PropertyReplacer replacer) throws DeploymentUnitProcessingException {
 179.            final String methodName = methodInfo.name();
 180.            if (!methodName.startsWith("set") || methodInfo.args().length != 1) {
 181.                throw EeLogger.ROOT_LOGGER.setterMethodOnly("@Resource", methodInfo);
 182.            }
 183.    
 184.            final String contextNameSuffix = methodName.substring(3, 4).toLowerCase(Locale.ENGLISH) + methodName.substring(4);
 185.            final String localContextName = isEmpty(name) ? methodInfo.declaringClass().name().toString() + "/" + contextNameSuffix : name;
 186.    
 187.            final String injectionType = isEmpty(type) || type.equals(Object.class.getName()) ? methodInfo.args()[0].name().toString() : type;
 188.            final InjectionTarget targetDescription = new MethodInjectionTarget(methodInfo.declaringClass().name().toString(), methodName, methodInfo.args()[0].name().toString());
 189.            process(phaseContext, classDescription, annotation, injectionType, localContextName, targetDescription, eeModuleDescription, module, applicationClasses, replacer);
 190.        }
 191.    
 192.        protected void processClassResource(final DeploymentPhaseContext phaseContext, final String name, final String type, final EEModuleClassDescription classDescription, final AnnotationInstance annotation, final EEModuleDescription eeModuleDescription, final Module module, final EEApplicationClasses applicationClasses, final PropertyReplacer replacer) throws DeploymentUnitProcessingException {
 193.            if (isEmpty(name)) {
 194.                throw EeLogger.ROOT_LOGGER.annotationAttributeMissing("@Resource", "name");
 195.            }
 196.            final String realType;
 197.            if (isEmpty(type)) {
 198.                realType = Object.class.getName();
 199.            } else {
 200.                realType = type;
 201.            }
 202.            process(phaseContext, classDescription, annotation, realType, name, null, eeModuleDescription, module, applicationClasses, replacer);
 203.        }
 204.    
 205.        protected void process(final DeploymentPhaseContext phaseContext, final EEModuleClassDescription classDescription, final AnnotationInstance annotation, final String injectionType, final String localContextName, final InjectionTarget targetDescription, final EEModuleDescription eeModuleDescription, final Module module, final EEApplicationClasses applicationClasses, final PropertyReplacer replacer) throws DeploymentUnitProcessingException {
 206.            final EEResourceReferenceProcessorRegistry registry = phaseContext.getDeploymentUnit().getAttachment(Attachments.RESOURCE_REFERENCE_PROCESSOR_REGISTRY);
 207.            final AnnotationValue lookupAnnotation = annotation.value("lookup");
 208.            String lookup = (lookupAnnotation == null) ? null : replacer.replaceProperties(lookupAnnotation.asString());
 209.            // if "lookup" hasn't been specified then fallback on "mappedName" which we treat the same as "lookup"
 210.            if (isEmpty(lookup)) {
 211.                final AnnotationValue mappedNameAnnotationValue = annotation.value("mappedName");
 212.                lookup = (mappedNameAnnotationValue == null) ? null : replacer.replaceProperties(mappedNameAnnotationValue.asString());
 213.            }
 214.    
 215.            if (isEmpty(lookup) && FIXED_LOCATIONS.containsKey(injectionType)) {
 216.                lookup = FIXED_LOCATIONS.get(injectionType);
 217.            }
 218.            InjectionSource valueSource = null;
 219.            final boolean isEnvEntryType = this.isEnvEntryType(injectionType, module);
 220.            if (!isEmpty(lookup)) {
 221.                valueSource = new LookupInjectionSource(lookup, JAVAX_NAMING_CONTEXT.equals(injectionType));
 222.            } else if (isEnvEntryType) {
 223.                // if it's an env-entry type then we do *not* create a BindingConfiguration to bind to the ENC
 224.                // since the binding (value) for env-entry is always driven from a deployment descriptor.
 225.                // The deployment descriptor processing and subsequent binding in the ENC is taken care off by a
 226.                // different Deployment unit processor. If the value isn't specified in the deployment descriptor,
 227.                // then there will be no binding the ENC and that's what is expected by the Java EE 6 spec. Furthermore,
 228.                // if the @Resource is an env-entry binding then the injection target will be optional since in the absence of
 229.                // an env-entry-value, there won't be a binding and effectively no injection. This again is as expected by spec.
 230.            } else {
 231.                //otherwise we just try and handle it
 232.                //if we don't have a value source we will try and inject from a lookup
 233.                //and the user has to configure the value in a deployment descriptor
 234.                final EEResourceReferenceProcessor resourceReferenceProcessor = registry.getResourceReferenceProcessor(injectionType);
 235.                if (resourceReferenceProcessor != null) {
 236.                    valueSource = resourceReferenceProcessor.getResourceReferenceBindingSource();
 237.                }
 238.            }
 239.    
 240.            // EE.5.2.4
 241.            // Each injection of an object corresponds to a JNDI lookup. Whether a new
 242.            // instance of the requested object is injected, or whether a shared instance is
 243.            // injected, is determined by the rules described above.
 244.    
 245.            // Because of performance we allow any type of InjectionSource.
 246.    
 247.            if (valueSource == null) {
 248.                // the ResourceInjectionConfiguration is created by LazyResourceInjection
 249.                if (targetDescription != null) {
 250.                    final LookupInjectionSource optionalInjection = new LookupInjectionSource(localContextName, true);
 251.                    final ResourceInjectionConfiguration injectionConfiguration = new ResourceInjectionConfiguration(targetDescription, optionalInjection, true);
 252.                    classDescription.addResourceInjection(injectionConfiguration);
 253.                }
 254.            } else {
 255.                // our injection comes from the local lookup, no matter what.
 256.                final InjectionSource injectionSource = new LookupInjectionSource(localContextName);
 257.                final ResourceInjectionConfiguration injectionConfiguration = targetDescription != null ?
 258.                        new ResourceInjectionConfiguration(targetDescription, injectionSource) : null;
 259.    
 260.                final BindingConfiguration bindingConfiguration = new BindingConfiguration(localContextName, valueSource);
 261.                classDescription.getBindingConfigurations().add(bindingConfiguration);
 262.    
 263.                if (injectionConfiguration != null) {
 264.                    classDescription.addResourceInjection(injectionConfiguration);
 265.                }
 266.            }
 267.        }
 268.    
 269.        private boolean isEmpty(final String string) {
 270.            return string == null || string.isEmpty();
 271.        }
 272.    
 273.        private boolean isEnvEntryType(final String type, final Module module) {
 274.            if (SIMPLE_ENTRIES.contains(type)) {
 275.                return true;
 276.            }
 277.            try {
 278.                return module.getClassLoader().loadClass(type).isEnum();
 279.            } catch (ClassNotFoundException e) {
 280.                return false;
 281.            }
 282.        }
