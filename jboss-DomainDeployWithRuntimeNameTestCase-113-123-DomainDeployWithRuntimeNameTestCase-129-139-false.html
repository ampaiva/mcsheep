 152.    
 153.            for (String host : CLITestSuite.hostAddresses.keySet()) {

 155.                for (String server : CLITestSuite.hostServers.get(host)) {
 156.                    if (!groupServers.contains(server)) {
 157.                        continue;  // server not in the group
 158.                    }
 159.                    if (!CLITestSuite.serverStatus.get(server)) {
 160.                        continue; // server not started
 161.                    }
 162.                    int port = 8080 + CLITestSuite.portOffsets.get(server);
 163.    
 164.                    URL url = new URL("http", address, port, path);
 165.                    boolean failed = false;
 166.                    try {
 167.                        String response = HttpRequest.get(url.toString(), 10, TimeUnit.SECONDS);
 168.                        assertThat(response, containsString(content));
 169.                    } catch (Exception e) {
 170.                        failed = true;
 171.                        if (!shouldFail) {
 172.                            throw new Exception("Http request failed.", e);
 173.                        }
 174.                    }
 175.                    if (shouldFail) {
 176.                        assertThat(url.toString(), failed, is(true));
 177.                    }
 178.                }
 179.            }

 146.    
 147.        private void checkURL(String path, String content, String serverGroup, boolean shouldFail) throws Exception {
 148.            List groupServers = new ArrayList();
 149.            for (String server : CLITestSuite.serverGroups.get(serverGroup)) {
 150.                groupServers.add(server);
 151.            }
 152.    
 153.            for (String host : CLITestSuite.hostAddresses.keySet()) {
 154.                String address = CLITestSuite.hostAddresses.get(host);
 155.                for (String server : CLITestSuite.hostServers.get(host)) {
 156.                    if (!groupServers.contains(server)) {
 157.                        continue;  // server not in the group
 158.                    }
 159.                    if (!CLITestSuite.serverStatus.get(server)) {
 160.                        continue; // server not started
 161.                    }
 162.                    int port = 8080 + CLITestSuite.portOffsets.get(server);
 163.    
 164.                    URL url = new URL("http", address, port, path);
 165.                    boolean failed = false;
 166.                    try {
 167.                        String response = HttpRequest.get(url.toString(), 10, TimeUnit.SECONDS);
 168.                        assertThat(response, containsString(content));
 169.                    } catch (Exception e) {
 170.                        failed = true;
 171.                        if (!shouldFail) {
 172.                            throw new Exception("Http request failed.", e);
 173.                        }
 174.                    }
 175.                    if (shouldFail) {
 176.                        assertThat(url.toString(), failed, is(true));
 177.                    }
 178.                }
 179.            }
 180.        }
