 199.        public void testShutdown(@ArquillianResource @OperateOnDeployment("singleton") ManagementClient client) throws Exception {
 200.            Connection connection = null;
 201.    
 202.            try {
 203.                deployer.deploy("mdb");
 204.                
 205.                ConnectionFactory cf = (ConnectionFactory) ctx.lookup("jms/RemoteConnectionFactory");
 206.                Queue queue = (Queue) ctx.lookup(QUEUE_SEND);
 207.    
 208.                connection = cf.createConnection("guest", "guest");
 209.                Session session = connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
 210.                Queue replyQueue = session.createTemporaryQueue();
 211.    
 212.                MessageProducer sender = session.createProducer(queue);
 213.                MessageConsumer receiver = session.createConsumer(replyQueue);
 214.    
 215.                connection.start();
 216.    
 217.                // we do not assume message order since the 1st message will be redelivered
 218.                // after redeployment (as the MDB is interrupted on 1st delivery)
 219.                Set expected = new TreeSet();
 220.                sendMessage(session, sender, replyQueue, "await");
 221.                expected.add("Reply: await");
 222.    
 223.                // synchronize receiving message
 224.                int awaitInt = awaitSingleton("await before undeploy");
 225.                log.debug("testsuite: first awaitSingleton() returned: " + awaitInt);
 226.    
 227.                Future undeployed = executor.submit(undeployTask());
 228.    
 229.                for (int i = 1; i <= 50; i++) {
 230.                    String msg = this.getClass().getSimpleName() + " 50loop: " + i;
 231.                    sendMessage(session, sender, replyQueue, msg); // should be bounced by BlockContainerShutdownInterceptor
 232.                    expected.add("Reply: " + msg);
 233.                }
 234.                log.debug("Sent 50 messages during MDB is undeploying");
 235.    
 236.                // synchronize with transaction timeout
 237.                awaitInt = awaitSingleton("await after undeploy");
 238.                log.debug("testsuite: second awaitSingleton() returned: " + awaitInt);
 239.    
 240.                undeployed.get(UNDEPLOYED_WAIT_S, SECONDS);
 241.    
 242.                // deploying via management client, arquillian deployer does not work for some reason
 243.                final ModelNode deployAddr = new ModelNode();
 244.                deployAddr.get(ClientConstants.OP_ADDR).add("deployment", MESSAGE_DRIVEN_BEAN + ".jar");
 245.                deployAddr.get(ClientConstants.OP).set("deploy");
 246.                applyUpdate(deployAddr, managementClient.getControllerClient());
 247.    
 248.                for (int i = 1; i <= 10; i++) {
 249.                    String msg = this.getClass().getSimpleName() + "10loop: " + i;

 256.                for (int i = 1; i <= (1 + 50 + 10 + 1); i++) {
 257.                    Message msg = receiver.receive(SECONDS.toMillis(RECEIVE_WAIT_S));
 258.                    assertNotNull("did not receive message with ordered number " + i +
 259.                            " in " + SECONDS.toMillis(RECEIVE_WAIT_S) + " seconds", msg);
 260.                    String text = ((TextMessage) msg).getText();
 261.                    received.add(text);
 262.                    log.info(i + ": " + text);
 263.                }
 264.                assertNull(receiver.receiveNoWait());
 265.    
 266.                assertEquals(expected, received);
 267.    
 268.            } finally {

 270.                    connection.close();
 271.                }

 273.            }
