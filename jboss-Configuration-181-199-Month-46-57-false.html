 215.         * if the persistence provider is Hibernate and use_class_enhancer is not true, don't need a class transformer.
 216.         * for other persistence providers, the transformer is assumed to be needed.
 217.         *

 219.         * @return true if class file transformer support is needed for pu
 220.         */
 221.        public static boolean needClassFileTransformer(PersistenceUnitMetadata pu) {
 222.            boolean result = true;
 223.            String provider = pu.getPersistenceProviderClassName();
 224.            if (pu.getProperties().containsKey(Configuration.JPA_CONTAINER_CLASS_TRANSFORMER)) {
 225.                result = Boolean.parseBoolean(pu.getProperties().getProperty(Configuration.JPA_CONTAINER_CLASS_TRANSFORMER));
 226.            }
 227.            else if (provider == null
 228.                || provider.equals(Configuration.PROVIDER_CLASS_HIBERNATE)) {
 229.                String useHibernateClassEnhancer = pu.getProperties().getProperty("hibernate.ejb.use_class_enhancer");
 230.                result = "true".equals(useHibernateClassEnhancer);
 231.            }
 232.            return result;
 233.        }

  64.        private static final int OFFSET = MONTH_ALIAS.get("jan") - Calendar.JANUARY;

  66.        private SortedSet offsetAdjustedMonths = new TreeSet();
  67.    
  68.        public Month(String value) {
  69.            super(value);
  70.            if (OFFSET != 0) {
  71.                for (Integer month : this.absoluteValues) {
  72.                    this.offsetAdjustedMonths.add(month - OFFSET);
  73.                }
  74.            } else {
  75.                this.offsetAdjustedMonths = this.absoluteValues;
  76.            }
  77.        }

 123.        public Integer getNextMatch(Calendar currentCal) {
 124.            if (this.scheduleExpressionType == ScheduleExpressionType.WILDCARD) {
 125.                return currentCal.get(Calendar.MONTH);
 126.            }
 127.            if (this.offsetAdjustedMonths.isEmpty()) {
 128.                return null;
 129.            }
 130.            int currentMonth = currentCal.get(Calendar.MONTH);
 131.            for (Integer month : this.offsetAdjustedMonths) {
 132.                if (currentMonth == month) {
 133.                    return currentMonth;
 134.                }
 135.                if (month > currentMonth) {
 136.                    return month;
 137.                }
 138.            }
 139.            return this.offsetAdjustedMonths.first();
 140.        }

 142.        public Integer getFirstMatch() {

 144.                return Calendar.JANUARY;
 145.            }
 146.            if (this.offsetAdjustedMonths.isEmpty()) {
 147.                throw EjbLogger.ROOT_LOGGER.invalidExpressionSeconds(this.origValue);
 148.            }
 149.            return this.offsetAdjustedMonths.first();
 150.        }
