 171.            //security=client is not allowed on 7.1.2
 172.    
 173.            KernelServicesBuilder builder = createKernelServicesBuilder(AdditionalInitialization.MANAGEMENT);
 174.    
 175.            // Add legacy subsystems
 176.            ModelVersion version_1_1_0 = ModelVersion.create(1, 1, 0);
 177.            builder.createLegacyKernelServicesBuilder(AdditionalInitialization.MANAGEMENT, controllerVersion, version_1_1_0)

 179.    
 180.            KernelServices mainServices = builder.build();

 187.            .addFailedAttribute(PathAddress.pathAddress(PathElement.pathElement(SUBSYSTEM, "jacorb")),
 188.                    new FailedOperationTransformationConfig.AttributesPathAddressConfig(JacORBSubsystemConstants.ORB_INIT_SECURITY) {
 189.    
 190.                        @Override
 191.                        protected boolean isAttributeWritable(String attributeName) {
 192.                            return true;
 193.                        }
 194.    
 195.                        @Override
 196.                        protected boolean checkValue(String attrName, ModelNode attribute, boolean isWriteAttribute) {
 197.                            return attribute.asString().equals("client");
 198.                        }
 199.    
 200.                        @Override
 201.                        protected ModelNode correctValue(ModelNode toResolve, boolean isWriteAttribute) {
 202.                            return new ModelNode("off");
 203.                        }
 204.                    });
 205.    
 206.            ModelTestUtils.checkFailedTransformedBootOperations(mainServices, version_1_1_0, builder.parseXmlResource("subsystem-security-client.xml"), config);
 207.            checkSubsystemModelTransformation(mainServices, version_1_1_0, new ModelFixer() {
 208.                @Override
 209.                public ModelNode fixModel(ModelNode modelNode) {
 210.                    //Set back to the value before the failed operation config kicked in
 211.                    Assert.assertEquals("off", modelNode.get(JacORBSubsystemConstants.ORB_INIT_SECURITY).asString());
 212.                    modelNode.get(JacORBSubsystemConstants.ORB_INIT_SECURITY).set("client");
 213.                    return modelNode;
 214.                }
 215.            });
 216.        }
