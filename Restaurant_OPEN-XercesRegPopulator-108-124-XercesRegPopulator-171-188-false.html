  42.    /**
  43.     * Simple implementation of the JposRegPopulator that loads and saves 
  44.     * the entries in XML using the "jpos/res/jcl.dtd" DTD and the XML4J 
  45.     * (Xerces) API 
  46.     * NOTE: this class must define a public no-argument ctor so that it may be 
  47.     * created via reflection when its defined in the jpos.properties as 
  48.     * the jpos.config.regPopulatorClass
  49.     * @see jpos.util.JposProperties#JPOS_REG_POPULATOR_CLASS_PROP_NAME
  50.     * @since 1.2 (NY 2K meeting)
  51.     * @author E. Michael Maximilien (maxim@us.ibm.com)
  52.     */
  53.    public class XercesRegPopulator extends AbstractXercesRegPopulator {
  54.        //-------------------------------------------------------------------------
  55.        // Ctor(s)
  56.        //
  57.    
  58.        /**
  59.         * Default ctor
  60.         * @since 1.2 (NY 2K meeting)
  61.         */
  62.        public XercesRegPopulator() {
  63.            super(XercesRegPopulator.class.getName());
  64.        }
  65.    
  66.        /**
  67.         * 1-arg constructor that takes the unique ID
  68.         * @param s the unique ID string
  69.         * @since 1.3 (Washington DC 2001)
  70.         */
  71.        public XercesRegPopulator(String s) {
  72.            super(s);
  73.        }
  74.    
  75.        //-------------------------------------------------------------------------
  76.        // Public methods
  77.        //
  78.    
  79.        /**
  80.         * @return the fully qualified class name implementing the 
  81.         * JposRegPopulator interface
  82.         * @since 1.3 (Washington DC 2001 meeting)
  83.         */
  84.        public String getClassName() {
  85.            return XercesRegPopulator.class.getName();
  86.        }
  87.    
  88.        /**
  89.         * Tell the populator to load the entries 
  90.         * @since 1.2 (NY 2K meeting)
  91.         */
  92.        public void load() {
  93.            tracer.println("load(): isPopulatorFileDefined=" + isPopulatorFileDefined());
  94.    
  95.            if (isPopulatorFileDefined() == false) {
  96.                getJposEntries().clear();
  97.                xmlFileName = DEFAULT_XML_FILE_NAME;
  98.                load(xmlFileName);
  99.    
 100.                return;
 101.            }
 102.    
 103.            try {
 104.                getJposEntries().clear();
 105.    
 106.                domParser.setEntityResolver(new XercesRegPopulator.JPOSDTDEntityResolver());
 107.    
 108.                domParser.parse(new InputSource(getPopulatorFileIS()));
 109.    
 110.                Document document = domParser.getDocument();
 111.    
 112.                Enumeration entries = extractJposEntries(document);
 113.    
 114.                while (entries.hasMoreElements()) {
 115.                    JposEntry jposEntry = (JposEntry) entries.nextElement();
 116.    
 117.                    if (jposEntry.hasPropertyWithName(JposEntry.LOGICAL_NAME_PROP_NAME))
 118.                        getJposEntries().put(jposEntry.getLogicalName(), jposEntry);
 119.                }
 120.    
 121.                lastLoadException = null;
 122.            } catch (Exception e) {
 123.                lastLoadException = e;
 124.                tracer.println("Error loading XML file.  Exception.msg = " + e.getMessage());
 125.            } finally {
 126.            }
 127.        }
 128.    
 129.        /**
 130.         * Loads the entries specified in the xmlFileName
 131.         * @param xmlFileName the XML file name
 132.         * @since 1.3 (SF 2K meeting)
 133.         */
 134.        public void load(String xmlFileName) {
 135.            tracer.println("load: xmlFileName=" + xmlFileName);
 136.    
 137.            InputStream is = null;
 138.            File xmlFile = new File(xmlFileName);
 139.    
 140.            try {
 141.                if (xmlFile.exists())
 142.                    is = new FileInputStream(xmlFile);
 143.                else
 144.                    is = findFileInClasspath(xmlFileName);
 145.    
 146.                if (is == null) {
 147.                    is = getClass().getClassLoader().getResourceAsStream(xmlFileName);
 148.                }
 149.    
 150.                if (is == null) {
 151.                    getJposEntries().clear();
 152.    
 153.                    tracer.println("Could not find file: " + xmlFileName + " in path or CLASSPATH");
 154.    
 155.                    lastLoadException = new FileNotFoundException(xmlFileName);
 156.    
 157.                    return;
 158.                }
 159.    
 160.                lastLoadException = null;
 161.            } catch (Exception e) {
 162.                lastLoadException = e;
 163.                tracer.println("Error loading XML file.  Exception.message = " + e.getMessage());
 164.            }
 165.    
 166.            try {
 167.                getJposEntries().clear();
 168.    
 169.                domParser.setEntityResolver(new XercesRegPopulator.JPOSDTDEntityResolver());
 170.    
 171.                domParser.parse(new InputSource(is));
 172.    
 173.                Document document = domParser.getDocument();
 174.    
 175.                Enumeration entries = extractJposEntries(document);
 176.    
 177.                while (entries.hasMoreElements()) {
 178.                    JposEntry jposEntry = (JposEntry) entries.nextElement();
 179.    
 180.                    if (jposEntry.hasPropertyWithName(JposEntry.LOGICAL_NAME_PROP_NAME))
 181.                        getJposEntries().put(jposEntry.getLogicalName(), jposEntry);
 182.                }
 183.    
 184.                lastLoadException = null;
 185.    
 186.            } catch (Exception e) {
 187.                lastLoadException = e;
 188.                tracer.println("Error loading XML file.  Exception.message = " + e.getMessage());
 189.            } finally {
 190.            }
 191.        }
 192.    
 193.        /**
 194.         * @return the name of this populator.  This should be a short descriptive name
 195.         * @since 1.3 (Washington DC 2001 meeting)
 196.         */
 197.        public String getName() {
 198.            return XERCES_REG_POPULATOR_NAME_STRING;
 199.        }
 200.    
 201.        //--------------------------------------------------------------------------
 202.        // Protected methods
 203.        //
 204.    
 205.        /**
 206.         * @return an enumeration of JposEntry objects read from the XML document object
 207.         * @param document the XML document object
 208.         * @since 1.2 (NY 2K meeting)
 209.         */
 210.        protected Enumeration extractJposEntries(Document document) {
 211.            Vector entries = new Vector();
 212.    
 213.            NodeList nodeList = document.getElementsByTagName("JposEntry");
 214.    
 215.            String currentEntryLogicalName = "";
 216.    
 217.            try {
 218.                for (int i = 0; i < nodeList.getLength(); ++i) {
 219.                    Node node = nodeList.item(i);
 220.    
 221.                    if (node.getNodeType() != Node.ELEMENT_NODE)
 222.                        continue;
 223.    
 224.                    JposEntry jposEntry = new SimpleEntry();
 225.    
 226.                    Element jposEntryElement = (Element) node;
 227.    
 228.                    currentEntryLogicalName = jposEntryElement.getAttribute("logicalName");
 229.                    jposEntry.addProperty("logicalName", currentEntryLogicalName);
 230.    
 231.                    NodeList childList = nodeList.item(i).getChildNodes();
 232.    
 233.                    for (int j = 0; j < childList.getLength(); ++j) {
 234.                        Node child = childList.item(j);
 235.    
 236.                        if (child.getNodeType() != Node.ELEMENT_NODE)
 237.                            continue;
 238.    
 239.                        Element element = (Element) child;
 240.    
 241.                        String elementName = element.getNodeName();
 242.    
 243.                        if (elementName.equals("creation"))
 244.                            extractCreationAttr(jposEntry, element);
 245.                        else if (elementName.equals("vendor"))
 246.                            extractVendorAttr(jposEntry, element);
 247.                        else if (elementName.equals("jpos"))
 248.                            extractJposAttr(jposEntry, element);
 249.                        else if (elementName.equals("product"))
 250.                            extractProductAttr(jposEntry, element);
 251.                        else
 252.                            extractPropAttr(jposEntry, element);
 253.                    }
 254.    
 255.                    if (JposEntryUtility.isValidJposEntry(jposEntry))
 256.                        entries.addElement(jposEntry);
 257.                    else {
 258.                        String msg = "JposEntry with logicalName: " + currentEntryLogicalName + " is not valid (missing required properties)";
 259.                        throw new JposConfigException(msg);
 260.                    }
 261.                }
 262.            } catch (JposConfigException jce) {
 263.                tracer.println("Skipping invalid entry with logicalName: " + currentEntryLogicalName);
 264.                tracer.println("--->JposConfigException.message = " + jce.getMessage());
 265.    
 266.                tracer.print(jce);
 267.    
 268.                if (jce.getOrigException() != null)
 269.                    tracer.print(jce.getOrigException());
 270.            }
 271.    
 272.            return entries.elements();
 273.        }
 274.    
 275.        /**
 276.         * Get the  element attributes and adds corresponding 
 277.         * properties to JposEntry
 278.         * @param jposEntry the entry to add properties to
 279.         * @param element the  XML element
 280.         * @since 1.2 (NY 2K meeting)
 281.         */
 282.        protected void extractCreationAttr(JposEntry jposEntry, Element element) {
 283.            jposEntry.addProperty("serviceInstanceFactoryClass", element.getAttribute("factoryClass"));
 284.    
 285.            jposEntry.addProperty("serviceClass", element.getAttribute("serviceClass"));
 286.        }
 287.    
 288.        /**
 289.         * Get the  element attributes and adds corresponding 
 290.         * properties to JposEntry
 291.         * @param jposEntry the entry to add properties to
 292.         * @param element the  XML element
 293.         * @since 1.2 (NY 2K meeting)
 294.         */
 295.        protected void extractVendorAttr(JposEntry jposEntry, Element element) {
 296.            jposEntry.addProperty("vendorName", element.getAttribute("name"));
 297.            jposEntry.addProperty("vendorURL", element.getAttribute("url"));
 298.        }
 299.    
 300.        /**
 301.         * Get the  element attributes and adds corresponding properties 
 302.         * to JposEntry
 303.         * @param jposEntry the entry to add properties to
 304.         * @param element the  XML element
 305.         * @since 1.2 (NY 2K meeting)
 306.         */
 307.        protected void extractJposAttr(JposEntry jposEntry, Element element) {
 308.            jposEntry.addProperty("jposVersion", element.getAttribute("version"));
 309.    
 310.            jposEntry.addProperty("deviceCategory", element.getAttribute("category"));
 311.        }
 312.    
 313.        /**
 314.         * Get the  element attributes and adds corresponding 
 315.         * properties to JposEntry
 316.         * @param jposEntry the entry to add properties to
 317.         * @param element the  XML element
 318.         * @since 1.2 (NY 2K meeting)
 319.         */
 320.        protected void extractProductAttr(JposEntry jposEntry, Element element) {
 321.            jposEntry.addProperty("productName", element.getAttribute("name"));
 322.    
 323.            jposEntry.addProperty("productDescription", element.getAttribute("description"));
 324.    
 325.            jposEntry.addProperty("productURL", element.getAttribute("url"));
 326.        }
 327.    
 328.        /**
 329.         * Get the  element attributes and adds corresponding properties 
 330.         * to JposEntry
 331.         * @param jposEntry the entry to add properties to
 332.         * @param element the  XML element
 333.         * @since 1.2 (NY 2K meeting)
 334.         * @throws jpos.config.JposConfigException if the property value does 
 335.         * not match the type or is not a valid value (like for instance 
 336.         * an invalid number)
 337.         */
 338.        protected void extractPropAttr(JposEntry jposEntry, Element element) throws JposConfigException {
 339.            String propName = element.getAttribute("name");
 340.            String propValueString = element.getAttribute("value");
 341.            String propTypeString = element.getAttribute("type");
 342.    
 343.            if (propTypeString.equals(""))
 344.                propTypeString = "String";
 345.    
 346.            Object propValue = null;
 347.            Class propType = null;
 348.    
 349.            try {
 350.                propType = Class.forName((propTypeString.startsWith("java.lang.") ? propTypeString : "java.lang." + propTypeString));
 351.            } catch (ClassNotFoundException cnfe) {
 352.                throw new JposConfigException("Invalid property type: " + propTypeString + " for property named: " + propName, cnfe);
 353.            }
 354.    
 355.            if (JposEntryUtility.isValidPropType(propType) == false)
 356.                throw new JposConfigException("Invalid property type: " + propTypeString + " for property named: " + propName);
 357.    
 358.            propValue = JposEntryUtility.parsePropValue(propValueString, propType);
 359.    
 360.            if (JposEntryUtility.validatePropValue(propValue, propType) == false)
 361.                throw new JposConfigException("Invalid property type: " + propTypeString + " for property named: " + propName);
 362.    
 363.            jposEntry.add(jposEntry.createProp(propName, propValue, propType));
 364.        }
 365.    
 366.        /**
 367.         * JposDTDEntityResolver to resolve DTD
 368.         */
 369.        public class JPOSDTDEntityResolver implements org.xml.sax.EntityResolver {
 370.            /**
 371.             * @return the DTD as an InputSource if it is found in a JAR
 372.             * file in the CLASSPATH otherwise return null
 373.             */
 374.            public org.xml.sax.InputSource resolveEntity(String publicId, String systemId) throws org.xml.sax.SAXException, java.io.IOException {
 375.                if (publicId.equals("-//JavaPOS//DTD//EN")) {
 376.                    InputStream is = getClass().getResourceAsStream(DTD_JAR_FILE_NAME);
 377.    
 378.                    if (is != null) {
 379.                        return (new org.xml.sax.InputSource(new InputStreamReader(is)));
 380.                    }
 381.                }
 382.    
 383.                return null;
 384.            }
 385.    
 386.        }
