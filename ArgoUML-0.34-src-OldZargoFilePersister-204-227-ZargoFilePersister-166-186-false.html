 123.         * @see org.argouml.persistence.ProjectFilePersister#save(
 124.         *      org.argouml.kernel.Project, java.io.File)
 125.         */
 126.        @Override
 127.        public void doSave(Project project, File file) throws SaveException, 
 128.        InterruptedException {
 129.    
 130.            LOG.info("Saving");
 131.            ProgressMgr progressMgr = new ProgressMgr();
 132.            progressMgr.setNumberOfPhases(4);
 133.            progressMgr.nextPhase();
 134.            
 135.            File lastArchiveFile = new File(file.getAbsolutePath() + "~");
 136.            File tempFile = null;
 137.    
 138.            try {
 139.                tempFile = createTempFile(file);
 140.            } catch (FileNotFoundException e) {
 141.                throw new SaveException(
 142.                        "Failed to archive the previous file version", e);
 143.            } catch (IOException e) {
 144.                throw new SaveException(
 145.                        "Failed to archive the previous file version", e);
 146.            }
 147.    
 148.            ZipOutputStream stream = null;
 149.            try {
 150.    
 151.                project.setFile(file);
 152.                project.setVersion(ApplicationVersion.getVersion());
 153.                project.setPersistenceVersion(PERSISTENCE_VERSION);
 154.    
 155.                stream = new ZipOutputStream(new FileOutputStream(file));
 156.    
 157.                for (ProjectMember projectMember : project.getMembers()) {
 158.                    if (projectMember.getType().equalsIgnoreCase("xmi")) {
 159.                        if (LOG.isInfoEnabled()) {
 160.                            LOG.info("Saving member of type: "
 161.                                    + projectMember.getType());
 162.                        }
 163.                        stream.putNextEntry(
 164.                                new ZipEntry(projectMember.getZipName()));
 165.                        MemberFilePersister persister =
 166.                            getMemberFilePersister(projectMember);
 167.                        persister.save(projectMember, stream);
 168.                    }
 169.                }
 170.                // if save did not raise an exception
 171.                // and name+"#" exists move name+"#" to name+"~"
 172.                // this is the correct backup file
 173.                if (lastArchiveFile.exists()) {
 174.                    lastArchiveFile.delete();
 175.                }
 176.                if (tempFile.exists() && !lastArchiveFile.exists()) {
 177.                    tempFile.renameTo(lastArchiveFile);
 178.                }
 179.                if (tempFile.exists()) {
 180.                    tempFile.delete();
 181.                }
 182.    
 183.                progressMgr.nextPhase();
 184.    
 185.            } catch (Exception e) {
 186.                LOG.error("Exception occured during save attempt", e);
 187.                try {
 188.                    if (stream != null) {
 189.                        stream.close();
 190.                    }
 191.                } catch (Exception ex) {
 192.                    // Do nothing.
 193.                }
 194.    
 195.                // frank: in case of exception
 196.                // delete name and mv name+"#" back to name if name+"#" exists
 197.                // this is the "rollback" to old file
 198.                file.delete();
 199.                tempFile.renameTo(file);
 200.                // we have to give a message to user and set the system to unsaved!
 201.                throw new SaveException(e);
 202.            }
 203.    
 204.            try {
 205.                stream.close();
 206.            } catch (IOException ex) {
 207.                LOG.error("Failed to close save output writer", ex);
 208.            }
 209.        }

 625.        /**
 626.         * Get a list of zip file entries which end with the given extension.
 627.         * If the extension is null, all entries are returned.
 628.         */
 629.        private List getEntryNames(File file, String extension)
 630.            throws IOException, MalformedURLException {
 631.            
 632.            ZipInputStream zis = new ZipInputStream(toURL(file).openStream());
 633.            List result = new ArrayList();
 634.            ZipEntry entry = zis.getNextEntry();
 635.            while (entry != null) {
 636.                String name = entry.getName();
 637.                if (extension == null || name.endsWith(extension)) {
 638.                    result.add(name);
 639.                }
 640.                entry = zis.getNextEntry();
 641.            }
 642.            zis.close();
 643.            return result;
 644.        }
