 215.        private void triggerRequestResponseCycleOnQueue() throws Exception {
 216.            final InitialContext ctx = new InitialContext();
 217.            final QueueConnectionFactory factory = (QueueConnectionFactory) ctx.lookup("java:/JmsXA");
 218.            final QueueConnection connection = factory.createQueueConnection();
 219.            try {
 220.                connection.start();
 221.                final QueueSession session = connection.createQueueSession(false, Session.AUTO_ACKNOWLEDGE);
 222.                final Queue replyDestination = session.createTemporaryQueue();
 223.                final String requestMessage = "test";
 224.                final Message message = session.createTextMessage(requestMessage);
 225.                message.setJMSReplyTo(replyDestination);
 226.                final Destination destination = (Destination) ctx.lookup(Constants.QUEUE_JNDI_NAME);
 227.                final MessageProducer producer = session.createProducer(destination);
 228.                producer.send(message);
 229.                producer.close();
 230.    
 231.                // wait for a reply
 232.                final QueueReceiver receiver = session.createReceiver(replyDestination);
 233.                final Message reply = receiver.receive(TimeoutUtil.adjust(1000));
 234.                assertNotNull("Did not receive a reply on the reply queue. Perhaps the original (request) message didn't make it to the MDB?", reply);
 235.                final String result = ((TextMessage) reply).getText();
 236.                assertEquals("Unexpected reply messsage", Constants.REPLY_MESSAGE_PREFIX + requestMessage, result);
 237.            } finally {
 238.                if (connection != null) {
 239.                    // just closing the connection will close the session and other related resources (@see javax.jms.Connection)
 240.                    safeClose(connection);
 241.                }
 242.            }
 243.    
 244.        }

 268.        private static void safeClose(final Connection connection) {

 270.                return;
 271.            }
 272.            try {
 273.                connection.close();
 274.            } catch (Throwable t) {
