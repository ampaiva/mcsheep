 320.         * @throws MalformedURLException

 322.         * @throws IOException
 323.         * @throws URISyntaxException

 325.        private void testAccess(URL url, boolean ignoreCase) throws MalformedURLException, ClientProtocolException, IOException,

 332.            Utils.makeCallWithBasicAuthn(servletUrl, MARCUS, MARCUS_PWD, 403);
 333.            //tests related to case (in)sensitiveness
 334.            if (ignoreCase) {
 335.                assertEquals("Response body is not correct.", SimpleSecuredServlet.RESPONSE_BODY,
 336.                        Utils.makeCallWithBasicAuthn(servletUrl, ANIL, ANIL_PWD.toUpperCase(Locale.ENGLISH), 200));
 337.                Utils.makeCallWithBasicAuthn(servletUrl, MARCUS, MARCUS_PWD.toLowerCase(Locale.ENGLISH), 403);
