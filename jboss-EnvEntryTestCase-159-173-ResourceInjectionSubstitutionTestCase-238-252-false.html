 182.                } else if (result.hasDefined(ClientConstants.FAILURE_DESCRIPTION)) {
 183.                    final String failureDesc = result.get(ClientConstants.FAILURE_DESCRIPTION).toString();
 184.                    throw new RuntimeException(failureDesc);
 185.                } else {
 186.                    throw new RuntimeException("Operation not successful; outcome = " + result.get("outcome"));
 187.                }
 188.            }
