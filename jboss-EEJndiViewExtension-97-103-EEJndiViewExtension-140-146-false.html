 156.                        final NamingStore compStore = NamingStore.class.cast(compContextController.getValue());
 157.                        try {
 158.                            context.addEntries(componentNode.get("java:comp"), new NamingContext(compStore, null));
 159.                        } catch (NamingException e) {
 160.                            throw new OperationFailedException(e, new ModelNode().set(EeLogger.ROOT_LOGGER.failedToRead("java:comp", appName, moduleName, componentName)));
 161.                        }
 162.                    }
 163.                }
 164.            }
 165.        }

 170.                cleaned = name.substring(0, name.length() - 4);
 171.            } else {
 172.                cleaned = name;
 173.            }
 174.            return cleaned;
 175.        }
