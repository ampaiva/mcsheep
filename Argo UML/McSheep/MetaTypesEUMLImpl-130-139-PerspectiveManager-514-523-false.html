 485.        public String getName(Object element) {
 486.            Class clazz;
 487.            if (element instanceof Class) {
 488.                clazz = (Class) element;
 489.            } else {
 490.                clazz = element.getClass();
 491.            }
 492.            String name = clazz.getName();
 493.    
 494.            // The name of the meta type is the class name (after the last .)
 495.            // and before the "Impl" or end of class name.
 496.            int startName = name.lastIndexOf('.') + 1;
 497.    
 498.            // Eclipse UML2 implementation classes often start with "UML"
 499.            final String prefix = "UML"; //$NON-NLS-1$
 500.            if (name.substring(startName).startsWith(prefix)) {
 501.                startName += prefix.length();
 502.            }
 503.    
 504.            // Eclipse UML2 implementation classes end with "Impl"
 505.            final String suffix = "Impl"; //$NON-NLS-1$
 506.            int endName = name.length();
 507.            if (name.endsWith(suffix)) {
 508.                endName -= suffix.length();
 509.            }
 510.    
 511.            return name.substring(startName, endName);
 512.        }

 248.        /**
 249.         * Tries to load user defined perspectives, if it can't it loads the
 250.         * (predefined) default perspectives.
 251.         */
 252.        public void loadUserPerspectives() {
 253.    
 254.            String userPerspectives =
 255.                Configuration.getString(
 256.                        Argo.KEY_USER_EXPLORER_PERSPECTIVES, "");
 257.    
 258.            StringTokenizer pst = new StringTokenizer(userPerspectives, ";");
 259.    
 260.            if (pst.hasMoreTokens()) {
 261.    
 262.                // load user perspectives
 263.                while (pst.hasMoreTokens()) {
 264.                    String perspective = pst.nextToken();
 265.                    StringTokenizer perspectiveDetails =
 266.                        new StringTokenizer(perspective, ",");
 267.    
 268.                    // get the perspective name
 269.                    String perspectiveName = perspectiveDetails.nextToken();
 270.    
 271.                    ExplorerPerspective userDefinedPerspective =
 272.                        new ExplorerPerspective(perspectiveName);
 273.    
 274.                    // make sure there are some rules...
 275.                    if (perspectiveDetails.hasMoreTokens()) {
 276.    
 277.                        // get the rules
 278.                        while (perspectiveDetails.hasMoreTokens()) {
 279.    
 280.                            // get the rule name
 281.                            String ruleName = perspectiveDetails.nextToken();
 282.    
 283.                            // create the rule
 284.                            try {
 285.                                Class ruleClass = Class.forName(ruleName);
 286.    
 287.                                PerspectiveRule rule =
 288.                                    (PerspectiveRule) ruleClass.newInstance();
 289.    
 290.                                userDefinedPerspective.addRule(rule);
 291.                            } catch (ClassNotFoundException e) {
 292.                                LOG.error(
 293.                                        "could not create rule " + ruleName 
 294.                                        + " you can try to "
 295.                                        + "refresh the perspectives to the "
 296.                                        + "default settings.",
 297.                                        e);
 298.                            } catch (InstantiationException e) {
 299.                                LOG.error(
 300.                                        "could not create rule " + ruleName 
 301.                                        + " you can try to "
 302.                                        + "refresh the perspectives to the "
 303.                                        + "default settings.",
 304.                                        e);
 305.                            } catch (IllegalAccessException e) {
 306.                                LOG.error(
 307.                                        "could not create rule " + ruleName 
 308.                                        + " you can try to "
 309.                                        + "refresh the perspectives to the "
 310.                                        + "default settings.",
 311.                                        e);
 312.                            }
 313.                        }
 314.                    } else {
 315.                        // rule name but no rules
 316.                        continue;
 317.                    }
 318.    
 319.                    // add the perspective
 320.                    addPerspective(userDefinedPerspective);
 321.                }
 322.            } else {
 323.                // no user defined perspectives
 324.                loadDefaultPerspectives();
 325.            }
 326.    
 327.            // one last check that some loaded.
 328.            if (getPerspectives().size() == 0) {
 329.                loadDefaultPerspectives();
 330.            }
 331.        }

 620.            if (first.length == 0) {
