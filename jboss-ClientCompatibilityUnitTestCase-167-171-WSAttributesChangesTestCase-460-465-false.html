 381.        private String getAttribute(final String attribute, final ModelControllerClient client, final boolean checkDefined) throws Exception {
 382.            ModelNode op = createOpNode("subsystem=webservices/", READ_ATTRIBUTE_OPERATION);
 383.            op.get(NAME).set(attribute);
 384.            final ModelNode result = client.execute(new OperationBuilder(op).build());
 385.            if (result.hasDefined(OUTCOME) && SUCCESS.equals(result.get(OUTCOME).asString())) {
 386.                if (checkDefined) {
 387.                    Assert.assertTrue(result.hasDefined(RESULT));
 388.                }
 389.                return result.get(RESULT).asString();
 390.            } else if (result.hasDefined(FAILURE_DESCRIPTION)) {
 391.                throw new Exception(result.get(FAILURE_DESCRIPTION).toString());
 392.            } else {
 393.                throw new Exception("Operation not successful; outcome = " + result.get(OUTCOME));
 394.            }
 395.        }

 397.        private static ModelNode createOpNode(String address, String operation) {
 398.            ModelNode op = new ModelNode();
 399.            // set address
 400.            ModelNode list = op.get(ADDRESS).setEmptyList();
 401.            if (address != null) {
 402.                String[] pathSegments = address.split("/");
 403.                for (String segment : pathSegments) {
 404.                    String[] elements = segment.split("=");
 405.                    list.add(elements[0], elements[1]);
 406.                }
 407.            }
 408.            op.get("operation").set(operation);
 409.            return op;
 410.        }

 412.        private static ModelNode applyUpdate(final ModelControllerClient client, final ModelNode update, final boolean expectReloadRequired) throws Exception {
 413.            final ModelNode result = client.execute(new OperationBuilder(update).build());
 414.            if (result.hasDefined(OUTCOME) && SUCCESS.equals(result.get(OUTCOME).asString())) {
 415.                if (expectReloadRequired) {
 416.                    Assert.assertTrue(result.hasDefined(RESPONSE_HEADERS));
 417.                    ModelNode responseHeaders = result.get(RESPONSE_HEADERS);
 418.                    Assert.assertTrue(responseHeaders.hasDefined(OPERATION_REQUIRES_RELOAD));
 419.                    Assert.assertEquals("true", responseHeaders.get(OPERATION_REQUIRES_RELOAD).asString());
 420.                } else {
 421.                    Assert.assertFalse(result.hasDefined(RESPONSE_HEADERS));
 422.                }
 423.                return result;
 424.            } else if (result.hasDefined(FAILURE_DESCRIPTION)) {
 425.                throw new Exception(result.get(FAILURE_DESCRIPTION).toString());
 426.            } else {
 427.                throw new Exception("Operation not successful; outcome = " + result.get(OUTCOME));
 428.            }
 429.        }

 431.        private void checkWsdl(URL wsdlURL, String hostOrPort) throws IOException {
 432.            HttpURLConnection connection = (HttpURLConnection) wsdlURL.openConnection();
 433.            try {
 434.                connection.connect();
 435.                Assert.assertEquals(200, connection.getResponseCode());
 436.                connection.getInputStream();
 437.    
 438.                BufferedReader in = new BufferedReader(new InputStreamReader(connection.getInputStream()));
 439.                String line;
 440.                while ((line = in.readLine()) != null) {
 441.                    if (line.contains("address location")) {
 442.                        Assert.assertTrue(line.contains(hostOrPort));
 443.                        return;
 444.                    }
 445.                }
 446.                fail("Could not check soap:address!");
 447.            } finally {
 448.                connection.disconnect();
 449.            }

 452.        private void checkWSDLUriScheme(final ModelControllerClient managementClient, String deploymentName, String expectedScheme) throws Exception {
 453.            final ModelNode address = new ModelNode();
 454.            address.add(DEPLOYMENT, deploymentName);
 455.            address.add(SUBSYSTEM, "webservices"); 
 456.            address.add("endpoint", "*"); // get all endpoints
 457.           
 458.            final ModelNode operation = new ModelNode();
 459.            operation.get(OP).set(READ_RESOURCE_OPERATION);
 460.            operation.get(OP_ADDR).set(address);
 461.            operation.get(INCLUDE_RUNTIME).set(true);
 462.            operation.get(RECURSIVE).set(true);
 463.                    
 464.            ModelNode result = managementClient.execute(operation);
 465.            Assert.assertEquals(SUCCESS, result.get(OUTCOME).asString());
 466.            for (final ModelNode endpointResult : result.get("result").asList()) {
 467.                final ModelNode endpoint = endpointResult.get("result");
 468.                final URL wsdlURL = new URL (endpoint.get("wsdl-url").asString());
 469.                HttpURLConnection connection = (HttpURLConnection) wsdlURL.openConnection();
 470.                try {
 471.                    connection.connect();
 472.                    Assert.assertEquals(200, connection.getResponseCode());
 473.                    BufferedReader in = new BufferedReader(new InputStreamReader(connection.getInputStream()));
 474.                    String line;
 475.                    while ((line = in.readLine()) != null) {
 476.                        if (line.contains("address location")) {
 477.                            if ("https".equals(expectedScheme)) {
 478.                                Assert.assertTrue(line, line.contains("https"));
 479.                                return;
 480.                            } else {
 481.                                Assert.assertTrue(line, line.contains("http") && !line.contains("https"));
 482.                                return;
 483.                            }
 484.                        }
 485.                    }
 486.                    fail(line + " Could not check soap:address!");
 487.                } finally {
 488.                    connection.disconnect();
 489.                }
 490.            }
