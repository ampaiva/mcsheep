 209.        private String modifyResource(final TxSupport txn, final String pUrl, final String pid, final String name,
 210.                final String value) {
 211.    
 212.            final int[] expected = new int[] { HttpURLConnection.HTTP_OK };

 219.        private String getResourceProperty(final TxSupport txn, final String pUrl, final String pid, final String name) {
 220.            final int[] expected = new int[] { HttpURLConnection.HTTP_OK, HttpURLConnection.HTTP_NO_CONTENT };
 221.            final String updateUrl = getResourceUpdateUrl(pUrl, pid, name, null).toString();
 222.            final String method = "GET";
 223.    
 224.            return txn.httpRequest(expected, updateUrl, method, TxMediaType.PLAIN_MEDIA_TYPE);
 225.        }

 227.        private StringBuilder getResourceUpdateUrl(final String pUrl, final String pid, final String name, final String value) {
 228.            final StringBuilder sb = new StringBuilder(pUrl);
 229.    
 230.            if (pid != null) {
 231.                sb.append("?pId=").append(pid).append("&name=");
 232.            } else {
 233.                sb.append("?name=");
 234.            }
 235.    
 236.            sb.append(name);
 237.    
 238.            if (value != null) {
 239.                sb.append("&value=").append(value);
 240.            }
 241.    
 242.            return sb;
 243.        }
