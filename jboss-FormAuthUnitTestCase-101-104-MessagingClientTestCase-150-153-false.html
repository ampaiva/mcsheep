  91.        private void doMessagingClient(ClientSessionFactory sf) throws Exception {
  92.            final String queueName = "queue.standalone";
  93.    
  94.            final ModelControllerClient client = managementClient.getControllerClient();
  95.    
  96.            // Check that the queue does not exists
  97.            if (queueExists(queueName, sf)) {
  98.                throw new IllegalStateException();
  99.            }
 100.    
 101.            // Create a new core queue using the standalone client
 102.            ModelNode op = new ModelNode();
 103.            op.get("operation").set("add");
 104.            op.get("address").add("subsystem", "messaging");
 105.            op.get("address").add("hornetq-server", "default");
 106.            op.get("address").add("queue", queueName);
 107.            op.get("queue-address").set(queueName);
 108.            applyUpdate(op, client);
 109.            // Check if the queue exists
 110.            if (!queueExists(queueName, sf)) {
 111.                throw new IllegalStateException();
 112.            }
 113.    
 114.            ClientSession session = null;
 115.            try {
 116.                session = sf.createSession("guest", "guest", false, true, true, false, 1);
 117.                ClientProducer producer = session.createProducer(queueName);
 118.                ClientMessage message = session.createMessage(false);
 119.    
 120.                final String propName = "myprop";
 121.                message.putStringProperty(propName, "Hello sent at " + new Date());
 122.    
 123.                producer.send(message);
 124.    
 125.                ClientConsumer messageConsumer = session.createConsumer(queueName);
 126.                session.start();
 127.    
 128.                ClientMessage messageReceived = messageConsumer.receive(1000);
 129.                assertNotNull("a message MUST have been received", messageReceived);
 130.            } finally {
 131.                if (session != null) {
 132.                    session.close();
 133.                }
 134.            }

 149.        static void applyUpdate(ModelNode update, final ModelControllerClient client) throws IOException {
 150.            ModelNode result = client.execute(new OperationBuilder(update).build());
 151.            if (result.hasDefined("outcome") && "success".equals(result.get("outcome").asString())) {
 152.                if (result.hasDefined("result")) {
 153.                    System.out.println(result.get("result"));
 154.                }
 155.            } else if (result.hasDefined("failure-description")) {
 156.                throw new RuntimeException(result.get("failure-description").toString());
 157.            } else {
 158.                throw new RuntimeException("Operation not successful; outcome = " + result.get("outcome"));
 159.            }
 160.        }
