 121.            // Handle case where no infinispan subsystem exists or does not define an ejb cache-container
 122.            Resource rootResource = context.readResourceFromRoot(PathAddress.EMPTY_ADDRESS);
 123.            PathElement infinispanPath = PathElement.pathElement(ModelDescriptionConstants.SUBSYSTEM, "infinispan");
 124.            if (!rootResource.hasChild(infinispanPath) || !rootResource.getChild(infinispanPath).hasChild(PathElement.pathElement("cache-container", BeanManagerFactoryBuilderConfiguration.DEFAULT_CONTAINER_NAME))) {
 125.                // Install services that would normally be installed by this container/cache
 126.                ModuleIdentifier module = Module.forClass(this.getClass()).getIdentifier();
 127.                for (GroupBuilderProvider provider : ServiceLoader.load(LocalGroupBuilderProvider.class, LocalGroupBuilderProvider.class.getClassLoader())) {
 128.                    for (Builder builder : provider.getBuilders(BeanManagerFactoryBuilderConfiguration.DEFAULT_CONTAINER_NAME, module)) {
 129.                        builder.build(target).install();
 130.                    }
 131.                }
 132.                for (CacheGroupBuilderProvider provider : ServiceLoader.load(LocalCacheGroupBuilderProvider.class, LocalCacheGroupBuilderProvider.class.getClassLoader())) {
 133.                    for (Builder builder : provider.getBuilders(BeanManagerFactoryBuilderConfiguration.DEFAULT_CONTAINER_NAME, CacheGroupServiceNameFactory.DEFAULT_CACHE)) {
 134.                        builder.build(target).install();
 135.                    }
 136.                }
 137.            }
 138.    
 139.            final OptionMap channelCreationOptions = this.getChannelCreationOptions(context);
 140.            // Install the EJB remoting connector service which will listen for client connections on the remoting channel
 141.            // TODO: Externalize (expose via management API if needed) the version and the marshalling strategy
 142.            final EJBRemoteConnectorService ejbRemoteConnectorService = new EJBRemoteConnectorService((byte) 0x02, new String[]{"river"}, channelCreationOptions);
 143.            target.addService(EJBRemoteConnectorService.SERVICE_NAME, ejbRemoteConnectorService)
 144.                    // add dependency on the Remoting subsystem endpoint
 145.                    .addDependency(RemotingServices.SUBSYSTEM_ENDPOINT, Endpoint.class, ejbRemoteConnectorService.getEndpointInjector())
 146.                    // add dependency on the remoting server (which allows remoting connector to connect to it)
 147.                    // add rest of the dependencies
 148.                    .addDependency(EJB3SubsystemModel.BASE_THREAD_POOL_SERVICE_NAME.append(threadPoolName), ExecutorService.class, ejbRemoteConnectorService.getExecutorService())
 149.                    .addDependency(DeploymentRepository.SERVICE_NAME, DeploymentRepository.class, ejbRemoteConnectorService.getDeploymentRepositoryInjector())
 150.                    .addDependency(EJBRemoteTransactionsRepository.SERVICE_NAME, EJBRemoteTransactionsRepository.class, ejbRemoteConnectorService.getEJBRemoteTransactionsRepositoryInjector())
 151.                    .addDependency(RegistryCollectorService.SERVICE_NAME, RegistryCollector.class, ejbRemoteConnectorService.getClusterRegistryCollectorInjector())
 152.                    .addDependency(TransactionManagerService.SERVICE_NAME, TransactionManager.class, ejbRemoteConnectorService.getTransactionManagerInjector())
 153.                    .addDependency(TransactionSynchronizationRegistryService.SERVICE_NAME, TransactionSynchronizationRegistry.class, ejbRemoteConnectorService.getTxSyncRegistryInjector())
 154.                    .addDependency(RemoteAsyncInvocationCancelStatusService.SERVICE_NAME, RemoteAsyncInvocationCancelStatusService.class, ejbRemoteConnectorService.getAsyncInvocationCancelStatusInjector())
 155.                    .addDependency(remotingServerInfoServiceName, RemotingConnectorBindingInfoService.RemotingConnectorInfo.class, ejbRemoteConnectorService.getRemotingConnectorInfoInjectedValue())
 156.                    .setInitialMode(ServiceController.Mode.ACTIVE)
 157.                    .install();
 158.        }

 166.        private OptionMap getChannelCreationOptions(final OperationContext context) throws OperationFailedException {
 167.            // read the full model of the current resource
 168.            final ModelNode fullModel = Resource.Tools.readModel(context.readResource(PathAddress.EMPTY_ADDRESS));
 169.            final ModelNode channelCreationOptions = fullModel.get(EJB3SubsystemModel.CHANNEL_CREATION_OPTIONS);
 170.            if (channelCreationOptions.isDefined() && channelCreationOptions.asInt() > 0) {
 171.                final ClassLoader loader = this.getClass().getClassLoader();
 172.                final OptionMap.Builder builder = OptionMap.builder();
 173.                for (final Property optionProperty : channelCreationOptions.asPropertyList()) {
 174.                    final String name = optionProperty.getName();
 175.                    final ModelNode propValueModel = optionProperty.getValue();
 176.                    final String type = RemoteConnectorChannelCreationOptionResource.CHANNEL_CREATION_OPTION_TYPE.resolveModelAttribute(context,propValueModel).asString();
 177.                    final String optionClassName = this.getClassNameForChannelOptionType(type);
 178.                    final String fullyQualifiedOptionName = optionClassName + "." + name;
 179.                    final Option option = Option.fromString(fullyQualifiedOptionName, loader);
 180.                    final String value = RemoteConnectorChannelCreationOptionResource.CHANNEL_CREATION_OPTION_VALUE.resolveModelAttribute(context, propValueModel).asString();
 181.                    builder.set(option, option.parseValue(value, loader));
 182.                }
 183.                return builder.getMap();
 184.            }
 185.            return OptionMap.EMPTY;
 186.        }
 187.    
 188.        private String getClassNameForChannelOptionType(final String optionType) {
 189.            if ("remoting".equals(optionType)) {
 190.                return RemotingOptions.class.getName();
 191.            }
 192.            if ("xnio".equals(optionType)) {
 193.                return Options.class.getName();
 194.            }
 195.            throw EjbLogger.ROOT_LOGGER.unknownChannelCreationOptionType(optionType);
 196.        }
