 124.                            }
 125.                        }
 126.                        break;
 127.                    }
 128.                    default: {
 129.                        throw unexpectedContent(reader);
 130.                    }
 131.                }
 132.            }
 133.            throw endOfDocument(reader.getLocation());
 134.        }

 136.        private static PermissionFactory parsePermission(final XMLStreamReader reader, final ModuleLoader loader, final ModuleIdentifier identifier)
 137.                throws XMLStreamException {
 138.    
 139.            // permission element has no attributes.
 140.            MetaDataElementParser.requireNoAttributes(reader);
 141.    
 142.            String permissionClass = null;
 143.            String permissionName = null;
 144.            String permissionActions = null;
 145.    
 146.            EnumSet requiredElements = EnumSet.of(Element.CLASS_NAME);
 147.            while (reader.hasNext()) {
 148.                switch (reader.nextTag()) {
 149.                    case XMLStreamConstants.END_ELEMENT: {
 150.                        // check if all required permission elements have been processed.
 151.                        if (!requiredElements.isEmpty())
 152.                            throw missingRequiredElement(reader, requiredElements);
 153.    
 154.                        // build a permission and add it to the list.
 155.                        PermissionFactory factory = new ModularPermissionFactory(loader, identifier, permissionClass,
 156.                                permissionName, permissionActions);
 157.                        return factory;
 158.                    }
 159.                    case XMLStreamConstants.START_ELEMENT: {
 160.                        Element element = Element.forName(reader.getLocalName());
 161.                        requiredElements.remove(element);
 162.                        switch (element) {
 163.                            case CLASS_NAME: {
 164.                                MetaDataElementParser.requireNoAttributes(reader);
 165.                                permissionClass = reader.getElementText();
 166.                                break;
 167.                            }
 168.                            case NAME: {
 169.                                MetaDataElementParser.requireNoAttributes(reader);
 170.                                permissionName = reader.getElementText();
 171.                                break;
 172.                            }
 173.                            case ACTIONS: {
 174.                                MetaDataElementParser.requireNoAttributes(reader);
 175.                                permissionActions = reader.getElementText();
 176.                                break;
 177.                            }
 178.                            default: {
 179.                                throw MetaDataElementParser.unexpectedElement(reader);
 180.                            }
 181.                        }
 182.                        break;
 183.                    }
 184.                    default: {
 185.                        throw unexpectedContent(reader);
 186.                    }
 187.                }
 188.            }
 189.            throw endOfDocument(reader.getLocation());
 190.        }

 192.        private static XMLStreamException unexpectedContent(final XMLStreamReader reader) {
 193.            final String kind;
 194.            switch (reader.getEventType()) {
 195.                case XMLStreamConstants.ATTRIBUTE:
 196.                    kind = "attribute";
 197.                    break;
 198.                case XMLStreamConstants.CDATA:
 199.                    kind = "cdata";
 200.                    break;
 201.                case XMLStreamConstants.CHARACTERS:
 202.                    kind = "characters";
 203.                    break;
 204.                case XMLStreamConstants.COMMENT:
 205.                    kind = "comment";
 206.                    break;
 207.                case XMLStreamConstants.DTD:
 208.                    kind = "dtd";
 209.                    break;
 210.                case XMLStreamConstants.END_DOCUMENT:
 211.                    kind = "document end";
 212.                    break;
 213.                case XMLStreamConstants.END_ELEMENT:
 214.                    kind = "element end";
 215.                    break;
 216.                case XMLStreamConstants.ENTITY_DECLARATION:
 217.                    kind = "entity declaration";
 218.                    break;
 219.                case XMLStreamConstants.ENTITY_REFERENCE:
 220.                    kind = "entity ref";
 221.                    break;
 222.                case XMLStreamConstants.NAMESPACE:
 223.                    kind = "namespace";
 224.                    break;
 225.                case XMLStreamConstants.NOTATION_DECLARATION:
 226.                    kind = "notation declaration";
 227.                    break;
 228.                case XMLStreamConstants.PROCESSING_INSTRUCTION:
 229.                    kind = "processing instruction";
 230.                    break;
 231.                case XMLStreamConstants.SPACE:
 232.                    kind = "whitespace";
 233.                    break;
 234.                case XMLStreamConstants.START_DOCUMENT:
 235.                    kind = "document start";
 236.                    break;
 237.                case XMLStreamConstants.START_ELEMENT:
 238.                    kind = "element start";
 239.                    break;
 240.                default:
 241.                    kind = "unknown";
 242.                    break;
 243.            }
 244.            final StringBuilder b = new StringBuilder("Unexpected content of type '").append(kind).append('\'');
 245.            if (reader.hasName()) {
 246.                b.append(" named '").append(reader.getName()).append('\'');
 247.            }
 248.            if (reader.hasText()) {
 249.                b.append(", text is: '").append(reader.getText()).append('\'');
 250.            }
 251.            return new XMLStreamException(b.toString(), reader.getLocation());
 252.        }
