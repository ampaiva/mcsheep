  73.

  75.         * Construct a new instance.
  76.         *
  77.         * @param componentName        the component name
  78.         * @param componentClassName   the component instance class name
  79.         * @param ejbModuleDescription the module description
  80.         */
  81.        public StatelessComponentDescription(final String componentName, final String componentClassName, final EjbJarDescription ejbModuleDescription,
  82.                                             final ServiceName deploymentUnitServiceName, final SessionBeanMetaData descriptorData) {
  83.            super(componentName, componentClassName, ejbModuleDescription, deploymentUnitServiceName, descriptorData);
  84.        }

  86.        @Override
  87.        public ComponentConfiguration createConfiguration(final ClassIndex classIndex, final ClassLoader moduleClassLoader, final ModuleLoader moduleLoader) {
  88.            final ComponentConfiguration statelessComponentConfiguration = new ComponentConfiguration(this, classIndex, moduleClassLoader, moduleLoader);
  89.            // setup the component create service
  90.            statelessComponentConfiguration.setComponentCreateServiceFactory(new StatelessComponentCreateServiceFactory());
  91.    
  92.            // setup the configurator to inject the PoolConfig in the StatelessSessionComponentCreateService
  93.            final StatelessComponentDescription statelessComponentDescription = (StatelessComponentDescription) statelessComponentConfiguration.getComponentDescription();
  94.            statelessComponentConfiguration.getCreateDependencies().add(new PoolInjectingConfigurator(statelessComponentDescription));
  95.    
  96.            // add the bmt interceptor
  97.            if (TransactionManagementType.BEAN.equals(this.getTransactionManagementType())) {
  98.                getConfigurators().add(new ComponentConfigurator() {
  99.                    @Override
 100.                    public void configure(final DeploymentPhaseContext context, final ComponentDescription description, final ComponentConfiguration configuration) throws DeploymentUnitProcessingException {
 101.    
 102.                        // add the bmt interceptor factory
 103.                        configuration.addComponentInterceptor(EjbBMTInterceptor.FACTORY, InterceptorOrder.Component.BMT_TRANSACTION_INTERCEPTOR, false);
 104.                    }
 105.                });
 106.            }
 107.            getConfigurators().add(new ComponentConfigurator() {
 108.                @Override
 109.                public void configure(final DeploymentPhaseContext context, final ComponentDescription description, final ComponentConfiguration configuration) throws DeploymentUnitProcessingException {
 110.                    if (TransactionManagementType.CONTAINER.equals(getTransactionManagementType())) {
 111.    
 112.                        final EEApplicationClasses applicationClasses = context.getDeploymentUnit().getAttachment(Attachments.EE_APPLICATION_CLASSES_DESCRIPTION);
 113.                        InterceptorClassDescription interceptorConfig = ComponentDescription.mergeInterceptorConfig(configuration.getComponentClass(), applicationClasses.getClassByName(description.getComponentClassName()), description, MetadataCompleteMarker.isMetadataComplete(context.getDeploymentUnit()));
 114.    
 115.                        configuration.addPostConstructInterceptor(new LifecycleCMTTxInterceptor.Factory(interceptorConfig.getPostConstruct(), true), InterceptorOrder.ComponentPostConstruct.TRANSACTION_INTERCEPTOR);
 116.                        configuration.addPreDestroyInterceptor(new LifecycleCMTTxInterceptor.Factory(interceptorConfig.getPreDestroy(), true), InterceptorOrder.ComponentPreDestroy.TRANSACTION_INTERCEPTOR);
 117.    
 118.                        configuration.addTimeoutViewInterceptor(TimerCMTTxInterceptor.FACTORY, InterceptorOrder.View.CMT_TRANSACTION_INTERCEPTOR);
 119.                    }
 120.                    configuration.addTimeoutViewInterceptor(StatelessComponentInstanceAssociatingFactory.instance(), InterceptorOrder.View.ASSOCIATING_INTERCEPTOR);
 121.                }
 122.            });
 123.    
 124.            return statelessComponentConfiguration;
 125.        }

 137.        @Override
 138.        protected void setupViewInterceptors(EJBViewDescription view) {
 139.            // let super do its job first
 140.            super.setupViewInterceptors(view);
 141.            addViewSerializationInterceptor(view);
 142.    
 143.            // add the instance associating interceptor at the start of the interceptor chain
 144.            view.getConfigurators().addFirst(new ViewConfigurator() {
 145.                @Override
 146.                public void configure(DeploymentPhaseContext context, ComponentConfiguration componentConfiguration, ViewDescription description, ViewConfiguration configuration) throws DeploymentUnitProcessingException {
 147.    
 148.                    //add equals/hashCode interceptor
 149.                    //add equals/hashCode interceptor
 150.                    for (Method method : configuration.getProxyFactory().getCachedMethods()) {
 151.                        if ((method.getName().equals("hashCode") && method.getParameterTypes().length == 0) ||
 152.                                method.getName().equals("equals") && method.getParameterTypes().length == 1 &&
 153.                                        method.getParameterTypes()[0] == Object.class) {
 154.                            configuration.addClientInterceptor(method, ComponentTypeIdentityInterceptorFactory.INSTANCE, InterceptorOrder.Client.EJB_EQUALS_HASHCODE);
 155.                        }
 156.                    }
 157.    
 158.                    // add the stateless component instance associating interceptor
 159.                    configuration.addViewInterceptor(StatelessComponentInstanceAssociatingFactory.instance(), InterceptorOrder.View.ASSOCIATING_INTERCEPTOR);
 160.                }
 161.            });
 162.    
 163.    
 164.            if (view instanceof EJBViewDescription) {
 165.                EJBViewDescription ejbViewDescription = (EJBViewDescription) view;
 166.                if (ejbViewDescription.getMethodIntf() == MethodIntf.REMOTE) {
 167.                    view.getConfigurators().add(new ViewConfigurator() {
 168.                        @Override
 169.                        public void configure(final DeploymentPhaseContext context, final ComponentConfiguration componentConfiguration, final ViewDescription description, final ViewConfiguration configuration) throws DeploymentUnitProcessingException {
 170.                            final String earApplicationName = componentConfiguration.getComponentDescription().getModuleDescription().getEarApplicationName();
 171.                            configuration.setViewInstanceFactory(new StatelessRemoteViewInstanceFactory(earApplicationName, componentConfiguration.getModuleName(), componentConfiguration.getComponentDescription().getModuleDescription().getDistinctName(), componentConfiguration.getComponentName()));
 172.                        }
 173.                    });
 174.                }
 175.            }
 176.    
 177.        }
 178.    
 179.        private void addViewSerializationInterceptor(final ViewDescription view) {
 180.            view.setSerializable(true);
 181.            view.setUseWriteReplace(true);
 182.            view.getConfigurators().add(new ViewConfigurator() {
 183.                @Override
 184.                public void configure(final DeploymentPhaseContext context, final ComponentConfiguration componentConfiguration, final ViewDescription description, final ViewConfiguration configuration) throws DeploymentUnitProcessingException {
 185.                    final DeploymentReflectionIndex index = context.getDeploymentUnit().getAttachment(org.jboss.as.server.deployment.Attachments.REFLECTION_INDEX);
 186.                    ClassReflectionIndex classIndex = index.getClassIndex(WriteReplaceInterface.class);
 187.                    for (Method method : classIndex.getMethods()) {
 188.                        configuration.addClientInterceptor(method, StatelessWriteReplaceInterceptor.factory(configuration.getViewServiceName().getCanonicalName()), InterceptorOrder.Client.WRITE_REPLACE);
 189.                    }
 190.                }
 191.            });
 192.        }

 212.        private class PoolInjectingConfigurator implements DependencyConfigurator> {
 213.    
 214.            private final StatelessComponentDescription statelessComponentDescription;
 215.    
 216.            PoolInjectingConfigurator(final StatelessComponentDescription statelessComponentDescription) {
 217.                this.statelessComponentDescription = statelessComponentDescription;
 218.            }
 219.    
 220.            @Override
 221.            public void configureDependency(ServiceBuilder serviceBuilder, Service service) throws DeploymentUnitProcessingException {
 222.                final StatelessSessionComponentCreateService statelessSessionComponentService = (StatelessSessionComponentCreateService) service;
 223.                final String poolName = this.statelessComponentDescription.getPoolConfigName();
 224.                // if no pool name has been explicitly set, then inject the *optional* "default slsb pool config".
 225.                // If the default slsb pool config itself is not configured, then the pooling is disabled for the bean
 226.                if (poolName == null) {
 227.                    serviceBuilder.addDependency(ServiceBuilder.DependencyType.OPTIONAL, PoolConfigService.DEFAULT_SLSB_POOL_CONFIG_SERVICE_NAME,
 228.                            PoolConfig.class, statelessSessionComponentService.getPoolConfigInjector());
 229.                } else {
 230.                    // pool name has been explicitly set so the pool config is a required dependency
 231.                    serviceBuilder.addDependency(PoolConfigService.EJB_POOL_CONFIG_BASE_SERVICE_NAME.append(poolName),
 232.                            PoolConfig.class, statelessSessionComponentService.getPoolConfigInjector());
 233.                }
 234.            }
 235.        }
