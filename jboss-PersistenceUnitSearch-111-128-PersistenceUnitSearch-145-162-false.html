  82.        private static PersistenceUnitMetadata findWithinApplication(DeploymentUnit unit, String persistenceUnitName) {

 108.        private static PersistenceUnitMetadata findWithinLibraryJar(DeploymentUnit unit, ResourceRoot moduleResourceRoot, String persistenceUnitName) {
 109.    
 110.            final ResourceRoot deploymentRoot = moduleResourceRoot;
 111.            PersistenceUnitMetadataHolder holder = deploymentRoot.getAttachment(PersistenceUnitMetadataHolder.PERSISTENCE_UNITS);
 112.            if (holder == null || holder.getPersistenceUnits() == null) {
 113.                if (traceEnabled) {
 114.                    ROOT_LOGGER.tracef("findWithinLibraryJar checking for '%s' found no persistence units", persistenceUnitName);
 115.                }
 116.                return null;
 117.            }
 118.    
 119.            ambiguousPUError(unit, persistenceUnitName, holder);
 120.            persistenceUnitName = defaultPersistenceUnitName(persistenceUnitName, holder);
 121.    
 122.            for (PersistenceUnitMetadata persistenceUnit : holder.getPersistenceUnits()) {
 123.                if (traceEnabled) {
 124.                    ROOT_LOGGER.tracef("findWithinLibraryJar check '%s' against pu '%s'", persistenceUnitName, persistenceUnit.getPersistenceUnitName());
 125.                }
 126.                if (persistenceUnitName == null || persistenceUnitName.length() == 0 || persistenceUnit.getPersistenceUnitName().equals(persistenceUnitName)) {
 127.                    if (traceEnabled) {
 128.                        ROOT_LOGGER.tracef("findWithinLibraryJar matched '%s' against pu '%s'", persistenceUnitName, persistenceUnit.getPersistenceUnitName());
 129.                    }
 130.                    return persistenceUnit;
 131.                }
 132.            }
 133.            return null;
 134.        }
 135.    
 136.        /*
 137.         * When finding the default persistence unit, the first persistence unit encountered is returned.
 138.         */
 139.        private static PersistenceUnitMetadata findWithinDeployment(DeploymentUnit unit, String persistenceUnitName) {
 140.            if (traceEnabled) {
 141.                ROOT_LOGGER.tracef("pu findWithinDeployment searching for %s", persistenceUnitName);
 142.            }
 143.    
 144.            for (ResourceRoot root : DeploymentUtils.allResourceRoots(unit)) {
 145.                PersistenceUnitMetadataHolder holder = root.getAttachment(PersistenceUnitMetadataHolder.PERSISTENCE_UNITS);
 146.                if (holder == null || holder.getPersistenceUnits() == null) {
 147.                    if (traceEnabled) {
 148.                        ROOT_LOGGER.tracef("pu findWithinDeployment skipping empty pu holder for %s", persistenceUnitName);
 149.                    }
 150.                    continue;
 151.                }
 152.    
 153.                ambiguousPUError(unit, persistenceUnitName, holder);
 154.                persistenceUnitName = defaultPersistenceUnitName(persistenceUnitName, holder);
 155.    
 156.                for (PersistenceUnitMetadata persistenceUnit : holder.getPersistenceUnits()) {
 157.                    if (traceEnabled) {
 158.                        ROOT_LOGGER.tracef("findWithinDeployment check '%s' against pu '%s'", persistenceUnitName, persistenceUnit.getPersistenceUnitName());
 159.                    }
 160.                    if (persistenceUnitName == null || persistenceUnitName.length() == 0 || persistenceUnit.getPersistenceUnitName().equals(persistenceUnitName)) {
 161.                        if (traceEnabled) {
 162.                            ROOT_LOGGER.tracef("findWithinDeployment matched '%s' against pu '%s'", persistenceUnitName, persistenceUnit.getPersistenceUnitName());
 163.                        }
 164.                        return persistenceUnit;
 165.                    }
 166.                }
 167.            }
 168.            return null;
 169.        }

 171.        private static void ambiguousPUError(DeploymentUnit unit, String persistenceUnitName, PersistenceUnitMetadataHolder holder) {
 172.            if (holder.getPersistenceUnits().size() > 1 &&  (persistenceUnitName == null || persistenceUnitName.length() == 0)) {
 173.                int numberOfDefaultPersistenceUnits = 0;
 174.    
 175.                // get number of persistence units that are marked as default
 176.                for (PersistenceUnitMetadata persistenceUnit : holder.getPersistenceUnits()) {
 177.                    String defaultPU = persistenceUnit.getProperties().getProperty(Configuration.JPA_DEFAULT_PERSISTENCE_UNIT);
 178.                    if(Boolean.TRUE.toString().equals(defaultPU)) {
 179.                        numberOfDefaultPersistenceUnits++;
 180.                    }
 181.                }
 182.                ROOT_LOGGER.tracef("checking for ambiguous persistence unit injection error, " +
 183.                        "number of persistence units marked default (%s) = %d", Configuration.JPA_DEFAULT_PERSISTENCE_UNIT,  numberOfDefaultPersistenceUnits);
 184.                // don't throw an error if there is exactly one default persistence unit
 185.                if (numberOfDefaultPersistenceUnits != 1) {
 186.                    // AS7-2275 no unitName and there is more than one persistence unit;
 187.                    throw JpaLogger.ROOT_LOGGER.noPUnitNameSpecifiedAndMultiplePersistenceUnits(holder.getPersistenceUnits().size(), unit);
 188.                }
 189.            }
 190.        }
 191.    
 192.        /**
 193.         * if no persistence unit name is specified, return name of default persistence unit
 194.         *
 195.         * @param persistenceUnitName that was specified to be used (null means to use the default persistence unit)
 196.         * @param holder
 197.         * @return
 198.         */
 199.        private static String defaultPersistenceUnitName(String persistenceUnitName, PersistenceUnitMetadataHolder holder) {
 200.            if ((persistenceUnitName == null || persistenceUnitName.length() == 0)) {
 201.                for (PersistenceUnitMetadata persistenceUnit : holder.getPersistenceUnits()) {
 202.                    String defaultPU = persistenceUnit.getProperties().getProperty(Configuration.JPA_DEFAULT_PERSISTENCE_UNIT);
 203.                    if(Boolean.TRUE.toString().equals(defaultPU)) {
 204.                        persistenceUnitName = persistenceUnit.getPersistenceUnitName();
 205.                    }
 206.                }
 207.            }
 208.            return persistenceUnitName;
 209.        }

 211.        private static PersistenceUnitMetadata getPersistenceUnit(DeploymentUnit current, final String absolutePath, String puName) {
 212.            final String path;
 213.            if (absolutePath.startsWith("../")) {
 214.                path = absolutePath.substring(3);
 215.            } else {
 216.                path = absolutePath;
 217.            }
 218.            final VirtualFile parent = current.getAttachment(Attachments.DEPLOYMENT_ROOT).getRoot().getParent();
 219.            final VirtualFile resolvedPath = parent.getChild(path);
 220.    
 221.            List resourceRoots = DeploymentUtils.allResourceRoots(DeploymentUtils.getTopDeploymentUnit(current));
 222.    
 223.            for (ResourceRoot resourceRoot : resourceRoots) {
 224.                if (resourceRoot.getRoot().equals(resolvedPath)) {
 225.                    PersistenceUnitMetadataHolder holder = resourceRoot.getAttachment(PersistenceUnitMetadataHolder.PERSISTENCE_UNITS);
 226.                    if (holder != null) {
 227.                        for (PersistenceUnitMetadata pu : holder.getPersistenceUnits()) {
 228.                            if (traceEnabled) {
 229.                                ROOT_LOGGER.tracef("getPersistenceUnit check '%s' against pu '%s'", puName, pu.getPersistenceUnitName());
 230.                            }
 231.                            if (pu.getPersistenceUnitName().equals(puName)) {
 232.                                if (traceEnabled) {
 233.                                    ROOT_LOGGER.tracef("getPersistenceUnit matched '%s' against pu '%s'", puName, pu.getPersistenceUnitName());
 234.                                }
 235.                                return pu;
 236.                            }
 237.                        }
 238.                    }
 239.                }
 240.            }
 241.    
 242.    
 243.            throw JpaLogger.ROOT_LOGGER.persistenceUnitNotFound(absolutePath, puName, current);
 244.        }
