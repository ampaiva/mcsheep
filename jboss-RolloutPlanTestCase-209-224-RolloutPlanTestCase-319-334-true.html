 391.            for (Object group : groups.values()) {
 392.                Map hosts = (Map)((Map)group).get("host");

 397.                        if (serverResult != null) {

 399.                            String serverOutcome = (String) serverResponse.get("outcome");
 400.                            return "success".equals(serverOutcome);
 401.                        }
 402.                    }
 403.                }
 404.            }
 405.            throw new NoResponseException(serverName);

 389.        private boolean getServerStatus(String serverName, CLIOpResult result) throws Exception {
 390.            Map  groups = (Map) result.getServerGroups();
 391.            for (Object group : groups.values()) {
 392.                Map hosts = (Map)((Map)group).get("host");
 393.                if (hosts != null) {
 394.                    for (Object value : hosts.values()) {
 395.                        Map serverResults = (Map)value;
 396.                        Map serverResult = (Map)serverResults.get(serverName);
 397.                        if (serverResult != null) {
 398.                            Map serverResponse  = (Map)serverResult.get("response");
 399.                            String serverOutcome = (String) serverResponse.get("outcome");
 400.                            return "success".equals(serverOutcome);
 401.                        }
 402.                    }
 403.                }
 404.            }
 405.            throw new NoResponseException(serverName);
 406.        }

 412.        private static String checkURL(String server, boolean shouldFail, String path) throws Exception {
 413.            String address = CLITestSuite.hostAddresses.get(getServerHost(server));
 414.            Integer portOffset = CLITestSuite.portOffsets.get(server);
 415.    
 416.            URL url = new URL("http", address, 8080 + portOffset, path);
 417.            boolean failed = false;
 418.            String response = null;
 419.            try {
 420.                response = HttpRequest.get(url.toString(), 60, TimeUnit.SECONDS);
 421.            } catch (Exception e) {
 422.                failed = true;
 423.                if (!shouldFail) throw new Exception("Http request failed.", e);
 424.            }
 425.            if (shouldFail) Assert.assertTrue(failed);
 426.            return response;
 427.    
 428.        }

 430.        private static String getServerHost(String server) {
 431.            for(Entry hostEntry : CLITestSuite.hostServers.entrySet()) {
 432.                for (String hostServer : hostEntry.getValue()) if (hostServer.equals(server)) return hostEntry.getKey();
 433.            }
 434.            return null;
 435.        }

 437.        private static void waitUntilState(final String serverName, final String state) throws TimeoutException {
 438.            final String serverHost = CLITestSuite.getServerHost(serverName);
 439.            RetryTaskExecutor taskExecutor = new RetryTaskExecutor();
 440.            taskExecutor.retryTask(new Callable() {
 441.                public Void call() throws Exception {
 442.                    cli.sendLine("/host=" + serverHost + "/server-config=" + serverName + ":read-attribute(name=status)");
 443.                    CLIOpResult res = cli.readAllAsOpResult();
 444.                    if (! res.getResult().equals(state)) throw new Exception("Server not in state.");
 445.                    return null;
 446.                }
 447.            });
