 120.        private static ContextSelector setupEJBClientContextSelector() throws IOException {
 121.            log.info("*** reading EJBClientContextSelector properties");
 122.            // setup the selector
 123.            final String clientPropertiesFile = "org/jboss/as/test/manualmode/ejb/ssl/jboss-ejb-client.properties";
 124.            final InputStream inputStream = SSLEJBRemoteClientTestCase.class.getClassLoader().getResourceAsStream(clientPropertiesFile);
 125.            if (inputStream == null) {
 126.                throw new IllegalStateException("Could not find " + clientPropertiesFile + " in classpath");
 127.            }
 128.            final Properties properties = new Properties();
 129.            properties.load(inputStream);
 130.            final EJBClientConfiguration ejbClientConfiguration = new PropertiesBasedEJBClientConfiguration(properties);
 131.            log.info("*** creating EJBClientContextSelector");
 132.            final ConfigBasedEJBClientContextSelector selector = new ConfigBasedEJBClientContextSelector(ejbClientConfiguration);
 133.            log.info("*** applying EJBClientContextSelector");
 134.            return EJBClientContext.setSelector(selector);
 135.        }

 138.        @Before
 139.        public void prepareServerOnce() throws Exception {
 140.            if(!serverConfigDone) {
 141.                // prepare server config and then restart
 142.                log.info("*** preparing server configuration");
 143.                ManagementClient managementClient;
 144.                log.info("*** starting server");
 145.                container.start(DEFAULT_JBOSSAS);
 146.                final ModelControllerClient client = TestSuiteEnvironment.getModelControllerClient();
 147.                managementClient = new ManagementClient(client, TestSuiteEnvironment.getServerAddress(), TestSuiteEnvironment.getServerPort(), "http-remoting");
 148.                log.info("*** will configure server now");
 149.                SSLRealmSetupTool.setup(managementClient);
 150.                log.info("*** restarting server");
 151.                container.stop(DEFAULT_JBOSSAS);
 152.                container.start(DEFAULT_JBOSSAS);
 153.                managementClient = new ManagementClient(client, TestSuiteEnvironment.getServerAddress(), TestSuiteEnvironment.getServerPort(), "http-remoting");
 154.                // write SSL realm config to output - debugging purposes
 155.                SSLRealmSetupTool.readSSLRealmConfig(managementClient);
 156.                previousClientContextSelector = setupEJBClientContextSelector();
 157.                serverConfigDone = true;
 158.            } else {
 159.                log.info("*** Server already prepared, skipping config procedure");
 160.            }
 161.        }

 166.                EJBClientContext.setSelector(previousClientContextSelector);
 167.            }
 168.            final ModelControllerClient client = TestSuiteEnvironment.getModelControllerClient();
 169.            final ManagementClient mClient = new ManagementClient(client, TestSuiteEnvironment.getServerAddress(), TestSuiteEnvironment.getServerPort(), "http-remoting");
 170.            SSLRealmSetupTool.tearDown(mClient, container);
 171.        }

 180.        @Test
 181.        public void testStatelessBean() throws Exception {
 182.            log.info("**** deploying deployment with stateless beans");
 183.            deployer.deploy(DEPLOYMENT_STATELESS);
 184.            log.info("**** creating InitialContext");
 185.            InitialContext ctx = new InitialContext(getEjbClientContextProperties());
 186.            try {
 187.                log.info("**** looking up StatelessBean through JNDI");
 188.                StatelessBeanRemote bean = (StatelessBeanRemote)
 189.                        ctx.lookup("ejb:/" + MODULE_NAME_STATELESS + "/" + StatelessBean.class.getSimpleName() + "!" + StatelessBeanRemote.class.getCanonicalName());
 190.                log.info("**** About to perform synchronous call on stateless bean");
 191.                String response = bean.sayHello();
 192.                log.info("**** The answer is: " + response);
 193.                Assert.assertEquals("Remote invocation of EJB was not successful", StatelessBeanRemote.ANSWER, response);
 194.                deployer.undeploy(DEPLOYMENT_STATELESS);
 195.            } finally {
 196.                ctx.close();
 197.            }
 198.        }
