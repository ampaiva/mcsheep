 261.        /**
 262.         * Run this one in the container so it can lookup the queue
 263.         * @throws Exception
 264.         */
 265.        @OperateOnDeployment("test")
 266.        @Test
 267.        public void testMDBLifecycle() throws Exception {
 268.            deployer.deploy("mdb");
 269.            SecurityClient client = this.login();
 270.            ITestResultsSingleton results = this.getResultsSingleton();
 271.    
 272.            MessageProducer producer = null;
 273.            MessageConsumer consumer = null;
 274.            QueueConnection conn = null;
 275.            Session session = null;
 276.    
 277.            try {
 278.                QueueConnectionFactory qcf = (QueueConnectionFactory) new InitialContext().lookup("java:/ConnectionFactory");
 279.                Queue queue = (Queue) new InitialContext().lookup("java:jboss/" + QUEUE_NAME);
 280.    
 281.                conn = qcf.createQueueConnection("guest", "guest");
 282.                conn.start();
 283.                session = conn.createQueueSession(false, QueueSession.AUTO_ACKNOWLEDGE);
 284.    
 285.                TemporaryQueue replyQueue = session.createTemporaryQueue();
 286.    
 287.                TextMessage msg = session.createTextMessage("Hello world");
 288.                msg.setJMSDeliveryMode(DeliveryMode.NON_PERSISTENT);
 289.                msg.setJMSReplyTo(replyQueue);
 290.    
 291.                producer = session.createProducer(queue);
 292.                producer.send(msg);
 293.                consumer = session.createConsumer(replyQueue);
 294.                Message replyMsg = consumer.receive(5000);
 295.    
 296.                Object obj = ((ObjectMessage) replyMsg).getObject();
 297.                log.info("MDB message get: " + obj);
 298.    
 299.                Assert.assertEquals(OK + "start", results.getMdb("postconstruct"));
 300.    
 301.                deployer.undeploy("mdb");
 302.    
 303.                Assert.assertEquals(OK + "stop", results.getMdb("predestroy"));
 304.            } finally {
 305.                if(consumer != null) {
 306.                    consumer.close();
 307.                }
 308.                if(producer!=null) {
 309.                    producer.close();
 310.                }
 311.                if(session!=null) {
 312.                    session.close();
 313.                }
 314.                if(conn!=null) {
 315.                    conn.close();
 316.                }
 317.                client.logout();
 318.            }
 319.        }
