 232.        private void parseHandlerChain(final XMLExtendedStreamReader reader, final String configName, final List operationList, final boolean isPreHandlerChain) throws XMLStreamException {
 233.            String handlerChainId = null;
 234.            final EnumSet encountered = EnumSet.noneOf(Element.class);
 235.            final int count = reader.getAttributeCount();
 236.            for (int i = 0; i < count; i++) {
 237.                requireNoNamespaceAttribute(reader, i);
 238.                final String value = reader.getAttributeValue(i);
 239.                final Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
 240.                switch (attribute) {
 241.                case ID:
 242.                    handlerChainId = value;
 243.                    break;
 244.                default:
 245.                    throw unexpectedAttribute(reader, i);
 246.                }
 247.            }
 248.            if (handlerChainId == null) {
 249.                handlerChainId = "auto-generated-" + System.currentTimeMillis();
 250.            }
 251.            String protocolBindings = null;
 252.            final List addHandlerOperations = new LinkedList();
 253.            while (reader.nextTag() != END_ELEMENT) {
 254.                final Element element = Element.forName(reader.getLocalName());
 255.                if (element != Element.HANDLER && !encountered.add(element)) {
 256.                    throw unexpectedElement(reader);
 257.                }
 258.                switch (element) {
 259.                    case PROTOCOL_BINDINGS: {
 260.                        protocolBindings = parseElementNoAttributes(reader);
 261.                        break;
 262.                    }
 263.                    case HANDLER: {
 264.                        parseHandler(reader, configName, handlerChainId, isPreHandlerChain, addHandlerOperations);
 265.                        break;
 266.                    }
 267.                    default: {
 268.                        throw unexpectedElement(reader);
 269.                    }
 270.                }
 271.            }
 272.            final ModelNode operation = new ModelNode();
 273.            final String handlerChainType = isPreHandlerChain ? PRE_HANDLER_CHAIN : POST_HANDLER_CHAIN;
 274.            operation.get(OP).set(ADD);
 275.            operation.get(OP_ADDR).add(SUBSYSTEM, WSExtension.SUBSYSTEM_NAME).add(ENDPOINT_CONFIG, configName).add(handlerChainType, handlerChainId);
 276.            if (protocolBindings != null) {
 277.                operation.get(Constants.PROTOCOL_BINDINGS).set(protocolBindings);
 278.            }
 279.            operationList.add(operation);
 280.            operationList.addAll(addHandlerOperations);
 281.        }
