 445.        /**
 446.         * Executes the operation and returns the result if successful. Else throws an exception
 447.         *
 448.         * @param modelControllerClient
 449.         * @param operation
 450.         * @return
 451.         * @throws IOException
 452.         */
 453.        private static ModelNode execute(final ModelControllerClient modelControllerClient, final ModelNode operation) throws IOException {
 454.            final ModelNode result = modelControllerClient.execute(operation);
 455.            if (result.hasDefined(ClientConstants.OUTCOME) && ClientConstants.SUCCESS.equals(result.get(ClientConstants.OUTCOME).asString())) {
 456.                logger.info("Operation " + operation.toString() + " successful");
 457.                return result;
 458.            } else if (result.hasDefined(ClientConstants.FAILURE_DESCRIPTION)) {
 459.                final String failureDesc = result.get(ClientConstants.FAILURE_DESCRIPTION).toString();
 460.                throw new RuntimeException(failureDesc);
 461.            } else {
 462.                throw new RuntimeException("Operation not successful; outcome = " + result.get(ClientConstants.OUTCOME));
 463.            }
 464.    
 465.        }

 467.        // TODO: This method is temporary hack till we figure out the management operation to get the
 468.        // jboss.node.name system property from the server http://lists.jboss.org/pipermail/jboss-as7-dev/2011-November/004434.html
 469.        public static String getNodeName() {
 470.            // Logic copied from org.jboss.as.server.ServerEnvironment constructor
 471.            final Properties props = System.getProperties();
 472.            final Map env = System.getenv();
 473.            // Calculate host and default server name
 474.            String hostName = props.getProperty("jboss.host.name");
 475.            String qualifiedHostName = props.getProperty("jboss.qualified.host.name");
 476.            if (qualifiedHostName == null) {
 477.                // if host name is specified, don't pick a qualified host name that isn't related to it
 478.                qualifiedHostName = hostName;
 479.                if (qualifiedHostName == null) {
 480.                    // POSIX-like OSes including Mac should have this set
 481.                    qualifiedHostName = env.get("HOSTNAME");
 482.                }
 483.                if (qualifiedHostName == null) {
 484.                    // Certain versions of Windows
 485.                    qualifiedHostName = env.get("COMPUTERNAME");
 486.                }
 487.                if (qualifiedHostName == null) {
 488.                    try {
 489.                        qualifiedHostName = NetworkUtils.canonize(InetAddress.getLocalHost().getHostName());
 490.                    } catch (UnknownHostException e) {
 491.                        qualifiedHostName = null;
 492.                    }
 493.                }
 494.                if (qualifiedHostName != null && qualifiedHostName.matches("^\\d+\\.\\d+\\.\\d+\\.\\d+$|:")) {
 495.                    // IP address is not acceptable
 496.                    qualifiedHostName = null;
 497.                }
 498.                if (qualifiedHostName == null) {
 499.                    // Give up
 500.                    qualifiedHostName = "unknown-host.unknown-domain";
 501.                }
 502.                qualifiedHostName = qualifiedHostName.trim().toLowerCase(Locale.ENGLISH);
 503.            }
 504.            if (hostName == null) {
 505.                // Use the host part of the qualified host name
 506.                final int idx = qualifiedHostName.indexOf('.');
 507.                hostName = idx == -1 ? qualifiedHostName : qualifiedHostName.substring(0, idx);
 508.            }
 509.    
 510.            // Set up the server name for management purposes
 511.            String serverName = props.getProperty("jboss.server.name");
 512.            if (serverName == null) {
 513.                serverName = hostName;
 514.            }
 515.            // Set up the clustering node name
 516.            String nodeName = props.getProperty("jboss.node.name");
 517.            if (nodeName == null) {
 518.                nodeName = serverName;
 519.            }
 520.            return nodeName;
 521.        }
