 215.        private void triggerRequestResponseCycleOnQueue() throws Exception {
 216.            final InitialContext ctx = new InitialContext();
 217.            final QueueConnectionFactory factory = (QueueConnectionFactory) ctx.lookup("java:/JmsXA");
 218.            final QueueConnection connection = factory.createQueueConnection();
 219.            try {
 220.                connection.start();
 221.                final QueueSession session = connection.createQueueSession(false, Session.AUTO_ACKNOWLEDGE);
 222.                final Queue replyDestination = session.createTemporaryQueue();
 223.                final String requestMessage = "test";
 224.                final Message message = session.createTextMessage(requestMessage);
 225.                message.setJMSReplyTo(replyDestination);
 226.                final Destination destination = (Destination) ctx.lookup(Constants.QUEUE_JNDI_NAME);
 227.                final MessageProducer producer = session.createProducer(destination);
 228.                producer.send(message);
 229.                producer.close();
 230.    
 231.                // wait for a reply
 232.                final QueueReceiver receiver = session.createReceiver(replyDestination);
 233.                final Message reply = receiver.receive(TimeoutUtil.adjust(1000));
 234.                assertNotNull("Did not receive a reply on the reply queue. Perhaps the original (request) message didn't make it to the MDB?", reply);
 235.                final String result = ((TextMessage) reply).getText();
 236.                assertEquals("Unexpected reply messsage", Constants.REPLY_MESSAGE_PREFIX + requestMessage, result);
 237.            } finally {
 238.                if (connection != null) {
 239.                    // just closing the connection will close the session and other related resources (@see javax.jms.Connection)
 240.                    safeClose(connection);
 241.                }
 242.            }

 244.        }

 268.        private static void safeClose(final Connection connection) {
 269.            if (connection == null) {
 270.                return;
 271.            }
 272.            try {
 273.                connection.close();
 274.            } catch (Throwable t) {
 275.                // just log

 141.        @SuppressWarnings("static-access")
 142.        @Test
 143.        public void testMDB() throws Exception {
 144.            boolean requiresUndeploy = false;
 145.            try {
 146.                // do the deployment of the MDB
 147.                log.info("About to deploy MDB archive " + MDB_DEPLOYMENT_NAME);
 148.                deployer.deploy(MDB_DEPLOYMENT_NAME);
 149.                // we keep track of this to make sure we undeploy before leaving this method
 150.                requiresUndeploy = true;
 151.                log.info("deployed " + MDB_DEPLOYMENT_NAME);
 152.    
 153.                // now send a messag to the queue on which the MDB is listening
 154.                log.info("Sending a message to the queue on which the MDB " + " is listening");
 155.                triggerRequestResponseCycleOnQueue();
 156.    
 157.                assertTrue("@PostConstruct wasn't invoked on MDB", lifecycleTracker.wasPostConstructInvokedOn(this.getClass().getPackage().getName() + ".LifecycleCounterMDB"));
 158.    
 159.                // undeploy
 160.                log.info("About to undeploy MDB archive " + MDB_DEPLOYMENT_NAME);
 161.                deployer.undeploy(MDB_DEPLOYMENT_NAME);
 162.                // we have undeployed successfully, there's no need anymore to trigger an undeployment before returning from this method
 163.                requiresUndeploy = false;
 164.    
 165.                assertTrue("@PreDestroy wasn't invoked on MDB", lifecycleTracker.wasPreDestroyInvokedOn(this.getClass().getPackage().getName() + ".LifecycleCounterMDB"));
 166.            } finally {
 167.                if (requiresUndeploy) {
 168.                    try {
 169.                        deployer.undeploy(MDB_DEPLOYMENT_NAME);
 170.                    } catch (Throwable t) {
 171.                        // log and return since we don't want to corrupt any original exceptions that might have caused the test to fail
 172.                        log.info("Ignoring the undeployment failure of " + MDB_DEPLOYMENT_NAME, t);
 173.                    }
 174.                }
 175.            }
 176.        }

 215.        private void triggerRequestResponseCycleOnQueue() throws Exception {
 216.            final InitialContext ctx = new InitialContext();
 217.            final QueueConnectionFactory factory = (QueueConnectionFactory) ctx.lookup("java:/JmsXA");
 218.            final QueueConnection connection = factory.createQueueConnection();
 219.            try {
 220.                connection.start();
 221.                final QueueSession session = connection.createQueueSession(false, Session.AUTO_ACKNOWLEDGE);
 222.                final Queue replyDestination = session.createTemporaryQueue();
 223.                final String requestMessage = "test";
 224.                final Message message = session.createTextMessage(requestMessage);
 225.                message.setJMSReplyTo(replyDestination);
 226.                final Destination destination = (Destination) ctx.lookup(Constants.QUEUE_JNDI_NAME);
 227.                final MessageProducer producer = session.createProducer(destination);
 228.                producer.send(message);
 229.                producer.close();
 230.    
 231.                // wait for a reply
 232.                final QueueReceiver receiver = session.createReceiver(replyDestination);
 233.                final Message reply = receiver.receive(TimeoutUtil.adjust(1000));
 234.                assertNotNull("Did not receive a reply on the reply queue. Perhaps the original (request) message didn't make it to the MDB?", reply);
 235.                final String result = ((TextMessage) reply).getText();
 236.                assertEquals("Unexpected reply messsage", Constants.REPLY_MESSAGE_PREFIX + requestMessage, result);
 237.            } finally {
 238.                if (connection != null) {
 239.                    // just closing the connection will close the session and other related resources (@see javax.jms.Connection)
 240.                    safeClose(connection);
 241.                }
 242.            }
 243.    
 244.        }

 246.        /**
 247.         * Responsible for creating and removing the queue required by this testcase
 248.         */
 249.        static class CreateQueueForPooledEJBLifecycleTestCase implements ServerSetupTask {
 250.    
 251.            private static final String QUEUE_NAME = "Queue-for-" + PooledEJBLifecycleTestCase.class.getName();
 252.    
 253.            @Override
 254.            public void setup(ManagementClient managementClient, String containerId) throws Exception {
 255.                // create the JMS queue
 256.                final JMSOperations jmsOperations = JMSOperationsProvider.getInstance(managementClient);
 257.                jmsOperations.createJmsQueue(QUEUE_NAME, Constants.QUEUE_JNDI_NAME);
 258.            }
 259.    
 260.            @Override
 261.            public void tearDown(ManagementClient managementClient, String containerId) throws Exception {
 262.                // destroy the JMS queue
 263.                final JMSOperations jmsOperations = JMSOperationsProvider.getInstance(managementClient);
 264.                jmsOperations.removeJmsQueue(QUEUE_NAME);
 265.            }
 266.        }

 268.        private static void safeClose(final Connection connection) {
 269.            if (connection == null) {
 270.                return;
 271.            }
 272.            try {
 273.                connection.close();
 274.            } catch (Throwable t) {
 275.                // just log
 276.                log.info("Ignoring a problem which occurred while closing: " + connection, t);
 277.            }
 278.        }
