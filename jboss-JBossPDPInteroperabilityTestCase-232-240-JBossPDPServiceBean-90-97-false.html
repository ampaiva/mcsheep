  77.        private JBossPDP createPDPForInterop() {
  78.            final String policiesPath = getClass().getPackage().getName().replace('.', '/') + "/"
  79.                    + XACMLTestUtils.TESTOBJECTS_POLICIES;

  91.            pdp.setPolicies(policies);
