 109.         * as state by SFSB Foo are expected to correcty end up on the remote server Y and return the correct state information

 115.        public void testSFSBPassivationWithScopedEJBProxyMemberInstances() throws Exception {
 116.            final StatefulBeanA sfsbOnLocalServer = InitialContext.doLookup("java:module/" + StatefulBeanA.class.getSimpleName() + "!" + StatefulBeanA.class.getName());
 117.            final int initialCount = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 118.            Assert.assertEquals("Unexpected initial count from stateful bean", 0, initialCount);
 119.            // just increment a few times
 120.            final int NUM_TIMES_BEFORE_PASSIVATION = 5;
 121.            for (int i = 0; i < NUM_TIMES_BEFORE_PASSIVATION; i++) {
 122.                sfsbOnLocalServer.incrementCountByInvokingOnRemoteServerBean();
 123.            }
 124.            final int countAfterIncrement = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 125.            Assert.assertEquals("Unexpected count after increment, from stateful bean", NUM_TIMES_BEFORE_PASSIVATION, countAfterIncrement);
 126.    
 127.            // let the SFSB invoke an SLSB on a remote server
 128.            final String message = "foo";
 129.            final String firstEcho = sfsbOnLocalServer.getEchoByInvokingOnRemoteServerBean(message);
 130.            Assert.assertEquals("Unexpected echo from remote server SLSB", message, firstEcho);
 131.    
 132.            // now let's wait for passivation of the SFSB on local server
 133.            final CountDownLatch passivationLatch = new CountDownLatch(1);
 134.            sfsbOnLocalServer.registerPassivationNotificationLatch(passivationLatch);
 135.    
 136.            logger.info("Triggering passivation of " + StatefulBeanA.class.getSimpleName() + " bean");
 137.            InitialContext.doLookup("java:module/" + StatefulBeanA.class.getSimpleName() + "!" + StatefulBeanA.class.getName());
 138.    
 139.            final boolean passivated = passivationLatch.await(2, TimeUnit.SECONDS);
 140.            if (passivated) {
 141.                logger.info("pre-passivate invoked on " + StatefulBeanA.class.getSimpleName() + " bean");
 142.            } else {
 143.                Assert.fail(sfsbOnLocalServer + " was not passivated");
 144.            }
 145.            // just wait a little while longer since the acknowledgement that the pre-passivate was invoked
 146.            // doesn't mean the passivation process is complete
 147.            Thread.sleep(1000);
 148.    
 149.            // let's activate the passivated SFSB on local server
 150.            final int countAfterActivate = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 151.            Assert.assertEquals("Unexpected count from stateful bean after it was activated", NUM_TIMES_BEFORE_PASSIVATION, countAfterActivate);
 152.            // just make sure @PostActivate was invoked
 153.            Assert.assertTrue("Post-activate method was not invoked on bean " + StatefulBeanA.class.getSimpleName(), sfsbOnLocalServer.wasPostActivateInvoked());
 154.            // now increment on the remote server SFSB via the local server SFSB
 155.            final int NUM_TIMES_AFTER_ACTIVATION = 2;
 156.            for (int i = 0; i < NUM_TIMES_AFTER_ACTIVATION; i++) {
 157.                sfsbOnLocalServer.incrementCountByInvokingOnRemoteServerBean();
 158.            }
 159.            final int counterAfterIncrementOnPostActivate = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 160.            Assert.assertEquals("Unexpected count after increment, on the post activated stateful bean", NUM_TIMES_BEFORE_PASSIVATION + NUM_TIMES_AFTER_ACTIVATION, counterAfterIncrementOnPostActivate);
 161.    
 162.            // let's also invoke on the remote server SLSB via the local server SFSB
 163.            final String echoAfterPostActivate = sfsbOnLocalServer.getEchoByInvokingOnRemoteServerBean(message);
 164.            Assert.assertEquals("Unexpected echo message from remote server SLSB", message, echoAfterPostActivate);
 165.    
 166.        }

  44.     * allows applications to pass JNDI context properties during JNDI context creation for (scoped) EJB client

  64.            jar.addClasses(StatefulRemoteHomeForBeanOnOtherServer.class);
  65.            jar.addAsManifestResource(DynamicJNDIContextEJBInvocationTestCase.class.getPackage(), "MANIFEST.MF", "MANIFEST.MF");

  79.        /**
  80.         * Tests that a SFSB hosted on server X can lookup and invoke a SFSB and SLSB hosted on a different server,
  81.         * by using a JNDI context which was created by passing the EJB client context creation properties
  82.         *
  83.         * @throws Exception
  84.         */
  85.        @Test
  86.        @OperateOnDeployment("local-server-deployment")
  87.        public void testServerToServerSFSBInvocation() throws Exception {
  88.            final StatefulBeanA sfsbOnLocalServer = InitialContext.doLookup("java:module/" + StatefulBeanA.class.getSimpleName() + "!" + StatefulBeanA.class.getName());
  89.            final int initialCount = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
  90.            Assert.assertEquals("Unexpected initial count from stateful bean", 0, initialCount);
  91.            // just increment a few times
  92.            final int NUM_TIMES = 5;
  93.            for (int i = 0; i < NUM_TIMES; i++) {
  94.                sfsbOnLocalServer.incrementCountByInvokingOnRemoteServerBean();
  95.            }
  96.            final int countAfterIncrement = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
  97.            Assert.assertEquals("Unexpected count after increment, from stateful bean", NUM_TIMES, countAfterIncrement);
  98.    
  99.            // let the SFSB invoke an SLSB on a remote server
 100.            final String message = "foo";
 101.            final String firstEcho = sfsbOnLocalServer.getEchoByInvokingOnRemoteServerBean(message);
 102.            Assert.assertEquals("Unexpected echo from remote server SLSB", message, firstEcho);
 103.        }

 105.        /**
 106.         * Tests that a SFSB Foo hosted on server X can lookup and store a SFSB and a SLSB hosted on a different server Y,
 107.         * by using a JNDI context which was created by passing the EJB client context creation properties. The SFSB Foo
 108.         * on server X is then allowed to passivate and after activation the invocations on the SFSB and SLSB members held
 109.         * as state by SFSB Foo are expected to correcty end up on the remote server Y and return the correct state information
 110.         *
 111.         * @throws Exception
 112.         */
 113.        @Test
 114.        @OperateOnDeployment("local-server-deployment")
 115.        public void testSFSBPassivationWithScopedEJBProxyMemberInstances() throws Exception {
 116.            final StatefulBeanA sfsbOnLocalServer = InitialContext.doLookup("java:module/" + StatefulBeanA.class.getSimpleName() + "!" + StatefulBeanA.class.getName());
 117.            final int initialCount = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 118.            Assert.assertEquals("Unexpected initial count from stateful bean", 0, initialCount);
 119.            // just increment a few times
 120.            final int NUM_TIMES_BEFORE_PASSIVATION = 5;
 121.            for (int i = 0; i < NUM_TIMES_BEFORE_PASSIVATION; i++) {
 122.                sfsbOnLocalServer.incrementCountByInvokingOnRemoteServerBean();
 123.            }
 124.            final int countAfterIncrement = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 125.            Assert.assertEquals("Unexpected count after increment, from stateful bean", NUM_TIMES_BEFORE_PASSIVATION, countAfterIncrement);
 126.    
 127.            // let the SFSB invoke an SLSB on a remote server
 128.            final String message = "foo";
 129.            final String firstEcho = sfsbOnLocalServer.getEchoByInvokingOnRemoteServerBean(message);
 130.            Assert.assertEquals("Unexpected echo from remote server SLSB", message, firstEcho);
 131.    
 132.            // now let's wait for passivation of the SFSB on local server
 133.            final CountDownLatch passivationLatch = new CountDownLatch(1);
 134.            sfsbOnLocalServer.registerPassivationNotificationLatch(passivationLatch);
 135.    
 136.            logger.info("Triggering passivation of " + StatefulBeanA.class.getSimpleName() + " bean");
 137.            InitialContext.doLookup("java:module/" + StatefulBeanA.class.getSimpleName() + "!" + StatefulBeanA.class.getName());
 138.    
 139.            final boolean passivated = passivationLatch.await(2, TimeUnit.SECONDS);
 140.            if (passivated) {
 141.                logger.info("pre-passivate invoked on " + StatefulBeanA.class.getSimpleName() + " bean");
 142.            } else {
 143.                Assert.fail(sfsbOnLocalServer + " was not passivated");
 144.            }
 145.            // just wait a little while longer since the acknowledgement that the pre-passivate was invoked
 146.            // doesn't mean the passivation process is complete
 147.            Thread.sleep(1000);
 148.    
 149.            // let's activate the passivated SFSB on local server
 150.            final int countAfterActivate = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 151.            Assert.assertEquals("Unexpected count from stateful bean after it was activated", NUM_TIMES_BEFORE_PASSIVATION, countAfterActivate);
 152.            // just make sure @PostActivate was invoked
 153.            Assert.assertTrue("Post-activate method was not invoked on bean " + StatefulBeanA.class.getSimpleName(), sfsbOnLocalServer.wasPostActivateInvoked());
 154.            // now increment on the remote server SFSB via the local server SFSB
 155.            final int NUM_TIMES_AFTER_ACTIVATION = 2;
 156.            for (int i = 0; i < NUM_TIMES_AFTER_ACTIVATION; i++) {
 157.                sfsbOnLocalServer.incrementCountByInvokingOnRemoteServerBean();
 158.            }
 159.            final int counterAfterIncrementOnPostActivate = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 160.            Assert.assertEquals("Unexpected count after increment, on the post activated stateful bean", NUM_TIMES_BEFORE_PASSIVATION + NUM_TIMES_AFTER_ACTIVATION, counterAfterIncrementOnPostActivate);
 161.    
 162.            // let's also invoke on the remote server SLSB via the local server SFSB
 163.            final String echoAfterPostActivate = sfsbOnLocalServer.getEchoByInvokingOnRemoteServerBean(message);
 164.            Assert.assertEquals("Unexpected echo message from remote server SLSB", message, echoAfterPostActivate);
 165.    
 166.        }

 168.        /**
 169.         * Tests that if a bean exposes a remote view with a method returning a remote business interface of either
 170.         * the same bean or a different bean, then that returned business object doesn't hold on to the EJB client
 171.         * context identifier (if any) which was used to create the proxy.
 172.         *
 173.         * @throws Exception
 174.         */
 175.        @Test
 176.        @OperateOnDeployment("local-server-deployment")
 177.        public void testEJBClientContextIdentiferIsNotSerialized() throws Exception {
 178.            final LocalServerStatefulRemote sfsbOnLocalServer = InitialContext.doLookup("java:module/" + StatefulBeanA.class.getSimpleName() + "!" + LocalServerStatefulRemote.class.getName());
 179.            final int initialCount = sfsbOnLocalServer.getCountByInvokingOnRemoteServerBean();
 180.            Assert.assertEquals("Unexpected initial count from stateful bean", 0, initialCount);
 181.            // let the SFSB invoke an SLSB on a remote server
 182.            final String message = "foo";
 183.            final String firstEcho = sfsbOnLocalServer.getEchoByInvokingOnRemoteServerBean(message);
 184.            Assert.assertEquals("Unexpected echo from remote server SLSB", message, firstEcho);
 185.    
 186.            // now let's fetch the SFSB proxy which was created using a scoped EJB client context
 187.            final StatefulRemoteOnOtherServer sfsbOnRemoteServer = sfsbOnLocalServer.getSFSBCreatedWithScopedEJBClientContext();
 188.            Assert.assertNotNull("Stateful bean proxy returned by local server SFSB was null", sfsbOnRemoteServer);
 189.            // try invoking on that returned proxy. It should fail since it MUST NOT know the EJB client context
 190.            // identifier with which it was created and should work as if it's using the "current" EJB client context
 191.            // which in this case has no knowledge of the remote server (== remote EJB receiver of the other server)
 192.            try {
 193.                sfsbOnRemoteServer.getCount();
 194.                Assert.fail("Invocation on SFSB proxy returned by the local server SFSB was expected to fail");
 195.            } catch (IllegalStateException ise) {
 196.                // expected
 197.                logger.debug("Invocation on returned SFSB failed with expected exception", ise);
 198.            }
 199.    
 200.            // now let's fetch the SLSB proxy which was created using a scoped EJB client context
 201.            final StatelessRemoteOnOtherServer slsbOnRemoteServer = sfsbOnLocalServer.getSLSBCreatedWithScopedEJBClientContext();
 202.            Assert.assertNotNull("Stateless bean proxy returned by local server SFSB was null", slsbOnRemoteServer);
 203.            // try invoking on that returned proxy. It should fail since it MUST NOT know the EJB client context
 204.            // identifier with which it was created and should work as if it's using the "current" EJB client context
 205.            // which in this case has no knowledge of the remote server (== remote EJB receiver of the other server)
 206.            try {
 207.                slsbOnRemoteServer.echo("foo-bar");
 208.                Assert.fail("Invocation on SLSB proxy returned by the local server SFSB was expected to fail");
 209.            } catch (IllegalStateException ise) {
 210.                // expected
 211.                logger.debug("Invocation on returned SLSB failed with expected exception", ise);
 212.            }
 213.    
 214.        }
