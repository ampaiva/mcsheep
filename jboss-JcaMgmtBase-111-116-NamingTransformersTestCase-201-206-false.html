 330.        private static ModelFixer MODEL_FIXER_1_1_0 = new ModelFixer() {
 331.            //The legacy subsystem does not seem to set expressions correctly so fix them here to match the transformed resource
 332.            @Override
 333.            public ModelNode fixModel(ModelNode modelNode) {
 334.                for (Property property : modelNode.get("binding").asPropertyList()) {
 335.                    ModelNode entry = property.getValue();
 336.                    for (Property nestedProp : entry.asPropertyList()) {
 337.                        ModelNode value = nestedProp.getValue();
 338.                        if (value.getType() == ModelType.STRING && value.asString().startsWith("$") && value.asString().endsWith("}")) {
 339.                            ModelNode fixed = new ModelNode();
 340.                            fixed.set(new ValueExpression(value.asString()));
 341.                            modelNode.get("binding", property.getName(), nestedProp.getName()).set(fixed);
 342.                        }
 343.                    }
 344.                }
 345.                return modelNode;
 346.            }
