 178.        /**
 179.         * If a Response contains results that contain paging information, returns a new
 180.         * Request that will retrieve the next page of results, in whichever direction
 181.         * is desired. If no paging information is available, returns null.
 182.         *
 183.         * @param direction enum indicating whether to page forward or backward
 184.         * @return a Request that will retrieve the next page of results in the desired
 185.         *         direction, or null if no paging information is available
 186.         */
 187.        public GraphRequest getRequestForPagedResults(PagingDirection direction) {
 188.            String link = null;
 189.            if (graphObject != null) {
 190.                JSONObject pagingInfo = graphObject.optJSONObject("paging");
 191.                if (pagingInfo != null) {
 192.                    if (direction == PagingDirection.NEXT) {
 193.                        link = pagingInfo.optString("next");
 194.                    } else {
 195.                        link = pagingInfo.optString("previous");
 196.                    }
 197.                }
 198.            }
 199.            if (Utility.isNullOrEmpty(link)) {
 200.                return null;
 201.            }
 202.    
 203.            if (link != null && link.equals(request.getUrlForSingleRequest())) {
 204.                // We got the same "next" link as we just tried to retrieve. This could happen if cached
 205.                // data is invalid. All we can do in this case is pretend we have finished.
 206.                return null;
 207.            }
 208.    
 209.            GraphRequest pagingRequest;
 210.            try {
 211.                pagingRequest = new GraphRequest(request.getAccessToken(), new URL(link));
 212.            } catch (MalformedURLException e) {
 213.                return null;
 214.            }
 215.    
 216.            return pagingRequest;
 217.        }

 330.        private static List createResponsesFromObject(
 331.                HttpURLConnection connection,
 332.                List requests,
 333.                Object object
 334.        ) throws FacebookException, JSONException {
 335.            int numRequests = requests.size();
 336.            List responses = new ArrayList(numRequests);
 337.            Object originalResult = object;
 338.    
 339.            if (numRequests == 1) {
 340.                GraphRequest request = requests.get(0);
 341.                try {
 342.                    // Single request case -- the entire response is the result, wrap it as "body" so we
 343.                    // can handle it the same as we do in the batched case. We get the response code
 344.                    // from the actual HTTP response, as opposed to the batched case where it is
 345.                    // returned as a "code" element.
 346.                    JSONObject jsonObject = new JSONObject();
 347.                    jsonObject.put(BODY_KEY, object);
 348.                    int responseCode = (connection != null) ? connection.getResponseCode() : 200;
 349.                    jsonObject.put(CODE_KEY, responseCode);
 350.    
 351.                    JSONArray jsonArray = new JSONArray();
 352.                    jsonArray.put(jsonObject);
 353.    
 354.                    // Pretend we got an array of 1 back.
 355.                    object = jsonArray;
 356.                } catch (JSONException e) {
 357.                    responses.add(
 358.                            new GraphResponse(
 359.                                    request,
 360.                                    connection,
 361.                                    new FacebookRequestError(connection, e)));
 362.                } catch (IOException e) {
 363.                    responses.add(
 364.                            new GraphResponse(
 365.                                    request,
 366.                                    connection,
 367.                                    new FacebookRequestError(connection, e)));
 368.                }
 369.            }
 370.    
 371.            if (!(object instanceof JSONArray) || ((JSONArray) object).length() != numRequests) {
 372.                FacebookException exception = new FacebookException("Unexpected number of results");
 373.                throw exception;
 374.            }
 375.    
 376.            JSONArray jsonArray = (JSONArray) object;
 377.    
 378.            for (int i = 0; i < jsonArray.length(); ++i) {
 379.                GraphRequest request = requests.get(i);
 380.                try {
 381.                    Object obj = jsonArray.get(i);
 382.                    responses.add(
 383.                            createResponseFromObject(
 384.                                    request,
 385.                                    connection,
 386.                                    obj,
 387.                                    originalResult));
 388.                } catch (JSONException e) {
 389.                    responses.add(
 390.                            new GraphResponse(
 391.                                    request,
 392.                                    connection,
 393.                                    new FacebookRequestError(connection, e)));
 394.                } catch (FacebookException e) {
 395.                    responses.add(
 396.                            new GraphResponse(
 397.                                    request,
 398.                                    connection,
 399.                                    new FacebookRequestError(connection, e)));
 400.                }
 401.            }
 402.    
 403.            return responses;
 404.        }

 406.        private static GraphResponse createResponseFromObject(
 407.                GraphRequest request,
 408.                HttpURLConnection connection,
 409.                Object object,
 410.                Object originalResult
 411.        ) throws JSONException {
 412.            if (object instanceof JSONObject) {
 413.                JSONObject jsonObject = (JSONObject) object;
 414.    
 415.                FacebookRequestError error =
 416.                        FacebookRequestError.checkResponseAndCreateError(
 417.                                jsonObject,
 418.                                originalResult,
 419.                                connection);
 420.                if (error != null) {
 421.                    if (error.getErrorCode() == FacebookRequestErrorClassification.EC_INVALID_TOKEN
 422.                            && Utility.isCurrentAccessToken(request.getAccessToken())) {
 423.                        AccessToken.setCurrentAccessToken(null);
 424.                    }
 425.                    return new GraphResponse(request, connection, error);
 426.                }
 427.    
 428.                Object body = Utility.getStringPropertyAsJSON(
 429.                        jsonObject,
 430.                        BODY_KEY,
 431.                        NON_JSON_RESPONSE_PROPERTY);
 432.    
 433.                if (body instanceof JSONObject) {
 434.                    return new GraphResponse(request, connection, body.toString(), (JSONObject)body);
 435.                } else if (body instanceof JSONArray) {
 436.                    return new GraphResponse(request, connection, body.toString(), (JSONArray)body);
 437.                }
 438.                // We didn't get a body we understand how to handle, so pretend we got nothing.
 439.                object = JSONObject.NULL;
 440.            }
 441.    
 442.            if (object == JSONObject.NULL) {
 443.                return new GraphResponse(request, connection, object.toString(), (JSONObject)null);
 444.            } else {
 445.                throw new FacebookException("Got unexpected object type in response, class: "
 446.                        + object.getClass().getSimpleName());
 447.            }
 448.        }
