 321.        private abstract class AbstractTimerHandler implements OperationStepHandler {
 322.    
 323.            public void execute(OperationContext context, ModelNode operation) throws OperationFailedException {
 324.    
 325.                if (context.isNormalServer()) {
 326.                    context.addStep(new OperationStepHandler() {
 327.                        @Override
 328.                        public void execute(OperationContext context, ModelNode operation) throws OperationFailedException {
 329.                            executeRuntime(context, operation);
 330.                        }
 331.                    }, OperationContext.Stage.RUNTIME);
 332.                }
 333.                context.stepCompleted();
 334.            }
 335.            protected TimerImpl getTimer(final OperationContext context, final ModelNode operation) throws OperationFailedException {
 336.                final T ejbcomponent = parentHandler.getComponent(context, operation);
 337.                final TimerServiceImpl timerService = (TimerServiceImpl) ejbcomponent.getTimerService();
 338.    
 339.                final PathAddress address = PathAddress.pathAddress(operation.require(ModelDescriptionConstants.OP_ADDR));
 340.                final String timerId = address.getLastElement().getValue();
 341.                final String timedInvokerObjectId = timerService.getTimedObjectInvoker().getValue().getTimedObjectId();
 342.                final TimerHandle handle = new TimerHandleImpl(timerId, timedInvokerObjectId, timerService);
 343.                try {
 344.                    return timerService.getTimer(handle);
 345.                } catch (Exception e) {
 346.                    throw new OperationFailedException(e, null);
 347.                }
 348.            }
 349.    
 350.            abstract void executeRuntime(OperationContext context, ModelNode operation) throws OperationFailedException;
 351.        }
 352.    
 353.        private abstract class AbstractReadAttributeHandler extends AbstractTimerHandler {
 354.    
 355.            void executeRuntime(final OperationContext context, final ModelNode operation) throws OperationFailedException {
 356.                final String opName = operation.require(ModelDescriptionConstants.OP).asString();
 357.                if (!opName.equals(ModelDescriptionConstants.READ_ATTRIBUTE_OPERATION)) {
 358.                    throw EjbLogger.ROOT_LOGGER.unknownOperations(opName);
 359.                }
 360.    
 361.                final TimerImpl timer = getTimer(context, operation);
 362.    
 363.                if(timer != null) {
 364.                    //the timer can expire at any point, so protect against an NPE
 365.                    readAttribute(timer, context.getResult());
 366.                }
 367.                context.completeStep(OperationContext.RollbackHandler.NOOP_ROLLBACK_HANDLER);
 368.            }
 369.    
 370.            protected abstract void readAttribute(final TimerImpl timer, final ModelNode toSet);
 371.        }
