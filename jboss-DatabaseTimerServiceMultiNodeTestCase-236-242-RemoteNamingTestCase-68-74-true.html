 188.        @Test
 189.        public void testEjbTimeoutOnOtherNode() throws Exception {
 190.    
 191.            Context clientContext = getRemoteContext(clientClient);
 192.            try {
 193.                RemoteTimedBean clientBean = (RemoteTimedBean) clientContext.lookup(ARCHIVE_NAME + "/" + TimedObjectTimerServiceBean.class.getSimpleName() + "!" + RemoteTimedBean.class.getName());
 194.                Set names = new HashSet<>();
 195.                long time = System.currentTimeMillis() + TIMER_DELAY;
 196.                for (int i = 0; i < TIMER_COUNT; ++i) {
 197.                    String name = "timer" + i;
 198.                    clientBean.scheduleTimer(time, name);
 199.                    names.add(name);
 200.                }
 201.    
 202.                final Context remoteContext = getRemoteContext(serverClient);
 203.                try {
 204.                    Collector serverBean = (Collector) remoteContext.lookup(ARCHIVE_NAME + "/" + CollectionSingleton.class.getSimpleName() + "!" + Collector.class.getName());
 205.                    List res = serverBean.collect(TIMER_COUNT);
 206.                    Assert.assertEquals("Expected " + TIMER_COUNT + " was " + res.size() + " " + res, TIMER_COUNT, res.size());
 207.                    boolean server = false;
 208.                    boolean client = false;
 209.                    final Set newNames = new HashSet<>(names);
 210.                    for (TimerData r : res) {
 211.                        if (!newNames.remove(r.getInfo())) {
 212.                            if (!names.contains(r.getInfo())) {
 213.                                throw new RuntimeException("Timer " + r.getInfo() + " not run " + res);
 214.                            } else {
 215.                                throw new RuntimeException("Timer " + r.getInfo() + " run twice " + res);
 216.                            }
 217.                        }
 218.                        if (r.getNode().equals("client")) {
 219.                            client = true;
 220.                        } else if (r.getNode().equals("server")) {
 221.                            server = true;
 222.                        }
 223.                    }
 224.                    Assert.assertTrue(client);
 225.                    Assert.assertTrue(server);
 226.                } finally {
 227.                    remoteContext.close();
 228.                }
 229.            } finally {
 230.                clientContext.close();
 231.            }
 232.        }
