 186.        protected  T lookup(String beanName, Class interfaceType) throws NamingException {
 187.            try {
 188.                return interfaceType.cast(iniCtx.lookup("java:global/" + ARCHIVE_NAME + "/" + "beans/" + beanName + "!" + interfaceType.getName()));
 189.            } catch (NamingException e) {
 190.                dumpJndi("");
 191.                throw e;
 192.            }
 193.        }

 195.        // TODO: move this logic to a common base class (might be helpful for writing new tests)
 196.        private void dumpJndi(String s) {
 197.            try {
 198.                dumpTreeEntry(iniCtx.list(s), s);
 199.            } catch (NamingException ignore) {
 200.            }
 201.        }
 202.    
 203.        private void dumpTreeEntry(NamingEnumeration list, String s) throws NamingException {
 204.            System.out.println("\ndump " + s);
 205.            while (list.hasMore()) {
 206.                NameClassPair ncp = list.next();
 207.                System.out.println(ncp.toString());
 208.                if (s.length() == 0) {
 209.                    dumpJndi(ncp.getName());
 210.                } else {
 211.                    dumpJndi(s + "/" + ncp.getName());
 212.                }
 213.            }
