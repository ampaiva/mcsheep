 431.        private void addHttpsConnector(String verifyClient, String httpsName, int httpsPort, ModelControllerClient client)
 432.                throws Exception {
 433.            final ModelNode compositeOp = Util.createOperation(COMPOSITE, PathAddress.EMPTY_ADDRESS);
 434.            final ModelNode steps = compositeOp.get(STEPS);
 435.

 437.            ModelNode op = Util.createAddOperation(PathAddress.pathAddress(SOCKET_BINDING_GROUP, STANDARD_SOCKETS).append(
 438.                    SOCKET_BINDING, httpsName));
 439.            op.get(PORT).set(httpsPort);
 440.            steps.add(op);
 441.    
 442.            ModelNode operation = createOpNode("subsystem=undertow/server=default-server/https-listener=" + httpsName,
 443.                    ModelDescriptionConstants.ADD);
 444.            operation.get("socket-binding").set(httpsName);
 445.            operation.get("security-realm").set(HTTPS_REALM);
 446.            operation.get("verify-client").set(verifyClient);
 447.    
 448.            steps.add(operation);
 449.    
 450.            Utils.applyUpdate(compositeOp, client);
 451.        }
