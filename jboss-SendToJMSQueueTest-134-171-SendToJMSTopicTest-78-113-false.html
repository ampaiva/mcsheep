  90.                logger.info("Creating session..");
  91.                senderSession = senderConnection.createSession(false, Session.AUTO_ACKNOWLEDGE);
  92.                MessageProducer producer = senderSession.createProducer(topic);
  93.                TextMessage message = senderSession.createTextMessage("Hello world!");
  94.                logger.info("Sending..");
  95.                producer.send(message);
  96.                logger.info("Message sent");
  97.                senderConnection.start();
  98.            } catch (Exception ex) {
  99.                ex.printStackTrace();
 100.            } finally {
 101.                logger.info("Closing connections and sessions");
 102.                if (senderSession != null) {
 103.                    senderSession.close();
 104.                }
 105.                if (senderConnection != null) {
 106.                    senderConnection.close();
 107.                }
 108.            }
 109.    
 110.            Message receivedMessage = null;
 111.            try {
 112.                logger.info("Receiving");
 113.                receivedMessage = consumer.receive(5000);
 114.                logger.info("Received: " + ((TextMessage) receivedMessage).getText());
 115.            } catch (Exception e) {
 116.                e.printStackTrace();
 117.                Assert.fail(e.getMessage());
 118.            } finally {
 119.                if (receivedMessage == null) {
 120.                    Assert.fail("received null instead of a TextMessage");
 121.                }
 122.                if (consumerSession != null) {
 123.                    consumerSession.close();
 124.                }
 125.                if (consumerConnection != null) {
 126.                    consumerConnection.close();
 127.                }
 128.            }
 129.    
 130.            Assert.assertTrue("received a " + receivedMessage.getClass().getName() + " instead of a TextMessage", receivedMessage instanceof TextMessage);

 132.        }
