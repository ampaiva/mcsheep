  47.            ModelNode operation = new ModelNode();
  48.            operation.get(OP).set("reload");
  49.            executeOperation(operation);
  50.            boolean reloaded = false;
  51.            int i = 0;
  52.            while (!reloaded) {
  53.                try {
  54.                    Thread.sleep(5000);
  55.                    if (getManagementClient().isServerInRunningState())
  56.                        reloaded = true;
  57.                } catch (Throwable t) {
  58.                    // nothing to do, just waiting
  59.                } finally {
  60.                    if (!reloaded && i++ > 10)
  61.                        throw new Exception("Server reloading failed");
  62.                }
  63.            }
