 124.         * (formerly: "name: base" )
 125.         *
 126.         * @param cls the classifier role to apply any changes to
 127.         * @param s the String to parse
 128.         * @return the classifier role with the applied changes
 129.         * @throws ParseException when it detects an error in the attribute string. 
 130.         *                  See also ParseError.getErrorOffset().
 131.         */
 132.        protected Object parseClassifierRole(Object cls, String s)
 133.            throws ParseException {
 134.            
 135.            String name = null;
 136.            String token;
 137.            String role = null;
 138.            String base = null;
 139.            List bases = null;
 140.            boolean hasColon = false;
 141.            boolean hasSlash = false;
 142.    
 143.            try {
 144.                MyTokenizer st = new MyTokenizer(s, " ,\t,/,:,\\,");
 145.    
 146.                while (st.hasMoreTokens()) {
 147.                    token = st.nextToken();
 148.                    if (" ".equals(token) || "\t".equals(token)) {
 149.                        /* Do nothing. */
 150.                    } else if ("/".equals(token)) {
 151.                        hasSlash = true;
 152.                        hasColon = false;
 153.    
 154.                        if (base != null) {
 155.                            if (bases == null) {
 156.                                bases = new ArrayList();
 157.                            }
 158.                            bases.add(base);
 159.                        }
 160.                        base = null;
 161.                    } else if (":".equals(token)) {
 162.                        hasColon = true;
 163.                        hasSlash = false;
 164.    
 165.                        if (bases == null) {
 166.                            bases = new ArrayList();
 167.                        }
 168.                        if (base != null) {
 169.                            bases.add(base);
 170.                        }
 171.                        base = null;
 172.                    } else if (",".equals(token)) {
 173.                        if (base != null) {
 174.                            if (bases == null) {
 175.                                bases = new ArrayList();
 176.                            }
 177.                            bases.add(base);
 178.                        }
 179.                        base = null;
 180.                    } else if (hasColon) {
 181.                        if (base != null) {
 182.                            String msg = "parsing.error.classifier.extra-test";
 183.                            throw new ParseException(
 184.                                    Translator.localize(msg), 
 185.                                    st.getTokenIndex());
 186.                        }
 187.    
 188.                        base = token;
 189.                    } else if (hasSlash) {
 190.                        if (role != null) {
 191.                            String msg = "parsing.error.classifier.extra-test";
 192.                            throw new ParseException(
 193.                                    Translator.localize(msg), 
 194.                                    st.getTokenIndex());
 195.                        }
 196.    
 197.                        role = token;
 198.                    } else {
 199.                        if (name != null) {
 200.                            String msg = "parsing.error.classifier.extra-test";
 201.                            throw new ParseException(
 202.                                    Translator.localize(msg), 
 203.                                    st.getTokenIndex());
 204.                        }
 205.    
 206.                        name = token;
 207.                    }
 208.                }
 209.            } catch (NoSuchElementException nsee) {
 210.                String msg = "parsing.error.classifier.unexpected-end-attribute";
 211.                throw new ParseException(Translator.localize(msg), s.length());
 212.            }
 213.    
 214.            if (base != null) {
 215.                if (bases == null) {
 216.                    bases = new ArrayList();
 217.                }
 218.                bases.add(base);
 219.            }
 220.    
 221.            // TODO: What to do about object name???
 222.            //    if (name != null)
 223.            //      ;
 224.    
 225.            if (role != null) {
 226.                Model.getCoreHelper().setName(cls, role.trim());
 227.            }
 228.    
 229.            if (bases != null) {
 230.                // Remove bases that aren't there anymore
 231.    
 232.                // copy - can't iterate modify live collection while iterating it
 233.                Collection b = new ArrayList(Model.getFacade().getBases(cls));
 234.                Iterator it = b.iterator();
 235.                Object c;
 236.                Object ns = Model.getFacade().getNamespace(cls);
 237.                if (ns != null && Model.getFacade().getNamespace(ns) != null) {
 238.                    ns = Model.getFacade().getNamespace(ns);
 239.                } else {
 240.                    ns = Model.getFacade().getRoot(cls);
 241.                }
 242.    
 243.                while (it.hasNext()) {
 244.                    c = it.next();
 245.                    if (!bases.contains(Model.getFacade().getName(c))) {
 246.                        Model.getCollaborationsHelper().removeBase(cls, c);
 247.                    }
 248.                }
 249.    
 250.                it = bases.iterator();
 251.            addBases:
 252.                while (it.hasNext()) {
 253.                    String d = ((String) it.next()).trim();
 254.    
 255.                    Iterator it2 = b.iterator();
 256.                    while (it2.hasNext()) {
 257.                        c = it2.next();
 258.                        if (d.equals(Model.getFacade().getName(c))) {
 259.                            continue addBases;
 260.                        }
 261.                    }
 262.                    c = NotationUtilityUml.getType(d, ns);
 263.                    if (Model.getFacade().isACollaboration(
 264.                            Model.getFacade().getNamespace(c))) {
 265.                        Model.getCoreHelper().setNamespace(c, ns);
 266.                    }
 267.                    Model.getCollaborationsHelper().addBase(cls, c);
 268.                }
 269.            }
 270.            
 271.            return cls;
 272.        }
 273.    
 274.        private String toString(Object modelElement) {
 275.            String nameString = Model.getFacade().getName(modelElement);
 276.            if (nameString == null) { 
 277.                nameString = "";
 278.            }
 279.            nameString = nameString.trim();
 280.            // Loop through all base classes, building a comma separated list
 281.            StringBuilder baseString = NotationUtilityUml.formatNameList(
 282.                    Model.getFacade().getBases(modelElement));
 283.            baseString = new StringBuilder(baseString.toString().trim());       
 284.            // Build the final string
 285.            if (nameString.length() != 0) {
 286.                nameString = "/" + nameString;
 287.            }
 288.            if (baseString.length() != 0) {
 289.                baseString = baseString.insert(0, ":");
 290.            }
 291.            return nameString + baseString.toString();
 292.        }
 293.    
 294.        @Override
 295.        public String toString(Object modelElement, NotationSettings settings) {
 296.            return toString(modelElement);
 297.        }
