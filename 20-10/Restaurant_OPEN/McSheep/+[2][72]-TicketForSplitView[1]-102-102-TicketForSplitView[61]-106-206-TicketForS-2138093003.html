 563.        private void updateSelectionView() {
 564.            Object selectedObject = ticketViewerTable.getSelected();
 565.    
 566.            OrderView orderView = OrderView.getInstance();
 567.    
 568.            TicketItem selectedItem = null;
 569.            if (selectedObject instanceof TicketItem) {
 570.                selectedItem = (TicketItem) selectedObject;
 571.                MenuItemDAO dao = new MenuItemDAO();
 572.                MenuItem menuItem = dao.get(selectedItem.getItemId());
 573.                MenuGroup menuGroup = menuItem.getParent();
 574.                MenuItemView itemView = OrderView.getInstance().getItemView();
 575.                if (!menuGroup.equals(itemView.getMenuGroup())) {
 576.                    itemView.setMenuGroup(menuGroup);
 577.                }
 578.                orderView.showView(MenuItemView.VIEW_NAME);
 579.    
 580.                MenuCategory menuCategory = menuGroup.getParent();
 581.                orderView.getCategoryView().setSelectedCategory(menuCategory);
 582.                return;
 583.            }
 584.            else if (selectedObject instanceof TicketItemModifier) {
 585.                selectedItem = ((TicketItemModifier) selectedObject).getParent().getParent();
 586.            }
 587.            if (selectedItem == null)
 588.                return;
 589.    
 590.            ModifierView modifierView = orderView.getModifierView();
 591.    
 592.            if (selectedItem.isHasModifiers()) {
 593.                MenuItemDAO dao = new MenuItemDAO();
 594.                MenuItem menuItem = dao.get(selectedItem.getItemId());
 595.                if (!menuItem.equals(modifierView.getMenuItem())) {
 596.                    menuItem = dao.initialize(menuItem);
 597.                    modifierView.setMenuItem(menuItem, selectedItem);
 598.                }
 599.    
 600.                MenuCategory menuCategory = menuItem.getParent().getParent();
 601.                orderView.getCategoryView().setSelectedCategory(menuCategory);
 602.    
 603.                orderView.showView(ModifierView.VIEW_NAME);
 604.            }
 605.        }
