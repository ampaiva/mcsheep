 482.        public boolean isConnectionValid(Object connectionType, Object fromElement,
 483.                Object toElement, boolean checkWFR) {
 484.            // Get the list of valid model item pairs for the given connection type
 485.            List validItems = (ArrayList) validConnectionMap.get(connectionType);
 486.            if (validItems == null) {
 487.                return false;
 488.            }
 489.            // See if there's a pair in this list that match the given
 490.            // model elements
 491.            Iterator it = validItems.iterator();
 492.            while (it.hasNext()) {
 493.                Class[] modeElementPair = (Class[]) it.next();
 494.                if (modeElementPair[0].isInstance(fromElement)
 495.                    && modeElementPair[1].isInstance(toElement)) {
 496.                    if (checkWFR) {
 497.                        return isConnectionWellformed(

 512.        public boolean isContainmentValid(Object metaType, Object container) {
 513.            
 514.            // find the passed in container in validContainmentMap

 537.         * @param connectionType
 538.         * @param fromElement
 539.         * @param toElement
 540.         * @return true if the connection satisfies the wellformedness rules
 541.         */
 542.        private boolean isConnectionWellformed(
 543.                Class connectionType,
 544.                Element fromElement,
 545.                Element toElement) {
 546.            
 547.            if (fromElement == null || toElement == null) {
 548.                return false;

 550.                
 551.            if (connectionType == Generalization.class) {
 552.                /*
 553.                 * UML 1.4.2 Spec section 4.5.3.20 [5]
 554.                 * A GeneralizableElement may only be a child of
 555.                 * GeneralizableElement of the same kind.
 556.                 */
 557.                if (fromElement.getClass() != toElement.getClass()) {
 558.                    return false;
 559.                }
 560.            }
 561.            
 562.            return true;
 563.        }

 565.        private void buildValidConnectionMap() {
 566.            // A list of valid connections between elements, the
 567.            // connection type first and then the elements to be connected
 568.    
 569.            Class<Element> connection = null;
 570.            for (int i = 0; i < VALID_CONNECTIONS.length; ++i) {
 571.                connection = VALID_CONNECTIONS[i][0];
 572.                List validItems = (ArrayList) validConnectionMap.get(connection);
 573.                if (validItems == null) {
 574.                    validItems = new ArrayList();
 575.                    validConnectionMap.put(connection, validItems);
 576.                }
 577.                if (VALID_CONNECTIONS[i].length < 3) {
 578.                    // If there isn't a 3rd column then this represents a connection
 579.                    // of elements of the same type.
 580.                    Object[] modeElementPair = new Class[2];
 581.                    modeElementPair[0] = VALID_CONNECTIONS[i][1];
 582.                    modeElementPair[1] = VALID_CONNECTIONS[i][1];
 583.                    validItems.add(modeElementPair);
 584.                } else {
 585.                    // If there is a 3rd column then this represents a connection
 586.                    // of between 2 different types of element.
 587.                    Object[] modeElementPair = new Class[2];
 588.                    modeElementPair[0] = VALID_CONNECTIONS[i][1];
 589.                    modeElementPair[1] = VALID_CONNECTIONS[i][2];
 590.                    validItems.add(modeElementPair);
 591.                    // If the array hasn't been flagged to indicate otherwise
 592.                    // swap elements the elemnts and add again.
 593.                    if (VALID_CONNECTIONS[i].length < 4
 594.                            && VALID_CONNECTIONS[i][1] != VALID_CONNECTIONS[i][2]) {
 595.                        Object[] reversedModeElementPair = new Class[2];
 596.                        reversedModeElementPair[0] = VALID_CONNECTIONS[i][2];
 597.                        reversedModeElementPair[1] = VALID_CONNECTIONS[i][1];
 598.                        validItems.add(reversedModeElementPair);
 599.                    }
 600.                }
 601.            }

 298.        private void buildValidConnectionMap() {
 299.            // A list of valid connections between elements, the
 300.            // connection type first and then the elements to be connected
 301.    
 302.            for (int i = 0; i < VALID_CONNECTIONS.length; ++i) {
 303.                final Class<?> connection = VALID_CONNECTIONS[i][0];
 304.                List<Class<?>[]> validItems = validConnectionMap.get(connection);
 305.                if (validItems == null) {
 306.                    validItems = new ArrayList<Class<?>[]>();
 307.                    validConnectionMap.put(connection, validItems);
 308.                }
 309.                if (VALID_CONNECTIONS[i].length < 3) {
 310.                    // If there isn't a 3rd column then this represents a connection
 311.                    // of elements of the same type.
 312.                    Class<?>[] modeElementPair = new Class[2];
 313.                    modeElementPair[0] = VALID_CONNECTIONS[i][1];
 314.                    modeElementPair[1] = VALID_CONNECTIONS[i][1];
 315.                    validItems.add(modeElementPair);
 316.                } else {
 317.                    // If there is a 3rd column then this represents a connection
 318.                    // of between 2 different types of element.
 319.                    Class<?>[] modeElementPair = new Class[2];
 320.                    modeElementPair[0] = VALID_CONNECTIONS[i][1];
 321.                    modeElementPair[1] = VALID_CONNECTIONS[i][2];
 322.                    validItems.add(modeElementPair);
 323.                    // If the array hasn't been flagged to indicate otherwise
 324.                    // swap elements the elements and add again.
 325.                    if (VALID_CONNECTIONS[i].length < 4) {
 326.                        Class<?>[] reversedModeElementPair = new Class[2];
 327.                        reversedModeElementPair[0] = VALID_CONNECTIONS[i][2];
 328.                        reversedModeElementPair[1] = VALID_CONNECTIONS[i][1];
 329.                        validItems.add(reversedModeElementPair);
 330.                    }
 331.                }
 332.            }
 333.        }
