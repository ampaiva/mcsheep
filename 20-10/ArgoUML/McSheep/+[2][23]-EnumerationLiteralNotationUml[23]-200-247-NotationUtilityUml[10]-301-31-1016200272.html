 279.        /**
 280.         * Parse a string on the format:
 281.         * <pre>
 282.         *     [ << stereotype >>] [+|-|#|~] [full_pathname ::] [name]
 283.         * </pre>
 284.         * 
 285.         * @param me   The ModelElement <em>text</em> describes.
 286.         * @param text A String on the above format.
 287.         * @throws ParseException
 288.         *             when it detects an error in the attribute string. See also
 289.         *             ParseError.getErrorOffset().
 290.         */
 291.        protected static void parseModelElement(Object me, String text)
 292.            throws ParseException {
 293.            MyTokenizer st;
 294.    
 295.            List<String> path = null;
 296.            String name = null;
 297.            StringBuilder stereotype = null;
 298.            String token;
 299.    
 300.            try {
 301.                st = new MyTokenizer(text, "<<,\u00AB,\u00BB,>>,::");
 302.                while (st.hasMoreTokens()) {
 303.                    token = st.nextToken();
 304.    
 305.                    if ("<<".equals(token) || "\u00AB".equals(token)) {
 306.                        if (stereotype != null) {
 307.                            String msg = 
 308.                                "parsing.error.model-element-name.twin-stereotypes";
 309.                            throw new ParseException(Translator.localize(msg),
 310.                                    st.getTokenIndex());
 311.                        }
 312.    
 313.                        stereotype = new StringBuilder();
 314.                        while (true) {
 315.                            token = st.nextToken();
 316.                            if (">>".equals(token) || "\u00BB".equals(token)) {
 317.                                break;
 318.                            }
 319.                            stereotype.append(token);
 320.                        }
 321.                    } else if ("::".equals(token)) {
 322.                        if (name != null) {
 323.                            name = name.trim();
 324.                        }
 325.    
 326.                        if (path != null && (name == null || "".equals(name))) {
 327.                            String msg = 
 328.                                "parsing.error.model-element-name.anon-qualifiers";
 329.                            throw new ParseException(Translator.localize(msg), 
 330.                                    st.getTokenIndex());
 331.                        }
 332.    
 333.                        if (path == null) {
 334.                            path = new ArrayList<String>();
 335.                        }
 336.                        if (name != null) {
 337.                            path.add(name);
 338.                        }
 339.                        name = null;
 340.                    } else {
 341.                        if (name != null) {
 342.                            String msg = 
 343.                                "parsing.error.model-element-name.twin-names";
 344.                            throw new ParseException(Translator.localize(msg), 
 345.                                    st.getTokenIndex());
 346.                        }
 347.    
 348.                        name = token;
 349.                    }
 350.                }
 351.            } catch (NoSuchElementException nsee) {
 352.                String msg = 
 353.                    "parsing.error.model-element-name.unexpected-name-element";
 354.                throw new ParseException(Translator.localize(msg),
 355.                        text.length());
 356.            } catch (ParseException pre) {
 357.                throw pre;
 358.            }
 359.    
 360.            if (name != null) {
 361.                name = name.trim();
 362.            }
 363.    
 364.            if (path != null && (name == null || "".equals(name))) {
 365.                String msg = "parsing.error.model-element-name.must-end-with-name";
 366.                throw new ParseException(Translator.localize(msg), 0);
 367.            }
 368.    
 369.            if (name != null && name.startsWith("+")) {
 370.                name = name.substring(1).trim();
 371.                Model.getCoreHelper().setVisibility(me,
 372.                                Model.getVisibilityKind().getPublic());
 373.            }
 374.            if (name != null && name.startsWith("-")) {
 375.                name = name.substring(1).trim();
 376.                Model.getCoreHelper().setVisibility(me,
 377.                                Model.getVisibilityKind().getPrivate());
 378.            }
 379.            if (name != null && name.startsWith("#")) {
 380.                name = name.substring(1).trim();
 381.                Model.getCoreHelper().setVisibility(me,
 382.                                Model.getVisibilityKind().getProtected());
 383.            }
 384.            if (name != null && name.startsWith("~")) {
 385.                name = name.substring(1).trim();
 386.                Model.getCoreHelper().setVisibility(me,
 387.                                Model.getVisibilityKind().getPackage());
 388.            }
 389.            if (name != null) {
 390.                Model.getCoreHelper().setName(me, name);
 391.            }
 392.    
 393.            StereotypeUtility.dealWithStereotypes(me, stereotype, false);
 394.    
 395.            if (path != null) {
 396.                Object nspe =
 397.                    Model.getModelManagementHelper().getElement(
 398.                            path,
 399.                            Model.getFacade().getRoot(me));
 400.    
 401.                if (nspe == null || !(Model.getFacade().isANamespace(nspe))) {
 402.                    String msg = 
 403.                            "parsing.error.model-element-name.namespace-unresolved";
 404.                    throw new ParseException(Translator.localize(msg), 
 405.                            0);
 406.                }
 407.                if (!Model.getCoreHelper().isValidNamespace(me, nspe)) {
 408.                    String msg = 
 409.                            "parsing.error.model-element-name.namespace-invalid";
 410.                    throw new ParseException(Translator.localize(msg), 
 411.                            0);
 412.                }
 413.    
 414.                Model.getCoreHelper().addOwnedElement(nspe, me);
 415.            }
 416.        }
 417.        
 418.        /**
 419.         * Utility function to determine the presence of a key. 
 420.         * The default is false.
 421.         * 
 422.         * @param key the string for the key
 423.         * @param map the Map to check for the presence 
 424.         * and value of the key
 425.         * @return true if the value for the key is true, otherwise false
 426.         */
 427.        public static boolean isValue(final String key, final Map map) {
 428.            if (map == null) {
 429.                return false;
 430.            }
 431.            Object o = map.get(key);
 432.            if (!(o instanceof Boolean)) {
 433.                return false;
 434.            }
 435.            return ((Boolean) o).booleanValue();
 436.        }

 438.        /**
 439.         * Returns a visibility String either for a VisibilityKind or a model
 440.         * element.
 441.         * 
 442.         * @param o a modelelement or a visibilitykind
 443.         * @return a string. May be the empty string, but guaranteed not to be null
 444.         */
 445.        public static String generateVisibility2(Object o) {
 446.            if (o == null) {
 447.                return "";
 448.            }
 449.            if (Model.getFacade().isANamedElement(o)) {
 450.                if (Model.getFacade().isPublic(o)) {
 451.                    return "+";
 452.                }
 453.                if (Model.getFacade().isPrivate(o)) {
 454.                    return "-";
 455.                }
 456.                if (Model.getFacade().isProtected(o)) {
 457.                    return "#";
 458.                }
 459.                if (Model.getFacade().isPackage(o)) {
 460.                    return "~";
 461.                }
 462.            }

 466.                }

 469.                }

 472.                }

 503.        /**
 504.         * Parses a parameter list and aligns the parameter list in op to that
 505.         * specified in param. A parameter list generally has the following syntax:
 506.         *
 507.         * <pre>
 508.         * param := [inout] [name] [: type] [= initial value]
 509.         * list := [param] [, param]*
 510.         * </pre>
 511.         *
 512.         * <code>inout</code> is optional and if omitted the old value preserved.
 513.         * If no value has been assigned, then <code>in </code> is assumed.<p>
 514.         *
 515.         * <code>name</code>, <code>type</code> and <code>initial value</code>
 516.         * are optional and if omitted the old value preserved.<p>
 517.         *
 518.         * <code>type</code> and <code>initial value</code> can be given
 519.         * in any order.<p>
 520.         *
 521.         * Unspecified properties is carried over by position, so if a parameter is
 522.         * inserted into the list, then it will inherit properties from the
 523.         * parameter that was there before for unspecified properties.<p>
 524.         *
 525.         * This syntax is compatible with the UML 1.3 specification.
 526.         *
 527.         * @param op
 528.         *            The operation the parameter list belongs to.
 529.         * @param param
 530.         *            The parameter list, without enclosing parentheses.
 531.         * @param paramOffset
 532.         *            The offset to the beginning of the parameter list. Used for
 533.         *            error reports.
 534.         * @throws java.text.ParseException
 535.         *             when it detects an error in the attribute string. See also
 536.         *             ParseError.getErrorOffset().
 537.         */
 538.        static void parseParamList(Object op, String param, int paramOffset)
 539.            throws ParseException {
 540.            MyTokenizer st =
 541.                new MyTokenizer(param, " ,\t,:,=,\\,", parameterCustomSep);
 542.            // Copy returned parameters because it will be a live collection for MDR
 543.            Collection origParam =
 544.                new ArrayList(Model.getFacade().getParameters(op));
 545.            Object ns = Model.getFacade().getRoot(op);
 546.            if (Model.getFacade().isAOperation(op)) {
 547.                Object ow = Model.getFacade().getOwner(op);
 548.    
 549.                if (ow != null && Model.getFacade().getNamespace(ow) != null) {
 550.                    ns = Model.getFacade().getNamespace(ow);
 551.                }
 552.            }
 553.    
 554.            Iterator it = origParam.iterator();
 555.            while (st.hasMoreTokens()) {
 556.                String kind = null;
 557.                String name = null;
 558.                String tok;
 559.                String type = null;
 560.                StringBuilder value = null;
 561.                Object p = null;
 562.                boolean hasColon = false;
 563.                boolean hasEq = false;
 564.    
 565.                while (it.hasNext() && p == null) {
 566.                    p = it.next();
 567.                    if (Model.getFacade().isReturn(p)) {
 568.                        p = null;
 569.                    }
 570.                }
 571.    
 572.                while (st.hasMoreTokens()) {
 573.                    tok = st.nextToken();
 574.    
 575.                    if (",".equals(tok)) {
 576.                        break;
 577.                    } else if (" ".equals(tok) || "\t".equals(tok)) {
 578.                        if (hasEq) {
 579.                            value.append(tok);
 580.                        }
 581.                    } else if (":".equals(tok)) {
 582.                        hasColon = true;
 583.                        hasEq = false;
 584.                    } else if ("=".equals(tok)) {
 585.                        if (value != null) {
 586.                            String msg =
 587.                                "parsing.error.notation-utility.two-default-values";
 588.                            throw new ParseException(Translator.localize(msg),
 589.                                    paramOffset + st.getTokenIndex());
 590.                        }
 591.                        hasEq = true;
 592.                        hasColon = false;
 593.                        value = new StringBuilder();
 594.                    } else if (hasColon) {
 595.                        if (type != null) {
 596.                            String msg = "parsing.error.notation-utility.two-types";
 597.                            throw new ParseException(Translator.localize(msg),
 598.                                    paramOffset + st.getTokenIndex());
 599.                        }
 600.    
 601.                        if (tok.charAt(0) == '\'' || tok.charAt(0) == '\"') {
 602.                            String msg =
 603.                                "parsing.error.notation-utility.type-quoted";
 604.                            throw new ParseException(Translator.localize(msg),
 605.                                    paramOffset + st.getTokenIndex());
 606.                        }
 607.    
 608.                        if (tok.charAt(0) == '(') {
 609.                            String msg =
 610.                                "parsing.error.notation-utility.type-expr";
 611.                            throw new ParseException(Translator.localize(msg),
 612.                                    paramOffset + st.getTokenIndex());
 613.                        }
 614.    
 615.                        type = tok;
 616.                    } else if (hasEq) {
 617.                        value.append(tok);
 618.                    } else {
 619.                        if (name != null && kind != null) {
 620.                            String msg =
 621.                                "parsing.error.notation-utility.extra-text";
 622.                            throw new ParseException(Translator.localize(msg),
 623.                                    paramOffset + st.getTokenIndex());
 624.                        }
 625.    
 626.                        if (tok.charAt(0) == '\'' || tok.charAt(0) == '\"') {
 627.                            String msg =
 628.                                "parsing.error.notation-utility.name-kind-quoted";
 629.                            throw new ParseException(
 630.                                    Translator.localize(msg),
 631.                                    paramOffset + st.getTokenIndex());
 632.                        }
 633.    
 634.                        if (tok.charAt(0) == '(') {
 635.                            String msg =
 636.                                "parsing.error.notation-utility.name-kind-expr";
 637.                            throw new ParseException(
 638.                                    Translator.localize(msg),
 639.                                    paramOffset + st.getTokenIndex());
 640.                        }
 641.    
 642.                        kind = name;
 643.                        name = tok;
 644.                    }
 645.                }
 646.    
 647.                if (p == null) {
 648.                    /* Leave the type undefined (see issue 6145): */
 649.                    p = Model.getCoreFactory().buildParameter(op, null);
 650.                }
 651.    
 652.                if (name != null) {
 653.                    Model.getCoreHelper().setName(p, name.trim());
 654.                }
 655.    
 656.                if (kind != null) {
 657.                    setParamKind(p, kind.trim());
 658.                }
 659.    
 660.                if (type != null) {
 661.                    Model.getCoreHelper().setType(p, getType(type.trim(), ns));
 662.                }
 663.    
 664.                if (value != null) {
 665.                    // TODO: Find a better default language
 666.                    // TODO: We should know the notation language, since it is us
 667.                    Project project =
 668.                        ProjectManager.getManager().getCurrentProject();
 669.                    ProjectSettings ps = project.getProjectSettings();
 670.                    String notationLanguage = ps.getNotationLanguage();
 671.    
 672.                    Object initExpr =
 673.                        Model.getDataTypesFactory()
 674.                            .createExpression(
 675.                                    notationLanguage,
 676.                                    value.toString().trim());
 677.                    Model.getCoreHelper().setDefaultValue(p, initExpr);
 678.                }
 679.            }
 680.    
 681.            while (it.hasNext()) {
 682.                Object p = it.next();
 683.                if (!Model.getFacade().isReturn(p)) {
 684.                    Model.getCoreHelper().removeParameter(op, p);
 685.                    Model.getUmlFactory().delete(p);
 686.                }
 687.            }
 688.        }

 730.         * Applies a List of name/value pairs of properties to a model element.
 731.         * The name is treated as the tag of a tagged value unless it is one of the
 732.         * PropertySpecialStrings, in which case the action of the
 733.         * PropertySpecialString is invoked.
 734.         *
 735.         * @param elem
 736.         *            An model element to apply the properties to.
 737.         * @param prop
 738.         *            A List with name, value pairs of properties.
 739.         * @param spec
 740.         *            An array of PropertySpecialStrings to use.
 741.         */
 742.        static void setProperties(Object elem, List<String> prop,
 743.                PropertySpecialString[] spec) {
 744.            String name;
 745.            String value;
 746.            int i, j;
 747.    
 748.        nextProp:
 749.            for (i = 0; i + 1 < prop.size(); i += 2) {
 750.                name = prop.get(i);
 751.                value = prop.get(i + 1);
 752.    
 753.                if (name == null) {
 754.                    continue;
 755.                }
 756.    
 757.                name = name.trim();
 758.                if (value != null) {
 759.                    value = value.trim();
 760.                }
 761.    
 762.                /* If the current property occurs a second time
 763.                 * in the given list of properties, then skip it: */
 764.                for (j = i + 2; j < prop.size(); j += 2) {
 765.                    String s = prop.get(j);
 766.                    if (s != null && name.equalsIgnoreCase(s.trim())) {
 767.                        continue nextProp;
 768.                    }
 769.                }
 770.    
 771.                if (spec != null) {
 772.                    for (j = 0; j < spec.length; j++) {
 773.                        if (spec[j].invoke(elem, name, value)) {
 774.                            continue nextProp;
 775.                        }
 776.                    }
 777.                }
 778.    
 779.                Model.getCoreHelper().setTaggedValue(elem, name, value);
 780.            }
 781.        }

 850.        /**
 851.         * Declares a string that should take special action when it is found
 852.         * as a property in
 853.         * {@link ParserDisplay#setProperties ParserDisplay.setProperties}.<p>
 854.         *
 855.         * <em>Example:</em>
 856.         *
 857.         * <pre>

 869.         *

 871.         * It creates a PropertySpecialString that is invoken when the String

 874.         * defined on the 2nd line is invoked and performs
 875.         * a custom action on the element on which the property was
 876.         * specified by the user. In this case it does a setChangeability
 877.         * on an attribute instead of setting a tagged value,
 878.         * which would not have the desired effect.

 880.         * @author Michael Stockman
 881.         * @since 0.11.2
 882.         * @see PropertyOperation
 883.         * @see ParserDisplay#setProperties
 884.         */
 885.        static class PropertySpecialString {
 886.            private String name;
 887.    
 888.            private PropertyOperation op;
 889.    
 890.            /**
 891.             * Constructs a new PropertySpecialString that will invoke the
 892.             * action in propop when {@link #invoke(Object, String, String)} is
 893.             * called with name equal to str and then return true from invoke.
 894.             *
 895.             * @param str
 896.             *            The name of this PropertySpecialString.
 897.             * @param propop
 898.             *            An object containing the method to invoke on a match.
 899.             */
 900.            public PropertySpecialString(String str, PropertyOperation propop) {
