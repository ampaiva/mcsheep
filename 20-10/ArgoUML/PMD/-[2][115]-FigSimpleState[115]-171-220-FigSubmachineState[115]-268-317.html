 331.        /*
 332.         * Update the text labels and listeners.
 333.         *
 334.         * @see org.argouml.uml.diagram.ui.FigNodeModelElement#modelChanged(java.beans.PropertyChangeEve...
 335.         */
 336.        protected void modelChanged(PropertyChangeEvent mee) {
 337.            super.modelChanged(mee);
 338.            
 339.            // TODO: Rather than specifically ignore some item maybe it would be better
 340.            // to specifically state what items are of interest. Otherwise we may still
 341.            // be acting on other events we don't need
 342.            if (!Model.getFacade().isATransition(mee.getNewValue())
 343.                    && getOwner() != null) {
 344.                if ((mee.getSource().equals(getOwner()))) {
 345.                    if ((mee.getPropertyName()).equals("submachine")) {
 346.                        updateInclude();
 347.                        if (mee.getOldValue() != null) {
 348.                            updateListenersX(getOwner(), mee.getOldValue());
 349.                        }
 350.                    }
 351.                } else {
 352.                    if (mee.getSource()
 353.                            == Model.getFacade().getSubmachine(getOwner())) {
 354.                        // The Machine State has got a new name
 355.                        if (mee.getPropertyName().equals("name")) {
 356.                            updateInclude();
 357.                        }
 358.                        // The Machine State has been deleted from model
 359.                        if (mee.getPropertyName().equals("top")) {
 360.                            updateListeners(getOwner(), null);
 361.                        }
 362.                    }
 363.                }
 364.            }
 365.        }
 366.    
 367.        /*
 368.         * @see org.argouml.uml.diagram.ui.FigNodeModelElement#updateListeners(java.lang.Object)
 369.         */
 370.        @Override
 371.        protected void updateListeners(Object oldOwner, Object newOwner) {
 372.            super.updateListeners(oldOwner, newOwner);
 373.            if (newOwner != null) {
 374.                Object newSm = Model.getFacade().getSubmachine(newOwner);
 375.                if (newSm != null) {
 376.                    addElementListener(newSm);
 377.                }
 378.            }
 379.        }

 386.        }

 395.        private String generateSubmachine(Object m) {
 396.            Object c = Model.getFacade().getSubmachine(m);
 397.            String s = "include / ";
 398.            if ((c == null)
 399.                    || (Model.getFacade().getName(c) == null)
 400.                    || (Model.getFacade().getName(c).length() == 0)) {
 401.                return s;
 402.            }
 403.            return (s + Model.getFacade().getName(c));
 404.        }
