 264.         * namespace is not yet known.

 267.         * model namespace. There is no package statement so the lookup will
 268.         * always work.
 269.         * 
 270.         * </ol>
 271.         * Therefore in the case of (1), we need to delete duplicate artifacts in
 272.         * the addPackage() method.
 273.         * <p>
 274.         *

 290.                // remove the java specific ending (per JSR 26).
 291.                // BUT we can't do this because then the component will be confused
 292.                // with its class with the same name when invoking
 293.                // Model.getFacade().lookupIn(Object,String)
 294.                /*
 295.                 * if(fileName.endsWith(".java")) fileName = fileName.substring(0,

 298.    
 299.                if (Model.getFacade().getUmlVersion().charAt(0) == '1') {

 410.            // only do imports on the 2nd pass.
 411.            if (getLevel() == 0) {
 412.                return;
 413.            }
 414.    
 415.            String packageName = getPackageName(name);
 416.    
 417.            if (packageName == null || "".equals(packageName)) {
 418.                // TODO: This won't happen and can be removed when there is a

 420.                // on parsing "." strings

 427.            // outer class name
 428.            String classifierName = getClassifierName(name);
 429.            Object mPackage = getPackage(packageName, true);

 435.                buildImport(mPackage, srcFile);

 480.        private Object buildImport(Object element, Object srcFile) {

 482.            Collection dependencies = Model.getCoreHelper().getDependencies(
 483.                    element, srcFile);

 486.                    if ("javaImport".equals(
 487.                            Model.getFacade().getName(stereotype))) {
 488.                        return dep;
 489.                    }
 490.                }
 491.            }
 492.    
 493.            // Didn't find it. Let's create one.
 494.            Object pkgImport = Model.getCoreFactory().buildDependency(srcFile,
 495.                    element);

 502.            String newName = makeDependencyName(srcFile, element);
 503.            Model.getCoreHelper().setName(pkgImport, newName);
 504.            newElements.add(pkgImport);
 505.            return pkgImport;
 506.        }
