  40.    
  41.    import java.beans.PropertyChangeEvent;
  42.    
  43.    import javax.swing.text.AttributeSet;
  44.    import javax.swing.text.BadLocationException;
  45.    import javax.swing.text.PlainDocument;
  46.    
  47.    import org.apache.log4j.Logger;
  48.    import org.argouml.model.AttributeChangeEvent;
  49.    import org.argouml.model.Model;
  50.    import org.argouml.model.ModelEventPump;
  51.    
  52.    /**
  53.     * Model for a text property on a model element. It listens to
  54.     * property change events for the given property name so that
  55.     * changes made to the underlying UML model are reflected here.
  56.     * <p>
  57.     * NOTE: If you override the insertString() or remove() methods
  58.     * be sure to preserve the flushEvents() calls to keep things
  59.     * synchronized.  Events caused by updates are delivered
  60.     * asynchronously to the actual update calls.
  61.     * <p>
  62.     * @since Oct 6, 2002
  63.     * @author jaap.branderhorst@xs4all.nl
  64.     */
  65.    abstract class UMLPlainTextDocument
  66.        extends PlainDocument
  67.        implements UMLDocument {
  68.    
  69.        private static final Logger LOG =
  70.            Logger.getLogger(UMLPlainTextDocument.class);
  71.    
  72.        /**
  73.         * True if an event should be fired when the text of the document is changed
  74.         */
  75.        private boolean firing = true;
  76.    
  77.        /**
  78.         * True if an user edits the document directly (by typing in text)
  79.         */
  80.        @Deprecated
  81.        private boolean editing = false;
  82.    
  83.        /**
  84.         * The model element target
  85.         */
  86.        private Object target = null;
  87.    
  88.        /**
  89.         * The name of the property set event that will change the
  90.         * property this document shows.
  91.         */
  92.        private String eventName = null;
  93.    
  94.        /**
  95.         * Constructor for UMLPlainTextDocument. This takes a panel to set the
  96.         * thirdpartyeventlistener to the given list of events to listen to.
  97.         *
  98.         * @param name the event
  99.         */
 100.        public UMLPlainTextDocument(
 101.                final String name,
 102.                final Object target) {
 103.            super();
 104.            setEventName(name);
 105.            setTarget(target);
 106.        }
 107.    
 108.        /*
 109.         * @see java.beans.PropertyChangeListener#propertyChange(java.beans.PropertyChangeEvent)
 110.         */
 111.        public void propertyChange(PropertyChangeEvent evt) {
 112.            // NOTE: This may be called from a different thread, so we need to be
 113.            // careful of the threading restrictions imposed by AbstractDocument
 114.            // for mutators to be sure we don't deadlock.
 115.            if (evt instanceof AttributeChangeEvent 
 116.                    && eventName.equals(evt.getPropertyName())) {
 117.                updateText((String) evt.getNewValue());
 118.            }
 119.        }
 120.    
 121.        /**
 122.         * Returns the target.
 123.         * @return Object
 124.         */
 125.        public final Object getTarget() {
 126.            return target;
 127.        }
 128.    
 129.        /**
 130.         * Sets the target.
 131.         * @param target The target to set
 132.         */
 133.        public final void setTarget(Object target) {
 134.            ModelEventPump eventPump = Model.getPump();
 135.            this.target = target;
 136.            eventPump.addModelEventListener(this, this.target,
 137.                    getEventName());
 138.            updateText(getProperty());
 139.        }
 140.    
 141.        /*
 142.         * @see javax.swing.text.Document#insertString(
 143.         *         int, java.lang.String, javax.swing.text.AttributeSet)
 144.         */
 145.        public void insertString(int offset, String str, AttributeSet a)
 146.            throws BadLocationException {
 147.            
 148.            // Mutators hold write lock & will deadlock if use is not thread safe 
 149.            super.insertString(offset, str, a);
 150.            
 151.            setPropertyInternal(getText(0, getLength()));
 152.        }
 153.    
 154.    
 155.        /*
 156.         * @see javax.swing.text.Document#remove(int, int)
 157.         */
 158.        public void remove(int offs, int len) throws BadLocationException {
 159.    
 160.            // Mutators hold write lock & will deadlock if use is not thread safe 
 161.            super.remove(offs, len);
 162.            
 163.            setPropertyInternal(getText(0, getLength()));
 164.        }
 165.    
 166.        /**
 167.         * Wrapped version of setProperty which attempts to keep us from hearing
 168.         * our own echo on the event listener when we change something.  Also
 169.         * skips updates equal the current value. 
 170.         */
 171.        private void setPropertyInternal(String newValue) {
 172.            // TODO: This is updating model on a per character basis as
 173.            // well as unregistering/reregistering event listeners every
 174.            // character - very wasteful - tfm
 175.            if (isFiring() && !newValue.equals(getProperty())) {
 176.                setFiring(false);
 177.                setProperty(newValue);
 178.                Model.getPump().flushModelEvents();
 179.                setFiring(true);
 180.            }
 181.        }
 182.    
 183.        /**
 184.         * @param text the value of the property
 185.         */
 186.        protected abstract void setProperty(String text);
 187.    
 188.        /**
 189.         * @return the value of the property
 190.         */
 191.        protected abstract String getProperty();
 192.    
 193.        /**
 194.         * Enable/disable firing of updates.  As a side effect, it unregisters
 195.         * model event listeners during disable and registers them again during
 196.         * enable.
 197.         * 
 198.         * @param f new firing state.  Pass false to disable updates.
 199.         */
 200.        final synchronized void setFiring(boolean f) {
 201.            ModelEventPump eventPump = Model.getPump();
 202.            if (f && target != null) {
 203.                eventPump.addModelEventListener(this, target, eventName);
 204.            } else {
 205.                eventPump.removeModelEventListener(this, target, eventName);
 206.            }
 207.            firing = f;
 208.        }
 209.    
 210.        /**
 211.         * Return the state of the firing flag.  Method is synchronized so it will
 212.         * return the correct value from any thread.
 213.         * 
 214.         * @return true firing of updates is allowed currently. Returns false if we
 215.         *         are in the process of doing an update so that we know to ignore
 216.         *         any resulting events.
 217.         */
 218.        private final synchronized boolean isFiring() {
 219.            return firing;
 220.        }
 221.    
 222.        private final void updateText(String textValue) {
 223.            try {
 224.                if (textValue == null) {
 225.                    textValue = "";
 226.                }
 227.                String currentValue = getText(0, getLength());
 228.                if (isFiring() && !textValue.equals(currentValue)) {
 229.                    setFiring(false);
 230.                    
 231.                    // Mutators hold write lock & will deadlock 
 232.                    // if use is not thread-safe 
 233.                    super.remove(0, getLength());
 234.                    super.insertString(0, textValue, null);
 235.                }
 236.            } catch (BadLocationException b) {
 237.                LOG.error(
 238.                  "A BadLocationException happened\n"
 239.                  + "The string to set was: "
 240.                  + getProperty(),
 241.                  b);
 242.            } finally {
 243.                setFiring(true);
 244.            }
 245.        }
 246.    
 247.    
 248.        /**
 249.         * Returns the eventName.
 250.         * @return String
 251.         */
 252.        public String getEventName() {
 253.            return eventName;
 254.        }
 255.    
 256.        /**
 257.         * Sets the eventName.
 258.         * @param en The eventName to set
 259.         */
 260.        protected void setEventName(String en) {
 261.            eventName = en;
 262.        }
 263.    }
