 471.                Dimension stereoMin = getStereotypeFig().getMinimumSize();
 472.                getNameFig().setTopMargin(stereoMin.height);

 474.    
 475.                getStereotypeFig().setBounds(xa, ya,
 476.                    tabWidth, stereoMin.height + 1);
 477.    
 478.                if (tabWidth < stereoMin.width + 1) {
 479.                    tabWidth = stereoMin.width + 2;
 480.                }
 481.            } else {
 482.                getNameFig().setBounds(xa, currentY, tabWidth + 1, minNameHeight);

 485.            // Advance currentY to where the start of the body box is,
 486.            // remembering that it overlaps the next box by 1 pixel. Calculate the
 487.            // size of the body box, and update the Y pointer past it if it is
 488.            // displayed.

 501.                 * and the figPoly next to the text.

 304.            Object modelElement = getOwner();
 305.    
 306.            if (modelElement == null) {
 307.                return;

 317.                } // else nothing changed
 318.            } else {

 320.                /* This populates the stereotypes area: */
 321.                super.updateStereotypeText();

 324.                    getNameFig().setTopMargin(0);

 328.                            getStereotypeFig().getMinimumSize().height);
 329.                    getStereotypeFig().setVisible(true);
 330.                } // else nothing changed
 331.            }

 340.         * incorrectly. This is not the best fix but is a workaround until the
 341.         * true cause is known. See issue 6135.

 379.         * @see org.tigris.gef.presentation.Fig#getMinimumSize()
 380.         */
 381.        @Override
 382.        public Dimension getMinimumSize() {
 383.            // Use "aSize" to build up the minimum size. Start with the size of the
 384.            // name fig and build up.
 385.            Dimension aSize = new Dimension(getNameFig().getMinimumSize());
 386.    
 387.            if (figPoly != null) {
 388.                /* The figPoly is located at the right of the name text. 
 389.                 * The nameFig size is increased, so that it fits its text, 
 390.                 * and the figPoly next to the text, all within the boundaries 
 391.                 * of the nameFig. */
 392.                Dimension symbol = figPoly.getSize();
 393.                aSize.width += symbol.width;
 394.                aSize.height = Math.max(aSize.height, symbol.height);
 395.            }
 396.            
 397.            aSize.height = Math.max(aSize.height, MIN_HEIGHT);
 398.            aSize.width = Math.max(aSize.width, MIN_WIDTH);
 399.    
 400.            // If we have any number of stereotypes displayed, then allow 
 401.            // some space for that (only width, height is included in nameFig):
 402.            if (isStereotypeVisible()) {
 403.                Dimension st = getStereotypeFig().getMinimumSize();
 404.                aSize.width =
 405.            Math.max(aSize.width, st.width);
 406.            }
 407.    
 408.            // take into account the tab is not as wide as the body:
 409.            aSize.width += indentX + 1; 
 410.    
 411.            // we want at least some of the package body to be displayed
 412.            aSize.height += 28 + 2 * getLineWidth();
 413.    
 414.            // And now aSize has the answer
 415.            return aSize;
 416.        }
 417.    
 418.        /**
 419.         * Sets the bounds, but the size will be at least the one returned by
 420.         * {@link #getMinimumSize()}.<p>
 421.         *
 422.         * If the required height is bigger, then the additional height is
 423.         * not distributed among all figs (i.e. compartments), 
 424.         * but goes into the body. Hence, the
 425.         * accumulated height of all visible figs equals the demanded height<p>.
 426.         *
 427.         * Some of this has "magic numbers" hardcoded in. In particular there is
 428.         * a knowledge that the minimum height of a name compartment is 21
 429.         * pixels. This height is needed to be able to display the "Clarifier"
 430.         * inside the name compartment.
 431.         *
 432.         * @param xa  Desired X coordinate of upper left corner
 433.         *
 434.         * @param ya  Desired Y coordinate of upper left corner
 435.         *
 436.         * @param w  Desired width of the FigClass
 437.         *
 438.         * @param h  Desired height of the FigClass
 439.         */
 440.        @Override
 441.        protected void setStandardBounds(int xa, int ya, int w, int h) {
 442.            // Save our old boundaries (needed later), and get minimum size
 443.            // info. "aSize" will be used to maintain a running calculation of our
 444.            // size at various points.
 445.    
 446.            Rectangle oldBounds = getBounds();
 447.    
 448.            // The new size can not be smaller than the minimum.
 449.            Dimension minimumSize = getMinimumSize();
 450.            int newW = Math.max(w, minimumSize.width);
 451.            int newH = Math.max(h, minimumSize.height);
 452.    
 453.            // Now resize all sub-figs, including not displayed figs. Start by the
 454.            // name. We override the getMinimumSize if it is less than our view (21
 455.            // pixels hardcoded!). Add in the shared extra, plus in this case the
 456.            // correction.
 457.    
 458.            Dimension nameMin = getNameFig().getMinimumSize();
 459.            int minNameHeight = Math.max(nameMin.height, MIN_HEIGHT);
 460.    
 461.            // Now sort out the stereotype display. If the stereotype is displayed,
 462.            // move the upper boundary of the name compartment up and set new
 463.            // bounds for the name and the stereotype compartments and the
 464.            // stereoLineBlinder that blanks out the line between the two
 465.    
 466.            int currentY = ya;
 467.    
 468.            int tabWidth = newW - indentX;
 469.    
 470.            if (isStereotypeVisible()) {
 471.                Dimension stereoMin = getStereotypeFig().getMinimumSize();
 472.                getNameFig().setTopMargin(stereoMin.height);
 473.                getNameFig().setBounds(xa, currentY, tabWidth + 1, minNameHeight);
 474.    
 475.                getStereotypeFig().setBounds(xa, ya,
 476.                    tabWidth, stereoMin.height + 1);
 477.    
 478.                if (tabWidth < stereoMin.width + 1) {
 479.                    tabWidth = stereoMin.width + 2;
 480.                }
 481.            } else {
 482.                getNameFig().setBounds(xa, currentY, tabWidth + 1, minNameHeight);
 483.            }
 484.            
 485.            // Advance currentY to where the start of the body box is,
 486.            // remembering that it overlaps the next box by 1 pixel. Calculate the
 487.            // size of the body box, and update the Y pointer past it if it is
 488.            // displayed.
 489.    
 490.            currentY += minNameHeight - 1; // -1 for 1 pixel overlap
 491.            body.setBounds(xa, currentY, newW, newH + ya - currentY);
 492.    
 493.            tabHeight = currentY - ya;
 494.            // set bounds of big box
 495.    
 496.            getBigPort().setBounds(xa, ya, newW, newH);
 497.    
 498.            if (figPoly != null) {
 499.                /* The figPoly is located at the right edge of the nameFig. 
 500.                 * The nameFig size is such that it at least fits its text, 
 501.                 * and the figPoly next to the text. 
 502.                 * Making the package bigger, causes the figPoly to stick to 
 503.                 * the right edge.*/
 504.                Rectangle previousBounds = figPoly.getBounds();
 505.                Rectangle name = getNameFig().getBounds();
 506.                int nx = name.x + name.width - figPoly.getWidth()
 507.                    - getLineWidth() - getNameFig().getRightMargin();
 508.                int ny = name.y + getLineWidth() + getNameFig().getTopMargin();
 509.                figPoly.translate((nx - previousBounds.x),
 510.                        ny - previousBounds.y);
 511.            }
 512.    
 513.            // Now force calculation of the bounds of the figure, update the edges
 514.            // and trigger anyone who's listening to see if the "bounds" property
 515.            // has changed.
 516.            
 517.            background.setBounds(xa, ya, w, h);
 518.    
 519.            calcBounds();
 520.            updateEdges();
 521.            firePropChange("bounds", oldBounds, getBounds());
 522.        }

 474.    
 475.                getStereotypeFig().setBounds(xa, ya,
 476.                    tabWidth, stereoMin.height + 1);
 477.    
 478.                if (tabWidth < stereoMin.width + 1) {
 479.                    tabWidth = stereoMin.width + 2;

 487.            // size of the body box, and update the Y pointer past it if it is
 488.            // displayed.

 508.                int ny = name.y + getLineWidth() + getNameFig().getTopMargin();

 510.                        ny - previousBounds.y);
 511.            }

 515.            // has changed.
 516.            
 517.            background.setBounds(xa, ya, w, h);
 518.

 520.            updateEdges();
 521.            firePropChange("bounds", oldBounds, getBounds());
