 284.                } else {

 286.                    value = namevaluepair.substring(equalsPos + 1);
 287.                }
 288.    
 289.                map.put(name, value);
 290.            }
 291.            return map;
 292.        }

 399.         * including the nodes contained within edges allowing edge to edge
 400.         * connections for comment edges, association classes and dependencies.
 401.         * 
 402.         * @param d the Diagram
 403.         */
 404.        private void attachEdges(Diagram d) {
 405.            for (EdgeData edgeData : figEdges) {
 406.                final FigEdge edge = edgeData.getFigEdge();
 407.    
 408.                Object modelElement = modelElementsByFigEdge.get(edge);
 409.                if (modelElement != null) {
 410.                    if (edge.getOwner() == null) {
 411.                        edge.setOwner(modelElement);
 412.                    }
 413.                }
 414.            }
 415.    
 416.            for (EdgeData edgeData : figEdges) {
 417.                final FigEdge edge = edgeData.getFigEdge();
 418.    
 419.                Fig sourcePortFig = findFig(edgeData.getSourcePortFigId());
 420.                Fig destPortFig = findFig(edgeData.getDestPortFigId());
 421.                final FigNode sourceFigNode = getFigNode(edgeData
 422.                        .getSourceFigNodeId());
 423.                final FigNode destFigNode = getFigNode(edgeData.getDestFigNodeId());
 424.    
 425.                if (sourceFigNode instanceof FigEdgePort) {
 426.                    sourcePortFig = sourceFigNode;
 427.                }
 428.    
 429.                if (destFigNode instanceof FigEdgePort) {
 430.                    destPortFig = destFigNode;
 431.                }
 432.    
 433.                if (sourcePortFig == null && sourceFigNode != null) {
 434.                    sourcePortFig = getPortFig(sourceFigNode);
 435.                }
 436.    
 437.                if (destPortFig == null && destFigNode != null) {
 438.                    destPortFig = getPortFig(destFigNode);
 439.                }
 440.    
 441.                if (sourcePortFig == null || destPortFig == null
 442.                        || sourceFigNode == null || destFigNode == null) {
 443.                    LOG.error("Can't find nodes for FigEdge: " + edge.getId() + ":"
 444.                            + edge.toString());
 445.                    edge.removeFromDiagram();
 446.                } else {
 447.                    edge.setSourcePortFig(sourcePortFig);
 448.                    edge.setDestPortFig(destPortFig);
 449.                    edge.setSourceFigNode(sourceFigNode);
 450.                    edge.setDestFigNode(destFigNode);
 451.                }
 452.            }

 456.            // Only do this after all edges are connected as compute route
 457.            // requires all edges to be connected to nodes.
 458.            // TODO: It would be nice not to have to do this and restore annotation
 459.            // positions instead.
 460.            for (Object edge : d.getLayer().getContentsEdgesOnly()) {
 461.                FigEdge figEdge = (FigEdge) edge;
 462.                figEdge.computeRouteImpl();
 463.            }
 464.        }

 490.         */
 491.        private FigNode getFigNode(String figId) throws IllegalStateException {
 492.            if (figId.contains(".")) {
 493.                // If the id does not look like a top-level Fig then we can assume
 494.                // that this is an id of a FigEdgePort inside some FigEdge.

 501.                            + figId);

 504.                return edge.getEdgePort();
 505.            } else {
 506.                // If there is no dot then this must be a top level Fig and can be

 631.         * <p>

 637.         * @param bounds position and size of figure

 639.         * @throws SAXException

 641.         *      java.lang.String, java.awt.Rectangle)
 642.         */
 643.        @Override
 644.        protected Fig constructFig(final String className, final String href,
 645.                final Rectangle bounds, final Attributes attributes)
 646.            throws SAXException {
 647.    
 648.            final DiagramSettings oldSettings = 
 649.                ((ArgoDiagram) getDiagram()).getDiagramSettings();
 650.    
 651.            Fig f = null;
 652.            try {
 653.                Class figClass = Class.forName(className);
 654.    
 655.                final Constructor[] constructors = figClass.getConstructors();
 656.    
 657.                // We are looking first to match with 3 different constructor
 658.                // types. We would not expect a Fig to have any mix of these.
 659.                // Any constructor other than these should be deprecated so we
 660.                // look for these first.
 661.                // Fig(DiagramEdgeSettings, DiagramSettings)
 662.                // Fig(Object, Rectangle, DiagramSettings)
 663.                // Fig(Rectangle, DiagramSettings)
 664.                for (Constructor constructor : constructors) {
 665.                    Class[] parameterTypes = constructor.getParameterTypes();
 666.                    if (parameterTypes.length == 3
 667.                            && parameterTypes[0].equals(Object.class)
 668.                            && parameterTypes[1].equals(Rectangle.class)
 669.                            && parameterTypes[2].equals(DiagramSettings.class)) {
 670.                        // FigNodeModelElements should match here
 671.                        final Object parameters[] = new Object[3];
 672.                        final Object owner = getOwner(className, href);
 673.                        if (owner == null) {
 674.                            return null;
 675.                        }
 676.                        parameters[0] = owner;
 677.                        parameters[1] = bounds;
 678.                        parameters[2] = oldSettings;
 679.    
 680.                        constructor.setAccessible(true);
 681.                        f = (Fig) constructor.newInstance(parameters);
 682.                    } else if (parameterTypes.length == 2
 683.                            && parameterTypes[0].equals(DiagramEdgeSettings.class)
 684.                            && parameterTypes[1].equals(DiagramSettings.class)) {
 685.                        // FigEdgeModelElements should match here (they have no
 686.                        // bounds)
 687.                        final Object parameters[] = new Object[2];
 688.                        final Object owner = getOwner(className, href);
 689.                        if (owner == null) {
 690.                            return null;
 691.                        }
 692.    
 693.                        String sourceUuid = attributes.getValue("sourceConnector");
 694.                        String destinationUuid = attributes
 695.                                .getValue("destConnector");
 696.    
 697.                        final Object source;
 698.                        final Object destination;
 699.                        if (sourceUuid != null && destinationUuid != null) {
 700.                            source = findOwner(sourceUuid);
 701.                            destination = findOwner(destinationUuid);
 702.                        } else {
 703.                            source = null;
 704.                            destination = null;
 705.                        }
 706.    
 707.                        DiagramEdgeSettings newSettings = new DiagramEdgeSettings(
 708.                                owner, source, destination);
 709.                        parameters[0] = newSettings;
 710.                        parameters[1] = oldSettings;
 711.    
 712.                        constructor.setAccessible(true);
 713.                        f = (Fig) constructor.newInstance(parameters);
 714.                    } else if (parameterTypes.length == 2
 715.                            && parameterTypes[0].equals(Rectangle.class)
 716.                            && parameterTypes[1].equals(DiagramSettings.class)) {
 717.                        // A FigNodeModelElement with no owner should match here
 718.                        // TODO: This is a temporary solution due to FigPool
 719.                        // extending
 720.                        // FigNodeModelElement when in fact it should not do so.
 721.                        Object parameters[] = new Object[2];
 722.                        parameters[0] = bounds;
 723.                        parameters[1] = oldSettings;
 724.    
 725.                        constructor.setAccessible(true);
 726.                        f = (Fig) constructor.newInstance(parameters);
 727.                    }
 728.                }
 729.                if (f == null) {
 730.                    // If no Fig was created by the code above then we must go
 731.                    // look for the old style constructor that should have fallen
 732.                    // into disuse by now.
 733.                    // Fig(Object, Rectangle, DiagramSettings)
 734.                    // All of these constructors should have been deprecated
 735.                    // at least and replaced with the new signature. This is
 736.                    // here for paranoia only until all Figs have been reviewed.
 737.                    for (Constructor constructor : constructors) {
 738.                        Class[] parameterTypes = constructor.getParameterTypes();
 739.                        if (parameterTypes.length == 2
 740.                                && parameterTypes[0].equals(Object.class)
 741.                                && parameterTypes[1].equals(DiagramSettings.class)) {
 742.                            Object parameters[] = new Object[2];
 743.    
 744.                            final Object owner = getOwner(className, href);
 745.                            // currently FigEdgeNote can be passed null
 746.    //                        if (owner == null) {
 747.    //                            return null;
 748.    //                        }
 749.                            parameters[0] = owner;
 750.                            parameters[1] = oldSettings;
 751.    
 752.                            constructor.setAccessible(true);
 753.                            f = (Fig) constructor.newInstance(parameters);
 754.                            LOG.warn("Fig created by old style constructor "
 755.                                    + f.getClass().getName());
 756.                            break;
 757.                        }
 758.                    }
 759.                }
 760.            } catch (ClassNotFoundException e) {
 761.                throw new SAXException(e);
 762.            } catch (IllegalAccessException e) {
 763.                throw new SAXException(e);
 764.            } catch (InstantiationException e) {
 765.                throw new SAXException(e);
 766.            } catch (InvocationTargetException e) {
 767.                throw new SAXException(e);
 768.            }
 769.    
 770.            // Fall back to GEF's handling if we couldn't find an appropriate
 771.            // constructor
 772.            if (f == null) {
 773.                LOG.warn("No ArgoUML constructor found for " + className
 774.                        + " falling back to GEF's default constructors");
 775.                f = super.constructFig(className, href, bounds, attributes);
 776.            }
 777.    
 778.            return f;
 779.        }
 780.    
 781.        /**
 782.         * Given the href extracted from the PGML return the model element with that
 783.         * uuid.
 784.         * 
 785.         * @param className Used only for logging should the href not be found
 786.         * @param href The href
 787.         * @return
 788.         */
 789.        private Object getOwner(String className, String id) {
 790.            if (id == null) {
 791.                LOG.warn("There is no href attribute provided for a " + className
 792.                        + " so the diagram element is ignored on load");
 793.                return null;

 275.         * destination and attach these to either end of the FigEdge

 277.         * @param newEdge The edge to attach
 278.         */
 279.        protected final void setPorts(Layer layer, FigEdge newEdge) {
 280.            Object modelElement = newEdge.getOwner();
 281.            if (newEdge.getSourcePortFig() == null) {
 282.                Object source;
 283.                if (modelElement instanceof CommentEdge) {
 284.                    source = ((CommentEdge) modelElement).getSource();
 285.                } else {
 286.                    source = Model.getUmlHelper().getSource(modelElement);
 287.                }
 288.                FigNode sourceNode = getNodePresentationFor(layer, source);
 289.                assert (sourceNode != null) : "No FigNode found for " + source;
 290.                setSourcePort(newEdge, sourceNode);
 291.            }
 292.    
 293.            if (newEdge.getDestPortFig() == null) {
 294.                Object dest;
 295.                if (modelElement instanceof CommentEdge) {
 296.                    dest = ((CommentEdge) modelElement).getDestination();
 297.                } else {
 298.                    dest = Model.getUmlHelper().getDestination(newEdge.getOwner());
 299.                }
 300.                setDestPort(newEdge, getNodePresentationFor(layer, dest));
 301.            }
 302.            
 303.            if (newEdge.getSourcePortFig() == null
 304.                    || newEdge.getDestPortFig() == null) {
 305.                throw new IllegalStateException("Edge of type "
 306.                    + newEdge.getClass().getName()
 307.                    + " created with no source or destination port");
 308.            }
 309.        }

 324.         * The FigNode portion of an association class is returned in preference
 325.         * to the FigEdge portion.
 326.         * If no FigNode is found then a FIgEdge is searched for and the FigNode
 327.         * that acts as its edge port is returned.
 328.         * @param lay the layer containing the Fig
 329.         * @param modelElement the model element to find presentation for
 330.         * @return the FigNode presentation of the model element
 331.         */
 332.        protected FigNode getNodePresentationFor(Layer lay, Object modelElement) {
 333.            assert modelElement != null : "A modelElement must be supplied";
 334.            for (Object fig : lay.getContentsNoEdges()) {
 335.     
 336.                if (fig instanceof FigNode
 337.                        && modelElement.equals(((FigNode) fig).getOwner())) {
 338.                    return ((FigNode) fig);
 339.                }
 340.            }
 341.            for (Object fig : lay.getContentsEdgesOnly()) {
 342.                if (fig instanceof FigEdgeModelElement
 343.                        && modelElement.equals(((FigEdgeModelElement) fig)
 344.                                .getOwner())) {
 345.                    return ((FigEdgeModelElement) fig).getEdgePort();
 346.                }
 347.            }
 348.            return null;
 349.        }
