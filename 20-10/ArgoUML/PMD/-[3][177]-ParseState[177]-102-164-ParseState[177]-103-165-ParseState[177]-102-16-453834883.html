  52.    class ParseState {
  53.    
  54.        /**
  55.         * Logger.
  56.         */
  57.        private static final Logger LOG = Logger.getLogger(ParseState.class);
  58.    
  59.        /**
  60.         * When the classifier parse is finished, these features will be
  61.         * removed from the model.
  62.         */
  63.        private Collection obsoleteFeatures;
  64.    
  65.        /**
  66.         * When the classifier parse is finished, these inner classes
  67.         * will be removed from the model.
  68.         */
  69.        private Collection obsoleteInnerClasses;
  70.    
  71.        /**
  72.         * This prefix is appended to inner classes, if any.
  73.         */
  74.        private String classnamePrefix;
  75.    
  76.        /**
  77.         * The available context for currentClassifier.
  78.         */
  79.        private Context context;
  80.    
  81.        /**
  82.         * The classifier that is parsed for the moment.
  83.         */
  84.        private Object classifier;
  85.    
  86.        /**
  87.         * Counter for anonymous innner classes.
  88.         */
  89.        private int anonymousClassCounter;
  90.    
  91.        /**
  92.         * Represents the source file being parsed. In UML1, this is a component.
  93.         * In UML2, this is an artifact.
  94.         */
  95.        private Object artifact;
  96.    
  97.        /**
  98.         * Create a new parse state.
  99.         *
 100.         * @param model The model.
 101.         * @param javaLangPackage The default package java.lang.
 102.         */
 103.        public ParseState(Object model, Object javaLangPackage) {
 104.        obsoleteInnerClasses = new ArrayList();
 105.        classifier = null;
 106.        context =
 107.            new PackageContext(new PackageContext(null, model),
 108.                       javaLangPackage);
 109.        anonymousClassCounter = 0;
 110.        }
 111.    
 112.        /**
 113.         * Create a new parse state based on another parse state.
 114.         *
 115.         * @param previousState The base parse state.
 116.         * @param mClassifier The new classifier being parsed.
 117.         * @param currentPackage The current package being parsed.
 118.         */
 119.        public ParseState(ParseState previousState,
 120.                          Object mClassifier,
 121.                          Object currentPackage) {
 122.    
 123.            LOG.info("Parsing the state of " + mClassifier);
 124.    
 125.            classnamePrefix =
 126.                previousState.classnamePrefix
 127.                + Model.getFacade().getName(mClassifier)
 128.                + "$";
 129.            obsoleteFeatures =
 130.                new ArrayList(Model.getFacade().getFeatures(mClassifier));
 131.            obsoleteInnerClasses =
 132.                new ArrayList(Model.getFacade().getOwnedElements(mClassifier));
 133.            context =
 134.                new OuterClassifierContext(
 135.                        previousState.context,
 136.                        mClassifier,
 137.                        currentPackage,
 138.                        classnamePrefix);
 139.            classifier = mClassifier;
 140.            anonymousClassCounter = previousState.anonymousClassCounter;
 141.        }

 226.        anonymousClassCounter = 0;
 227.        }

 246.        }

 250.         * classifier in the model.

 252.        public void removeObsoleteFeatures() {
 253.            if (obsoleteFeatures == null) {
 254.                return;
 255.            }
 256.            for (Object feature : obsoleteFeatures) {
 257.                Model.getCoreHelper().removeFeature(classifier, feature);
 258.                Model.getUmlFactory().delete(feature);
 259.            }
 260.        }
 261.    
 262.        /**
 263.         * Remove inner classes no longer in the source from the current
 264.         * classifier in the model.
 265.         */
 266.        public void removeObsoleteInnerClasses() {
 267.            if (obsoleteInnerClasses == null) {
 268.            return;
 269.        }
 270.        for (Object element : obsoleteInnerClasses) {
 271.            if (Model.getFacade().isAClassifier(element)) {
 272.            Model.getUmlFactory().delete(element);
 273.            }
 274.        }
 275.        }

 278.         * Tell the parse state that a feature belongs to the current
 279.         * classifier.

 285.        }

 294.            for (Object mFeature : obsoleteFeatures) {
 295.            if (name.equals(Model.getFacade().getName(mFeature))) {
 296.            return mFeature;
 297.            }
 298.        }
 299.        return null;
 300.        }

 328.            return mFeature;
 329.            }
 330.        }
 331.        return null;
 332.        }
 333.    
 334.        /**
 335.         * Get a operation from the current classifier not yet modeled.
 336.         *
 337.         * @param name The name of the operation.
 338.         * @return The found operation, null if not found.
 339.         */
 340.        public Object getOperation(String name) {
 341.            for (Object feature : obsoleteFeatures) {
 342.            if (Model.getFacade().isAOperation(feature)
 343.                        && name.equals(Model.getFacade().getName(feature))) {
 344.            return feature;
 345.            }
 346.        }
 347.        return null;
 348.        }
 349.    
 350.        /**
 351.         * Get a attribute from the current classifier not yet modeled.

 354.         * @return The found attribute, null if not found.
 355.         */
 356.        public Object getAttribute(String name) {
 357.            for (Object feature : obsoleteFeatures) {
 358.                if (Model.getFacade().isAAttribute(feature)
 359.                        && name.equals(Model.getFacade().getName(feature))) {

  67.         * will be removed from the model.
  68.         */
  69.        private Collection obsoleteInnerClasses;
  70.    
  71.        /**
  72.         * This prefix is appended to inner classes, if any.
  73.         */
  74.        private String classnamePrefix;
  75.    
  76.        /**
  77.         * The available context for currentClassifier.
  78.         */
  79.        private Context context;
  80.    
  81.        /**
  82.         * The classifier that is parsed for the moment.
  83.         */
  84.        private Object classifier;
  85.    
  86.        /**
  87.         * Counter for anonymous innner classes.
  88.         */
  89.        private int anonymousClassCounter;
  90.    
  91.        /**
  92.         * Represents the source file being parsed.
  93.         */
  94.        private Object component;
  95.    
  96.        /**
  97.         * Create a new parse state.
  98.         *
  99.         * @param model The model.
 100.         * @param javaLangPackage The default package java.lang.
 101.         */
 102.        public ParseState(Object model, Object javaLangPackage) {
 103.        obsoleteInnerClasses = new ArrayList();
 104.        classifier = null;
 105.        context =
 106.            new PackageContext(new PackageContext(null, model),
 107.                       javaLangPackage);
 108.        anonymousClassCounter = 0;
 109.        }
 110.    
 111.        /**
 112.         * Create a new parse state based on another parse state.
 113.         *
 114.         * @param previousState The base parse state.
 115.         * @param mClassifier The new classifier being parsed.
 116.         * @param currentPackage The current package being parsed.
 117.         */
 118.        public ParseState(ParseState previousState,
 119.                          Object mClassifier,
 120.                          Object currentPackage) {
 121.    
 122.            LOG.info("Parsing the state of " + mClassifier);
 123.    
 124.            classnamePrefix =
 125.                previousState.classnamePrefix
 126.                + Model.getFacade().getName(mClassifier)
 127.                + "$";
 128.            obsoleteFeatures =
 129.                new ArrayList(Model.getFacade().getFeatures(mClassifier));
 130.            obsoleteInnerClasses =
 131.                new ArrayList(Model.getFacade().getOwnedElements(mClassifier));
 132.            context =
 133.                new OuterClassifierContext(
 134.                        previousState.context,
 135.                        mClassifier,
 136.                        currentPackage,
 137.                        classnamePrefix);
 138.            classifier = mClassifier;
 139.            anonymousClassCounter = previousState.anonymousClassCounter;
 140.        }
 141.    
 142.        /**
 143.         * Add a package to the current context.
 144.         *
 145.         * @param mPackage The package to add.
 146.         */
 147.        public void addPackageContext(Object mPackage) {
 148.        context = new PackageContext(context, mPackage);
 149.        }
 150.    
 151.        /**
 152.         * Add a classifier to the current context.
 153.         *
 154.         * @param mClassifier The classifier to add.
 155.         */
 156.        public void addClassifierContext(Object mClassifier) {
 157.        context = new ClassifierContext(context, mClassifier);
 158.        }
 159.    
 160.        /**
 161.         * @param c the source file being parsed
 162.         */
 163.        public void addComponent(Object c) {
 164.            this.component = c;
 165.        }
 166.    
 167.        /**
 168.         * @return the source file being parsed
 169.         */
 170.        public Object getComponent() {
 171.            return component;
 172.        }
 173.    
 174.        /**
 175.         * Get the current context.
 176.         *
 177.         * @return The current context.
 178.         */
 179.        public Context getContext() {
 180.        return context;
 181.        }
 182.    
 183.        /**
 184.         * Get the current classifier.
 185.         *
 186.         * @return The current classifier.
 187.         */
 188.        public Object getClassifier() {
 189.        return classifier;
 190.        }
 191.    
 192.        /**
 193.         * Tell the parse state that an anonymous class is being parsed.
 194.         *
 195.         * @return The name of the anonymous class.
 196.         */
 197.        public String anonymousClass() {
 198.        classnamePrefix =
 199.            classnamePrefix.substring(0, classnamePrefix.indexOf("$") + 1);
 200.        anonymousClassCounter++;
 201.        return (Integer.valueOf(anonymousClassCounter)).toString();
 202.        }
 203.    
 204.        /**
 205.         * Tell the parse state that an outer class is being parsed.
 206.         */
 207.        public void outerClassifier() {
 208.        classnamePrefix = "";
 209.        anonymousClassCounter = 0;
 210.        }
 211.    
 212.        /**
 213.         * Get the current classname prefix.
 214.         *
 215.         * @return The current classname prefix.
 216.         */
 217.        public String getClassnamePrefix() {
 218.        return classnamePrefix;
 219.        }
 220.    
 221.        /**
 222.         * Tell the parse state that a classifier is an inner classifier
 223.         * to the current parsed classifier.
 224.         *
 225.         * @param mClassifier The inner classifier.
 226.         */
 227.        public void innerClassifier(Object mClassifier) {
 228.        obsoleteInnerClasses.remove(mClassifier);
 229.        }
 230.    
 231.        /**
 232.         * Remove features no longer in the source from the current
 233.         * classifier in the model.
 234.         */
 235.        public void removeObsoleteFeatures() {
 236.            if (obsoleteFeatures == null) {
 237.                return;
 238.            }
 239.            for (Object feature : obsoleteFeatures) {
 240.                Model.getCoreHelper().removeFeature(classifier, feature);
 241.                Model.getUmlFactory().delete(feature);
 242.            }
 243.        }
 244.    
 245.        /**
 246.         * Remove inner classes no longer in the source from the current
 247.         * classifier in the model.
 248.         */
 249.        public void removeObsoleteInnerClasses() {
 250.            if (obsoleteInnerClasses == null) {
 251.            return;
 252.        }
 253.        for (Object element : obsoleteInnerClasses) {
 254.            if (Model.getFacade().isAClassifier(element)) {
 255.            Model.getUmlFactory().delete(element);
 256.            }
 257.        }
 258.        }
 259.    
 260.        /**
 261.         * Tell the parse state that a feature belongs to the current
 262.         * classifier.
 263.         *
 264.         * @param feature The feature.
 265.         */
 266.        public void feature(Object feature) {
 267.        obsoleteFeatures.remove(feature);
 268.        }
 269.    
 270.        /**
 271.         * Get a feature from the current classifier not yet modeled.
 272.         *
 273.         * @param name The name of the feature.
 274.         * @return The found feature, null if not found.
 275.         */
 276.        public Object getFeature(String name) {
 277.            for (Object mFeature : obsoleteFeatures) {
 278.            if (name.equals(Model.getFacade().getName(mFeature))) {
 279.            return mFeature;
 280.            }
 281.        }
 282.        return null;
 283.        }
 284.    
 285.        /**
 286.         * Get a features from the current classifier not yet modeled.
 287.         *
 288.         * @param name The name of the feature.
 289.         * @return The collection of found features
 290.         */
 291.        public Collection getFeatures(String name) {
 292.            ArrayList list = new ArrayList();
 293.            for (Object mFeature : obsoleteFeatures) {
 294.            if (name.equals(Model.getFacade().getName(mFeature))) {
 295.            list.add(mFeature);
 296.            }
 297.        }
 298.        return list;
 299.        }
 300.    
 301.        /**
 302.         * Get a method from the current classifier not yet modeled.
 303.         *
 304.         * @param name The name of the method.
 305.         * @return The found method, null if not found.
 306.         */
 307.        public Object getMethod(String name) {
 308.            for (Object mFeature : obsoleteFeatures) {
 309.            if (Model.getFacade().isAMethod(mFeature)
 310.            && name.equals(Model.getFacade().getName(mFeature))) {
 311.            return mFeature;
 312.            }
 313.        }
 314.        return null;
 315.        }
 316.    
 317.        /**
 318.         * Get a operation from the current classifier not yet modeled.
 319.         *
 320.         * @param name The name of the operation.
 321.         * @return The found operation, null if not found.
 322.         */
 323.        public Object getOperation(String name) {
 324.            for (Object feature : obsoleteFeatures) {
 325.            if (Model.getFacade().isAOperation(feature)
 326.                        && name.equals(Model.getFacade().getName(feature))) {
 327.            return feature;
 328.            }
 329.        }
 330.        return null;
 331.        }
 332.    
 333.        /**
 334.         * Get a attribute from the current classifier not yet modeled.
 335.         *
 336.         * @param name The name of the attribute.
 337.         * @return The found attribute, null if not found.
 338.         */
 339.        public Object getAttribute(String name) {
 340.            for (Object feature : obsoleteFeatures) {
 341.                if (Model.getFacade().isAAttribute(feature)
 342.                        && name.equals(Model.getFacade().getName(feature))) {
 343.                    return feature;
 344.                }
 345.            }
 346.            return null;
 347.        }
