 213.    
 214.            logger.infof("String to be executed is: " + toExecute);
 215.    
 216.            Runtime.getRuntime().exec(toExecute).waitFor();
 217.            symbolic = new File(warDeployment.getAbsolutePath(), "symbolic.html");
 218.    
 219.    
 220.            logger.infof("Absolute path of symbolic file is located at %s. Checking to see if the file really exists.",

 197.        private static void createSymlink() throws IOException, InterruptedException {
 198.            // TODO: navssurtani: once AS7 is on a minimum of Java 7 then we can change the approach used to create the symlink.
 199.            File index = new File(warDeployment, "index.html");
 200.            logger.infof("Path to index file is: " + index.getAbsolutePath());
 201.            Assert.assertTrue(index.exists());
 202.            // Now set up the information to create the symlink.
 203.            String toExecute;
 204.    
 205.            if (SystemUtils.IS_OS_WINDOWS) {
 206.                logger.infof("Windows based OS detected.");
 207.                toExecute = "mklink \\D " + index.getAbsolutePath() + " " + warDeployment.getAbsolutePath()
 208.                        + "\\symbolic.html";
 209.            } else {
 210.                logger.infof("UNIX based OS detected.");
 211.                toExecute = "ln -s " + index.getAbsolutePath() + " " + warDeployment.getAbsolutePath() + "/symbolic.html";
 212.            }
 213.    
 214.            logger.infof("String to be executed is: " + toExecute);
 215.    
 216.            Runtime.getRuntime().exec(toExecute).waitFor();
 217.            symbolic = new File(warDeployment.getAbsolutePath(), "symbolic.html");
 218.    
 219.    
 220.            logger.infof("Absolute path of symbolic file is located at %s. Checking to see if the file really exists.",
 221.                    symbolic.getAbsolutePath());
 222.            Assert.assertTrue(symbolic.exists());
 223.        }
