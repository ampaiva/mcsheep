 235.        /**
 236.         * Controls if result of reparsing contains certain parameters
 237.         * @param model
 238.         * @param params
 239.         */
 240.        public static void checkModelParams(ModelNode node, Properties params){
 241.            String str;
 242.    
 243.            StringBuffer sb = new StringBuffer();
 244.            String par,child;
 245.            Enumeration e = params.propertyNames();
 246.    
 247.            while (e.hasMoreElements()) {
 248.                str=(String)e.nextElement();
 249.                par=params.getProperty(str);
 250.                if (node.get(str) == null) {
 251.                    sb.append("Parameter <"+str+"> is not set, but must be set to '"+par+"' \n");
 252.                }
 253.                else{
 254.                    child= node.get(str).asString();
 255.                    if (!child.equals(par)) {
 256.                        sb.append("Parameter <"+str+"> is set to '"+child+"', but must be set to '"+par+"' \n");
 257.                    }
 258.                }
 259.            }
 260.            if (sb.length()>0) Assert.fail("There are parsing errors:\n"+sb.toString()+"Parsed configuration:\n"+node);
 261.        }
