 235.        /**
 236.         * Handle a refresh of the style panel after the fig has moved.


 237.         *
 238.         * Warning. There is a circular trap here. Editing the
 239.         * boundary box will also trigger a refresh, and so we reset the
 240.         * boundary box, which causes funny behaviour (the cursor keeps
 241.         * jumping to the end of the text).
 242.         *
 243.         * The solution is to not reset the boundary box field if the boundaries
 244.         * have not changed.


 245.         */
 246.        public void refresh() {
 247.            if (TargetManager.getInstance().getTargets().size() > 1) {
 248.                // See issue 6109 - if we have multiple targets this method
 249.                // can result in a feedback problem where selecting a target
 250.                // changes the selection colour in the combo and as a result
 251.                // that trigger a change of colour of all selected Figs
 252.                return;
 253.            }
 254.            Fig target = getPanelTarget();
 255.             //TODO: How about FigAssociationClass?
 256.            if (target instanceof FigEdgeModelElement) {
 257.                hasEditableBoundingBox(false);
 258.            } else {
 259.                hasEditableBoundingBox(true);
 260.            }
 261.            if (target == null) {
 262.                return;
 263.            }
 264.    
 265.            // The boundary box as held in the target fig, and as listed in
 266.            // the
 267.            // boundary box style field (null if we don't have anything
 268.            // valid)
 269.    
 270.            Rectangle figBounds = target.getBounds();
 271.            Rectangle styleBounds = parseBBox();
 272.    
 273.            // Only reset the text if the two are not the same (i.e the fig
 274.            // has
 275.            // moved, rather than we've just edited the text, when
 276.            // setTargetBBox()
 277.            // will have made them the same). Note that styleBounds could
 278.            // be null,
 279.            // so we do the test this way round.
 280.    
 281.            if (!(figBounds.equals(styleBounds))) {
 282.                bboxField.setText(figBounds.x + "," + figBounds.y + ","
 283.                        + figBounds.width + "," + figBounds.height);
 284.            }
 285.    
 286.            // Change the fill colour
 287.    
 288.            if (target.isFilled()) {
 289.                Color c = target.getFillColor();
 290.                fillField.setSelectedItem(c);
 291.                if (c != null && !fillField.getSelectedItem().equals(c)) {
 292.                    fillField.insertItemAt(c, fillField.getItemCount() - 1);
 293.                    fillField.setSelectedItem(c);
 294.                }
 295.            } else {
 296.                fillField.setSelectedIndex(0);
 297.            }

 322.        /**
 323.         * Change the bounds of the target fig. Called whenever the bounds box is
 324.         * edited. 


 325.         *
 326.         * Format of the bounds is four integers representing x, y, width and height
 327.         * separated by spaces or commas. An empty field is treated as no change and
 328.         * leading and trailing spaces are ignored. 


 329.         *
 330.         * Note. There is a note in the old code that more work might be

 332.         * could ever be.

 334.        protected void setTargetBBox() {
 335.            Fig target = getPanelTarget();
 336.            // Can't do anything if we don't have a fig.
 337.            if (target == null) {
 338.                return;
 339.            }
 340.            // Parse the boundary box text. Null is
 341.            // returned if it is empty or
 342.            // invalid, which causes no change. Otherwise we tell
 343.            // GEF we are making
 344.            // a change, make the change and tell GEF we've
 345.            // finished.
 346.            Rectangle bounds = parseBBox();
 347.            if (bounds == null) {
 348.                return;
 349.            }
 350.    
 351.            if (!target.getBounds().equals(bounds)) {
 352.                target.setBounds(bounds.x, bounds.y, bounds.width,
 353.                        bounds.height);
 354.                target.endTrans();
 355.            }
 356.        }

 358.        /**
 359.         * Parse the boundary box string and return the rectangle it
 360.         * represents.


 361.         *
 362.         * The syntax are four integers separated by spaces or commas. We
 363.         * ignore leading and trailing blanks.


 364.         *
 365.         * If we have the empty string we return null.


 366.         *
 367.         * If we fail to parse, then we return null and print
 368.         * out a rude message.


 369.         *
 370.         * @return The size of the box, or null if the bounds string
 371.         *         is empty or invalid.
 372.         */
 373.        protected Rectangle parseBBox() {
 374.            Fig target = getPanelTarget();
 375.            // Get the text in the field, and don't do anything if the
 376.            // field is
 377.            // empty.
 378.            String bboxStr = bboxField.getText().trim();
 379.            if (bboxStr.length() == 0) {
 380.                return null;
 381.            } // Parse the string as if
 382.            // possible
 383.            Rectangle res = new Rectangle();
 384.            java.util.StringTokenizer st =
 385.                new java.util.StringTokenizer(bboxStr, ", ");
 386.            try {
 387.                boolean changed = false;
 388.                if (!st.hasMoreTokens()) {
 389.                    return target.getBounds();
 390.                }
 391.                res.x = Integer.parseInt(st.nextToken());
 392.                if (!st.hasMoreTokens()) {
 393.                    res.y = target.getBounds().y;
 394.                    res.width = target.getBounds().width;
 395.                    res.height = target.getBounds().height;
 396.                    return res;
 397.                }
 398.                res.y = Integer.parseInt(st.nextToken());
 399.                if (!st.hasMoreTokens()) {
 400.                    res.width = target.getBounds().width;
 401.                    res.height = target.getBounds().height;
 402.                    return res;
 403.                }

 408.                    res.width = 6000 - res.x;
 409.                    changed = true;
 410.                }
 411.                if (!st.hasMoreTokens()) {
 412.                    res.width = target.getBounds().width;
 413.                    return res;
 414.                }
 415.                res.height = Integer.parseInt(st.nextToken());
 416.                // TODO: What does the magic number 6000 represent here?
 417.                // Put it in an explanatory literal! - tfm - 20071205
 418.                if ((res.height + res.y) > 6000) {
 419.                    res.height = 6000 - res.y;
 420.                    changed = true;
 421.                }
 422.                if (res.x < 0 || res.y < 0) {
 423.                    // TODO: This exception will be thrown during autoscrolling
 424.                    // when the edge of the canvas is reached causing either
 425.                    // the width or height to be "adjusted" to a negative value
 426.                    LOG.warn("Part of bounding box is off screen " + res);
 427.                }
 428.                if (res.width < 0 || res.height < 0) {
 429.                    // TODO: This exception will be thrown during autoscrolling
 430.                    // when the edge of the canvas is reached causing either
 431.                    // the width or height to be "adjusted" to a negative value
 432.                    throw new IllegalArgumentException(
 433.                            "Bounding box has negative size " + res);
 434.                }
 435.                if (changed) {
 436.                    StringBuffer sb = new StringBuffer();
 437.                    sb.append(Integer.toString(res.x));
 438.                    sb.append(",");
 439.                    sb.append(Integer.toString(res.y));
 440.                    sb.append(",");
 441.                    sb.append(Integer.toString(res.width));
 442.                    sb.append(",");
 443.                    sb.append(Integer.toString(res.height));
 444.                    bboxField.setText(sb.toString());
 445.                }
 446.            } catch (NumberFormatException ex) {
 447.                bboxField.setBackground(Color.RED);
 448.                return null;
 449.            }
 450.            catch (IllegalArgumentException iae) {
 451.                bboxField.setBackground(Color.RED);
 452.                return null;
 453.            }
 454.            bboxField.setBackground(null);
 455.            return res;
 456.        }

 159.            textColorField.addItem(Color.red);
 160.            textColorField.addItem(Color.blue);
 161.            textColorField.addItem(Color.green);
 162.            textColorField.addItem(Color.orange);
 163.            textColorField.addItem(Color.pink);
 164.            textColorField.addItem(getCustomItemName());
 165.

 168.        /**
 169.         * Refresh the style panel when the Fig has been altered.
 170.         *
 171.         * @see org.argouml.ui.TabTarget#refresh()
 172.         */
 173.        public void refresh() {
 174.            if (TargetManager.getInstance().getTargets().size() > 1) {
 175.                // See issue 6109 - if we have multiple targets this method
 176.                // can result in a feedback problem where selecting a target
 177.                // changes the selection colour in the combo and as a result
 178.                // that trigger a change of colour of all selected Figs
 179.                return;
 180.            }
 181.            super.refresh();
 182.            FigText ft = (FigText) getPanelTarget();
 183.            if (ft == null) {
 184.                return;
 185.            }
 186.            String fontName = ft.getFontFamily().toLowerCase();
 187.            int size = ft.getFontSize();
 188.            String styleName = STYLES[0];
 189.    
 190.            fontField.setSelectedItem(fontName);
 191.            sizeField.setSelectedItem(Integer.valueOf(size));
 192.            if (ft.getBold()) {
 193.                styleName = STYLES[1];
 194.            }
 195.            if (ft.getItalic()) {
 196.                styleName = STYLES[2];
 197.            }
 198.            if (ft.getBold() && ft.getItalic()) {
 199.                styleName = STYLES[3];
 200.            }
 201.            styleField.setSelectedItem(styleName);
 202.    
 203.            String justName = JUSTIFIES[0];
 204.            int justCode = ft.getJustification();
 205.            if (justCode >= 0 && justCode <= JUSTIFIES.length) {
 206.                justName = JUSTIFIES[justCode];
 207.            }
 208.            justField.setSelectedItem(justName);
 209.    
 210.            Color c = ft.getTextColor();
 211.            textColorField.setSelectedItem(c);
 212.            if (c != null && !textColorField.getSelectedItem().equals(c)) {
 213.                textColorField.insertItemAt(c, textColorField.getItemCount() - 1);
 214.                textColorField.setSelectedItem(c);
 215.            }
 216.    
 217.            if (ft.isFilled()) {
 218.                Color fc = ft.getFillColor();
 219.                getFillField().setSelectedItem(fc);
 220.                if (fc != null && !getFillField().getSelectedItem().equals(fc)) {
 221.                    getFillField().insertItemAt(fc,
 222.                                                getFillField().getItemCount() - 1);
 223.                    getFillField().setSelectedItem(fc);
 224.                }
 225.            } else {
 226.                getFillField().setSelectedIndex(0);
 227.            }
 228.        }

 230.        /**
 231.         * Set the font of the text element to the selected value.
 232.         */
 233.        protected void setTargetFont() {
 234.            if (getPanelTarget() == null) {
 235.                return;
 236.            }
 237.            String fontStr = (String) fontField.getSelectedItem();
 238.            if (fontStr.length() == 0) {
 239.                return;
 240.            }
 241.            ((FigText) getPanelTarget()).setFontFamily(fontStr);
 242.            getPanelTarget().endTrans();
 243.        }
 244.    
 245.        /**
 246.         * Change font size of the text element according to the selected value.
 247.         */
 248.        protected void setTargetSize() {
 249.            if (getPanelTarget() == null) {
 250.                return;
 251.            }
 252.            Integer size = (Integer) sizeField.getSelectedItem();
 253.            ((FigText) getPanelTarget()).setFontSize(size.intValue());
 254.            getPanelTarget().endTrans();
 255.        }
 256.    
 257.        /**
 258.         * Change style of the text element (bold/italic) according to the selected
 259.         * value.
 260.         */
 261.        protected void setTargetStyle() {
 262.            if (getPanelTarget() == null) {
 263.                return;
 264.            }
 265.            String styleStr = (String) styleField.getSelectedItem();
 266.            if (styleStr == null) {
 267.                return;
 268.            }
 269.            boolean bold = (styleStr.indexOf("Bold") != -1);
 270.            boolean italic = (styleStr.indexOf("Italic") != -1);
 271.            ((FigText) getPanelTarget()).setBold(bold);
 272.            ((FigText) getPanelTarget()).setItalic(italic);
 273.            getPanelTarget().endTrans();
 274.        }

 276.        /**
 277.         * Change the justification of the text according to the selected value.
 278.         */
 279.        protected void setTargetJustification() {
 280.            if (getPanelTarget() == null) {
 281.                return;
 282.            }
 283.            String justStr = (String) justField.getSelectedItem();
 284.            if (justStr == null) {
 285.                return;
 286.            }
 287.            ((FigText) getPanelTarget()).setJustificationByName(justStr);
 288.            getPanelTarget().endTrans();
 289.        }

 294.         * Change the color of the text element according to the selected value.
 295.         */
 296.        protected void setTargetTextColor() {
 297.            if (getPanelTarget() == null) {
 298.                return;
 299.            }
 300.            Object c = textColorField.getSelectedItem();

 303.            }

 305.        }

 307.        /*
 308.         * @see java.awt.event.ItemListener#itemStateChanged(java.awt.event.ItemEvent)
 309.         */
 310.        public void itemStateChanged(ItemEvent e) {
 311.            Object src = e.getSource();
 312.            Fig target = getPanelTarget();
 313.            if (e.getStateChange() == ItemEvent.SELECTED
 314.                    && target instanceof FigText) {
 315.                if (src == fontField) {
 316.                    setTargetFont();
 317.                } else if (src == sizeField) {
 318.                    setTargetSize();
 319.                } else if (src == styleField) {
 320.                    setTargetStyle();
 321.                } else if (src == justField) {
 322.                    setTargetJustification();
 323.                } else if (src == textColorField) {
 324.                    if (e.getItem() == getCustomItemName()) {
 325.                        handleCustomColor(textColorField, 
 326.                                "label.stylepane.custom-text-color",
 327.                                ((FigText) target).getTextColor());
 328.                    }
 329.                    setTargetTextColor();
 330.                } else {
 331.                    super.itemStateChanged(e);
 332.                }
 333.            }
 334.        }
