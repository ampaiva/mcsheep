 271.        public static Cookie copyCookie(Cookie toCopy, String targetServer) {
 272.            // Parse the target server down to a domain name
 273.            int index = targetServer.indexOf("://");
 274.            if (index > -1) {
 275.                targetServer = targetServer.substring(index + 3);
 276.            }
 277.            // JBAS-8540
 278.            // need to be able to parse IPv6 URLs which have enclosing brackets
 279.            // HttpClient 3.1 creates cookies which oinclude the square brackets
 280.            // index = targetServer.indexOf(":");
 281.            index = targetServer.lastIndexOf(":");
 282.            if (index > -1) {
 283.                targetServer = targetServer.substring(0, index);
 284.            }
 285.            index = targetServer.indexOf("/");
 286.            if (index > -1) {
 287.                targetServer = targetServer.substring(0, index);
 288.            }
 289.    
 290.            // Cookie copy = new Cookie(targetServer, toCopy.getName(), toCopy.getValue(), "/", null, false);
 291.            BasicClientCookie copy = new BasicClientCookie(toCopy.getName(), toCopy.getValue());
 292.            copy.setDomain(targetServer);
 293.            return copy;
 294.        }

 370.        public static void applyUpdates(final List updates, final ModelControllerClient client) throws Exception {
 371.            for (ModelNode update : updates) {
 372.                log.info("+++ Update on " + client + ":\n" + update.toString());
 373.                ModelNode result = client.execute(new OperationBuilder(update).build());
 374.                if (result.hasDefined("outcome") && "success".equals(result.get("outcome").asString())) {
 375.                    if (result.hasDefined("result"))
 376.                        log.info(result.get("result"));
 377.                } else if (result.hasDefined("failure-description")) {
 378.                    throw new RuntimeException(result.get("failure-description").toString());
 379.                } else {
 380.                    throw new RuntimeException("Operation not successful; outcome = " + result.get("outcome"));
 381.                }
 382.            }
 383.        }
 384.    
 385.        // Reload operation is not handled well by Arquillian
 386.        // See ARQ-791: JMX: Arquillian is unable to reconnect to JMX server if the connection is lost
 387.        public static void restartServer(final ModelControllerClient client) {
 388.            try {
 389.                applyUpdates(Arrays.asList(createOpNode(null, "reload")), client);
 390.            } catch (Exception e) {
 391.                throw new RuntimeException("Restart operation not successful. " + e.getMessage());
 392.            }
 393.            try {
 394.                RetryTaskExecutor rte = new RetryTaskExecutor();
 395.                rte.retryTask(new Callable() {
 396.                    public Boolean call() throws Exception {
 397.                        ModelNode readOp = createOpNode(null, READ_ATTRIBUTE_OPERATION);
 398.                        readOp.get("name").set("server-state");
 399.                        ModelNode result = client.execute(new OperationBuilder(readOp).build());
 400.                        if (result.hasDefined("outcome") && "success".equals(result.get("outcome").asString())) {
 401.                            if ((result.hasDefined("result")) && (result.get("result").asString().equals("running")))
 402.                                return true;
 403.                        }
 404.                        log.info("Server is down.");
 405.                        throw new Exception("Connector not available.");
 406.                    }
 407.                });
 408.            } catch (TimeoutException e) {
 409.                throw new RuntimeException("Timeout on restart operation. " + e.getMessage());
 410.            }
 411.            log.info("Server is up.");
 412.        }

 414.        public static DefaultHttpClient relaxedCookieHttpClient() {
