 118.        /**
 119.         * Start a server (A) which has a remote outbound connection to another server (B). Server (B) is down.
 120.         * Deploy (X) to server A. X contains a jboss-ejb-client.xml pointing to server B (which is down). The deployment
 121.         * must succeed. However invocations on beans which depend on server B should fail.
 122.         * Then start server B and deploy Y to it. Invoke again on server A beans which depend on server B and this time
 123.         * they should pass
 124.         *
 125.         * @throws Exception
 126.         */
 127.        @Test
 128.        public void testRemoteServerStartsLate() throws Exception {
 129.            // First start the server which has a remote-outbound-connection
 130.            this.container.start(JBOSSAS_WITH_REMOTE_OUTBOUND_CONNECTION);
 131.            boolean defaultContainerStarted = false;
 132.            try {
 133.                // deploy a deployment which contains jboss-ejb-client.xml that contains an EJB receiver pointing
 134.                // to a server which hasn't yet started. Should succeed without throwing deployment error
 135.                this.deployer.deploy(DEPLOYMENT_WITH_JBOSS_EJB_CLIENT_XML);
 136.                // To make sure deployment succeeded and invocations are possible, call an independent bean
 137.                final RemoteEcho independentBean = (RemoteEcho) context.lookup("ejb:/" + SERVER_ONE_MODULE_NAME + "//" + IndependentBean.class.getSimpleName() + "!" + RemoteEcho.class.getName());
 138.                final String msg = "Hellooooo!";
 139.                final String echoFromIndependentBean = independentBean.echo(msg);
 140.                Assert.assertEquals("Unexpected echo from independent bean", msg, echoFromIndependentBean);
 141.    
 142.                // now try invoking the EJB (which calls a delegate bean on other server) on this server. 
 143.                // should fail with no EJB receivers, since the other server
 144.                // which can handle the delegate bean invocation hasn't yet started.
 145.                try {
 146.                    final RemoteEcho dependentBean = (RemoteEcho) context.lookup("ejb:/" + SERVER_ONE_MODULE_NAME + "//" + EchoOnServerOne.class.getSimpleName() + "!" + RemoteEcho.class.getName());
 147.                    final String echoBeforeOtherServerStart = dependentBean.echo(msg);
 148.                    Assert.fail("Invocation on bean when was expected to fail due to other server being down");
 149.                } catch (Exception e) {
 150.                    // expected 
 151.                    logger.info("Got the expected exception on invoking a bean when other server was down", e);
 152.                }
 153.                // now start the main server
 154.                this.container.start(DEFAULT_JBOSSAS);
 155.                defaultContainerStarted = true;
 156.                // deploy to this container
 157.                this.deployer.deploy(DEFAULT_AS_DEPLOYMENT);
 158.    
 159.                // now invoke the EJB (which had failed earlier)
 160.                final RemoteEcho dependentBean = (RemoteEcho) context.lookup("ejb:/" + SERVER_ONE_MODULE_NAME + "//" + EchoOnServerOne.class.getSimpleName() + "!" + RemoteEcho.class.getName());
 161.                final String echoAfterOtherServerStarted = dependentBean.echo(msg);
 162.                Assert.assertEquals("Unexpected echo from bean", EchoOnServerTwo.ECHO_PREFIX + msg, echoAfterOtherServerStarted);
 163.    
 164.            } finally {
 165.                try {
 166.                    this.deployer.undeploy(DEPLOYMENT_WITH_JBOSS_EJB_CLIENT_XML);
 167.                    this.container.stop(JBOSSAS_WITH_REMOTE_OUTBOUND_CONNECTION);
 168.                } catch (Exception e) {
 169.                    logger.debug("Exception during container shutdown", e);
 170.                }
 171.                if (defaultContainerStarted) {
 172.                    try {
 173.                        this.deployer.undeploy(DEFAULT_AS_DEPLOYMENT);
 174.                        this.container.stop(DEFAULT_JBOSSAS);
 175.                    } catch (Exception e) {
 176.                        logger.debug("Exception during container shutdown", e);
 177.                    }
 178.                }
 179.            }
 180.        }

 182.        /**
 183.         * Start a server (A) which has a remote outbound connection to another server (B). Also start Server (B).
 184.         * Deploy (X) to server A. X contains a jboss-ejb-client.xml pointing to server B. The deployment and invocations
 185.         * must succeed.
 186.         * Now stop server (B). Invoke again on the bean. Invocation should fail since server B is down. Now
 187.         * restart server B and invoke again on the bean. Invocation should pass since the EJB client context is
 188.         * expected to reconnect to the restarted server B.
 189.         *
 190.         * @throws Exception
 191.         */
 192.        @Test
 193.        public void testRemoteServerRestarts() throws Exception {
 194.            // Start the main server
 195.            this.container.start(DEFAULT_JBOSSAS);
 196.            // deploy to this container
 197.            this.deployer.deploy(DEFAULT_AS_DEPLOYMENT);
 198.    
 199.            // Now start the server which has a remote-outbound-connection
 200.            this.container.start(JBOSSAS_WITH_REMOTE_OUTBOUND_CONNECTION);
 201.            this.deployer.deploy(DEPLOYMENT_WITH_JBOSS_EJB_CLIENT_XML);
 202.    
 203.            boolean defaultContainerStarted = true;
 204.            try {
 205.                // To make sure deployment succeeded and invocations are possible, call an independent bean
 206.                final RemoteEcho independentBean = (RemoteEcho) context.lookup("ejb:/" + SERVER_ONE_MODULE_NAME + "//" + IndependentBean.class.getSimpleName() + "!" + RemoteEcho.class.getName());
 207.                final String msg = "Hellooooo!";
 208.                final String echoFromIndependentBean = independentBean.echo(msg);
 209.                Assert.assertEquals("Unexpected echo from independent bean", msg, echoFromIndependentBean);
 210.    
 211.                // now try invoking the EJB (which calls a delegate bean on other server) on this server.
 212.                // should fail with no EJB receivers, since the other server
 213.                // which can handle the delegate bean invocation hasn't yet started.
 214.                final RemoteEcho dependentBean = (RemoteEcho) context.lookup("ejb:/" + SERVER_ONE_MODULE_NAME + "//" + EchoOnServerOne.class.getSimpleName() + "!" + RemoteEcho.class.getName());
 215.                final String echoBeforeShuttingDownServer = dependentBean.echo(msg);
 216.                Assert.assertEquals("Unexpected echo from bean", EchoOnServerTwo.ECHO_PREFIX + msg, echoBeforeShuttingDownServer);
 217.    
 218.                // now stop the main server
 219.                this.container.stop(DEFAULT_JBOSSAS);
 220.                defaultContainerStarted = false;
 221.    
 222.                try {
 223.                    final String echoAfterServerShutdown = dependentBean.echo(msg);
 224.                    Assert.fail("Invocation on bean when was expected to fail due to other server being down");
 225.                } catch (Exception e) {
 226.                    // expected
 227.                    logger.info("Got the expected exception on invoking a bean when other server was down", e);
 228.                }
 229.    
 230.                // now restart the main server
 231.                this.container.start(DEFAULT_JBOSSAS);
 232.                defaultContainerStarted = true;
 233.    
 234.                final String echoAfterServerRestart = dependentBean.echo(msg);
 235.                Assert.assertEquals("Unexpected echo from bean after server restart", EchoOnServerTwo.ECHO_PREFIX + msg, echoAfterServerRestart);
 236.    
 237.    
 238.            } finally {
 239.                try {
 240.                    this.deployer.undeploy(DEPLOYMENT_WITH_JBOSS_EJB_CLIENT_XML);
 241.                    this.container.stop(JBOSSAS_WITH_REMOTE_OUTBOUND_CONNECTION);
 242.                } catch (Exception e) {
 243.                    logger.debug("Exception during container shutdown", e);
 244.                }
 245.                if (defaultContainerStarted) {
 246.                    try {
 247.                        this.deployer.undeploy(DEFAULT_AS_DEPLOYMENT);
 248.                        this.container.stop(DEFAULT_JBOSSAS);
 249.                    } catch (Exception e) {
 250.                        logger.debug("Exception during container shutdown", e);
 251.                    }
 252.                }
 253.            }
 254.    
 255.        }

 257.        /**
 258.         * Sets up the EJB client context to use a selector which processes and sets up EJB receivers
 259.         * based on this testcase specific jboss-ejb-client.properties file
 260.         *
 261.         * @return
 262.         * @throws java.io.IOException
 263.         */
 264.        private static ContextSelector setupEJBClientContextSelector() throws IOException {
 265.            // setup the selector
 266.            final String clientPropertiesFile = "org/jboss/as/test/manualmode/ejb/client/outbound/connection/jboss-ejb-client.properties";
 267.            final InputStream inputStream = RemoteOutboundConnectionReconnectTestCase.class.getClassLoader().getResourceAsStream(clientPropertiesFile);
 268.            if (inputStream == null) {
 269.                throw new IllegalStateException("Could not find " + clientPropertiesFile + " in classpath");
 270.            }
 271.            final Properties properties = new Properties();
 272.            properties.load(inputStream);
 273.            final EJBClientConfiguration ejbClientConfiguration = new PropertiesBasedEJBClientConfiguration(properties);
 274.            final ConfigBasedEJBClientContextSelector selector = new ConfigBasedEJBClientContextSelector(ejbClientConfiguration);
 275.    
 276.            return EJBClientContext.setSelector(selector);
 277.        }
