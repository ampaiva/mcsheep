  98.      private boolean populate(boolean forcefully) throws PopulateException {
  99.        XMLReader reader = null;
 100.        boolean alreadyPopulated = false;
 101.        try {
 102.          SAXParserFactory parserFactory = SAXParserFactory.newInstance();
 103.          //parserFactory.setValidating(true);
 104.          parserFactory.setNamespaceAware(true);
 105.          reader = parserFactory.newSAXParser().getXMLReader();
 106.          InventoryPopulator inventoryPopulator = new InventoryPopulator();
 107.          if (!forcefully) {
 108.            try {
 109.              alreadyPopulated = inventoryPopulator.check();
 110.            } catch (PopulateException exception) {}
 111.            //System.err.println("Already populated: " + alreadyPopulated);
 112.          }
 113.          if (forcefully || !alreadyPopulated) {
 114.            try {
 115.              inventoryPopulator.setup(reader).parse(new InputSource(getResource(populateDataPath)));
 116.            } catch (Exception exception) {
 117.              throw new PopulateException(exception);
 118.            }
 119.          }
 120.        } catch (Exception exception) {
 121.          throw new PopulateException(exception);
 122.        }
 123.        return forcefully || !alreadyPopulated;
 124.      }

 126.      private String getResource(String path) throws IOException {
 127.        String url;

 130.        } catch (MalformedURLException exception) {
 131.          URL u = getServletContext().getResource(path);
 132.          url = u != null ? u.toString() : path;
 133.        }
 134.        //System.err.println("Made " + url + " from " + path);
 135.        return url;
 136.      }
