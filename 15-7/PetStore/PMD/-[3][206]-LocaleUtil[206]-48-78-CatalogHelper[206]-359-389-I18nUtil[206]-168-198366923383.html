 359.        private Locale getLocaleFromString(String localeString) {
 360.            if (localeString == null) return null;
 361.            if (localeString.toLowerCase().equals("default")) return Locale.getDefault();
 362.            int languageIndex = localeString.indexOf('_');

 375.            String language = localeString.substring(0, languageIndex);
 376.            String variant = null;
 377.            if (variantIndex  != -1) {
 378.
