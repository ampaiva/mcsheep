 162.        @Override
 163.        public boolean predicate2(Object dm, Designer dsgr) {
 164.        if (!(Model.getFacade().isPrimaryObject(dm))) {
 165.                return NO_PROBLEM;
 166.            }
 167.    
 168.            if (!(Model.getFacade().isAModelElement(dm))) {
 169.                return NO_PROBLEM;
 170.            }
 171.            String meName = Model.getFacade().getName(dm);
 172.            if (meName == null || meName.equals("")) {
 173.                return NO_PROBLEM;
 174.            }
 175.            String nameStr = meName;
 176.            if (nameStr == null || nameStr.length() == 0) {
 177.                return NO_PROBLEM;
 178.            }
 179.    
 180.        if (isBuiltin(nameStr)) {
 181.                return NO_PROBLEM;
 182.            }
 183.    
 184.            for (String name : names) {
 185.                if (name.equalsIgnoreCase(nameStr)) {
 186.                    return PROBLEM_FOUND;
 187.                }
 188.            }
 189.    
 190.            return NO_PROBLEM;
 191.        }
