  79.        /*
  80.         * Build the panel to be used for our settings tab.
  81.         */
  82.        private JPanel buildPanel() {
  83.            LOG.debug("SettingsTabJava being created...");
  84.            JPanel top = new JPanel();
  85.            top.setLayout(new BorderLayout());
  86.            JPanel panel = new JPanel();
  87.            panel.setLayout(new GridBagLayout());
  88.    
  89.            GridBagConstraints constraints = new GridBagConstraints();
  90.            constraints.anchor = GridBagConstraints.WEST;
  91.            constraints.fill = GridBagConstraints.HORIZONTAL;
  92.            constraints.gridy = GridBagConstraints.RELATIVE;
  93.            constraints.gridx = 0;
  94.            constraints.gridwidth = 1;
  95.            constraints.gridheight = 1;
  96.            constraints.weightx = 1.0;
  97.            constraints.insets = new Insets(0, 30, 0, 5);
  98.    
  99.            // adds indent width spinner
 100.            JLabel label = new JLabel(Translator.localize("java.indent"));
 101.            // The actual value is loaded in handleSettingsTabRefresh()
 102.            Integer spinVal = Integer.valueOf(4);
 103.            Integer spinMin = Integer.valueOf(0);
 104.            Integer spinStep = Integer.valueOf(1);
 105.            indent = new JSpinner(
 106.                    new SpinnerNumberModel(spinVal, spinMin, null, spinStep));
 107.            label.setLabelFor(indent);
 108.    
 109.            JPanel indentPanel = new JPanel();
 110.            indentPanel.setLayout(new BoxLayout(indentPanel, BoxLayout.LINE_AXIS));
 111.            indentPanel.add(label);
 112.            indentPanel.add(Box.createRigidArea(new Dimension(5, 0)));
 113.            indentPanel.add(indent);
 114.            indentPanel.add(Box.createHorizontalGlue());
 115.            panel.add(indentPanel, constraints);
 116.    
 117.            verboseDocs = new JCheckBox(Translator.localize("java.verbose-docs"));
 118.            panel.add(verboseDocs, constraints);
 119.    
 120.            lfBeforeCurly = new JCheckBox(Translator
 121.                    .localize("java.lf-before-curly"));
 122.            panel.add(lfBeforeCurly, constraints);
 123.    
 124.            headerGuardUpperCase = new JCheckBox(Translator
 125.                    .localize("java.header-guard-case"));
 126.            panel.add(headerGuardUpperCase, constraints);
 127.    
 128.            headerGuardGUID = new JCheckBox(Translator
 129.                    .localize("java.header-guard-guid"));
 130.            panel.add(headerGuardGUID, constraints);
 131.    
 132.            // TODO: add more options
 133.    
 134.            top.add(panel, BorderLayout.NORTH);
 135.    
 136.            LOG.debug("SettingsTabJava created!");
 137.            return top;
 138.        }
