  61.    /**

  87.        public void actionPerformed(ActionEvent arg0) {
  88.            // get one of the default profile dirs, if available

  90.            ProfileManager profileManager = ProfileFacade.getManager();
  91.            StringBuffer path = new StringBuffer();
  92.            Collection dirs = profileManager.getSearchPathDirectories();
  93.            if (dirs != null && !dirs.isEmpty()) {
  94.                String s = (String) dirs.iterator().next();
  95.                path.append(s);
  96.                if (!(s.endsWith("/") || s.endsWith("\\"))) {
  97.                    path.append('/');
  98.                }
  99.            }
 100.            // save profile
 101.            path.append(Model.getFacade().getName(undeployedProfile));
 102.            File f = saveProfile(path.toString());
 103.            if (f == null) {
 104.                return;
 105.            }
 106.            // register it as a user profile
 107.            try {
 108.                profileManager.registerProfile(new UserDefinedProfile(f,
 109.                    profileManager));
 110.            } catch (ProfileException e) {
 111.                LOG.warn("failed to load profile from file " + f.getPath(), e);
 112.            }
 113.        }
 114.

 149.        }
