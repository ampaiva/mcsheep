 504.        public void handleSettingsTabSave() {
 505.            List<Profile> toRemove = new ArrayList<Profile>();
 506.            List<Profile> usedItens = new ArrayList<Profile>();
 507.    
 508.            MutableComboBoxModel modelUsd = ((MutableComboBoxModel) defaultList
 509.                    .getModel());
 510.            MutableComboBoxModel modelDir = ((MutableComboBoxModel) directoryList
 511.                    .getModel());
 512.    
 513.            for (int i = 0; i < modelUsd.getSize(); ++i) {
 514.                usedItens.add((Profile) modelUsd.getElementAt(i));
 515.            }
 516.    
 517.            for (Profile profile 
 518.                    : ProfileFacade.getManager().getDefaultProfiles()) {
 519.                if (!usedItens.contains(profile)) {
 520.                    toRemove.add(profile);
 521.                }
 522.            }
 523.    
 524.            for (Profile profile : toRemove) {
 525.                ProfileFacade.getManager().removeFromDefaultProfiles(profile);
 526.            }
 527.    
 528.            for (Profile profile : usedItens) {
 529.                if (!ProfileFacade.getManager().getDefaultProfiles().contains(
 530.                        profile)) {
 531.                    ProfileFacade.getManager().addToDefaultProfiles(profile);
 532.                }
 533.            }
 534.    
 535.            List<String> toRemoveDir = new ArrayList<String>();
 536.            List<String> usedItensDir = new ArrayList<String>();
 537.    
 538.            for (int i = 0; i < modelDir.getSize(); ++i) {
 539.                usedItensDir.add((String) modelDir.getElementAt(i));
 540.            }
 541.    
 542.            for (String dirEntry : ProfileFacade.getManager()
 543.                    .getSearchPathDirectories()) {
 544.                if (!usedItensDir.contains(dirEntry)) {
 545.                    toRemoveDir.add(dirEntry);
 546.                }
 547.            }
 548.    
 549.            for (String dirEntry : toRemoveDir) {
 550.                ProfileFacade.getManager().removeSearchPathDirectory(dirEntry);
 551.            }
 552.    
 553.            for (String dirEntry : usedItensDir) {
 554.                if (!ProfileFacade.getManager().getSearchPathDirectories()
 555.                        .contains(dirEntry)) {
 556.                    ProfileFacade.getManager().addSearchPathDirectory(dirEntry);
 557.                }
 558.            }
 559.    
 560.        }

 215.        /*
 216.         * @see org.argouml.ui.GUISettingsTabInterface#handleSettingsTabRefresh()
 217.         */
 218.        public void handleSettingsTabRefresh() {
 219.            if (scope == Argo.SCOPE_APPLICATION) {
 220.                showBoldNames.setSelected(getBoolean(
 221.                        DiagramAppearance.KEY_SHOW_BOLD_NAMES));
 222.                selectedDiagramFontName = DiagramAppearance.getInstance()
 223.                        .getConfiguredFontName();
 224.                selectedDiagramFontSize = Configuration
 225.                        .getInteger(DiagramAppearance.KEY_FONT_SIZE);
 226.                /*
 227.                 * The next one defaults to TRUE, despite that this is
 228.                 * NOT compatible with older ArgoUML versions
 229.                 * (before 0.28?) that did
 230.                 * not have this setting - see issue 535
 231.                 */
 232.                hideBidirectionalArrows.setSelected(Configuration.getBoolean(
 233.                        DiagramAppearance.KEY_HIDE_BIDIRECTIONAL_ARROWS, true));
 234.                defaultShadowWidth.setSelectedIndex(Configuration.getInteger(
 235.                        DiagramAppearance.KEY_DEFAULT_SHADOW_WIDTH, 1));
 236.            }
 237.            if (scope == Argo.SCOPE_PROJECT) {
 238.                assert p != null;
 239.                ProjectSettings ps = p.getProjectSettings();
 240.                DiagramSettings ds = ps.getDefaultDiagramSettings();
 241.    
 242.                showBoldNames.setSelected(ds.isShowBoldNames());
 243.                selectedDiagramFontName = ds.getFontName();
 244.                selectedDiagramFontSize = ds.getFontSize();
 245.                hideBidirectionalArrows.setSelected(!
 246.                        ds.isShowBidirectionalArrows());
 247.                defaultShadowWidth.setSelectedIndex(
 248.                        ds.getDefaultShadowWidth());
 249.            }
 250.        }
 251.    
 252.        /**
 253.         * Get a boolean from the configuration.
 254.         *
 255.         * @param key a notation key.
 256.         * @return a boolean
 257.         */
 258.        protected static boolean getBoolean(ConfigurationKey key) {
 259.            return Configuration.getBoolean(key, false);
 260.        }

 262.        /*
 263.         * @see org.argouml.ui.GUISettingsTabInterface#handleSettingsTabSave()
 264.         */
 265.        public void handleSettingsTabSave() {
 266.            if (scope == Argo.SCOPE_APPLICATION) {
 267.                Configuration.setBoolean(DiagramAppearance.KEY_SHOW_BOLD_NAMES,
 268.                        showBoldNames.isSelected());
 269.                Configuration.setString(DiagramAppearance.KEY_FONT_NAME,
 270.                        selectedDiagramFontName);
 271.                Configuration.setInteger(DiagramAppearance.KEY_FONT_SIZE,
 272.                        selectedDiagramFontSize);
 273.                Configuration.setBoolean(
 274.                        DiagramAppearance.KEY_HIDE_BIDIRECTIONAL_ARROWS,
 275.                        hideBidirectionalArrows.isSelected());            
 276.                Configuration.setInteger(DiagramAppearance.KEY_DEFAULT_SHADOW_WIDTH,
 277.                        defaultShadowWidth.getSelectedIndex());
 278.            }
 279.            if (scope == Argo.SCOPE_PROJECT) {
 280.                assert p != null;
 281.                ProjectSettings ps = p.getProjectSettings();
 282.                final DiagramSettings ds = ps.getDefaultDiagramSettings();
 283.    
 284.                ds.setShowBoldNames(showBoldNames.isSelected());
 285.                ds.setFontName(selectedDiagramFontName);
 286.                ds.setFontSize(selectedDiagramFontSize);
 287.                ds.setDefaultShadowWidth(defaultShadowWidth.getSelectedIndex());
 288.                ds.setShowBidirectionalArrows(
 289.                        !hideBidirectionalArrows.isSelected());
 290.    
 291.                /* Update the diagram, and other users of these settings: */
 292.                Memento memento = new Memento() {
 293.                    public void redo() {
 294.                        ds.notifyOfChangedSettings();
 295.                    }
 296.    
 297.                    public void undo() {
 298.                        ds.notifyOfChangedSettings();
 299.                    }
 300.                };
 301.                // TODO: Undo should be managed externally or we should be given 
 302.                // an Undo manager to use (the project's) rather than using a 
 303.                // global one
 304.    //            if (DiagramUndoManager.getInstance().isGenerateMementos()) {
 305.    //                DiagramUndoManager.getInstance().addMemento(memento);
 306.    //            }
 307.                memento.redo();
 308.            }
 309.        }
