  74.         */

  79.        /*
  80.         * @see org.argouml.notation.providers.NotationProvider#parse(java.lang.Object, java.lang.String...
  81.         */
  82.        public void parse(Object modelElement, String text) {
  83.            try {
  84.                parseOperationFig(Model.getFacade().getOwner(modelElement), 
  85.                        modelElement, text);
  86.            } catch (ParseException pe) {
  87.                String msg = "statusmsg.bar.error.parsing.operation";

  90.                    Integer.valueOf(pe.getErrorOffset()),
  91.                };

  93.                        ArgoEventTypes.HELP_CHANGED, this,
  94.                        Translator.messageFormat(msg, args)));
  95.            }
  96.        }

  98.        /**
  99.         * Parse a string representing one ore more ';' separated operations. The
 100.         * case that a String or char contains a ';' (e.g. in an initializer) is
 101.         * handled, but not other occurences of ';'.
 102.         *
 103.         * @param classifier  Classifier The classifier the operation(s) belong to
 104.         * @param operation   Operation The operation on which the editing happened
 105.         * @param text The string to parse
 106.         * @throws ParseException for invalid input
 107.         */
 108.        public void parseOperationFig(
 109.                Object classifier,
 110.                Object operation,
 111.                String text) throws ParseException {
 112.    
 113.            if (classifier == null || operation == null) {
 114.                return;
 115.            }
 116.            ParseException pex = null;
 117.            int start = 0;
 118.            int end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 119.            Project currentProject =
 120.                ProjectManager.getManager().getCurrentProject();
 121.            if (end == -1) {
 122.                //no text? remove op!
 123.                currentProject.moveToTrash(operation);
 124.                return;
 125.            }
 126.            String s = text.substring(start, end).trim();
 127.            if (s.length() == 0) {
 128.                //no non-whitechars in text? remove op!
 129.                currentProject.moveToTrash(operation);
 130.                return;
 131.            }
 132.            parseOperation(s, operation);
 133.            int i = Model.getFacade().getFeatures(classifier).indexOf(operation);
 134.            // check for more operations (';' separated):
 135.            start = end + 1;
 136.            end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 137.            while (end > start && end <= text.length()) {
 138.                s = text.substring(start, end).trim();
 139.                if (s.length() > 0) {
 140.                    // yes, there are more:
 141.                    Object returnType = currentProject.getDefaultReturnType();
 142.                    Object newOp =
 143.                        Model.getCoreFactory()
 144.                            .buildOperation(classifier, returnType);
 145.                    if (newOp != null) {
 146.                        try {
 147.                            parseOperation(s, newOp);
 148.                            //newOp.setOwnerScope(op.getOwnerScope()); //
 149.                            //not needed in case of operation
 150.                            if (i != -1) {
 151.                                Model.getCoreHelper().addFeature(
 152.                                        classifier, ++i, newOp);
 153.                            } else {
 154.                                Model.getCoreHelper().addFeature(
 155.                                        classifier, newOp);
 156.                            }
 157.                        } catch (ParseException ex) {
 158.                            if (pex == null) {
 159.                                pex = ex;
 160.                            }
 161.                        }
 162.                    }
 163.                }
 164.                start = end + 1;
 165.                end = NotationUtilityUml.indexOfNextCheckedSemicolon(text, start);
 166.            }
 167.            if (pex != null) {
 168.                throw pex;
 169.            }
 170.        }
 171.    
 172.    
 173.        /**
 174.         * Parse a line of text and aligns the Operation to the specification
 175.         * given. The line should be on the following form:<ul>
 176.         * <li> visibility name (parameter list) : return-type-expression
 177.         * {property-string}
 178.         * </ul>
 179.         *
 180.         * All elements are optional and, if left unspecified, will preserve their
 181.         * old values.<p>
 182.         * 
 183.         * <em>Stereotypes</em> can be given between any element in the line on the
 184.         * form: <<stereotype1,stereotype2,stereotype3>><p>
 185.         *
 186.         * The following properties are recognized to have special meaning:
 187.         * abstract, concurrency, concurrent, guarded, leaf, query, root and
 188.         * sequential.<p>
 189.         *
 190.         * This syntax is compatible with the UML 1.3 spec.<p>
 191.         *
 192.         * (formerly visibility name (parameter list) : return-type-expression
 193.         * {property-string} ) (formerly 2nd: [visibility] [keywords] returntype
 194.         * name(params)[;] )
 195.         *
 196.         * @param s   The String to parse.
 197.         * @param op  The Operation to adjust to the specification in s.
 198.         * @throws ParseException
 199.         *             when it detects an error in the attribute string. See also
 200.         *             ParseError.getErrorOffset().
 201.         */
 202.        public void parseOperation(String s, Object op) throws ParseException {
 203.            MyTokenizer st;
 204.            boolean hasColon = false;
 205.            String name = null;
 206.            String parameterlist = null;
 207.            StringBuilder stereotype = null;
 208.            String token;
 209.            String type = null;
 210.            String visibility = null;
 211.            List<String> properties = null;
 212.            int paramOffset = 0;
 213.    
 214.            s = s.trim();
 215.            if (s.length() > 0 
 216.                    && NotationUtilityUml.VISIBILITYCHARS.indexOf(s.charAt(0)) 
 217.                        >= 0) {
 218.                visibility = s.substring(0, 1);
 219.                s = s.substring(1);
 220.            }
 221.    
 222.            try {
 223.                st = new MyTokenizer(s, " ,\t,<<,\u00AB,\u00BB,>>,:,=,{,},\\,",
 224.                        NotationUtilityUml.operationCustomSep);
 225.                while (st.hasMoreTokens()) {
 226.                    token = st.nextToken();
 227.                    if (" ".equals(token) || "\t".equals(token)
 228.                            || ",".equals(token)) {
 229.                        continue; // Do nothing
 230.                    } else if ("<<".equals(token) || "\u00AB".equals(token)) {
 231.                        if (stereotype != null) {
 232.                            parseError("operation.stereotypes", 
 233.                                    st.getTokenIndex());
 234.                        }
 235.                        stereotype = new StringBuilder();
 236.                        while (true) {
 237.                            token = st.nextToken();
 238.                            if (">>".equals(token) || "\u00BB".equals(token)) {
 239.                                break;
 240.                            }
 241.                            stereotype.append(token);
 242.                        }
 243.                    } else if ("{".equals(token)) {
 244.                        properties = tokenOpenBrace(st, properties);
 245.                    } else if (":".equals(token)) {
 246.                        hasColon = true;
 247.                    } else if ("=".equals(token)) {
 248.                        parseError("operation.default-values", st.getTokenIndex());
 249.                    } else if (token.charAt(0) == '(' && !hasColon) {
 250.                        if (parameterlist != null) {
 251.                            parseError("operation.two-parameter-lists", 
 252.                                    st.getTokenIndex());
 253.                        }
 254.    
 255.                        parameterlist = token;
 256.                    } else {
 257.                        if (hasColon) {
 258.                            if (type != null) {
 259.                                parseError("operation.two-types", 
 260.                                        st.getTokenIndex());
 261.                            }
 262.    
 263.                            if (token.length() > 0
 264.                                    && (token.charAt(0) == '\"'
 265.                                        || token.charAt(0) == '\'')) {
 266.                                parseError("operation.type-quoted",
 267.                                        st.getTokenIndex());
 268.                            }
 269.    
 270.                            if (token.length() > 0 && token.charAt(0) == '(') {
 271.                                parseError("operation.type-expr", 
 272.                                        st.getTokenIndex());
 273.                            }
 274.    
 275.                            type = token;
 276.                        } else {
 277.                            if (name != null && visibility != null) {
 278.                                parseError("operation.extra-text",
 279.                                        st.getTokenIndex());
 280.                            }
 281.    
 282.                            if (token.length() > 0
 283.                                    && (token.charAt(0) == '\"'
 284.                                        || token.charAt(0) == '\'')) {
 285.                                parseError("operation.name-quoted",
 286.                                        st.getTokenIndex());
 287.                            }
 288.    
 289.                            if (token.length() > 0 && token.charAt(0) == '(') {
 290.                                parseError("operation.name-expr", 
 291.                                        st.getTokenIndex());
 292.                            }
 293.    
 294.                            if (name == null
 295.                                    && visibility == null
 296.                                    && token.length() > 1
 297.                                    && NotationUtilityUml.VISIBILITYCHARS.indexOf(
 298.                                            token.charAt(0))
 299.                                                        >= 0) {
 300.                                visibility = token.substring(0, 1);
 301.                                token = token.substring(1);
 302.                            }
 303.    
 304.                            if (name != null) {
 305.                                visibility = name;
 306.                                name = token;
 307.                            } else {
 308.                                name = token;
 309.                            }
 310.                        }
 311.                    }
 312.                } // end while loop
 313.            } catch (NoSuchElementException nsee) {
 314.                parseError("operation.unexpected-end-operation", 
 315.                        s.length());
 316.            } catch (ParseException pre) {
 317.                throw pre;
 318.            }
 319.            
 320.            if (parameterlist != null) {
 321.                // parameterlist is guaranteed to contain at least "("
 322.                if (parameterlist.charAt(parameterlist.length() - 1) != ')') {
 323.                    parseError("operation.parameter-list-incomplete",
 324.                            paramOffset + parameterlist.length() - 1);
 325.                }
 326.    
 327.                paramOffset++;
 328.                parameterlist = parameterlist.substring(1,
 329.                        parameterlist.length() - 1);
 330.                NotationUtilityUml.parseParamList(op, parameterlist, paramOffset);
 331.            }
 332.    
 333.            if (visibility != null) {
 334.                Model.getCoreHelper().setVisibility(op,
 335.                        NotationUtilityUml.getVisibility(visibility.trim()));
 336.            }
 337.    
 338.            if (name != null) {
 339.                Model.getCoreHelper().setName(op, name.trim());
 340.            } else if (Model.getFacade().getName(op) == null
 341.                    || "".equals(Model.getFacade().getName(op))) {
 342.                Model.getCoreHelper().setName(op, "anonymous");
 343.            }
 344.    
 345.            if (type != null) {
 346.                Object ow = Model.getFacade().getOwner(op);
 347.                Object ns = null;
 348.                if (ow != null && Model.getFacade().getNamespace(ow) != null) {
 349.                    ns = Model.getFacade().getNamespace(ow);
 350.                } else {
 351.                    ns = Model.getFacade().getRoot(op);
 352.                }
 353.                Object mtype = NotationUtilityUml.getType(type.trim(), ns);
 354.                setReturnParameter(op, mtype);
 355.            }
 356.    
 357.            if (properties != null) {
 358.                NotationUtilityUml.setProperties(op, properties, 
 359.                        NotationUtilityUml.operationSpecialStrings);
 360.            }
 361.    
 362.            // Don't create a stereotype for <<signal>> on a Reception
 363.            // but create any other parsed stereotypes as needed
 364.            if (!Model.getFacade().isAReception(op) 
 365.                    || !RECEPTION_KEYWORD.equals(stereotype.toString())) {
 366.                StereotypeUtility.dealWithStereotypes(op, stereotype, true);
 367.            }
 368.        }
 369.    
 370.        /**
 371.         * Convenience method to signal a parser error.
 372.         * 
 373.         * @param message
 374.         *            string containing error message literal. It will be appended
 375.         *            to the base "parser.error." and localized.
 376.         * @param offset
 377.         *            offset to where error occurred
 378.         * @throws ParseException
 379.         */
 380.        private void parseError(String message, int offset)
 381.            throws ParseException {
 382.    
 383.            throw new ParseException(
 384.                    Translator.localize("parsing.error." + message), 
 385.                    offset);
 386.        }
 387.    
 388.        /**
 389.         * Parse tokens following an open brace (properties).
 390.         * 
 391.         * @param st tokenizer being used
 392.         * @param properties current properties list
 393.         * @return updated list of properties
 394.         * @throws ParseException
 395.         */
 396.        private List<String> tokenOpenBrace(MyTokenizer st, List<String> properties)
 397.            throws ParseException {
 398.            String token;
 399.            StringBuilder propname = new StringBuilder();
 400.            String propvalue = null;
 401.    
 402.            if (properties == null) {
 403.                properties = new ArrayList<String>();
 404.            }
 405.            while (true) {
 406.                token = st.nextToken();
 407.                if (",".equals(token) || "}".equals(token)) {
 408.                    if (propname.length() > 0) {
 409.                        properties.add(propname.toString());
 410.                        properties.add(propvalue);
 411.                    }
 412.                    propname = new StringBuilder();
 413.                    propvalue = null;
 414.    
 415.                    if ("}".equals(token)) {
 416.                        break;
 417.                    }
 418.                } else if ("=".equals(token)) {
 419.                    if (propvalue != null) {
 420.                        String msg = 
 421.                            "parsing.error.operation.prop-stereotypes";
 422.                        Object[] args = {propname};
 423.                        throw new ParseException(
 424.                                Translator.localize(msg, 
 425.                                args), 
 426.                                st.getTokenIndex());
 427.                    }
 428.                    propvalue = "";
 429.                } else if (propvalue == null) {
 430.                    propname.append(token);
 431.                } else {
 432.                    propvalue += token;
 433.                }
 434.            }
 435.            if (propname.length() > 0) {
 436.                properties.add(propname.toString());
 437.                properties.add(propvalue);
 438.            }
 439.            return properties;
 440.        }
 441.

 443.        /**
 444.         * Sets the return parameter of op to be of type type. If there is none, one
 445.         * is created. If there are many, all but one are removed.
 446.         *
 447.         * @param op the operation
 448.         * @param type the type of the return parameter
 449.         */
 450.        private void setReturnParameter(Object op, Object type) {
 451.            Object param = null;
 452.            Iterator it = Model.getFacade().getParameters(op).iterator();
 453.            while (it.hasNext()) {
 454.                Object p = it.next();
 455.                if (Model.getFacade().isReturn(p)) {
 456.                    param = p;
 457.                    break;
 458.                }
 459.            }
 460.            while (it.hasNext()) {
 461.                Object p = it.next();
 462.                if (Model.getFacade().isReturn(p)) {
 463.                    ProjectManager.getManager().getCurrentProject().moveToTrash(p);
 464.                }
 465.            }
 466.            if (param == null) {
 467.                Object returnType =
 468.                    ProjectManager.getManager()
 469.                            .getCurrentProject().getDefaultReturnType();
 470.                param = Model.getCoreFactory().buildParameter(op, returnType);
 471.            }
 472.            Model.getCoreHelper().setType(param, type);
 473.        }

 482.        /**
 483.         * Generate an operation according to the UML notation:
 484.         * <pre>
 485.         *         stereotype visibility name (parameter-list) :
 486.         *                         return-type-expression {property-string}
 487.         * </pre>
 488.         * For the return-type-expression: only the types of the return parameters
 489.         * are shown.  Depending on settings in Notation, visibility and
 490.         * properties are shown/not shown.
 491.         * 
 492.         * @param modelElement UML Operation element
 493.         * @param settings notation settings
 494.         * @return a formatted text string
 495.         * @see org.argouml.notation.NotationProvider#toString(java.lang.Object, org.argouml.notation.No...
 496.         */
 497.        public String toString(Object modelElement, NotationSettings settings) {
 498.            return toString(modelElement, settings.isUseGuillemets(), 
 499.                    settings.isShowVisibilities(), settings.isShowTypes(),
 500.                    settings.isShowProperties());
 501.        }
 502.        
 503.        /**
 504.         * Generate an operation according to the UML notation:
 505.         * <pre>
 506.         *         stereotype visibility name (parameter-list) :
 507.         *                         return-type-expression {property-string}
 508.         * </pre>
 509.         * For the return-type-expression: only the types of the return parameters
 510.         * are shown.  Depending on settings in Notation, visibility and
 511.         * properties are shown/not shown.
 512.         *
 513.         * @author jaap.branderhorst@xs4all.nl
 514.         */
 515.        private String toString(Object modelElement, boolean useGuillemets, 
 516.                boolean showVisibility,
 517.                boolean showTypes, boolean showProperties) {
 518.            try {
 519.                String stereoStr = NotationUtilityUml.generateStereotype(
 520.                        Model.getFacade().getStereotypes(modelElement), 
 521.                        useGuillemets);
 522.                boolean isReception = Model.getFacade().isAReception(modelElement);
 523.                // TODO: needs I18N
 524.                if (isReception) {
 525.                    stereoStr =
 526.                            NotationUtilityUml
 527.                                    .generateStereotype(RECEPTION_KEYWORD, 
 528.                                            useGuillemets)
 529.                                    + " " + stereoStr;
 530.                }
 531.    
 532.                // Unused currently
 533.    //            StringBuffer taggedValuesSb = getTaggedValues(modelElement);
 534.                
 535.                // lets concatenate it to the resulting string (genStr)
 536.                StringBuffer genStr = new StringBuffer(30);
 537.                if ((stereoStr != null) && (stereoStr.length() > 0)) {
 538.                    genStr.append(stereoStr).append(" ");
 539.                }
 540.                if (showVisibility) {
 541.                    String visStr = NotationUtilityUml
 542.                            .generateVisibility2(modelElement);
 543.                    if (visStr != null) {
 544.                        genStr.append(visStr);
 545.                    }
 546.                }
 547.                
 548.                String nameStr = Model.getFacade().getName(modelElement);
 549.                if ((nameStr != null) && (nameStr.length() > 0)) {
 550.                    genStr.append(nameStr);
 551.                }
 552.                
 553.                /* The "show types" defaults to TRUE, to stay compatible with older
 554.                 * ArgoUML versions that did not have this setting: */
 555.                if (showTypes) {
 556.                    // the parameters
 557.                    StringBuffer parameterStr = new StringBuffer();
 558.                    parameterStr.append("(").append(getParameterList(modelElement))
 559.                            .append(")");
 560.    
 561.                    // the returnparameters
 562.                    StringBuffer returnParasSb = getReturnParameters(modelElement,
 563.                            isReception);
 564.                    genStr.append(parameterStr).append(" ");
 565.                    if ((returnParasSb != null) && (returnParasSb.length() > 0)) {
 566.                        genStr.append(returnParasSb).append(" ");
 567.                    }
 568.                } else {
 569.                    genStr.append("()");
 570.                }
 571.                if (showProperties) {
 572.                    StringBuffer propertySb = getProperties(modelElement,
 573.                            isReception);
 574.                    if (propertySb.length() > 0) {
 575.                        genStr.append(propertySb);
 576.                    }
 577.                }
 578.                return genStr.toString().trim();
 579.            } catch (InvalidElementException e) {
 580.                // The model element was deleted while we were working on it
 581.                return "";   
 582.            }
 583.    
 584.        }

 587.        private StringBuffer getParameterList(Object modelElement) {
 588.            StringBuffer parameterListBuffer = new StringBuffer();
 589.            Collection coll = Model.getFacade().getParameters(modelElement);
 590.            Iterator it = coll.iterator();
 591.            int counter = 0;
 592.            while (it.hasNext()) {
 593.                Object parameter = it.next();
 594.                if (!Model.getFacade().hasReturnParameterDirectionKind(
 595.                        parameter)) {
 596.                    counter++;
 597.                    parameterListBuffer.append(
 598.                            NotationUtilityUml.generateParameter(parameter));
 599.                    parameterListBuffer.append(",");
 600.                }
 601.            }
 602.            if (counter > 0) {
 603.                parameterListBuffer.delete(
 604.                        parameterListBuffer.length() - 1,
 605.                        parameterListBuffer.length());
 606.            }
 607.            return parameterListBuffer;
 608.        }

 610.        private StringBuffer getReturnParameters(Object modelElement,
 611.                boolean isReception) {
 612.            StringBuffer returnParasSb = new StringBuffer();
 613.            if (!isReception) {
 614.                Collection coll = 
 615.                    Model.getCoreHelper().getReturnParameters(modelElement);
 616.                if (coll != null && coll.size() > 0) {
 617.                    returnParasSb.append(": ");
 618.                    Iterator it2 = coll.iterator();
 619.                    while (it2.hasNext()) {
 620.                        Object type = Model.getFacade().getType(it2.next());
 621.                        if (type != null) {
 622.                            returnParasSb.append(Model.getFacade()
 623.                                    .getName(type));
 624.                        }
 625.                        returnParasSb.append(",");
 626.                    }
 627.                    // if we have only one return value and without type,
 628.                    // the return param string is ": ,", we remove it
 629.                    if (returnParasSb.length() == 3) {
 630.                        returnParasSb.delete(0, returnParasSb.length());
 631.                    }
 632.                    // else: we remove only the extra ","
 633.                    else {
 634.                        returnParasSb.delete(
 635.                                returnParasSb.length() - 1,
 636.                                returnParasSb.length());
 637.                    }
 638.                }
 639.            }
 640.            return returnParasSb;
 641.        }

 644.        private StringBuffer getProperties(Object modelElement, 
 645.                boolean isReception) {
 646.            StringBuffer propertySb = new StringBuffer().append("{");
 647.            // the query state
 648.            if (Model.getFacade().isQuery(modelElement)) {
 649.                propertySb.append("query,");
 650.            }
 651.            /*
 652.             * Although Operation and Signal are peers in the UML type 
 653.             * hierarchy they share the attributes isRoot, isLeaf, 
 654.             * isAbstract, and  specification. Concurrency is *not* 
 655.             * shared and is specific to Operation.
 656.             */
 657.            if (Model.getFacade().isRoot(modelElement)) {
 658.                propertySb.append("root,");
 659.            }
 660.            if (Model.getFacade().isLeaf(modelElement)) {
 661.                propertySb.append("leaf,");
 662.            }
 663.            if (!isReception) {
 664.                if (Model.getFacade().getConcurrency(modelElement) != null) {
 665.                    propertySb.append(Model.getFacade().getName(
 666.                            Model.getFacade().getConcurrency(modelElement)));
 667.                    propertySb.append(',');
 668.                }
 669.            }
 670.            if (propertySb.length() > 1) {
 671.                propertySb.delete(propertySb.length() - 1, propertySb.length());
 672.                // remove last ,
 673.                propertySb.append("}");
 674.            } else {
 675.                propertySb = new StringBuffer();
 676.            }
 677.            return propertySb;
 678.        }
