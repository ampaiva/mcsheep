 135.        protected Class<?> findClass(String name, boolean interfacesOnly) {
 136.            Class<?> clazz = null;
 137.            try {
 138.                // TODO: Do we ever want to look up things on our own classpath?
 139.                // This really should only be used for default Java types and even
 140.                // then probably should be done after searching the user classpath
 141.                clazz = Class.forName(name);
 142.            } catch (ClassNotFoundException e) {
 143.                clazz = findClassOnUserClasspath(name, interfacesOnly);
 144.            } catch (LinkageError e) {
 145.                // We found the class, but we couldn't load it for some reason
 146.                // most likely a missing dependency on the class path, but could
 147.                // be wrong class file version or something else
 148.                // TODO: Need to make this visible to the user
 149.                LOG.warn("Linkage error loading found class " + name, e);
 150.                // We'll continue the search, but this is probably the one we wanted
 151.                clazz = findClassOnUserClasspath(name, interfacesOnly);
 152.            }
 153.            // Got something, but it wasn't what we wanted.  Try again.
 154.            if (clazz != null && interfacesOnly && !clazz.isInterface()) {
 155.                clazz = findClassOnUserClasspath(name, interfacesOnly);
 156.            }
 157.            return clazz;
 158.        }

 160.        private Class<?> findClassOnUserClasspath(String name,
 161.                boolean interfacesOnly) {
 162.            Class<?> clazz = null;
 163.            try {
 164.                clazz = ImportClassLoader.getInstance().loadClass(name);
 165.            } catch (MalformedURLException e) {
 166.                // TODO: Need to make this visible to the user
 167.                LOG.warn("Classpath configuration error", e);
 168.            } catch (ClassNotFoundException e) {
 169.                return null;
 170.            } catch (LinkageError e) {
 171.                // We found the class, but we couldn't load it for some reason
 172.                // most likely a missing dependency on the class path
 173.                // TODO: Need to make this visible to the user
 174.                LOG.warn("Linkage error loading found class " + name, e);
 175.                return null;
 176.            }
 177.            if (clazz != null && interfacesOnly && !clazz.isInterface()) {
 178.                return null;
 179.            }
 180.            return clazz;
 181.        }
