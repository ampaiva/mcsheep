 146.                        && Model.getFacade().isAPackage(encloser.getOwner())) {

 149.    
 150.                // If default Namespace is not already set
 151.                if (Model.getFacade().getNamespace(me) == null
 152.                        && (TargetManager.getInstance().getTarget()
 153.                        instanceof ArgoDiagram)) {
 154.                    m =
 155.                        ((ArgoDiagram) TargetManager.getInstance().getTarget())
 156.                            .getNamespace();
 157.                    Model.getCoreHelper().setNamespace(me, m);
 158.                }
 159.            } catch (Exception e) {
 160.                LOG.error("could not set package due to:" + e
 161.                        + "' at " + encloser, e);
 162.            }
 163.    
 164.            // The next if-clause is important for the Deployment-diagram
 165.            // it detects if the enclosing fig is a component, in this case
 166.            // the container will be set for the owning Interface
 167.            if (encloser != null
 168.                    && (Model.getFacade().isAComponent(encloser.getOwner()))) {
 169.                moveIntoComponent(encloser);
 170.                super.setEnclosingFig(encloser);
 171.            }
 172.        }

 189.            Set<Object[]> listeners = new HashSet<Object[]>();
 190.    
 191.            // Collect the set of model elements that we want to listen to
 192.            if (newOwner != null) {
 193.                // TODO: Because we get called on each and every change event, when
 194.                // the model is in a state of flux, we'll often get an
 195.                // InvalidElementException before we finish this collection. The
 196.                // only saving grace is that we're called SO many times that on the
 197.                // last time, things should be stable again and we'll get a good set
 198.                // of elements for the final update.  We need a better mechanism.
 199.                
 200.                // add the listeners to the newOwner
 201.                listeners.add(new Object[] {newOwner, null});
 202.                
 203.                // and its stereotypes
 204.                // TODO: Aren't stereotypes handled elsewhere?
 205.                for (Object stereotype 
 206.                        : Model.getFacade().getStereotypes(newOwner)) {

 208.                }
 209.    
 210.                // and its features
 211.                for (Object feat : Model.getFacade().getFeatures(newOwner)) {
 212.                    listeners.add(new Object[] {feat, null});
 213.                    // and the stereotypes of its features
 214.                    for (Object stereotype 
 215.                            : Model.getFacade().getStereotypes(feat)) {
 216.                        listeners.add(new Object[] {stereotype, null});
 217.                    }
 218.                    // and the parameter of its operations
 219.                    if (Model.getFacade().isAOperation(feat)) {
 220.                        for (Object param : Model.getFacade().getParameters(feat)) {
 221.                            listeners.add(new Object[] {param, null});
 222.                            /* Testing: Add a parameter to an operation on an Interface.
 223.                             * Does the Interface Fig adapt its width? */
 224.                        }
 225.                    }
 226.                }
 227.            }
 228.            
 229.            // Update the listeners to match the desired set using the minimal
 230.            // update facility
 231.            updateElementListeners(listeners);
 232.        }
