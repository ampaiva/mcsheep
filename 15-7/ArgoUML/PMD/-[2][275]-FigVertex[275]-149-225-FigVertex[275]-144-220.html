  94.        public void setEnclosingFig(Fig encloser) {
  95.            LayerPerspective lp = (LayerPerspective) getLayer();
  96.            if (lp == null) {
  97.                return;
  98.            }
  99.            
 100.            super.setEnclosingFig(encloser);
 101.    
 102.            Object region = null;
 103.            if (encloser != null) {
 104.                // Get the region as the first Region in the State.
 105.                // If there is no region in the StateMachine then create one.
 106.                List regions = Model.getStateMachinesHelper().getRegions(
 107.                        encloser.getOwner());
 108.                if (regions.isEmpty()) {
 109.                    // There are no regions so create one and 
 110.                    // place the vertex there.
 111.                    region = Model.getUmlFactory().buildNode(
 112.                            Model.getMetaTypes().getRegion(), encloser.getOwner());
 113.                } else {
 114.                    // There are one or more regions so find the one that the
 115.                    //vertex was dropped in
 116.                    FigVertex compositeState = (FigVertex) encloser;
 117.                    for (DiagramElement de : 
 118.                            compositeState.regionCompartment.getDiagramElements()) {
 119.                        if (((Fig) de).getBounds().contains(getBounds())) {
 120.                            region = de.getOwner();
 121.                            break;
 122.                        }
 123.                    }
 124.                }
 125.            } else {
 126.                // The vertex was dropped onto the diagram.
 127.                // Get the region as the first Region in the StateMachine.
 128.                // If there is no region in the StateMachine then create one.
 129.                ArgoDiagram diagram = (ArgoDiagram) lp.getDiagram();
 130.                Object stateMachine = diagram.getOwner();
 131.                List regions = 
 132.                    Model.getStateMachinesHelper().getRegions(stateMachine);
 133.                if (regions.isEmpty()) {
 134.                    region = Model.getUmlFactory().buildNode(
 135.                            Model.getMetaTypes().getRegion(), stateMachine);
 136.                } else {
 137.                    region = regions.get(0);
 138.                }
 139.            }
 140.            if (region != null
 141.                    && Model.getFacade().getContainer(getOwner()) != region) {
 142.                Model.getStateMachinesHelper().setContainer(getOwner(), region);
 143.            }
 144.        }

  93.        @Override
  94.        public void setEnclosingFig(Fig encloser) {
  95.            LayerPerspective lp = (LayerPerspective) getLayer();
  96.            if (lp == null) {
  97.                return;
  98.            }
  99.            
 100.            super.setEnclosingFig(encloser);
 101.    
 102.            Object region = null;
 103.            if (encloser != null) {
 104.                // Get the region as the first Region in the State.
 105.                // If there is no region in the StateMachine then create one.
 106.                List regions = Model.getStateMachinesHelper().getRegions(
 107.                        encloser.getOwner());
 108.                if (regions.isEmpty()) {
 109.                    // There are no regions so create one and place the vertex
 110.                    //there.
 111.                    region = Model.getUmlFactory().buildNode(
 112.                            Model.getMetaTypes().getRegion(), encloser.getOwner());
 113.                } else {
 114.                    // There are one or more regions so find the one that the
 115.                    //vertex was dropped in
 116.                    FigVertex compositeState = (FigVertex) encloser;
 117.                    for (DiagramElement de :
 118.                            compositeState.regionCompartment.getDiagramElements()) {
 119.                        if (((Fig) de).getBounds().contains(getBounds())) {
 120.                            region = de.getOwner();
 121.                            break;
 122.                        }
 123.                    }
 124.                }
 125.            } else {
 126.                // The vertex was dropped onto the diagram.
 127.                // Get the region as the first Region in the StateMachine.
 128.                // If there is no region in the StateMachine then create one.
 129.                ArgoDiagram diagram = (ArgoDiagram) lp.getDiagram();
 130.                Object stateMachine = diagram.getOwner();
 131.                List regions = 
 132.                    Model.getStateMachinesHelper().getRegions(stateMachine);
 133.                if (regions.isEmpty()) {
 134.                    region = Model.getUmlFactory().buildNode(
 135.                            Model.getMetaTypes().getRegion(), stateMachine);
 136.                } else {
 137.                    region = regions.get(0);
 138.                }
 139.            }
 140.            if (region != null
 141.                    && Model.getFacade().getContainer(getOwner()) != region) {
 142.                Model.getStateMachinesHelper().setContainer(getOwner(), region);
 143.            }
 144.        }
