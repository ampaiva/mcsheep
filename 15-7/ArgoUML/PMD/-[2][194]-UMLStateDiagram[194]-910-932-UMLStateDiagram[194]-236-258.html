 227.         */
 228.        private Object getNamespaceFromMachine(Object machine) {
 229.            if (!Model.getFacade().isAStateMachine(machine)) {
 230.                throw new IllegalStateException(
 231.                    "No StateMachine given to create a Statechart diagram");
 232.            }
 233.            
 234.            Object ns = Model.getFacade().getNamespace(machine);
 235.            if (ns != null) {
 236.                return ns;
 237.            }
 238.            
 239.            Object context = Model.getFacade().getContext(machine);
 240.            if (Model.getFacade().isAClassifier(context)) {
 241.                ns = context;
 242.            } else if (Model.getFacade().isABehavioralFeature(context)) {
 243.                ns = Model.getFacade().getNamespace( // or just the owner?
 244.                        Model.getFacade().getOwner(context));
 245.            }
 246.            if (ns == null) {
 247.                ns = getProject().getRoots().iterator().next();
 248.            }
 249.            if (ns == null || !Model.getFacade().isANamespace(ns)) {
 250.                throw new IllegalStateException(
 251.                        "Can not deduce a Namespace from a StateMachine");

 803.        @Override
 804.        public DiagramElement drop(Object droppedObject, Point location) {
 805.            FigNodeModelElement figNode = null;
 806.    
 807.            // If location is non-null, convert to a rectangle that we can use
 808.            Rectangle bounds = null;
 809.            if (location != null) {
 810.                bounds = new Rectangle(location.x, location.y, 0, 0);
 811.            }
 812.            DiagramSettings settings = getDiagramSettings();
 813.    
 814.            if (Model.getFacade().isAActionState(droppedObject)) {
 815.                figNode = new FigActionState(droppedObject, bounds, settings);
 816.            } else if (Model.getFacade().isAFinalState(droppedObject)) {
 817.                figNode = new FigFinalState(droppedObject, bounds, settings);
 818.            } else if (Model.getFacade().isAStubState(droppedObject)) {
 819.                figNode = new FigStubState(droppedObject, bounds, settings);
 820.            } else if (Model.getFacade().isASubmachineState(droppedObject)) {
 821.                figNode = new FigSubmachineState(droppedObject, bounds, settings);
 822.            } else if (Model.getFacade().isACompositeState(droppedObject)) {
 823.                figNode = new FigCompositeState(droppedObject, bounds, settings);
 824.            } else if (Model.getFacade().isASynchState(droppedObject)) {
 825.                figNode = new FigSynchState(droppedObject, bounds, settings);
 826.            } else if (Model.getFacade().isAState(droppedObject)) {
 827.                figNode = new FigSimpleState(droppedObject, bounds, settings);
 828.            } else if (Model.getFacade().isAComment(droppedObject)) {
 829.                figNode = new FigComment(droppedObject, bounds, settings);
 830.            } else if (Model.getFacade().isAPseudostate(droppedObject)) {
 831.                Object kind = Model.getFacade().getKind(droppedObject);
 832.                if (kind == null) {
 833.                    LOG.warn("found a null type pseudostate");
 834.                    return null;
 835.                }
 836.                if (kind.equals(Model.getPseudostateKind().getInitial())) {
 837.                    figNode = new FigInitialState(droppedObject, bounds, settings);
 838.                } else if (kind.equals(
 839.                        Model.getPseudostateKind().getChoice())) {
 840.                    figNode = new FigBranchState(droppedObject, bounds, settings);
 841.                } else if (kind.equals(
 842.                        Model.getPseudostateKind().getJunction())) {
 843.                    figNode = new FigJunctionState(droppedObject, bounds, settings);
 844.                } else if (kind.equals(
 845.                        Model.getPseudostateKind().getFork())) {
 846.                    figNode = new FigForkState(droppedObject, bounds, settings);
 847.                } else if (kind.equals(
 848.                        Model.getPseudostateKind().getJoin())) {
 849.                    figNode = new FigJoinState(droppedObject, bounds, settings);
 850.                } else if (kind.equals(
 851.                        Model.getPseudostateKind().getShallowHistory())) {
 852.                    figNode = new FigShallowHistoryState(droppedObject, bounds, 
 853.                            settings);
 854.                } else if (kind.equals(
 855.                        Model.getPseudostateKind().getDeepHistory())) {
 856.                    figNode = new FigDeepHistoryState(droppedObject, bounds,

 914.                } else if (kind.equals(
 915.                        Model.getPseudostateKind().getJunction())) {
 916.                    figNode = new FigJunctionState(modelElement, bounds, settings);
 917.                } else if (kind.equals(
 918.                        Model.getPseudostateKind().getFork())) {
 919.                    figNode = new FigForkState(modelElement, bounds, settings);
 920.                } else if (kind.equals(
 921.                        Model.getPseudostateKind().getJoin())) {

 924.                        Model.getPseudostateKind().getShallowHistory())) {

 926.                            settings);

 936.            if (figNode != null) {

 938.                        + figNode);

 941.            }
