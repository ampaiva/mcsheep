 276.        public void actionPerformed(ActionEvent e) {
 277.            super.actionPerformed(e);
 278.            if (e.getSource() instanceof JTextField) {
 279.                Object oldValue = getSelectedItem();
 280.                ComboBoxEditor editor = getEditor();
 281.                Object item = editor.getItem();
 282.                doOnEdit(item);
 283.                // next statement is necessary to update the textfield
 284.                // if the selection is equal to what was allready
 285.                // selected
 286.                if (oldValue == getSelectedItem()) {
 287.                    getEditor().setItem(getSelectedItem());
 288.                }
 289.            }

 233.            private UMLImagePanel panel;

 236.             * True if an icon should be shown.
 237.             */

 240.    
 241.            /**
 242.             * Constructor for UMLComboBoxEditor.
 243.             *
 244.             * @param showIcon true if an icon is to be shown
 245.             */

 247.                super();
 248.                panel = new UMLImagePanel(editor, showIcon);
 249.                setShowIcon(showIcon);
 250.            }

 252.            /*
 253.             * @see javax.swing.ComboBoxEditor#setItem(java.lang.Object)
 254.             */
 255.            public void setItem(Object anObject) {
 256.                if (((UMLComboBoxModel) getModel()).contains(anObject)) {
 257.                    editor.setText(((UMLListCellRenderer2) getRenderer())
 258.                            .makeText(anObject));
 259.                    if (theShowIcon && (anObject != null))
 260.                        panel.setIcon(ResourceLoaderWrapper.getInstance()
 261.                                .lookupIcon(anObject));
 262.                } else if (Model.getFacade().isAUMLElement(anObject)) {
 263.                    super.setItem(Model.getFacade().getName(anObject));
 264.                } else {
 265.                super.setItem(anObject);
 266.                }
 267.            }

 344.        /*
 345.         * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
 346.         * TODO: From ComboBox javadoc - "This method is public as an 
 347.         * implementation side effect. do not call or override."
 348.         * We should find some other way to implement this.
 349.         */
 350.        public void actionPerformed(ActionEvent e) {
 351.            super.actionPerformed(e);
 352.            if (e.getSource() instanceof JTextField) {
 353.                Object oldValue = getSelectedItem();
 354.                ComboBoxEditor editor = getEditor();
 355.                Object item = editor.getItem();
 356.                doOnEdit(item);
 357.                // next statement is necessary to update the textfield
 358.                // if the selection is equal to what was allready
 359.                // selected
 360.                if (oldValue == getSelectedItem()) {
 361.                    getEditor().setItem(getSelectedItem());
 362.                }
 363.            }
 364.        }

 369.         * editing the textfield and calls this method afterwards.
 370.         * @param item The item in the comboboxeditor. In this case it's the text of
 371.         * the editable textfield.
 372.         */
 373.        protected void doOnEdit(Object item) {}
