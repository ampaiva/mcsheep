  53.    /**
  54.     * This layout manager lines up components in 2 columns. All JLabels
  55.     * are the first column and any component the JLabel is registered
  56.     * with is in a second column next to the label. <p>
  57.     *
  58.     * Components are sized automatically to fill available space in the container
  59.     * when it is resized. <p>
  60.     * 
  61.     * All JLabel widths will be the largest of the JLabel preferred widths (unless
  62.     * the container is too narrow). <p>
  63.     * 
  64.     * The components will take up any left over width unless they are
  65.     * restricted themselves by a maximum width. <p>
  66.     * 
  67.     * The height of each component is either fixed or will resize to use up any
  68.     * available space in the container. Whether a components height is resizable
  69.     * is determined by checking whether the preferred height of that component is
  70.     * greater then its minimum height. This is the case for components such as
  71.     * JList which would require to expand to show the maximum number or items. <p>
  72.     * 
  73.     * If a component is not to have its height resized then its preferred
  74.     * height and minimum height should be the same. This is the case for
  75.     * components such as JTextField and JComboBox* which should always stay the
  76.     * same height. <p>
  77.     * 
  78.     * [There is known bug in JRE5 where the prefered height and minimum height of
  79.     * a JComboBox can differ. LabelledLayout has coded a workaround for this bug.
  80.     * See - http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6255154 ] <p>
  81.     * 
  82.     * LabelledLayout can show multiple panels of label/component
  83.     * pairs. The seperation of these panels is indicated by adding a
  84.     * Seperator component to the container. Labelled layout starts
  85.     * a new panel when detecting this Seperator. <p>

 180.            synchronized (parent.getTreeLock()) {
 181.                final Insets insets = parent.getInsets();
 182.                int preferredWidth = 0;
 183.                int preferredHeight = 0;
 184.                int widestLabel = 0;
 185.    
 186.                final int componentCount = parent.getComponentCount();
 187.                for (int i = 0; i < componentCount; ++i) {
 188.                    Component childComp = parent.getComponent(i);
 189.                    if (childComp.isVisible()
 190.                            && !(childComp instanceof Seperator)) {
 191.                        int childHeight = getPreferredHeight(childComp);
 192.                        if (childComp instanceof JLabel) {
 193.                            final JLabel jlabel = (JLabel) childComp;
 194.                            widestLabel = 
 195.                                Math.max(widestLabel, getPreferredWidth(jlabel));
 196.                            childComp = jlabel.getLabelFor();
 197.                            final int childWidth = getPreferredWidth(childComp);
 198.                            preferredWidth = 
 199.                                Math.max(preferredWidth, childWidth);
 200.    
 201.                            childHeight =
 202.                                Math.min(childHeight, getPreferredHeight(jlabel));
 203.                        }
 204.                        preferredHeight += childHeight + this.vgap;
 205.                    }
 206.                }
 207.                preferredWidth += insets.left + widestLabel + insets.right;
 208.                preferredHeight += insets.top + insets.bottom;
 209.                return new Dimension(
 210.                        insets.left + widestLabel + preferredWidth + insets.right,
 211.                        preferredHeight);
 212.            }
 213.        }
 214.        
 215.        /** 
 216.         * Required by LayoutManager.
 217.         * 
 218.         * @see java.awt.LayoutManager#minimumLayoutSize(java.awt.Container)
 219.         */
 220.        public Dimension minimumLayoutSize(Container parent) {
 221.            synchronized (parent.getTreeLock()) {
 222.                final Insets insets = parent.getInsets();
 223.                int minimumHeight = insets.top + insets.bottom;
 224.    
 225.                final int componentCount = parent.getComponentCount();
 226.                for (int i = 0; i < componentCount; ++i) {
 227.                    Component childComp = parent.getComponent(i);
 228.                    if (childComp instanceof JLabel) {
 229.                        final JLabel jlabel = (JLabel) childComp;
 230.                        childComp = jlabel.getLabelFor();
 231.                        
 232.                        final int childHeight = Math.max(
 233.                                getMinimumHeight(childComp),
 234.                                getMinimumHeight(jlabel));
 235.                        minimumHeight += childHeight + this.vgap;
 236.                    }
 237.                }
 238.                return new Dimension(0, minimumHeight);
 239.            }
 240.        }
 241.    
 242.        /**
 243.         * @see java.awt.LayoutManager#layoutContainer(java.awt.Container)
 244.         */
 245.        public void layoutContainer(Container parent) {
 246.            synchronized (parent.getTreeLock()) {
 247.                int sectionX = parent.getInsets().left;
 248.    
 249.                final ArrayList<Component> components = new ArrayList<Component>();
 250.                final int sectionCount = getSectionCount(parent);
 251.                final int sectionWidth = getSectionWidth(parent, sectionCount);
 252.                int sectionNo = 0;
 253.                for (int i = 0; i < parent.getComponentCount(); ++i) {
 254.                    final Component childComp = parent.getComponent(i);
 255.                    if (childComp instanceof Seperator) {
 256.                        if (!this.ignoreSplitters) {
 257.                            layoutSection(
 258.                                    parent, 
 259.                                    sectionX, 
 260.                                    sectionWidth, 
 261.                                    components, 
 262.                                    sectionNo++);
 263.                            sectionX += sectionWidth + this.hgap;
 264.                            components.clear();
 265.                        }
 266.                    } else {
 267.                        components.add(parent.getComponent(i));
 268.                    }
 269.                }
 270.                layoutSection(
 271.                        parent, 
 272.                        sectionX, 
 273.                        sectionWidth, 
 274.                        components, 
 275.                        sectionNo);
 276.            }
 277.        }
 278.    
 279.        /** Determine the number of sections.  There is only ever one
 280.         * section if oriented vertically.  If oriented horizontally the
 281.         * number of sections is deduced from the number of Splitters in
 282.         * the parent container.
 283.         */
 284.        private int getSectionCount(Container parent) {
 285.            int sectionCount = 1;
 286.            final int componentCount = parent.getComponentCount();
 287.            if (!ignoreSplitters) {
 288.                for (int i = 0; i < componentCount; ++i) {
 289.                    if (parent.getComponent(i) instanceof Seperator) {
 290.                        ++sectionCount;
 291.                    }
 292.                }
 293.            }
 294.            return sectionCount;
 295.        }
 296.        
 297.        /**
 298.         * Determine the width of each section from the section count.
 299.         * This is the working width minus the gaps between sections. This
 300.         * result is then divided equally by the section count.
 301.         */
 302.        private int getSectionWidth(Container parent, int sectionCount) {
 303.            return (getUsableWidth(parent) - (sectionCount - 1) * this.hgap)
 304.                    / sectionCount;
 305.        }
 306.    
 307.        /**
 308.         * Determine the usable width of the parent.
 309.         * This is the full width minus any borders.
 310.         */
 311.        private int getUsableWidth(Container parent) {
 312.            final Insets insets = parent.getInsets();
 313.            return parent.getWidth() - (insets.left + insets.right);
 314.        }
 315.        
 316.        /**
 317.         * Layout a single section
 318.         */
 319.        private void layoutSection(
 320.                final Container parent,
 321.                final int sectionX,
 322.                final int sectionWidth,
 323.                final ArrayList components,
 324.                final int sectionNo) {
 325.            final ArrayList<Integer> rowHeights = new ArrayList<Integer>();
 326.    
 327.            final int componentCount = components.size();
 328.            if (componentCount == 0) {
 329.                return;
 330.            }
 331.    
 332.            int labelWidth = 0;
 333.            int unknownHeightCount = 0;
 334.            int totalHeight = 0;
 335.    
 336.            // Build up an array list of the heights of each label/component pair.
 337.            // Heights of zero indicate a proportional height.
 338.            Component previousComp = null;
 339.            for (int i = 0; i < componentCount; ++i) {
 340.                final Component childComp = (Component) components.get(i);
 341.                final int childHeight;
 342.                if (childComp instanceof JLabel) {
 343.                    final JLabel jlabel = (JLabel) childComp;
 344.                    final Component labelledComp = jlabel.getLabelFor();
 345.                    
 346.                    labelWidth = Math.max(labelWidth, getPreferredWidth(jlabel));
 347.                    
 348.                    if (labelledComp != null) {
 349.                        ++i;
 350.                        childHeight = getChildHeight(labelledComp);
 351.                        if (childHeight == 0) {
 352.                            ++unknownHeightCount;
 353.                        }
 354.                    } else {
 355.                        childHeight = getPreferredHeight(jlabel);
 356.                    }
 357.                    
 358.                    totalHeight += childHeight + this.vgap;
 359.                    rowHeights.add(new Integer(childHeight));
 360.                } else {
 361.                    // to manage the case there are no label/component
 362.                    // pairs but just one component
 363.                    childHeight = getChildHeight(childComp);
 364.                    if (childHeight == 0) {
 365.                        ++unknownHeightCount;
 366.                    }
 367.                    
 368.                    totalHeight += childHeight + this.vgap;
 369.                    rowHeights.add(new Integer(childHeight));
 370.                }
 371.                
 372.                previousComp = childComp;
 373.            }
 374.            totalHeight -= this.vgap;
 375.            
 376.            final Insets insets = parent.getInsets();
 377.            final int parentHeight = 
 378.                parent.getHeight() - (insets.top + insets.bottom);
 379.            // Set the child components to the heights in the array list
 380.            // calculating the height of any proportional component on the
 381.            // fly.  FIXME - This assumes that the JLabel and the
 382.            // component it labels have been added to the parent component
 383.            // consecutively.
 384.            int y = insets.top;
 385.            int row = 0;
 386.            previousComp = null;
 387.            for (int i = 0; i < componentCount; ++i) {
 388.                Component childComp = (Component) components.get(i);
 389.                if (childComp.isVisible()) {
 390.                    int rowHeight;
 391.                    int componentWidth = sectionWidth;
 392.                    int componentX = sectionX;
 393.                    // If the component is a JLabel which has another
 394.                    // component assigned then position/size the label and
 395.                    // calculate the size of the registered component
 396.                    if (childComp instanceof JLabel
 397.                            && ((JLabel) childComp).getLabelFor() != null) {
 398.                        i++; // Assumes the next child is the labelled component
 399.                        final JLabel jlabel = (JLabel) childComp;
 400.                        childComp = jlabel.getLabelFor();
 401.                        jlabel.setBounds(sectionX, y, labelWidth,
 402.                                         getPreferredHeight(jlabel));
 403.                        componentWidth = sectionWidth - (labelWidth);
 404.                        componentX = sectionX + labelWidth;
 405.                    }
 406.                    rowHeight = rowHeights.get(row).intValue();
 407.                    if (rowHeight == 0) {
 408.                        try {
 409.                            rowHeight = calculateHeight(
 410.                                    parentHeight, 
 411.                                    totalHeight, 
 412.                                    unknownHeightCount--, 
 413.                                    childComp);
 414.                        } catch (ArithmeticException e) {
 415.                            String lookAndFeel = 
 416.                                UIManager.getLookAndFeel().getClass().getName();
 417.                            throw new IllegalStateException(
 418.                                    "Division by zero laying out "
 419.                                    + childComp.getClass().getName()
 420.                                    + " on " + parent.getClass().getName()
 421.                                    + " in section " + sectionNo
 422.                                    + " using "
 423.                                    + lookAndFeel,
 424.                                    e);
 425.                        }
 426.                        totalHeight += rowHeight;
 427.                    }
 428.                    // Make sure the component width isn't any greater
 429.                    // than its maximum allowed width
 430.                    if (childComp.getMaximumSize() != null
 431.                            && getMaximumWidth(childComp) < componentWidth) {
 432.                        componentWidth = getMaximumWidth(childComp);
 433.                    }
 434.                    childComp.setBounds(componentX, y, componentWidth, rowHeight);
 435.                    y += rowHeight + this.vgap;
 436.                    ++row;
 437.                    previousComp = childComp;
 438.                }
 439.            }
 440.        }
 441.        
 442.        /**
 443.         * @param childComp a component
 444.         * @return 0 for a resizable component or a positive value for its fixed
 445.         * height
 446.         */
 447.        private int getChildHeight(Component childComp) {
 448.            if (isResizable(childComp)) {
 449.                // If the child component is resizable then
 450.                // we don't know it's actual size yet.
 451.                // It will be calculated later as a
 452.                // proportion of the available left over
 453.                // space.  For now this is flagged as zero.
 454.                return 0;
 455.            } else {
 456.                // If a preferred height is not given or is
 457.                // the same as the minimum height then fix the
 458.                // height of this row.
 459.                return getMinimumHeight(childComp);
 460.            }
 461.        }
 462.    
 463.        /**
 464.         * A component is resizable if its minimum size is less than
 465.         * its preferred size.
 466.         * There is a workaround here for a bug introduced in JRE5
 467.         * where JComboBox minimum and preferred size now differ.
 468.         * JComboBox is not resizable.
 469.         * Anything in a JScrollPane is considered resizable
 470.         * @param comp the component to check for resizability.
 471.         * @return true if the given component should be resized to take u[p empty
 472.         * space.
 473.         */
 474.        private boolean isResizable(Component comp) {
 475.            if (comp == null) {
 476.                return false;
 477.            }
 478.            if (comp instanceof JComboBox) {
 479.                return false;
 480.            }
 481.            if (comp.getPreferredSize() == null) {
 482.                return false;
 483.            }
 484.            if (comp.getMinimumSize() == null) {
 485.                return false;
 486.            }
 487.            return (getMinimumHeight(comp) < getPreferredHeight(comp));
 488.        }
 489.    
 490.        private final int calculateHeight(
 491.                final int parentHeight, 
 492.                final int totalHeight,
 493.                final int unknownHeightsLeft, 
 494.                final Component childComp) {
 495.            return Math.max(
 496.                    (parentHeight - totalHeight) / unknownHeightsLeft,
 497.                    getMinimumHeight(childComp));
 498.        }
 499.        
 500.        private int getPreferredHeight(final Component comp) {
 501.            return (int) comp.getPreferredSize().getHeight();
 502.        }
 503.        
 504.        private int getPreferredWidth(final Component comp) {
 505.            return (int) comp.getPreferredSize().getWidth();
 506.        }
 507.    
 508.        private int getMinimumHeight(final Component comp) {
 509.            return (int) comp.getMinimumSize().getHeight();
 510.        }
 511.        
 512.        private int getMaximumWidth(final Component comp) {
 513.            return (int) comp.getMaximumSize().getWidth();
 514.        }

 519.         */

 521.            return new Seperator();
 522.        }

  52.    /**
  53.     * This layout manager lines up components in 2 columns. All JLabels
  54.     * are the first column and any component the JLabel is registered
  55.     * with is in a second column next to the label. <p>
  56.     *
  57.     * Components are sized automatically to fill available space in the container
  58.     * when it is resized. <p>
  59.     * 
  60.     * All JLabel widths will be the largest of the JLabel preferred widths (unless
  61.     * the container is too narrow). <p>
  62.     * 
  63.     * The components will take up any left over width unless they are
  64.     * restricted themselves by a maximum width. <p>
  65.     * 
  66.     * The height of each component is either fixed or will resize to use up any
  67.     * available space in the container. Whether a components height is resizable
  68.     * is determined by checking whether the preferred height of that component is
  69.     * greater then its minimum height. This is the case for components such as
  70.     * JList which would require to expand to show the maximum number or items. <p>
  71.     * 
  72.     * If a component is not to have its height resized then its preferred
  73.     * height and minimum height should be the same. This is the case for
  74.     * components such as JTextField and JComboBox* which should always stay the
  75.     * same height. <p>
  76.     * 
  77.     * [There is known bug in JRE5 where the prefered height and minimum height of
  78.     * a JComboBox can differ. LabelledLayout has coded a workaround for this bug.
  79.     * See - http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6255154 ] <p>
  80.     * 
  81.     * LabelledLayout can show multiple panels of label/component
  82.     * pairs. The seperation of these panels is indicated by adding a
  83.     * Seperator component to the container. Labelled layout starts
  84.     * a new panel when detecting this Seperator. <p>
  85.     * 
  86.     * When there are multiple panels, each panel is given equal width.
  87.     * The width restriction of JLabels and components described above are then
  88.     * dependent on panel width rather than container width.
  89.     *
  90.     * @author Bob Tarling
  91.     */
  92.    class LabelledLayout implements LayoutManager, java.io.Serializable {
  93.    
  94.        /**
  95.         * The class uid
  96.         */
  97.        private static final long serialVersionUID = -5596655602155151443L;
  98.    
  99.        /**
 100.         * This is the horizontal gap (in pixels) which specifies the space
 101.         * between sections.  They can be changed at any time.
 102.         * This should be a non negative integer.
 103.         *
 104.         * @serial
 105.         * @see #getHgap()
 106.         * @see #setHgap(int)
 107.         */

 109.        
 110.        /**
 111.         * This is the vertical gap (in pixels) which specifies the space
 112.         * between rows.  They can be changed at any time.
 113.         * This should be a non negative integer.
 114.         *
 115.         * @serial
 116.         * @see #getVgap()
 117.         * @see #setVgap(int)
 118.         */
 119.        private int vgap;
 120.    
 121.        private boolean ignoreSplitters;

 175.        /**
 176.         * Determines the preferred size of the container argument using
 177.         * this labelled layout.  The preferred size is that all child
 178.         * components are in one section at their own preferred size with
 179.         * gaps and border indents.
 180.         */
 181.        public Dimension preferredLayoutSize(Container parent) {
 182.            synchronized (parent.getTreeLock()) {
 183.                final Insets insets = parent.getInsets();
 184.                int preferredWidth = 0;
 185.                int preferredHeight = 0;
 186.                int widestLabel = 0;
 187.    
 188.                final int componentCount = parent.getComponentCount();
 189.                for (int i = 0; i < componentCount; ++i) {
 190.                    Component childComp = parent.getComponent(i);
 191.                    if (childComp.isVisible()
 192.                            && !(childComp instanceof Seperator)) {
 193.                        int childHeight = getPreferredHeight(childComp);
 194.                        preferredHeight += childHeight + this.vgap;
 195.                    }
 196.                }
 197.                preferredWidth += insets.left + widestLabel + insets.right;
 198.                preferredHeight += insets.top + insets.bottom;
 199.                return new Dimension(
 200.                        insets.left + widestLabel + preferredWidth + insets.right,
 201.                        preferredHeight);
 202.            }
 203.        }
 204.        
 205.        /** 
 206.         * Required by LayoutManager.
 207.         * 
 208.         * @see java.awt.LayoutManager#minimumLayoutSize(java.awt.Container)
 209.         */
 210.        public Dimension minimumLayoutSize(Container parent) {
 211.            synchronized (parent.getTreeLock()) {
 212.                final Insets insets = parent.getInsets();
 213.                int minimumHeight = insets.top + insets.bottom;
 214.                return new Dimension(0, minimumHeight);
 215.            }
 216.        }

 218.        /**
 219.         * @see java.awt.LayoutManager#layoutContainer(java.awt.Container)
 220.         */
 221.        public void layoutContainer(Container parent) {
 222.            synchronized (parent.getTreeLock()) {
 223.                int sectionX = parent.getInsets().left;
 224.    
 225.                final ArrayList<Component> components = new ArrayList<Component>();
 226.                final int sectionCount = getSectionCount(parent);
 227.                final int sectionWidth = getSectionWidth(parent, sectionCount);
 228.                int sectionNo = 0;
 229.                for (int i = 0; i < parent.getComponentCount(); ++i) {
 230.                    final Component childComp = parent.getComponent(i);
 231.                    if (childComp instanceof Seperator) {
 232.                        if (!this.ignoreSplitters) {
 233.                            layoutSection(
 234.                                    parent, 
 235.                                    sectionX, 
 236.                                    sectionWidth, 
 237.                                    components, 
 238.                                    sectionNo++);
 239.                            sectionX += sectionWidth + this.hgap;
 240.                            components.clear();
 241.                        }
 242.                    } else {
 243.                        components.add(parent.getComponent(i));
 244.                    }
 245.                }
 246.                layoutSection(
 247.                        parent, 
 248.                        sectionX, 
 249.                        sectionWidth, 
 250.                        components, 
 251.                        sectionNo);
 252.            }
 253.        }
 254.    
 255.        /** Determine the number of sections.  There is only ever one
 256.         * section if oriented vertically.  If oriented horizontally the
 257.         * number of sections is deduced from the number of Splitters in
 258.         * the parent container.
 259.         */
 260.        private int getSectionCount(Container parent) {
 261.            int sectionCount = 1;
 262.            final int componentCount = parent.getComponentCount();
 263.            if (!ignoreSplitters) {
 264.                for (int i = 0; i < componentCount; ++i) {
 265.                    if (parent.getComponent(i) instanceof Seperator) {
 266.                        ++sectionCount;
 267.                    }
 268.                }
 269.            }
 270.            return sectionCount;
 271.        }
 272.        
 273.        /**
 274.         * Determine the width of each section from the section count.
 275.         * This is the working width minus the gaps between sections. This
 276.         * result is then divided equally by the section count.
 277.         */
 278.        private int getSectionWidth(Container parent, int sectionCount) {
 279.            return (getUsableWidth(parent) - (sectionCount - 1) * this.hgap)
 280.                    / sectionCount;
 281.        }
 282.    
 283.        /**
 284.         * Determine the usable width of the parent.
 285.         * This is the full width minus any borders.
 286.         */
 287.        private int getUsableWidth(Container parent) {
 288.            final Insets insets = parent.getInsets();
 289.            return parent.getWidth() - (insets.left + insets.right);
 290.        }
 291.        
 292.        /**
 293.         * Layout a single section
 294.         */
 295.        private void layoutSection(
 296.                final Container parent,
 297.                final int sectionX,
 298.                final int sectionWidth,
 299.                final ArrayList components,
 300.                final int sectionNo) {
 301.            final ArrayList<Integer> rowHeights = new ArrayList<Integer>();
 302.    
 303.            final int componentCount = components.size();
 304.            if (componentCount == 0) {
 305.                return;
 306.            }
 307.    
 308.            int labelWidth = 0;
 309.            int unknownHeightCount = 0;
 310.            int totalHeight = 0;
 311.    
 312.            // Build up an array list of the heights of each label/component pair.
 313.            // Heights of zero indicate a proportional height.
 314.            for (int i = 0; i < componentCount; ++i) {
 315.                final Component childComp = (Component) components.get(i);
 316.                if (childComp instanceof LabelledComponent) {
 317.                    LabelledComponent lc = (LabelledComponent) childComp;
 318.                    final JLabel label = lc.getLabel();
 319.                    if (label != null) {
 320.                        if (label.getMinimumSize().width > labelWidth) {
 321.                            labelWidth = label.getMinimumSize().width;
 322.                        }
 323.                    }
 324.                }
 325.                final int childHeight = getChildHeight(childComp);
 326.                if (childHeight == 0) {
 327.                    ++unknownHeightCount;
 328.                }
 329.                
 330.                totalHeight += childHeight + this.vgap;
 331.                rowHeights.add(new Integer(childHeight));
 332.            }
 333.            totalHeight -= this.vgap;
 334.            
 335.            final Insets insets = parent.getInsets();
 336.            final int parentHeight = 
 337.                parent.getHeight() - (insets.top + insets.bottom);
 338.            // Set the child components to the heights in the array list
 339.            // calculating the height of any proportional component on the
 340.            // fly.
 341.            int y = insets.top;
 342.            int row = 0;
 343.            for (int i = 0; i < componentCount; ++i) {
 344.                Component childComp = (Component) components.get(i);
 345.                if (childComp.isVisible()) {
 346.                    if (childComp instanceof LabelledComponent) {
 347.                        final LabelledComponent lc = (LabelledComponent) childComp;
 348.                        final JLabel label = lc.getLabel();
 349.                        if (label != null) {
 350.                            label.setPreferredSize(
 351.                                    new Dimension(
 352.                                            labelWidth,
 353.                                            label.getPreferredSize().height));
 354.                        }
 355.                    }
 356.                    int rowHeight;
 357.                    int componentWidth = sectionWidth;
 358.                    int componentX = sectionX;
 359.                    // If the component is a JLabel which has another
 360.                    // component assigned then position/size the label and
 361.                    // calculate the size of the registered component
 362.                    rowHeight = rowHeights.get(row).intValue();
 363.                    if (rowHeight == 0) {
 364.                        try {
 365.                            rowHeight = calculateHeight(
 366.                                    parentHeight, 
 367.                                    totalHeight, 
 368.                                    unknownHeightCount--, 
 369.                                    childComp);
 370.                        } catch (ArithmeticException e) {
 371.                            final String lookAndFeel = 
 372.                                UIManager.getLookAndFeel().getClass().getName();
 373.                            throw new IllegalStateException(
 374.                                    "Division by zero laying out "
 375.                                    + childComp.getClass().getName()
 376.                                    + " on " + parent.getClass().getName()
 377.                                    + " in section " + sectionNo
 378.                                    + " using "
 379.                                    + lookAndFeel,
 380.                                    e);
 381.                        }
 382.                        totalHeight += rowHeight;
 383.                    }
 384.                    // Make sure the component width isn't any greater
 385.                    // than its maximum allowed width
 386.                    if (childComp.getMaximumSize() != null
 387.                            && getMaximumWidth(childComp) < componentWidth) {
 388.                        componentWidth = getMaximumWidth(childComp);
 389.                    }
 390.                    childComp.setBounds(componentX, y, componentWidth, rowHeight);
 391.                    y += rowHeight + this.vgap;
 392.                    ++row;
 393.                }
 394.            }
 395.        }
 396.        
 397.        /**
 398.         * @param childComp a component
 399.         * @return 0 for a resizable component or a positive value for its fixed
 400.         * height
 401.         */
 402.        private int getChildHeight(Component childComp) {
 403.            if (isResizable(childComp)) {
 404.                // If the child component is resizable then
 405.                // we don't know it's actual size yet.
 406.                // It will be calculated later as a
 407.                // proportion of the available left over
 408.                // space.  For now this is flagged as zero.
 409.                return 0;
 410.            } else {
 411.                // If a preferred height is not given or is
 412.                // the same as the minimum height then fix the
 413.                // height of this row.
 414.                return getMinimumHeight(childComp);
 415.            }
 416.        }
 417.    
 418.        /**
 419.         * A component is resizable if its minimum size is less than
 420.         * its preferred size.
 421.         * There is a workaround here for a bug introduced in JRE5
 422.         * where JComboBox minimum and preferred size now differ.
 423.         * JComboBox is not resizable.
 424.         * Anything in a JScrollPane is considered resizable
 425.         * @param comp the component to check for resizability.
 426.         * @return true if the given component should be resized to take u[p empty
 427.         * space.
 428.         */
 429.        private boolean isResizable(Component comp) {
 430.            if (comp == null) {
 431.                return false;
 432.            }
 433.            if (comp instanceof JComboBox) {
 434.                return false;
 435.            }
 436.            if (comp.getPreferredSize() == null) {
 437.                return false;
 438.            }
 439.            if (comp.getMinimumSize() == null) {
 440.                return false;
 441.            }
 442.            return (getMinimumHeight(comp) < getPreferredHeight(comp));
 443.        }
 444.    
 445.        private final int calculateHeight(
 446.                final int parentHeight, 
 447.                final int totalHeight,
 448.                final int unknownHeightsLeft, 
 449.                final Component childComp) {
 450.            return Math.max(
 451.                    (parentHeight - totalHeight) / unknownHeightsLeft,
 452.                    getMinimumHeight(childComp));
 453.        }
 454.        
 455.        private int getPreferredHeight(final Component comp) {
 456.            return (int) comp.getPreferredSize().getHeight();
 457.        }
 458.        
 459.        private int getPreferredWidth(final Component comp) {
 460.            return (int) comp.getPreferredSize().getWidth();
 461.        }
 462.    
 463.        private int getMinimumHeight(final Component comp) {
 464.            return (int) comp.getMinimumSize().getHeight();
 465.        }
 466.        
 467.        private int getMaximumWidth(final Component comp) {
 468.            return (int) comp.getMaximumSize().getWidth();
 469.        }
 470.        
 471.        /**
 472.         * Create a new instance of the Separator that splits the layout in columns
 473.         * @return the separator
 474.         */
 475.        public static Seperator getSeparator() {
 476.            return new Seperator();
 477.        }
 478.    
 479.        /**
 480.         * @return the horizontal gaps between components
 481.         */
 482.        public int getHgap() {
 483.            return this.hgap;
 484.        }
 485.    
 486.        /**
 487.         * Set the horizontal gaps between components
 488.         * @param hgap the horizontal gap
 489.         */
 490.        public void setHgap(int hgap) {
 491.            this.hgap = hgap;
 492.        }
 493.    
 494.        /**
 495.         * @return the vertical gaps between components
 496.         */
 497.        public int getVgap() {
 498.            return this.vgap;
 499.        }
 500.    
 501.        /**
 502.         * Set the vertical gaps between components
 503.         * @param vgap the horizontal gap
 504.         */
 505.        public void setVgap(int vgap) {
 506.            this.vgap = vgap;
 507.        }
 508.    }
