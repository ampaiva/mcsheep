 452.        private boolean doForAll(Map<String, Object> vt, Collection collection,
 453.                List<String> vars, Object exp, LambdaEvaluator eval) {
 454.            if (vars.isEmpty()) {

 462.                    vt.put(var, element);

 464.                    boolean ret = doForAll(vt, collection, vars, exp, eval);

 467.                        return false;
 468.                    }
 469.                }
 470.    
 471.                vt.put(var, oldval);
 472.            }

 452.        private boolean doForAll(Map<String, Object> vt, Collection collection,
 453.                List<String> vars, Object exp, LambdaEvaluator eval) {
 454.            if (vars.isEmpty()) {
 455.                return (Boolean) eval.evaluate(vt, exp);
 456.            } else {
 457.                String var = vars.get(0);
 458.                vars.remove(var);
 459.                Object oldval = vt.get(var);
 460.    
 461.                for (Object element : collection) {
 462.                    vt.put(var, element);
 463.    
 464.                    boolean ret = doForAll(vt, collection, vars, exp, eval);
 465.    
 466.                    if (!ret) {
 467.                        return false;
 468.                    }
 469.                }
 470.    
 471.                vt.put(var, oldval);
 472.            }
 473.            return true;
 474.        }
