 363.            /**
 364.             * Parse field descriptor according the grammar:
 365.             * 
 366.             * <pre>
 367.             * FieldDescriptor:=BaseType|ObjectType|ArrayType
 368.             *         BaseType:=B|C|D|F|J|S|Z
 369.             *         ObjectType:=L Classname;
 370.             *         ArrayType:=[FieldDescriptor
 371.             * </pre>
 372.             * 
 373.             */
 374.            public List<Token> parse() {
 375.                // Object type?
 376.                Matcher m = Pattern.compile("^(L)(.+)(;)").matcher(desc);
 377.                if (m.matches()) {
 378.                    // Classname
 379.                    result.add(new Token(Token.CLASS_NAME, m.group(2)));
 380.                    desc = desc.substring(m.group(0).length()); // the rest after ;
 381.                    return result;
 382.                }
 383.                // Array type?
 384.                m = Pattern.compile("^(\\[)((.+))").matcher(desc);
 385.                if (m.matches()) {
 386.                    result.add(new Token(Token.ARRAY_BRACKET, "[]"));
 387.                    desc = desc.substring(1);
 388.                    parse();
 389.                }
 390.                // Base type?
 391.                m = Pattern.compile("^(B|C|D|F|I|J|S|Z)((.*))").matcher(desc);
 392.                if (m.matches()) {
 393.                    result.add(new Token(Token.BASE_TYPE, m.group(1)));
 394.                    desc = desc.substring(1);
 395.                    return result;
 396.                }
 397.                if (desc.length() == 0) {
 398.                    return result;
 399.                }
 400.                throw new IllegalArgumentException(desc
 401.                        + " is not a FieldDescriptor");
 402.            }

 365.             * 
 366.             * <pre>
 367.             * FieldDescriptor:=BaseType|ObjectType|ArrayType
 368.             *         BaseType:=B|C|D|F|J|S|Z
 369.             *         ObjectType:=L Classname;

 375.                // Object type?
 376.                Matcher m = Pattern.compile("^(L)(.+)(;)").matcher(desc);
 377.                if (m.matches()) {
 378.                    // Classname
