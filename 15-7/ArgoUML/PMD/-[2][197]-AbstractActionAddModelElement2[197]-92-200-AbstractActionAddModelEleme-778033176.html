  56.     * Abstract action that is the parent to all add actions that add the
  57.     * modelelements via the UMLAddDialog.
  58.     * 
  59.     * @since Oct 2, 2002
  60.     * @author jaap.branderhorst@xs4all.nl
  61.     */
  62.    @UmlModelMutator
  63.    public abstract class AbstractActionAddModelElement extends UndoableAction {
  64.    
  65.        private Object target;
  66.        private boolean multiSelect = true;
  67.        private boolean exclusive = true;
  68.    
  69.        /**
  70.         * Construct an action to add a model element to some list.
  71.         */
  72.        protected AbstractActionAddModelElement() {
  73.            super(Translator.localize("menu.popup.add-modelelement"), null);
  74.            // Set the tooltip string:
  75.            putValue(Action.SHORT_DESCRIPTION, 
  76.                    Translator.localize("menu.popup.add-modelelement"));
  77.        }
  78.    
  79.        /**
  80.         * Construct a named action to add a model element to some list.
  81.         * @param name name for action
  82.         */
  83.        public AbstractActionAddModelElement(String name) {
  84.            super(name);
  85.        }
  86.    
  87.        /**
  88.         * Construct an action to add a model element to some list with the
  89.         * given name and icon.
  90.         * @param name name for action
  91.         * @param icon icon for action
  92.         */
  93.        public AbstractActionAddModelElement(String name, Icon icon) {
  94.            super(name, icon);
  95.        }
  96.        
  97.    
  98.        /*
  99.         * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
 100.         */
 101.        @Override
 102.        public void actionPerformed(ActionEvent e) {
 103.            super.actionPerformed(e);
 104.            UMLAddDialog dialog =
 105.                new UMLAddDialog(getChoices(), getSelected(), getDialogTitle(),
 106.                                 isMultiSelect(),
 107.                                 isExclusive());
 108.            int result = dialog.showDialog(ArgoFrame.getFrame());
 109.            if (result == JOptionPane.OK_OPTION) {
 110.                doIt(dialog.getSelected());
 111.            }
 112.        }
 113.        
 114.        /**
 115.         * Returns the choices the user has in the UMLAddDialog. The choices are
 116.         * depicted on the left side of the UMLAddDialog (sorry Arabic users) and
 117.         * can be moved via the buttons on the dialog to the right side. On the
 118.         * right side are the selected modelelements.
 119.         * @return List of choices
 120.         */
 121.        protected abstract List getChoices();
 122.    
 123.        
 124.        /**
 125.         * The modelelements already selected BEFORE the dialog is shown.
 126.         * @return List of model elements
 127.         */
 128.        protected abstract List getSelected();
 129.    
 130.        /**
 131.         * The action that has to be done by ArgoUml after the user clicks ok in the
 132.         * UMLAddDialog.
 133.         * @param selected The choices the user has selected in the UMLAddDialog
 134.         */
 135.        protected abstract void doIt(Collection selected);
 136.    
 137.        /*
 138.         * @see javax.swing.Action#isEnabled()
 139.         */
 140.        @Override
 141.        public boolean isEnabled() {
 142.            return !getChoices().isEmpty();
 143.        }
 144.        
 145.        
 146.        /**
 147.         * Returns the UML model target.
 148.         * @return UML ModelElement
 149.         */
 150.        protected Object getTarget() {
 151.            return target;
 152.        }
 153.    
 154.        /**
 155.         * Sets the UML model target.
 156.         * @param theTarget The target to set
 157.         */
 158.        public void setTarget(Object theTarget) {
 159.            target = theTarget;
 160.        }
 161.    
 162.        /**
 163.         * Returns the title of the dialog.
 164.         * @return String
 165.         */
 166.        protected abstract String getDialogTitle();
 167.    
 168.        /**
 169.         * Returns the exclusive.
 170.         * @return boolean
 171.         */
 172.        public boolean isExclusive() {
 173.            return exclusive;
 174.        }
 175.    
 176.        /**
 177.         * Returns the multiSelect.
 178.         * @return boolean
 179.         */
 180.        public boolean isMultiSelect() {
 181.            return multiSelect;
 182.        }
 183.    
 184.        /**
 185.         * Sets the exclusive.
 186.         * @param theExclusive The exclusive to set
 187.         */
 188.        public void setExclusive(boolean theExclusive) {
 189.            exclusive = theExclusive;
 190.        }
 191.    
 192.        /**
 193.         * Sets the multiSelect.
 194.         * @param theMultiSelect The multiSelect to set
 195.         */
 196.        public void setMultiSelect(boolean theMultiSelect) {
 197.            multiSelect = theMultiSelect;
 198.        }
 199.    
 200.    }
