  96.        private void initializeActionState() {
  97.            cover = new FigRRect(X0, Y0, STATE_WIDTH, HEIGHT, 
  98.                    LINE_COLOR, FILL_COLOR);
  99.            cover.setCornerRadius(getHeight() / 2);
 100.    
 101.            // overrule the single-line name-fig created by the parent
 102.            Rectangle bounds = new Rectangle(X0 + PADDING, Y0, 
 103.                    STATE_WIDTH - PADDING * 2, HEIGHT);
 104.            setNameFig(new FigMultiLineTextWithBold(
 105.                    getOwner(),
 106.                    bounds,
 107.                    getSettings(),
 108.                    true));
 109.            getNameFig().setText(placeString());
 110.            getNameFig().setBotMargin(7); // make space for the clarifier
 111.            getNameFig().setTopMargin(7); // for vertical symmetry
 112.            getNameFig().setRightMargin(4); // margin between text and border
 113.            getNameFig().setLeftMargin(4);
 114.            getNameFig().setJustification(FigText.JUSTIFY_CENTER);
 115.    
 116.            getBigPort().setLineWidth(0);
 117.    
 118.            // add Figs to the FigNode in back-to-front order
 119.            addFig(getBigPort());
 120.            addFig(cover);
 121.            addFig(getStereotypeFig());
 122.            addFig(getNameFig());
 123.    
 124.            //setBlinkPorts(false); //make port invisible unless mouse enters
 125.            Rectangle r = getBounds();

 158.        public Dimension getMinimumSize() {
 159.            Dimension stereoDim = getStereotypeFig().getMinimumSize();
 160.            Dimension nameDim = getNameFig().getMinimumSize();
 161.    
 162.            int w = Math.max(stereoDim.width, nameDim.width) + PADDING * 2;
 163.            /* The stereoDim has height=2, even if it is empty, 
 164.             * hence the -2 below: */
 165.            int h = stereoDim.height - 2 + nameDim.height + PADDING;
 166.            w = Math.max(w, h + 44); // the width needs to be > the height
 167.            return new Dimension(w, h);
 168.        }

 184.            getNameFig().setBounds(x + PADDING, y + stereoDim.height,
 185.                    w - PADDING * 2, nameDim.height);
 186.            getStereotypeFig().setBounds(x + PADDING, y,
 187.                    w - PADDING * 2, stereoDim.height);
 188.            getBigPort().setBounds(x + 1, y + 1, w - 2, h - 2);

 195.            firePropChange("bounds", oldBounds, getBounds());

  67.    /**
  68.     * Class to display graphics for a UML ObjectFlowState in a diagram.<p>
  69.     *
  70.     * The Fig of this modelElement may either represent the following UMLelements:
  71.     * <p>
  72.     * (1) an ObjectFlowState with a Classifier as type, or <p>
  73.     * (2) an ObjectFlowState with a ClassifierInState as type. <p>
  74.     *
  75.     * In both cases (1) and (2), the Fig shows
  76.     * the underlined name of the Classifier,
  77.     * and in the latter case (2), it shows also the names of the states
  78.     * of the ClassifierInState. <p>
  79.     *
  80.     * In the examples in the UML standard, this is written like<pre>
  81.     *      PurchaseOrder
  82.     *       [approved]
  83.     * </pre>
  84.     * i.e. in 2 lines. The first line is underlined,
  85.     * to indicate that it is an instance (object).<p>
  86.     *
  87.     * The fact that the first line is underlined, and the 2nd not, is the
  88.     * reason to implement them in 2 separate Figs.<p>
  89.     *
  90.     * TODO: Allow stereotypes to be shown.
  91.     *
  92.     * @author mvw
  93.     */
  94.    public class FigObjectFlowState extends FigNodeModelElement {
  95.    
  96.        private static final int PADDING = 8;
  97.        private static final int OFS_WIDTH = 70;
  98.        private static final int HEIGHT = 50;
  99.        private static final int STATE_HEIGHT = NAME_FIG_HEIGHT;
 100.    
 101.        private NotationProvider notationProviderState;
 102.        
 103.        private FigRect cover;
 104.        private FigText state;      // the state name
 105.    
 106.        
 107.        /**
 108.         * Construct a new FigObjectFlowState.
 109.         * 
 110.         * @param owner owning UML element
 111.         * @param bounds position and size
 112.         * @param settings rendering settings
 113.         */
 114.        public FigObjectFlowState(Object owner, Rectangle bounds,
 115.                DiagramSettings settings) {
 116.            super(owner, bounds, settings);
 117.            state = new FigSingleLineText(owner, new Rectangle(X0, Y0, OFS_WIDTH,
 118.                    STATE_HEIGHT), settings, true);
 119.            initFigs(bounds);
 120.        }
 121.    
 122.        private void initFigs(Rectangle bounds) {
 123.            cover =
 124.                new FigRect(X0, Y0, OFS_WIDTH, HEIGHT,
 125.                        LINE_COLOR, FILL_COLOR);
 126.    
 127.            getNameFig().setUnderline(true);
 128.            getNameFig().setLineWidth(0);
 129.    
 130.            // add Figs to the FigNode in back-to-front order
 131.            addFig(getBigPort());
 132.            addFig(cover);
 133.            addFig(getNameFig());
 134.            addFig(state);
 135.    
 136.            enableSizeChecking(false);
 137.    
 138.            /* Set the drop location in the case of D&D: */
 139.            if (bounds != null) {
 140.                setLocation(bounds.x, bounds.y);
 141.            }
 142.    
 143.            renderingChanged();
 144.            setSuppressCalcBounds(false);
 145.            setBounds(getBounds());
 146.            enableSizeChecking(true);
 147.        }
 148.    
 149.        /*
 150.         * @see org.argouml.uml.diagram.ui.FigNodeModelElement#initNotationProviders(java.lang.Object)
 151.         */
 152.        @Override
 153.        protected void initNotationProviders(Object own) {
 154.            super.initNotationProviders(own);
 155.            if (Model.getFacade().isAModelElement(own)) {
 156.                NotationName notationName = Notation
 157.                        .findNotation(getNotationSettings().getNotationLanguage());
 158.                notationProviderState =
 159.                    NotationProviderFactory2.getInstance().getNotationProvider(
 160.                            NotationProviderFactory2.TYPE_OBJECTFLOWSTATE_STATE,
 161.                            own, this, notationName);
 162.            }
 163.        }
 164.        
 165.        protected int getNotationProviderType() {
 166.            return NotationProviderFactory2.TYPE_OBJECTFLOWSTATE_TYPE;
 167.        }

 169.        @Override
 170.        protected void modelChanged(PropertyChangeEvent mee) {
 171.            super.modelChanged(mee);
 172.            // TODO: Rather than specifically ignore some item maybe it would be better
 173.            // to specifically state what items are of interest. Otherwise we may still
 174.            // be acting on other events we don't need
 175.            if (!Model.getFacade().isATransition(mee.getNewValue())) {
 176.                renderingChanged();
 177.                updateListeners(getOwner(), getOwner());
 178.            }
 179.        }
 180.    
 181.        @Override
 182.        protected void updateListeners(Object oldOwner, Object newOwner) {
 183.            Set<Object[]> l = new HashSet<Object[]>();
 184.    
 185.            if (newOwner != null) {
 186.                /* Don't listen to all property names
 187.                 * We only need to listen to its "type", and "remove". */
 188.                l.add(new Object[] {newOwner, new String[] {"type", "remove"}});
 189.                // register for events from the type
 190.                Object type = Model.getFacade().getType(newOwner);
 191.                if (Model.getFacade().isAClassifier(type)) {
 192.                    if (Model.getFacade().isAClassifierInState(type)) {
 193.                        Object classifier = Model.getFacade().getType(type);
 194.                        l.add(new Object[] {classifier, "name"});
 195.                        l.add(new Object[] {type, "inState"});
 196.                        Collection states = Model.getFacade().getInStates(type);
 197.                        Iterator i = states.iterator();
 198.                        while (i.hasNext()) {
 199.                            l.add(new Object[] {i.next(), "name"});
 200.                        }
 201.                    } else {
 202.                        l.add(new Object[] {type, "name"});
 203.                    }
 204.                }
 205.            }
 206.    
 207.            updateElementListeners(l);
 208.        }

 210.        @Override
 211.        public Object clone() {
 212.            FigObjectFlowState figClone = (FigObjectFlowState) super.clone();
 213.            Iterator it = figClone.getFigs().iterator();
 214.            figClone.setBigPort((FigRect) it.next());
 215.            figClone.cover = (FigRect) it.next();
 216.            figClone.setNameFig((FigText) it.next());
 217.            figClone.state = (FigText) it.next();
 218.            return figClone;
 219.        }

 232.        /*
 233.         * The space between the 2 text figs is: PADDING.
 234.         * @see org.tigris.gef.presentation.Fig#getMinimumSize()
 235.         */
 236.        @Override
 237.        public Dimension getMinimumSize() {
 238.            Dimension tempDim = getNameFig().getMinimumSize();
 239.            int w = tempDim.width + PADDING * 2;
 240.            int h = tempDim.height + PADDING;
 241.            tempDim = state.getMinimumSize();
 242.            w = Math.max(w, tempDim.width + PADDING * 2);
 243.            h = h + PADDING + tempDim.height + PADDING;
 244.    
 245.            return new Dimension(Math.max(w, OFS_WIDTH / 2), Math.max(h, HEIGHT / 2));
 246.        }
 247.    
 248.        /*
 249.         * Override setBounds to keep shapes looking right.
 250.         * The classifier and state Figs are nicely centered vertically,
 251.         * and stretched out over the full width,
 252.         * to allow easy selection with the mouse.
 253.         * The Fig can only be shrunk to half its original size - so that
 254.         * it is not reducible to a few pixels only.
 255.         *
 256.         * @see org.tigris.gef.presentation.Fig#setBoundsImpl(int, int, int, int)
 257.         */
 258.        @Override
 259.        protected void setStandardBounds(int x, int y, int w, int h) {
 260.            Rectangle oldBounds = getBounds();
 261.    
 262.            Dimension classDim = getNameFig().getMinimumSize();
 263.            Dimension stateDim = state.getMinimumSize();
 264.            /* the height of the blank space above and below the text figs: */
 265.            int blank = (h - PADDING - classDim.height - stateDim.height) / 2;
 266.            getNameFig().setBounds(x + PADDING,
 267.                    y + blank,
 268.                    w - PADDING * 2,
 269.                    classDim.height);
 270.            state.setBounds(x + PADDING,
 271.                    y + blank + classDim.height + PADDING,
 272.                    w - PADDING * 2,
 273.                    stateDim.height);
 274.    
 275.            getBigPort().setBounds(x, y, w, h);
 276.            cover.setBounds(x, y, w, h);
 277.    
 278.            calcBounds();
 279.            updateEdges();
 280.            firePropChange("bounds", oldBounds, getBounds());
 281.        }
