 209.        public DiagramElement createDiagramElement(
 210.                final Object modelElement,
 211.                final Rectangle bounds) {
 212.            
 213.            FigNodeModelElement figNode = null;
 214.            
 215.            DiagramSettings settings = getDiagramSettings();
 216.            
 217.            if (Model.getFacade().isAFinalState(modelElement)) {
 218.                figNode = new FigFinalState(modelElement, bounds, settings);
 219.            } else if (Model.getFacade().isAState(modelElement)) {
 220.                figNode = new FigVertex(modelElement, bounds, settings);
 221.            } else if (Model.getFacade().isAComment(modelElement)) {
 222.                figNode = new FigComment(modelElement, bounds, settings);
 223.            } else if (Model.getFacade().isAPseudostate(modelElement)) {
 224.                Object kind = Model.getFacade().getKind(modelElement);
 225.                if (kind == null) {
 226.                    return null;
 227.                }
 228.                if (kind.equals(Model.getPseudostateKind().getInitial())) {
 229.                    figNode = new FigInitialState(modelElement, bounds, settings);
 230.                } else if (kind.equals(
 231.                        Model.getPseudostateKind().getEntryPoint())) {
 232.                    figNode = 
 233.                            new FigEntryPoint(modelElement, bounds, settings);
 234.                } else if (kind.equals(
 235.                        Model.getPseudostateKind().getExitPoint())) {
 236.                    figNode = new FigExitPoint(modelElement, bounds, settings);
 237.                } else if (kind.equals(
 238.                        Model.getPseudostateKind().getChoice())) {
 239.                    figNode = new FigBranchState(modelElement, bounds, settings);
 240.                } else if (kind.equals(
 241.                        Model.getPseudostateKind().getJunction())) {
 242.                    figNode = new FigJunctionState(modelElement, bounds, settings);
 243.                } else if (kind.equals(
 244.                        Model.getPseudostateKind().getFork())) {
 245.                    figNode = new FigForkState(modelElement, bounds, settings);
 246.                } else if (kind.equals(
 247.                        Model.getPseudostateKind().getJoin())) {
 248.                    figNode = new FigJoinState(modelElement, bounds, settings);
 249.                } else if (kind.equals(
 250.                        Model.getPseudostateKind().getShallowHistory())) {
 251.                    figNode = new FigShallowHistoryState(modelElement, bounds, 
 252.                            settings);
 253.                } else if (kind.equals(
 254.                        Model.getPseudostateKind().getDeepHistory())) {
 255.                    figNode = new FigDeepHistoryState(modelElement, bounds, 
 256.                            settings);
 257.                }
 258.            }
 259.            return figNode;
 260.        }
