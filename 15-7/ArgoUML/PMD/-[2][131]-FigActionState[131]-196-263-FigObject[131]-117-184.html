  59.    /**
  60.     * Class to display graphics for a UML ActionState in a diagram.
  61.     * It contains a multiline textfield for the Entry Action Expression.
  62.     *
  63.     * @author ics 125b silverbullet team
  64.     */
  65.    public class FigActionState extends FigStateVertex {
  66.    
  67.        private static final int HEIGHT = 25;
  68.    
  69.        private static final int STATE_WIDTH = 90;
  70.    
  71.        private static final int PADDING = 8;
  72.    
  73.        private FigRRect cover;
  74.    
  75.        /**
  76.         * Constructor used by PGML parser.
  77.         * 
  78.         * @param owner the owning UML element
  79.         * @param bounds rectangle describing bounds
  80.         * @param settings rendering settings
  81.         */
  82.        public FigActionState(Object owner, Rectangle bounds, 
  83.                DiagramSettings settings) {
  84.            super(owner, bounds, settings);
  85.            initializeActionState();
  86.        }
  87.    
  88.        @Override
  89.        protected Fig createBigPortFig() {
  90.            FigRRect frr = new FigRRect(X0 + 1, Y0 + 1, STATE_WIDTH - 2, HEIGHT - 2,
  91.                    DEBUG_COLOR, DEBUG_COLOR);
  92.            frr.setCornerRadius(frr.getHeight() / 2);
  93.            return frr;
  94.        }
  95.    
  96.        private void initializeActionState() {
  97.            cover = new FigRRect(X0, Y0, STATE_WIDTH, HEIGHT, 
  98.                    LINE_COLOR, FILL_COLOR);
  99.            cover.setCornerRadius(getHeight() / 2);
 100.    
 101.            // overrule the single-line name-fig created by the parent
 102.            Rectangle bounds = new Rectangle(X0 + PADDING, Y0, 
 103.                    STATE_WIDTH - PADDING * 2, HEIGHT);
 104.            setNameFig(new FigMultiLineTextWithBold(
 105.                    getOwner(),
 106.                    bounds,
 107.                    getSettings(),
 108.                    true));
 109.            getNameFig().setText(placeString());
 110.            getNameFig().setBotMargin(7); // make space for the clarifier
 111.            getNameFig().setTopMargin(7); // for vertical symmetry
 112.            getNameFig().setRightMargin(4); // margin between text and border
 113.            getNameFig().setLeftMargin(4);
 114.            getNameFig().setJustification(FigText.JUSTIFY_CENTER);
 115.    
 116.            getBigPort().setLineWidth(0);
 117.    
 118.            // add Figs to the FigNode in back-to-front order
 119.            addFig(getBigPort());
 120.            addFig(cover);
 121.            addFig(getStereotypeFig());
 122.            addFig(getNameFig());
 123.    
 124.            //setBlinkPorts(false); //make port invisible unless mouse enters
 125.            Rectangle r = getBounds();
 126.            setBounds(r.x, r.y, r.width, r.height);
 127.        }
 128.

 141.         * @see java.lang.Object#clone()
 142.         */
 143.        @Override
 144.        public Object clone() {
 145.            FigActionState figClone = (FigActionState) super.clone();
 146.            Iterator it = figClone.getFigs().iterator();
 147.            figClone.setBigPort((FigRRect) it.next());
 148.            figClone.cover = (FigRRect) it.next();
 149.            figClone.setNameFig((FigText) it.next());
 150.            /* TODO: Do we need to clone the stereotype(s)? */
 151.            return figClone;
 152.        }

 154.        /*
 155.         * @see org.tigris.gef.presentation.Fig#getMinimumSize()
 156.         */
 157.        @Override
 158.        public Dimension getMinimumSize() {
 159.            Dimension stereoDim = getStereotypeFig().getMinimumSize();
 160.            Dimension nameDim = getNameFig().getMinimumSize();
 161.    
 162.            int w = Math.max(stereoDim.width, nameDim.width) + PADDING * 2;
 163.            /* The stereoDim has height=2, even if it is empty, 
 164.             * hence the -2 below: */
 165.            int h = stereoDim.height - 2 + nameDim.height + PADDING;
 166.            w = Math.max(w, h + 44); // the width needs to be > the height
 167.            return new Dimension(w, h);
 168.        }

 170.        /*
 171.         * @see org.tigris.gef.presentation.Fig#setBoundsImpl(int, int, int, int)
 172.         *
 173.         * Override setBounds to keep shapes looking right.
 174.         */
 175.        @Override
 176.        protected void setStandardBounds(int x, int y, int w, int h) {
 177.            if (getNameFig() == null) {
 178.                return;
 179.            }
 180.            Rectangle oldBounds = getBounds();
 181.    
 182.            Dimension stereoDim = getStereotypeFig().getMinimumSize();
 183.            Dimension nameDim = getNameFig().getMinimumSize();
 184.            getNameFig().setBounds(x + PADDING, y + stereoDim.height,
 185.                    w - PADDING * 2, nameDim.height);
 186.            getStereotypeFig().setBounds(x + PADDING, y,
 187.                    w - PADDING * 2, stereoDim.height);
 188.            getBigPort().setBounds(x + 1, y + 1, w - 2, h - 2);
 189.            cover.setBounds(x, y, w, h);
 190.            ((FigRRect) getBigPort()).setCornerRadius(h);
 191.            cover.setCornerRadius(h);
 192.    
 193.            calcBounds();
 194.            updateEdges();
 195.            firePropChange("bounds", oldBounds, getBounds());
 196.        }

 262.        @Override
 263.        protected void modelChanged(PropertyChangeEvent mee) {
 264.            super.modelChanged(mee);
 265.            if (mee instanceof AddAssociationEvent
 266.                    || mee instanceof AttributeChangeEvent) {
 267.                // TODO: Rather than specifically ignore some item maybe it would be better
 268.                // to specifically state what items are of interest. Otherwise we may still
 269.                // be acting on other events we don't need
 270.                if (!Model.getFacade().isATransition(mee.getNewValue())) {
 271.                    renderingChanged();
 272.                    damage();
 273.                }
 274.            }

  55.    /**
  56.     * Class to display graphics for a UML Object in a diagram.
  57.     *
  58.     * @author 5eichler
  59.     */
  60.    public class FigObject extends FigNodeModelElement {
  61.    
  62.        private FigRect cover;
  63.    
  64.        @Override
  65.        protected Fig createBigPortFig() {
  66.            return new FigRect(X0, Y0, 90, 50, DEBUG_COLOR, DEBUG_COLOR);
  67.        }
  68.    
  69.        private void initFigs() {
  70.            cover = new FigRect(X0, Y0, 90, 50, LINE_COLOR, FILL_COLOR);
  71.            getNameFig().setLineWidth(0);
  72.            getNameFig().setFilled(false);
  73.            getNameFig().setUnderline(true);
  74.            Dimension nameMin = getNameFig().getMinimumSize();
  75.            getNameFig().setBounds(X0, Y0, nameMin.width + 20, nameMin.height);
  76.    
  77.            // add Figs to the FigNode in back-to-front order
  78.            addFig(getBigPort());
  79.            addFig(cover);
  80.            addFig(getNameFig());
  81.    
  82.            Rectangle r = getBounds();
  83.            setBounds(r.x, r.y, nameMin.width, nameMin.height);
  84.        }
  85.    
  86.        /**
  87.         * Construct a new FigObject.
  88.         * 
  89.         * @param owner owning UML element
  90.         * @param bounds position and size
  91.         * @param settings render settings
  92.         */
  93.        public FigObject(Object owner, Rectangle bounds, DiagramSettings settings) {
  94.            super(owner, bounds, settings);
  95.            initFigs();
  96.        }
  97.        
  98.        /*
  99.         * @see org.argouml.uml.diagram.ui.FigNodeModelElement#getNotationProviderType()
 100.         */
 101.        @Override
 102.        protected int getNotationProviderType() {
 103.            return NotationProviderFactory2.TYPE_OBJECT;
 104.        }
 105.    
 106.        /*
 107.         * @see java.lang.Object#clone()
 108.         */
 109.        @Override
 110.        public Object clone() {
 111.            FigObject figClone = (FigObject) super.clone();
 112.            Iterator it = figClone.getFigs().iterator();
 113.            figClone.setBigPort((FigRect) it.next());
 114.            figClone.cover = (FigRect) it.next();
 115.            figClone.setNameFig((FigText) it.next());
 116.            return figClone;
 117.        }
 118.    
 119.        /*
 120.         * @see org.tigris.gef.presentation.Fig#setLineColor(java.awt.Color)
 121.         */
 122.        @Override
 123.        public void setLineColor(Color col) {
 124.            cover.setLineColor(col);
 125.        }
 126.    
 127.        /*
 128.         * @see org.tigris.gef.presentation.Fig#getLineColor()
 129.         */
 130.        @Override
 131.        public Color getLineColor() {
 132.            return cover.getLineColor();
 133.        }
 134.    
 135.        /*
 136.         * @see org.tigris.gef.presentation.Fig#setFillColor(java.awt.Color)
 137.         */
 138.        @Override
 139.        public void setFillColor(Color col) {
 140.            cover.setFillColor(col);
 141.        }
 142.    
 143.        /*
 144.         * @see org.tigris.gef.presentation.Fig#getFillColor()
 145.         */
 146.        @Override
 147.        public Color getFillColor() {
 148.            return cover.getFillColor();
 149.        }
 150.    
 151.        /*
 152.         * @see org.tigris.gef.presentation.Fig#setFilled(boolean)
 153.         */
 154.        @Override
 155.        public void setFilled(boolean f) {
 156.            cover.setFilled(f);
 157.        }
 158.    
 159.    
 160.        @Override
 161.        public boolean isFilled() {
 162.            return cover.isFilled();
 163.        }
 164.    
 165.        /*
 166.         * @see org.tigris.gef.presentation.Fig#setLineWidth(int)
 167.         */
 168.        @Override
 169.        public void setLineWidth(int w) {
 170.            cover.setLineWidth(w);
 171.        }
 172.    
 173.        /*
 174.         * @see org.tigris.gef.presentation.Fig#getLineWidth()
 175.         */
 176.        @Override
 177.        public int getLineWidth() {
 178.            return cover.getLineWidth();
 179.        }
 180.    
 181.        /*
 182.         * @see org.tigris.gef.presentation.Fig#makeSelection()
 183.         */
 184.        @Override
 185.        public Selection makeSelection() {
 186.            return new SelectionObject(this);
 187.        }
 188.    
 189.        /*
 190.         * @see org.tigris.gef.presentation.Fig#getMinimumSize()
 191.         */
 192.        @Override
 193.        public Dimension getMinimumSize() {
 194.            Dimension nameMin = getNameFig().getMinimumSize();
 195.    
 196.            int w = nameMin.width + 10;
 197.            int h = nameMin.height + 5;
 198.            
 199.            w = Math.max(60, w);
 200.            return new Dimension(w, h);
 201.        }
 202.    
 203.        /**
 204.         * Override ancestor behaviour by always calling setBounds even if the
 205.         * size hasn't changed. Without this override the Package bounds draw
 206.         * incorrectly. This is not the best fix but is a workaround until the
 207.         * true cause is known. See issue 6135.
 208.         * 
 209.         * @see org.argouml.uml.diagram.ui.FigNodeModelElement#updateBounds()
 210.         */
 211.        protected void updateBounds() {
 212.            if (!isCheckSize()) {
 213.                return;
 214.            }
 215.            Rectangle bbox = getBounds();
 216.            Dimension minSize = getMinimumSize();
 217.            bbox.width = Math.max(bbox.width, minSize.width);
 218.            bbox.height = Math.max(bbox.height, minSize.height);
 219.            setBounds(bbox.x, bbox.y, bbox.width, bbox.height);
 220.        }
 221.    
 222.        /*
 223.         * Override setBounds to keep shapes looking right.
 224.         *
 225.         * @see org.tigris.gef.presentation.Fig#setBounds(int, int, int, int)
 226.         */
 227.        @Override
 228.        protected void setStandardBounds(int x, int y, int w, int h) {
 229.            if (getNameFig() == null) {
 230.                return;
 231.            }
 232.    
 233.            Rectangle oldBounds = getBounds();
 234.    
 235.            Dimension nameMin = getNameFig().getMinimumSize();
 236.    
 237.            getBigPort().setBounds(x, y, w, h);
 238.            cover.setBounds(x, y, w, h);
 239.            getNameFig().setBounds(x, y, nameMin.width + 10, nameMin.height + 4);
 240.    
 241.            //_bigPort.setBounds(x+1, y+1, w-2, h-2);
 242.            _x = x; _y = y; _w = w; _h = h;
 243.    
 244.            firePropChange("bounds", oldBounds, getBounds());
 245.            calcBounds(); //_x = x; _y = y; _w = w; _h = h;
 246.            updateEdges();
 247.        }
 248.    
 249.        /*
 250.         * @see org.tigris.gef.presentation.Fig#setEnclosingFig(org.tigris.gef.presentation.Fig)
 251.         */
 252.        @Override
 253.        public void setEnclosingFig(Fig encloser) {
 254.    
 255.            Object owner = getOwner();
 256.    
 257.            if (encloser != null
 258.                    && (Model.getFacade()
 259.                            .isAComponentInstance(encloser.getOwner()))) {
 260.                Model.getCommonBehaviorHelper()
 261.                        .setComponentInstance(owner, encloser.getOwner());
 262.                super.setEnclosingFig(encloser);
 263.    
 264.            } else if (Model.getFacade().getUmlVersion().startsWith("1")
 265.                    && Model.getFacade().getComponentInstance(owner) != null) {
 266.                Model.getCommonBehaviorHelper().setComponentInstance(owner, null);
 267.                super.setEnclosingFig(null);
 268.            }
 269.            
 270.            
 271.            if (encloser != null
 272.                    && (Model.getFacade()
 273.                            .isAComponent(encloser.getOwner()))) {
 274.    
 275.                moveIntoComponent(encloser);
 276.                super.setEnclosingFig(encloser);
 277.            } else if (encloser != null
 278.                    && Model.getFacade().isANode(encloser.getOwner())) {
 279.                super.setEnclosingFig(encloser);
 280.            } else if (encloser == null) {
 281.                super.setEnclosingFig(null);
 282.            }
 283.        }
