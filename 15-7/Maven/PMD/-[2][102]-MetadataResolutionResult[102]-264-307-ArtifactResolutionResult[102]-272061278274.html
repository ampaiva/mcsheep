 339.        public String toString()
 340.        {
 341.            StringBuilder sb = new StringBuilder();
 342.    
 343.            if ( artifacts != null )
 344.            {

 346.                sb.append( "---------" ).append( "\n" );

 348.                for ( Artifact a : artifacts )
 349.                {

 352.                }
 353.                sb.append( "---------" ).append( "\n" );
 354.            }
 355.    
 356.            return sb.toString();
