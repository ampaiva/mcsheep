  64.     * {@code equals}, and {@code toArray} are <em>not</em> guaranteed
  65.     * to be performed atomically. For example, an iterator operating
  66.     * concurrently with an {@code addAll} operation might view only some
  67.     * of the added elements.
  68.     *
  69.     * <p>This class and its iterator implement all of the <em>optional</em>

  90.    public class FastConcurrentDirectDeque<E>
  91.        extends ConcurrentDirectDeque<E> implements Deque<E>, Serializable {
  92.    
  93.        /*
  94.         * This is an implementation of a concurrent lock-free deque
  95.         * supporting interior removes but not interior insertions, as
  96.         * required to support the entire Deque interface.
  97.         *
  98.         * We extend the techniques developed for ConcurrentLinkedQueue and
  99.         * LinkedTransferQueue (see the internal docs for those classes).
 100.         * Understanding the ConcurrentLinkedQueue implementation is a
 101.         * prerequisite for understanding the implementation of this class.
 102.         *
 103.         * The data structure is a symmetrical doubly-linked "GC-robust"
 104.         * linked list of nodes.  We minimize the number of volatile writes
 105.         * using two techniques: advancing multiple hops with a single CAS
 106.         * and mixing volatile and non-volatile writes of the same memory
 107.         * locations.
 108.         *
 109.         * A node contains the expected E ("item") and links to predecessor
 110.         * ("prev") and successor ("next") nodes:
 111.         *
 112.         * class Node<E> { volatile Node<E> prev, next; volatile E item; }
 113.         *
 114.         * A node p is considered "live" if it contains a non-null item
 115.         * (p.item != null).  When an item is CASed to null, the item is
 116.         * atomically logically deleted from the collection.
 117.         *
 118.         * At any time, there is precisely one "first" node with a null
 119.         * prev reference that terminates any chain of prev references
 120.         * starting at a live node.  Similarly there is precisely one
 121.         * "last" node terminating any chain of next references starting at
 122.         * a live node.  The "first" and "last" nodes may or may not be live.
 123.         * The "first" and "last" nodes are always mutually reachable.
 124.         *
 125.         * A new element is added atomically by CASing the null prev or
 126.         * next reference in the first or last node to a fresh node
 127.         * containing the element.  The element's node atomically becomes
 128.         * "live" at that point.
 129.         *
 130.         * A node is considered "active" if it is a live node, or the
 131.         * first or last node.  Active nodes cannot be unlinked.
 132.         *
 133.         * A "self-link" is a next or prev reference that is the same node:
 134.         *   p.prev == p  or  p.next == p
 135.         * Self-links are used in the node unlinking process.  Active nodes
 136.         * never have self-links.
 137.         *
 138.         * A node p is active if and only if:
 139.         *
 140.         * p.item != null ||
 141.         * (p.prev == null && p.next != p) ||
 142.         * (p.next == null && p.prev != p)
 143.         *
 144.         * The deque object has two node references, "head" and "tail".
 145.         * The head and tail are only approximations to the first and last
 146.         * nodes of the deque.  The first node can always be found by
 147.         * following prev pointers from head; likewise for tail.  However,
 148.         * it is permissible for head and tail to be referring to deleted
 149.         * nodes that have been unlinked and so may not be reachable from
 150.         * any live node.
 151.         *
 152.         * There are 3 stages of node deletion;
 153.         * "logical deletion", "unlinking", and "gc-unlinking".
 154.         *
 155.         * 1. "logical deletion" by CASing item to null atomically removes
 156.         * the element from the collection, and makes the containing node
 157.         * eligible for unlinking.
 158.         *
 159.         * 2. "unlinking" makes a deleted node unreachable from active
 160.         * nodes, and thus eventually reclaimable by GC.  Unlinked nodes
 161.         * may remain reachable indefinitely from an iterator.
 162.         *
 163.         * Physical node unlinking is merely an optimization (albeit a
 164.         * critical one), and so can be performed at our convenience.  At
 165.         * any time, the set of live nodes maintained by prev and next
 166.         * links are identical, that is, the live nodes found via next
 167.         * links from the first node is equal to the elements found via
 168.         * prev links from the last node.  However, this is not true for
 169.         * nodes that have already been logically deleted - such nodes may
 170.         * be reachable in one direction only.
 171.         *
 172.         * 3. "gc-unlinking" takes unlinking further by making active
 173.         * nodes unreachable from deleted nodes, making it easier for the
 174.         * GC to reclaim future deleted nodes.  This step makes the data
 175.         * structure "gc-robust", as first described in detail by Boehm
 176.         * (http://portal.acm.org/citation.cfm?doid=503272.503282).
 177.         *
 178.         * GC-unlinked nodes may remain reachable indefinitely from an
 179.         * iterator, but unlike unlinked nodes, are never reachable from
 180.         * head or tail.
 181.         *
 182.         * Making the data structure GC-robust will eliminate the risk of
 183.         * unbounded memory retention with conservative GCs and is likely
 184.         * to improve performance with generational GCs.
 185.         *
 186.         * When a node is dequeued at either end, e.g. via poll(), we would
 187.         * like to break any references from the node to active nodes.  We
 188.         * develop further the use of self-links that was very effective in
 189.         * other concurrent collection classes.  The idea is to replace
 190.         * prev and next pointers with special values that are interpreted
 191.         * to mean off-the-list-at-one-end.  These are approximations, but
 192.         * good enough to preserve the properties we want in our
 193.         * traversals, e.g. we guarantee that a traversal will never visit
 194.         * the same element twice, but we don't guarantee whether a
 195.         * traversal that runs out of elements will be able to see more
 196.         * elements later after enqueues at that end.  Doing gc-unlinking
 197.         * safely is particularly tricky, since any node can be in use
 198.         * indefinitely (for example by an iterator).  We must ensure that
 199.         * the nodes pointed at by head/tail never get gc-unlinked, since
 200.         * head/tail are needed to get "back on track" by other nodes that
 201.         * are gc-unlinked.  gc-unlinking accounts for much of the
 202.         * implementation complexity.
 203.         *
 204.         * Since neither unlinking nor gc-unlinking are necessary for
 205.         * correctness, there are many implementation choices regarding
 206.         * frequency (eagerness) of these operations.  Since volatile
 207.         * reads are likely to be much cheaper than CASes, saving CASes by
 208.         * unlinking multiple adjacent nodes at a time may be a win.
 209.         * gc-unlinking can be performed rarely and still be effective,
 210.         * since it is most important that long chains of deleted nodes
 211.         * are occasionally broken.
 212.         *
 213.         * The actual representation we use is that p.next == p means to
 214.         * goto the first node (which in turn is reached by following prev
 215.         * pointers from head), and p.next == null && p.prev == p means
 216.         * that the iteration is at an end and that p is a (static final)
 217.         * dummy node, NEXT_TERMINATOR, and not the last active node.
 218.         * Finishing the iteration when encountering such a TERMINATOR is
 219.         * good enough for read-only traversals, so such traversals can use
 220.         * p.next == null as the termination condition.  When we need to
 221.         * find the last (active) node, for enqueueing a new node, we need
 222.         * to check whether we have reached a TERMINATOR node; if so,
 223.         * restart traversal from tail.
 224.         *
 225.         * The implementation is completely directionally symmetrical,
 226.         * except that most public methods that iterate through the list
 227.         * follow next pointers ("forward" direction).
 228.         *
 229.         * We believe (without full proof) that all single-element deque
 230.         * operations (e.g., addFirst, peekLast, pollLast) are linearizable
 231.         * (see Herlihy and Shavit's book).  However, some combinations of
 232.         * operations are known not to be linearizable.  In particular,
 233.         * when an addFirst(A) is racing with pollFirst() removing B, it is
 234.         * possible for an observer iterating over the elements to observe
 235.         * A B C and subsequently observe A C, even though no interior
 236.         * removes are ever performed.  Nevertheless, iterators behave
 237.         * reasonably, providing the "weakly consistent" guarantees.
 238.         *
 239.         * Empirically, microbenchmarks suggest that this class adds about
 240.         * 40% overhead relative to ConcurrentLinkedQueue, which feels as
 241.         * good as we can hope for.
 242.         */

 346.        /**
 347.         * Links e as first element.
 348.         */
 349.        private Node linkFirst(E e) {
 350.            checkNotNull(e);
 351.            final Node<E> newNode = new Node<>(e);
 352.    
 353.            restartFromHead:
 354.            for (;;)
 355.                for (Node<E> h = head, p = h, q;;) {
 356.                    if ((q = p.prev) != null &&
 357.                        (q = (p = q).prev) != null)
 358.                        // Check for head updates every other hop.
 359.                        // If p == q, we are sure to follow head instead.
 360.                        p = (h != (h = head)) ? h : q;
 361.                    else if (p.next == p) // PREV_TERMINATOR
 362.                        continue restartFromHead;
 363.                    else {
 364.                        // p is first node
 365.                        newNode.lazySetNext(p); // CAS piggyback
 366.                        if (p.casPrev(null, newNode)) {
 367.                            // Successful CAS is the linearization point
 368.                            // for e to become an element of this deque,
 369.                            // and for newNode to become "live".
 370.                            if (p != h) // hop two nodes at a time
 371.                                casHead(h, newNode);  // Failure is OK.
 372.                            return newNode;
 373.                        }
 374.                        // Lost CAS race to another thread; re-read prev
 375.                    }
 376.                }
 377.        }
 378.    
 379.        /**
 380.         * Links e as last element.
 381.         */
 382.        private Node linkLast(E e) {
 383.            checkNotNull(e);
 384.            final Node<E> newNode = new Node<>(e);
 385.    
 386.            restartFromTail:
 387.            for (;;)
 388.                for (Node<E> t = tail, p = t, q;;) {
 389.                    if ((q = p.next) != null &&
 390.                        (q = (p = q).next) != null)
 391.                        // Check for tail updates every other hop.
 392.                        // If p == q, we are sure to follow tail instead.
 393.                        p = (t != (t = tail)) ? t : q;
 394.                    else if (p.prev == p) // NEXT_TERMINATOR
 395.                        continue restartFromTail;
 396.                    else {
 397.                        // p is last node
 398.                        newNode.lazySetPrev(p); // CAS piggyback
 399.                        if (p.casNext(null, newNode)) {
 400.                            // Successful CAS is the linearization point
 401.                            // for e to become an element of this deque,
 402.                            // and for newNode to become "live".
 403.                            if (p != t) // hop two nodes at a time
 404.                                casTail(t, newNode);  // Failure is OK.
 405.                            return newNode;
 406.                        }
 407.                        // Lost CAS race to another thread; re-read next
 408.                    }
 409.                }

 414.        /**
 415.         * Unlinks non-null node x.
 416.         */
 417.        void unlink(Node<E> x) {
 418.            // assert x != null;
 419.            // assert x.item == null;
 420.            // assert x != PREV_TERMINATOR;
 421.            // assert x != NEXT_TERMINATOR;
 422.    
 423.            final Node<E> prev = x.prev;
 424.            final Node<E> next = x.next;
 425.            if (prev == null) {
 426.                unlinkFirst(x, next);
 427.            } else if (next == null) {
 428.                unlinkLast(x, prev);
 429.            } else {
 430.                // Unlink interior node.
 431.                //
 432.                // This is the common case, since a series of polls at the
 433.                // same end will be "interior" removes, except perhaps for
 434.                // the first one, since end nodes cannot be unlinked.
 435.                //
 436.                // At any time, all active nodes are mutually reachable by
 437.                // following a sequence of either next or prev pointers.
 438.                //
 439.                // Our strategy is to find the unique active predecessor
 440.                // and successor of x.  Try to fix up their links so that
 441.                // they point to each other, leaving x unreachable from
 442.                // active nodes.  If successful, and if x has no live
 443.                // predecessor/successor, we additionally try to gc-unlink,
 444.                // leaving active nodes unreachable from x, by rechecking
 445.                // that the status of predecessor and successor are
 446.                // unchanged and ensuring that x is not reachable from
 447.                // tail/head, before setting x's prev/next links to their
 448.                // logical approximate replacements, self/TERMINATOR.
 449.                Node<E> activePred, activeSucc;
 450.                boolean isFirst, isLast;
 451.                int hops = 1;
 452.    
 453.                // Find active predecessor
 454.                for (Node<E> p = prev; ; ++hops) {
 455.                    if (p.item != null) {
 456.                        activePred = p;
 457.                        isFirst = false;
 458.                        break;
 459.                    }
 460.                    Node<E> q = p.prev;
 461.                    if (q == null) {
 462.                        if (p.next == p)
 463.                            return;
 464.                        activePred = p;
 465.                        isFirst = true;
 466.                        break;
 467.                    }
 468.                    else if (p == q)
 469.                        return;
 470.                    else
 471.                        p = q;
 472.                }
 473.    
 474.                // Find active successor
 475.                for (Node<E> p = next; ; ++hops) {
 476.                    if (p.item != null) {
 477.                        activeSucc = p;
 478.                        isLast = false;
 479.                        break;
 480.                    }
 481.                    Node<E> q = p.next;
 482.                    if (q == null) {
 483.                        if (p.prev == p)
 484.                            return;
 485.                        activeSucc = p;
 486.                        isLast = true;
 487.                        break;
 488.                    }
 489.                    else if (p == q)
 490.                        return;
 491.                    else
 492.                        p = q;
 493.                }
 494.    
 495.                // TODO: better HOP heuristics
 496.                if (hops < HOPS
 497.                    // always squeeze out interior deleted nodes
 498.                    && (isFirst | isLast))
 499.                    return;
 500.    
 501.                // Squeeze out deleted nodes between activePred and
 502.                // activeSucc, including x.
 503.                skipDeletedSuccessors(activePred);
 504.                skipDeletedPredecessors(activeSucc);
 505.    
 506.                // Try to gc-unlink, if possible
 507.                if ((isFirst | isLast) &&
 508.    
 509.                    // Recheck expected state of predecessor and successor
 510.                    (activePred.next == activeSucc) &&
 511.                    (activeSucc.prev == activePred) &&
 512.                    (isFirst ? activePred.prev == null : activePred.item != null) &&
 513.                    (isLast  ? activeSucc.next == null : activeSucc.item != null)) {
 514.    
 515.                    updateHead(); // Ensure x is not reachable from head
 516.                    updateTail(); // Ensure x is not reachable from tail
 517.    
 518.                    // Finally, actually gc-unlink
 519.                    x.lazySetPrev(isFirst ? prevTerminator() : x);
 520.                    x.lazySetNext(isLast  ? nextTerminator() : x);
 521.                }
 522.            }
 523.        }

 526.         * Unlinks non-null first node.
 527.         */
 528.        private void unlinkFirst(Node<E> first, Node<E> next) {
 529.            // assert first != null;
 530.            // assert next != null;
 531.            // assert first.item == null;
 532.            for (Node<E> o = null, p = next, q;;) {

 535.                        skipDeletedPredecessors(p);

 537.                            (p.next == null || p.item != null) &&

 539.    
 540.                            updateHead(); // Ensure o is not reachable from head
 541.                            updateTail(); // Ensure o is not reachable from tail
 542.

 545.                            o.lazySetPrev(prevTerminator());
 546.                        }
 547.                    }
 548.                    return;
 549.                }
 550.                else if (p == q)
 551.                    return;

 593.        /**
 594.         * Guarantees that any node which was unlinked before a call to
 595.         * this method will be unreachable from head after it returns.
 596.         * Does not guarantee to eliminate slack, only that head will
 597.         * point to a node that was active while this method was running.
 598.         */
 599.        private void updateHead() {
 600.            // Either head already points to an active node, or we keep
 601.            // trying to cas it to the first node until it does.
 602.            Node<E> h, p, q;
 603.            restartFromHead:
 604.            while ((h = head).item == null && (p = h.prev) != null) {
 605.                for (;;) {
 606.                    if ((q = p.prev) == null ||
 607.                        (q = (p = q).prev) == null) {
 608.                        // It is possible that p is PREV_TERMINATOR,
 609.                        // but if so, the CAS is guaranteed to fail.
 610.                        if (casHead(h, p))
 611.                            return;
 612.                        else
 613.                            continue restartFromHead;
 614.                    }
 615.                    else if (h != head)
 616.                        continue restartFromHead;
 617.                    else
 618.                        p = q;
 619.                }
 620.            }
 621.        }

 623.        /**
 624.         * Guarantees that any node which was unlinked before a call to

 626.         * Does not guarantee to eliminate slack, only that tail will
 627.         * point to a node that was active while this method was running.

 631.            // trying to cas it to the last node until it does.

 633.            restartFromTail:
 634.            while ((t = tail).item == null && (p = t.next) != null) {
 635.                for (;;) {
 636.                    if ((q = p.next) == null ||

 638.                        // It is possible that p is NEXT_TERMINATOR,
 639.                        // but if so, the CAS is guaranteed to fail.
 640.                        if (casTail(t, p))
 641.                            return;
 642.                        else
 643.                            continue restartFromTail;
 644.                    }
 645.                    else if (t != tail)
 646.                        continue restartFromTail;
 647.                    else
 648.                        p = q;
 649.                }
 650.            }

 653.        private void skipDeletedPredecessors(Node<E> x) {
 654.            whileActive:
 655.            do {
 656.                Node<E> prev = x.prev;
 657.                // assert prev != null;
 658.                // assert x != NEXT_TERMINATOR;
 659.                // assert x != PREV_TERMINATOR;
 660.                Node<E> p = prev;
 661.                findActive:
 662.                for (;;) {
 663.                    if (p.item != null)
 664.                        break findActive;
 665.                    Node<E> q = p.prev;
 666.                    if (q == null) {
 667.                        if (p.next == p)
 668.                            continue whileActive;
 669.                        break findActive;
 670.                    }
 671.                    else if (p == q)
 672.                        continue whileActive;
 673.                    else
 674.                        p = q;
 675.                }
 676.    
 677.                // found active CAS target
 678.                if (prev == p || x.casPrev(prev, p))
 679.                    return;
 680.    
 681.            } while (x.item != null || x.next == null);
 682.        }

  40.     * Concurrent insertion, removal, and access operations execute safely
  41.     * across multiple threads.
  42.     * A {@code ConcurrentLinkedDeque} is an appropriate choice when
  43.     * many threads will share access to a common collection.
  44.     * Like most other concurrent collection implementations, this class

  48.     * reflecting the state of the deque at some point at or since the

  87.    public class PortableConcurrentDirectDeque<E>
  88.        extends ConcurrentDirectDeque<E> implements Deque<E>, java.io.Serializable {
  89.    
  90.        /*
  91.         * This is an implementation of a concurrent lock-free deque
  92.         * supporting interior removes but not interior insertions, as
  93.         * required to support the entire Deque interface.
  94.         *
  95.         * We extend the techniques developed for ConcurrentLinkedQueue and
  96.         * LinkedTransferQueue (see the internal docs for those classes).
  97.         * Understanding the ConcurrentLinkedQueue implementation is a
  98.         * prerequisite for understanding the implementation of this class.
  99.         *
 100.         * The data structure is a symmetrical doubly-linked "GC-robust"
 101.         * linked list of nodes.  We minimize the number of volatile writes
 102.         * using two techniques: advancing multiple hops with a single CAS
 103.         * and mixing volatile and non-volatile writes of the same memory
 104.         * locations.
 105.         *
 106.         * A node contains the expected E ("item") and links to predecessor
 107.         * ("prev") and successor ("next") nodes:
 108.         *
 109.         * class Node<E> { volatile Node<E> prev, next; volatile E item; }
 110.         *
 111.         * A node p is considered "live" if it contains a non-null item
 112.         * (p.item != null).  When an item is CASed to null, the item is
 113.         * atomically logically deleted from the collection.
 114.         *
 115.         * At any time, there is precisely one "first" node with a null
 116.         * prev reference that terminates any chain of prev references
 117.         * starting at a live node.  Similarly there is precisely one
 118.         * "last" node terminating any chain of next references starting at
 119.         * a live node.  The "first" and "last" nodes may or may not be live.
 120.         * The "first" and "last" nodes are always mutually reachable.
 121.         *
 122.         * A new element is added atomically by CASing the null prev or
 123.         * next reference in the first or last node to a fresh node
 124.         * containing the element.  The element's node atomically becomes
 125.         * "live" at that point.
 126.         *
 127.         * A node is considered "active" if it is a live node, or the
 128.         * first or last node.  Active nodes cannot be unlinked.
 129.         *
 130.         * A "self-link" is a next or prev reference that is the same node:
 131.         *   p.prev == p  or  p.next == p
 132.         * Self-links are used in the node unlinking process.  Active nodes
 133.         * never have self-links.
 134.         *
 135.         * A node p is active if and only if:
 136.         *
 137.         * p.item != null ||
 138.         * (p.prev == null && p.next != p) ||
 139.         * (p.next == null && p.prev != p)
 140.         *
 141.         * The deque object has two node references, "head" and "tail".
 142.         * The head and tail are only approximations to the first and last
 143.         * nodes of the deque.  The first node can always be found by
 144.         * following prev pointers from head; likewise for tail.  However,
 145.         * it is permissible for head and tail to be referring to deleted
 146.         * nodes that have been unlinked and so may not be reachable from
 147.         * any live node.
 148.         *
 149.         * There are 3 stages of node deletion;
 150.         * "logical deletion", "unlinking", and "gc-unlinking".
 151.         *
 152.         * 1. "logical deletion" by CASing item to null atomically removes
 153.         * the element from the collection, and makes the containing node
 154.         * eligible for unlinking.
 155.         *
 156.         * 2. "unlinking" makes a deleted node unreachable from active
 157.         * nodes, and thus eventually reclaimable by GC.  Unlinked nodes
 158.         * may remain reachable indefinitely from an iterator.
 159.         *
 160.         * Physical node unlinking is merely an optimization (albeit a
 161.         * critical one), and so can be performed at our convenience.  At
 162.         * any time, the set of live nodes maintained by prev and next
 163.         * links are identical, that is, the live nodes found via next
 164.         * links from the first node is equal to the elements found via
 165.         * prev links from the last node.  However, this is not true for
 166.         * nodes that have already been logically deleted - such nodes may
 167.         * be reachable in one direction only.
 168.         *
 169.         * 3. "gc-unlinking" takes unlinking further by making active
 170.         * nodes unreachable from deleted nodes, making it easier for the
 171.         * GC to reclaim future deleted nodes.  This step makes the data
 172.         * structure "gc-robust", as first described in detail by Boehm
 173.         * (http://portal.acm.org/citation.cfm?doid=503272.503282).
 174.         *
 175.         * GC-unlinked nodes may remain reachable indefinitely from an
 176.         * iterator, but unlike unlinked nodes, are never reachable from
 177.         * head or tail.
 178.         *
 179.         * Making the data structure GC-robust will eliminate the risk of
 180.         * unbounded memory retention with conservative GCs and is likely
 181.         * to improve performance with generational GCs.
 182.         *
 183.         * When a node is dequeued at either end, e.g. via poll(), we would
 184.         * like to break any references from the node to active nodes.  We
 185.         * develop further the use of self-links that was very effective in
 186.         * other concurrent collection classes.  The idea is to replace
 187.         * prev and next pointers with special values that are interpreted
 188.         * to mean off-the-list-at-one-end.  These are approximations, but
 189.         * good enough to preserve the properties we want in our
 190.         * traversals, e.g. we guarantee that a traversal will never visit
 191.         * the same element twice, but we don't guarantee whether a
 192.         * traversal that runs out of elements will be able to see more
 193.         * elements later after enqueues at that end.  Doing gc-unlinking
 194.         * safely is particularly tricky, since any node can be in use
 195.         * indefinitely (for example by an iterator).  We must ensure that
 196.         * the nodes pointed at by head/tail never get gc-unlinked, since
 197.         * head/tail are needed to get "back on track" by other nodes that
 198.         * are gc-unlinked.  gc-unlinking accounts for much of the
 199.         * implementation complexity.
 200.         *
 201.         * Since neither unlinking nor gc-unlinking are necessary for
 202.         * correctness, there are many implementation choices regarding
 203.         * frequency (eagerness) of these operations.  Since volatile
 204.         * reads are likely to be much cheaper than CASes, saving CASes by
 205.         * unlinking multiple adjacent nodes at a time may be a win.
 206.         * gc-unlinking can be performed rarely and still be effective,
 207.         * since it is most important that long chains of deleted nodes
 208.         * are occasionally broken.
 209.         *
 210.         * The actual representation we use is that p.next == p means to
 211.         * goto the first node (which in turn is reached by following prev
 212.         * pointers from head), and p.next == null && p.prev == p means
 213.         * that the iteration is at an end and that p is a (static final)
 214.         * dummy node, NEXT_TERMINATOR, and not the last active node.
 215.         * Finishing the iteration when encountering such a TERMINATOR is
 216.         * good enough for read-only traversals, so such traversals can use
 217.         * p.next == null as the termination condition.  When we need to
 218.         * find the last (active) node, for enqueueing a new node, we need
 219.         * to check whether we have reached a TERMINATOR node; if so,
 220.         * restart traversal from tail.
 221.         *
 222.         * The implementation is completely directionally symmetrical,
 223.         * except that most public methods that iterate through the list
 224.         * follow next pointers ("forward" direction).
 225.         *
 226.         * We believe (without full proof) that all single-element deque
 227.         * operations (e.g., addFirst, peekLast, pollLast) are linearizable
 228.         * (see Herlihy and Shavit's book).  However, some combinations of
 229.         * operations are known not to be linearizable.  In particular,
 230.         * when an addFirst(A) is racing with pollFirst() removing B, it is
 231.         * possible for an observer iterating over the elements to observe
 232.         * A B C and subsequently observe A C, even though no interior
 233.         * removes are ever performed.  Nevertheless, iterators behave
 234.         * reasonably, providing the "weakly consistent" guarantees.
 235.         *
 236.         * Empirically, microbenchmarks suggest that this class adds about
 237.         * 40% overhead relative to ConcurrentLinkedQueue, which feels as
 238.         * good as we can hope for.
 239.         */

 330.         * Links e as first element.
 331.         */
 332.        private Node linkFirst(E e) {
 333.            checkNotNull(e);
 334.            final Node<E> newNode = new Node<>(e);
 335.    
 336.            restartFromHead:
 337.            for (;;)
 338.                for (Node<E> h = head, p = h, q;;) {
 339.                    if ((q = p.prev) != null &&
 340.                        (q = (p = q).prev) != null)
 341.                        // Check for head updates every other hop.
 342.                        // If p == q, we are sure to follow head instead.
 343.                        p = (h != (h = head)) ? h : q;
 344.                    else if (p.next == p) // PREV_TERMINATOR
 345.                        continue restartFromHead;
 346.                    else {
 347.                        // p is first node
 348.                        newNode.lazySetNext(p); // CAS piggyback
 349.                        if (p.casPrev(null, newNode)) {
 350.                            // Successful CAS is the linearization point
 351.                            // for e to become an element of this deque,
 352.                            // and for newNode to become "live".
 353.                            if (p != h) // hop two nodes at a time
 354.                                casHead(h, newNode);  // Failure is OK.
 355.                            return newNode;
 356.                        }
 357.                        // Lost CAS race to another thread; re-read prev
 358.                    }
 359.                }
 360.        }

 397.        /**
 398.         * Unlinks non-null node x.
 399.         */
 400.        void unlink(Node<E> x) {
 401.            // assert x != null;
 402.            // assert x.item == null;
 403.            // assert x != PREV_TERMINATOR;
 404.            // assert x != NEXT_TERMINATOR;
 405.    
 406.            final Node<E> prev = x.prev;
 407.            final Node<E> next = x.next;
 408.            if (prev == null) {
 409.                unlinkFirst(x, next);
 410.            } else if (next == null) {
 411.                unlinkLast(x, prev);
 412.            } else {
 413.                // Unlink interior node.
 414.                //
 415.                // This is the common case, since a series of polls at the
 416.                // same end will be "interior" removes, except perhaps for
 417.                // the first one, since end nodes cannot be unlinked.
 418.                //
 419.                // At any time, all active nodes are mutually reachable by
 420.                // following a sequence of either next or prev pointers.
 421.                //
 422.                // Our strategy is to find the unique active predecessor
 423.                // and successor of x.  Try to fix up their links so that
 424.                // they point to each other, leaving x unreachable from
 425.                // active nodes.  If successful, and if x has no live
 426.                // predecessor/successor, we additionally try to gc-unlink,
 427.                // leaving active nodes unreachable from x, by rechecking
 428.                // that the status of predecessor and successor are
 429.                // unchanged and ensuring that x is not reachable from
 430.                // tail/head, before setting x's prev/next links to their
 431.                // logical approximate replacements, self/TERMINATOR.
 432.                Node<E> activePred, activeSucc;
 433.                boolean isFirst, isLast;
 434.                int hops = 1;
 435.    
 436.                // Find active predecessor
 437.                for (Node<E> p = prev; ; ++hops) {
 438.                    if (p.item != null) {
 439.                        activePred = p;
 440.                        isFirst = false;
 441.                        break;
 442.                    }
 443.                    Node<E> q = p.prev;
 444.                    if (q == null) {
 445.                        if (p.next == p)
 446.                            return;
 447.                        activePred = p;
 448.                        isFirst = true;
 449.                        break;
 450.                    }
 451.                    else if (p == q)
 452.                        return;
 453.                    else
 454.                        p = q;
 455.                }
 456.    
 457.                // Find active successor
 458.                for (Node<E> p = next; ; ++hops) {
 459.                    if (p.item != null) {
 460.                        activeSucc = p;
 461.                        isLast = false;
 462.                        break;
 463.                    }
 464.                    Node<E> q = p.next;
 465.                    if (q == null) {
 466.                        if (p.prev == p)
 467.                            return;
 468.                        activeSucc = p;
 469.                        isLast = true;
 470.                        break;
 471.                    }
 472.                    else if (p == q)
 473.                        return;
 474.                    else
 475.                        p = q;
 476.                }
 477.    
 478.                // TODO: better HOP heuristics
 479.                if (hops < HOPS
 480.                    // always squeeze out interior deleted nodes
 481.                    && (isFirst | isLast))
 482.                    return;
 483.    
 484.                // Squeeze out deleted nodes between activePred and
 485.                // activeSucc, including x.
 486.                skipDeletedSuccessors(activePred);
 487.                skipDeletedPredecessors(activeSucc);
 488.    
 489.                // Try to gc-unlink, if possible
 490.                if ((isFirst | isLast) &&
 491.    
 492.                    // Recheck expected state of predecessor and successor
 493.                    (activePred.next == activeSucc) &&
 494.                    (activeSucc.prev == activePred) &&
 495.                    (isFirst ? activePred.prev == null : activePred.item != null) &&
 496.                    (isLast  ? activeSucc.next == null : activeSucc.item != null)) {
 497.    
 498.                    updateHead(); // Ensure x is not reachable from head
 499.                    updateTail(); // Ensure x is not reachable from tail
 500.    
 501.                    // Finally, actually gc-unlink
 502.                    x.lazySetPrev(isFirst ? prevTerminator() : x);
 503.                    x.lazySetNext(isLast  ? nextTerminator() : x);
 504.                }
 505.            }
 506.        }

 576.        /**
 577.         * Guarantees that any node which was unlinked before a call to
 578.         * this method will be unreachable from head after it returns.
 579.         * Does not guarantee to eliminate slack, only that head will
 580.         * point to a node that was active while this method was running.
 581.         */
 582.        private void updateHead() {
 583.            // Either head already points to an active node, or we keep
 584.            // trying to cas it to the first node until it does.
 585.            Node<E> h, p, q;
 586.            restartFromHead:
 587.            while ((h = head).item == null && (p = h.prev) != null) {
 588.                for (;;) {
 589.                    if ((q = p.prev) == null ||
 590.                        (q = (p = q).prev) == null) {
 591.                        // It is possible that p is PREV_TERMINATOR,
 592.                        // but if so, the CAS is guaranteed to fail.
 593.                        if (casHead(h, p))
 594.                            return;
 595.                        else
 596.                            continue restartFromHead;
 597.                    }
 598.                    else if (h != head)
 599.                        continue restartFromHead;
 600.                    else
 601.                        p = q;
 602.                }
 603.            }
 604.        }

 609.         * Does not guarantee to eliminate slack, only that tail will

 636.        private void skipDeletedPredecessors(Node<E> x) {
 637.            whileActive:
 638.            do {
 639.                Node<E> prev = x.prev;
 640.                // assert prev != null;
 641.                // assert x != NEXT_TERMINATOR;
 642.                // assert x != PREV_TERMINATOR;
 643.                Node<E> p = prev;
 644.                findActive:
 645.                for (;;) {
 646.                    if (p.item != null)
 647.                        break findActive;
 648.                    Node<E> q = p.prev;
 649.                    if (q == null) {
 650.                        if (p.next == p)
 651.                            continue whileActive;
 652.                        break findActive;
 653.                    }
 654.                    else if (p == q)
 655.                        continue whileActive;
 656.                    else
 657.                        p = q;
 658.                }
 659.    
 660.                // found active CAS target
 661.                if (prev == p || x.casPrev(prev, p))
 662.                    return;
 663.    
 664.            } while (x.item != null || x.next == null);
 665.        }

 721.         *     p.prev == null && p.next != p
 722.         * The returned node may or may not be logically deleted.
 723.         * Guarantees that head is set to the returned node.
 724.         */
 725.        Node<E> first() {

 735.                             // It is possible that p is PREV_TERMINATOR,
 736.                             // but if so, the CAS is guaranteed to fail.
 737.                             || casHead(h, p))
 738.                        return p;

 741.                }
 742.        }

 744.        /**
 745.         * Returns the last node, the unique node p for which:
 746.         *     p.next == null && p.prev != p
 747.         * The returned node may or may not be logically deleted.
 748.         * Guarantees that tail is set to the returned node.
 749.         */
 750.        Node<E> last() {
 751.            restartFromTail:

 753.                for (Node<E> t = tail, p = t, q;;) {
 754.                    if ((q = p.next) != null &&
 755.                        (q = (p = q).next) != null)
 756.                        // Check for tail updates every other hop.
 757.                        // If p == q, we are sure to follow tail instead.

 761.                             // but if so, the CAS is guaranteed to fail.
