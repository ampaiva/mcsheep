 116.         * @param contextID the JACC contextID to be set.
 117.         * @return the previous contextID as retrieved from the {@code PolicyContext}.
 118.         */
 119.        protected String setContextID(final String contextID) {
 120.            if (! WildFlySecurityManager.isChecking()) {
 121.                final String previousID = PolicyContext.getContextID();
 122.                PolicyContext.setContextID(contextID);
 123.                return previousID;
 124.            } else {
 125.                final PrivilegedAction<String> action = new SetContextIDAction(contextID);
 126.                return AccessController.doPrivileged(action);
 127.            }
 128.        }
 129.    
 130.        /**
 131.         * PrivilegedAction that sets the {@code PolicyContext} id.
 132.         */
 133.        private static class SetContextIDAction implements PrivilegedAction<String> {
 134.    
 135.            private String contextID;
 136.    
 137.            SetContextIDAction(final String contextID) {
 138.                this.contextID = contextID;
 139.            }
 140.    
 141.            @Override
 142.            public String run() {
 143.                final String previousID = PolicyContext.getContextID();
 144.                PolicyContext.setContextID(this.contextID);
 145.                return previousID;
 146.            }
 147.        }
