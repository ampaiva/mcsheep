 216.        private TldMetaData parseTLD(VirtualFile tld)
 217.                throws DeploymentUnitProcessingException {
 218.            if (IMPLICIT_TLD.equals(tld.getName())) {
 219.                // Implicit TLDs are different from regular TLDs
 220.                return new TldMetaData();
 221.            }
 222.            InputStream is = null;
 223.            try {
 224.                is = tld.openStream();
 225.                final XMLInputFactory inputFactory = XMLInputFactory.newInstance();
 226.                inputFactory.setXMLResolver(NoopXMLResolver.create());
 227.                XMLStreamReader xmlReader = inputFactory.createXMLStreamReader(is);
 228.                return TldMetaDataParser.parse(xmlReader);
 229.            } catch (XMLStreamException e) {
 230.                throw new DeploymentUnitProcessingException(UndertowLogger.ROOT_LOGGER.failToParseXMLDes...
 231.                        e.getLocation().getColumnNumber()), e);
 232.            } catch (IOException e) {
 233.                throw new DeploymentUnitProcessingException(UndertowLogger.ROOT_LOGGER.failToParseXMLDes...
 234.            } finally {
 235.                try {
 236.                    if (is != null) {
 237.                        is.close();
 238.                    }
 239.                } catch (IOException e) {
 240.                    // Ignore
 241.                }
 242.            }
