 130.        private String getVariableBundleKey(final String name, final String... variable) {
 131.            final String prefix = sharedAttributeResolver.get(name);
 132.            StringBuilder sb = new StringBuilder(InfinispanExtension.SUBSYSTEM_NAME);
 133.            // construct the key prefix
 134.            if (prefix != null) {
 135.                sb.append('.').append(prefix);
 136.            }
 137.            sb.append('.').append(name);
 138.            // construct the key suffix
 139.            if (variable != null) {
 140.                for (String arg : variable) {
 141.                    sb.append('.').append(arg);
 142.                }
 143.            }
 144.            return sb.toString();
 145.        }
