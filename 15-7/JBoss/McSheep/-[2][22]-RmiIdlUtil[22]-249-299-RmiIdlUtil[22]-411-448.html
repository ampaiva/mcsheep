 164.        public static boolean isRMIIIOPType(Class type) {
 165.    
 166.            /*
 167.            *  Java Language to IDL Mapping
 168.            *  ftp://ftp.omg.org/pub/docs/ptc/99-03-09.pdf
 169.            *
 170.            *  A conforming RMI/IDL type is a Java type whose values
 171.            *  may be transmitted across an RMI/IDL remote interface at
 172.            *  run-time. A Java data type is a conforming RMI/IDL type
 173.            *  if it is:
 174.            *
 175.            *  - one of the Java primitive types (see Primitive Types on page 28-2).
 176.            *  - a conforming remote interface (as defined in RMI/IDL Remote Interfaces on page 28-2).
 177.            *  - a conforming value type (as defined in RMI/IDL Value Types on page 28-4).
 178.            *  - an array of conforming RMI/IDL types (see RMI/IDL Arrays on page 28-5).
 179.            *  - a conforming exception type (see RMI/IDL Exception Types on page 28-5).
 180.            *  - a conforming CORBA object reference type (see CORBA Object Reference Types on page 28-6...
 181.            *  - a conforming IDL entity type see IDL Entity Types on page 28-6).
 182.            */
 183.    
 184.            /*
 185.            * Primitive types.
 186.            *
 187.            * Spec 28.2.2
 188.            */
 189.            if (type.isPrimitive())
 190.                return true;
 191.    
 192.            /*
 193.            * Conforming array.
 194.            *
 195.            * Spec 28.2.5
 196.            */
 197.            if (type.isArray())
 198.                return isRMIIIOPType(type.getComponentType());
 199.    
 200.            /*
 201.            * Conforming CORBA reference type
 202.            *
 203.            * Spec 28.2.7
 204.            */
 205.            if (org.omg.CORBA.Object.class.isAssignableFrom(type))
 206.                return true;
 207.    
 208.            /*
 209.            * Conforming IDL Entity type
 210.            *
 211.            * Spec 28.2.8
 212.            */
 213.            if (org.omg.CORBA.portable.IDLEntity.class.isAssignableFrom(type))
 214.                return true;
 215.    
 216.            /*
 217.            * Conforming remote interface.
 218.            *
 219.            * Spec 28.2.3
 220.            */
 221.            if (isRMIIDLRemoteInterface(type))
 222.                return true;
 223.    
 224.            /*
 225.            * Conforming exception.
 226.            *
 227.            * Spec 28.2.6
 228.            */
 229.            if (isRMIIDLExceptionType(type))
 230.                return true;
 231.    
 232.            /*
 233.            * Conforming value type.
 234.            *
 235.            * Spec 28.2.4
 236.            */
 237.            if (isRMIIDLValueType(type))
 238.                return true;
 239.    
 240.            return false;
 241.        }

 400.        public static boolean isAbstractValueType(Class type) {
 401.    
 402.            if (!type.isInterface())
 403.                return false;
 404.    
 405.            if (org.omg.CORBA.Object.class.isAssignableFrom(type))
 406.                return false;
 407.    
 408.            boolean cannotBeRemote = false;
 409.            boolean cannotBeAbstractInterface = false;
 410.    
 411.            if (java.rmi.Remote.class.isAssignableFrom(type)) {
 412.                cannotBeAbstractInterface = true;
 413.            } else {
 414.                cannotBeRemote = true;
 415.            }
 416.    
 417.            Iterator methodIterator = Arrays.asList(type.getMethods()).iterator();
 418.    
 419.            while (methodIterator.hasNext()) {
 420.                Method m = (Method) methodIterator.next();
 421.    
 422.                if (!throwsRemoteException(m)) {
 423.                    cannotBeAbstractInterface = true;
 424.                    cannotBeRemote = true;
 425.                    break;
 426.                }
 427.    
 428.                Iterator it = Arrays.asList(m.getExceptionTypes()).iterator();
 429.    
 430.                while (it.hasNext()) {
 431.                    Class exception = (Class) it.next();
 432.    
 433.                    if (!isRMIIDLExceptionType(exception)) {
 434.                        cannotBeRemote = true;
 435.                        break;
 436.                    }
 437.                }
 438.            }
 439.    
 440.            if (!cannotBeRemote) {
 441.                Iterator fieldIterator = Arrays.asList(type.getFields()).iterator();
 442.    
 443.                while (fieldIterator.hasNext()) {
 444.                    Field f = (Field) fieldIterator.next();
 445.    
 446.                    if (f.getType().isPrimitive())
 447.                        continue;
 448.                    if (f.getType().equals(java.lang.String.class))
 449.                        continue;
 450.                    cannotBeRemote = true;
 451.                    break;
 452.                }
 453.            }
 454.            return cannotBeRemote && cannotBeAbstractInterface;
 455.        }

 457.        public static void rethrowIfCorbaSystemException(Exception e) {
 458.            if (e instanceof java.rmi.MarshalException)
 459.                throw new org.omg.CORBA.MARSHAL(e.toString());
 460.            else if (e instanceof java.rmi.NoSuchObjectException)
 461.                throw new org.omg.CORBA.OBJECT_NOT_EXIST(e.toString());
 462.            else if (e instanceof java.rmi.AccessException)
 463.                throw new org.omg.CORBA.NO_PERMISSION(e.toString());
 464.            else if (e instanceof javax.transaction.TransactionRequiredException)
 465.                throw new org.omg.CORBA.TRANSACTION_REQUIRED(e.toString());
 466.            else if (e instanceof javax.transaction.TransactionRolledbackException)
 467.                throw new org.omg.CORBA.TRANSACTION_ROLLEDBACK(e.toString());
 468.            else if (e instanceof javax.transaction.InvalidTransactionException)
 469.                throw new org.omg.CORBA.INVALID_TRANSACTION(e.toString());
 470.            else if (e instanceof org.omg.CORBA.SystemException)
 471.                throw (org.omg.CORBA.SystemException) e;
 472.        }

 164.        public static boolean isRMIIIOPType(Class type) {
 165.    
 166.            /*
 167.            *  Java Language to IDL Mapping
 168.            *  ftp://ftp.omg.org/pub/docs/ptc/99-03-09.pdf
 169.            *
 170.            *  A conforming RMI/IDL type is a Java type whose values
 171.            *  may be transmitted across an RMI/IDL remote interface at
 172.            *  run-time. A Java data type is a conforming RMI/IDL type
 173.            *  if it is:
 174.            *
 175.            *  - one of the Java primitive types (see Primitive Types on page 28-2).
 176.            *  - a conforming remote interface (as defined in RMI/IDL Remote Interfaces on page 28-2).
 177.            *  - a conforming value type (as defined in RMI/IDL Value Types on page 28-4).
 178.            *  - an array of conforming RMI/IDL types (see RMI/IDL Arrays on page 28-5).
 179.            *  - a conforming exception type (see RMI/IDL Exception Types on page 28-5).
 180.            *  - a conforming CORBA object reference type (see CORBA Object Reference Types on page 28-6...
 181.            *  - a conforming IDL entity type see IDL Entity Types on page 28-6).
 182.            */
 183.    
 184.            /*
 185.            * Primitive types.
 186.            *
 187.            * Spec 28.2.2
 188.            */
 189.            if (type.isPrimitive())
 190.                return true;
 191.    
 192.            /*
 193.            * Conforming array.
 194.            *
 195.            * Spec 28.2.5
 196.            */
 197.            if (type.isArray())
 198.                return isRMIIIOPType(type.getComponentType());
 199.    
 200.            /*
 201.            * Conforming CORBA reference type
 202.            *
 203.            * Spec 28.2.7
 204.            */
 205.            if (org.omg.CORBA.Object.class.isAssignableFrom(type))
 206.                return true;
 207.    
 208.            /*
 209.            * Conforming IDL Entity type
 210.            *
 211.            * Spec 28.2.8
 212.            */
 213.            if (org.omg.CORBA.portable.IDLEntity.class.isAssignableFrom(type))
 214.                return true;
 215.    
 216.            /*
 217.            * Conforming remote interface.
 218.            *
 219.            * Spec 28.2.3
 220.            */
 221.            if (isRMIIDLRemoteInterface(type))
 222.                return true;
 223.    
 224.            /*
 225.            * Conforming exception.
 226.            *
 227.            * Spec 28.2.6
 228.            */
 229.            if (isRMIIDLExceptionType(type))
 230.                return true;
 231.    
 232.            /*
 233.            * Conforming value type.
 234.            *
 235.            * Spec 28.2.4
 236.            */
 237.            if (isRMIIDLValueType(type))
 238.                return true;
 239.    
 240.            return false;
 241.        }

 243.        public static boolean isRMIIDLRemoteInterface(Class type) {
 244.    
 245.            /*
 246.            * If does not implement java.rmi.Remote, cannot be valid RMI-IDL
 247.            * remote interface.
 248.            */
 249.            if (!java.rmi.Remote.class.isAssignableFrom(type))
 250.                return false;
 251.    
 252.            Iterator methodIterator = Arrays.asList(type.getMethods()).iterator();
 253.    
 254.            while (methodIterator.hasNext()) {
 255.                Method m = (Method) methodIterator.next();
 256.    
 257.                /*
 258.                * All methods in the interface MUST throw
 259.                * java.rmi.RemoteException or a superclass of it.
 260.                *
 261.                * Spec 28.2.3 (2)
 262.                */
 263.                if (!throwsRemoteException(m)) {
 264.                    return false;
 265.                }
 266.    
 267.                /*
 268.                * All checked exception classes used in method declarations
 269.                * (other than java.rmi.RemoteException) MUST be conforming
 270.                * RMI/IDL exception types.
 271.                *
 272.                * Spec 28.2.3 (4)
 273.                */
 274.                Iterator it = Arrays.asList(m.getExceptionTypes()).iterator();
 275.    
 276.                while (it.hasNext()) {
 277.                    Class exception = (Class) it.next();
 278.    
 279.                    if (!isRMIIDLExceptionType(exception))
 280.                        return false;
 281.                }
 282.            }
 283.    
 284.            /*
 285.            * The constant values defined in the interface MUST be
 286.            * compile-time types of RMI/IDL primitive types or String.
 287.            *
 288.            * Spec 28.2.3 (6)
 289.            */
 290.            Iterator fieldIterator = Arrays.asList(type.getFields()).iterator();
 291.    
 292.            while (fieldIterator.hasNext()) {
 293.    
 294.                Field f = (Field) fieldIterator.next();
 295.    
 296.                if (f.getType().isPrimitive())
 297.                    continue;
 298.    
 299.                if (f.getType().equals(java.lang.String.class))
 300.                    continue;
 301.    
 302.                return false;
 303.            }
 304.    
 305.            return true;
 306.        }

 308.        public static boolean isAbstractInterface(Class type) {
 309.    
 310.            /*
 311.            * It must be a Java interface.
 312.            */
 313.            if (!type.isInterface())
 314.                return false;
 315.    
 316.            /*
 317.            * It must not be the interface of a CORBA object.
 318.            */
 319.            if (org.omg.CORBA.Object.class.isAssignableFrom(type))
 320.                return false;
 321.    
 322.            /*
 323.            * It must not extend java.rmi.Remote directly or indirectly.
 324.            */
 325.            if (java.rmi.Remote.class.isAssignableFrom(type))
 326.                return false;
 327.    
 328.            Iterator methodIterator = Arrays.asList(type.getMethods()).iterator();
 329.    
 330.            while (methodIterator.hasNext()) {
 331.                Method m = (Method) methodIterator.next();
 332.    
 333.                /*
 334.                * All methods MUST throw java.rmi.RemoteException
 335.                * or a superclass of it.
 336.                */
 337.                if (!throwsRemoteException(m)) {
 338.                    return false;
 339.                }
 340.    
 341.            }
 342.    
 343.            return true;
 344.        }

 346.        public static boolean isRMIIDLExceptionType(Class type) {

 348.            /*
 349.            * A conforming RMI/IDL Exception class MUST be a checked
 350.            * exception class and MUST be a valid RMI/IDL value type.
 351.            *
 352.            * Spec 28.2.6
 353.            */
 354.            if (!Throwable.class.isAssignableFrom(type))
 355.                return false;
 356.    
 357.            if (Error.class.isAssignableFrom(type))
 358.                return false;
 359.    
 360.            if (RuntimeException.class.isAssignableFrom(type))
 361.                return false;
 362.    
 363.            if (!isRMIIDLValueType(type))
 364.                return false;
 365.    
 366.            return true;
 367.        }

 369.        public static boolean isRMIIDLValueType(Class type) {
 370.    
 371.            /*
 372.            * A value type MUST NOT either directly or indirectly implement the
 373.            * java.rmi.Remote interface.
 374.            *
 375.            * Spec 28.2.4 (4)
 376.            */
 377.            if (java.rmi.Remote.class.isAssignableFrom(type))
 378.                return false;
 379.    
 380.    
 381.            /*
 382.            * It cannot be a CORBA object.
 383.            */
 384.            if (org.omg.CORBA.Object.class.isAssignableFrom(type))
 385.                return false;
 386.    
 387.            /*
 388.            * If class is a non-static inner class then its containing class must
 389.            * also be a conforming RMI/IDL value type.
 390.            *
 391.            * Spec 2.8.4 (3)
 392.            */
 393.            if (type.getDeclaringClass() != null && isStatic(type))
 394.                if (!isRMIIDLValueType(type.getDeclaringClass()))
 395.                    return false;
 396.    
 397.            return true;
 398.        }

 400.        public static boolean isAbstractValueType(Class type) {

 403.                return false;

 406.                return false;
 407.    
 408.            boolean cannotBeRemote = false;
 409.            boolean cannotBeAbstractInterface = false;
 410.    
 411.            if (java.rmi.Remote.class.isAssignableFrom(type)) {
 412.                cannotBeAbstractInterface = true;
 413.            } else {
 414.                cannotBeRemote = true;
 415.            }
 416.    
 417.            Iterator methodIterator = Arrays.asList(type.getMethods()).iterator();
 418.    
 419.            while (methodIterator.hasNext()) {
 420.                Method m = (Method) methodIterator.next();
 421.    
 422.                if (!throwsRemoteException(m)) {
 423.                    cannotBeAbstractInterface = true;
 424.                    cannotBeRemote = true;
 425.                    break;
 426.                }
 427.    
 428.                Iterator it = Arrays.asList(m.getExceptionTypes()).iterator();
 429.    
 430.                while (it.hasNext()) {
 431.                    Class exception = (Class) it.next();
 432.    
 433.                    if (!isRMIIDLExceptionType(exception)) {
 434.                        cannotBeRemote = true;
 435.                        break;
 436.                    }
 437.                }
 438.            }
 439.    
 440.            if (!cannotBeRemote) {
 441.                Iterator fieldIterator = Arrays.asList(type.getFields()).iterator();
 442.    
 443.                while (fieldIterator.hasNext()) {
 444.                    Field f = (Field) fieldIterator.next();
 445.    
 446.                    if (f.getType().isPrimitive())
 447.                        continue;
 448.                    if (f.getType().equals(java.lang.String.class))
 449.                        continue;
 450.                    cannotBeRemote = true;
 451.                    break;
 452.                }
 453.            }
 454.            return cannotBeRemote && cannotBeAbstractInterface;
 455.        }
 456.    
 457.        public static void rethrowIfCorbaSystemException(Exception e) {
 458.            if (e instanceof java.rmi.MarshalException)
 459.                throw new org.omg.CORBA.MARSHAL(e.toString());
 460.            else if (e instanceof java.rmi.NoSuchObjectException)
 461.                throw new org.omg.CORBA.OBJECT_NOT_EXIST(e.toString());
 462.            else if (e instanceof java.rmi.AccessException)
 463.                throw new org.omg.CORBA.NO_PERMISSION(e.toString());
 464.            else if (e instanceof javax.transaction.TransactionRequiredException)
 465.                throw new org.omg.CORBA.TRANSACTION_REQUIRED(e.toString());
 466.            else if (e instanceof javax.transaction.TransactionRolledbackException)
 467.                throw new org.omg.CORBA.TRANSACTION_ROLLEDBACK(e.toString());
 468.            else if (e instanceof javax.transaction.InvalidTransactionException)
 469.                throw new org.omg.CORBA.INVALID_TRANSACTION(e.toString());
 470.            else if (e instanceof org.omg.CORBA.SystemException)
 471.                throw (org.omg.CORBA.SystemException) e;
 472.        }
