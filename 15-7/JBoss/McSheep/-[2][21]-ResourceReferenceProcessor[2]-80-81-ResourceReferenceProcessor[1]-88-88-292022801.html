 360.        private Class<?> loadClass(String className, ClassLoader cl) throws ClassNotFoundException {
 361.            if (className == null || className.trim().isEmpty()) {
 362.                throw EeLogger.ROOT_LOGGER.cannotBeNullOrEmpty("Classname", className);
 363.            }
 364.            if (className.equals(void.class.getName())) {
 365.                return void.class;
 366.            }
 367.            if (className.equals(byte.class.getName())) {
 368.                return byte.class;
 369.            }
 370.            if (className.equals(short.class.getName())) {
 371.                return short.class;
 372.            }
 373.            if (className.equals(int.class.getName())) {
 374.                return int.class;
 375.            }
 376.            if (className.equals(long.class.getName())) {
 377.                return long.class;
 378.            }
 379.            if (className.equals(char.class.getName())) {
 380.                return char.class;
 381.            }
 382.            if (className.equals(boolean.class.getName())) {
 383.                return boolean.class;
 384.            }
 385.            if (className.equals(float.class.getName())) {
 386.                return float.class;
 387.            }
 388.            if (className.equals(double.class.getName())) {
 389.                return double.class;
 390.            }
 391.            // Now that we know its not a primitive, lets just allow
 392.            // the passed classloader to handle the request.
 393.            return Class.forName(className, false, cl);
 394.        }
