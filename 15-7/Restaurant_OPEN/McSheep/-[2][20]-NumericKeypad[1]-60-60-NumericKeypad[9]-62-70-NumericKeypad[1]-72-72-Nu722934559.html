 296.            public void actionPerformed(ActionEvent e) {
 297.                PosButton button = (PosButton) e.getSource();
 298.                String s = button.getActionCommand();
 299.                if (s.equals("CLEAR")) {
 300.                    tfAmountTendered.setText("0");
 301.                }
 302.                else if (s.equals(".")) {
 303.                    if (tfAmountTendered.getText().indexOf('.') < 0) {
 304.                        tfAmountTendered.setText(tfAmountTendered.getText() + ".");
 305.                    }
 306.                }
 307.                else {
 308.                    String string = tfAmountTendered.getText();
 309.                    int index = string.indexOf('.');
 310.                    if (index < 0) {
 311.                        double value = 0;
 312.                        try {
 313.                            value = Double.parseDouble(string);
 314.                        } catch (NumberFormatException x) {
