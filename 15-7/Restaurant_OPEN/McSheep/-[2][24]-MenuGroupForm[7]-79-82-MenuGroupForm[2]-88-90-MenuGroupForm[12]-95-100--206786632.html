 158.        protected void updateView() {
 159.            MenuGroup foodGroup = (MenuGroup) getBean();
 160.            if(foodGroup == null) {
 161.                tfName.setText("");
 162.                cbCategory.setSelectedItem(null);
 163.                chkVisible.setSelected(false);
 164.                return;
 165.            }
 166.            tfName.setText(foodGroup.getName());
 167.            chkVisible.setSelected(foodGroup.isVisible());
 168.            
 169.            if(foodGroup.getParent() != null) {
 170.                cbCategory.setSelectedItem(foodGroup.getParent());
 171.            }
 172.        }

 174.        @Override
 175.        protected boolean updateModel() {
 176.            MenuGroup foodGroup = (MenuGroup) getBean();
 177.            if(foodGroup == null) {
 178.                return false;
 179.            }
 180.            
 181.            String name = tfName.getText();
 182.            if(POSUtil.isBlankOrNull(name)) {
 183.                MessageDialog.showError("Name is required");
 184.                return false;
 185.            }
 186.            
 187.            MenuCategory category = (MenuCategory) cbCategory.getSelectedItem();
 188.            if(category == null) {
 189.                MessageDialog.showError("Category is required");
 190.                return false;
 191.            }
 192.            
 193.            foodGroup.setName(tfName.getText());
 194.            
 195.            foodGroup.setParent(category);
 196.            foodGroup.setVisible(chkVisible.isSelected());
 197.            
 198.            return true;
 199.        }
