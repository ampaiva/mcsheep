 323.        public void actionPerformed(ActionEvent e) {
 324.            if ("scrollUP".equals(e.getActionCommand())) {
 325.                if (couponList == null || couponList.size() == 0)
 326.                    return;
 327.    
 328.                int selectedRow = listCoupons.getSelectedIndex();
 329.    
 330.                if (selectedRow <= 0) {
 331.                    selectedRow = 0;
 332.                }
 333.                else {
 334.                    --selectedRow;
 335.                }
 336.    
 337.                listCoupons.setSelectedIndex(selectedRow);
 338.                Rectangle cellRect = listCoupons.getCellBounds(selectedRow, selectedRow);
 339.                listCoupons.scrollRectToVisible(cellRect);
 340.            }
 341.            else if ("scrollDown".equals(e.getActionCommand())) {
 342.                if (couponList == null || couponList.size() == 0)
 343.                    return;
 344.    
 345.                int selectedRow = listCoupons.getSelectedIndex();
 346.    
 347.                if (selectedRow < 0) {
 348.                    selectedRow = 0;
 349.                }
 350.                else if (selectedRow >= couponList.size() - 1) {
 351.                    //return;
 352.                }
 353.                else {
 354.                    ++selectedRow;
 355.                }
 356.    
 357.                listCoupons.setSelectedIndex(selectedRow);
 358.                Rectangle cellRect = listCoupons.getCellBounds(selectedRow, selectedRow);
 359.                cellRect.y += 20;
 360.                listCoupons.scrollRectToVisible(cellRect);
 361.            }
 362.        }
