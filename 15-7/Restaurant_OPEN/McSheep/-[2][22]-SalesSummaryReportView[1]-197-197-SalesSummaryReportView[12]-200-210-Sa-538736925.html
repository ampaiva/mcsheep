 135.            toDate = toDatePicker.getDate();

 138.                POSMessageDialog.showError(Application.getInstance().getBackOfficeWindow(), com.floreantpos.POSCo...

 142.            dateDiff = (int) ((toDate.getTime() - fromDate.getTime()) * (1.15740741 * Math.pow(10, -8))) + 1;
 143.            userType = (UserType) cbUserType.getSelectedItem();
 144.    //        if (userType.equalsIgnoreCase(com.floreantpos.POSConstants.ALL)) {
 145.    //            userType = null;
 146.    //        }
 147.            terminal = null;
 148.            if (cbTerminal.getSelectedItem() instanceof Terminal) {
 149.                terminal = (Terminal) cbTerminal.getSelectedItem();
 150.            }
 151.    
 152.            Calendar calendar = Calendar.getInstance();
 153.            calendar.clear();
 154.    
 155.            Calendar calendar2 = Calendar.getInstance();
 156.            calendar2.setTime(fromDate);
 157.    
 158.            calendar.set(Calendar.YEAR, calendar2.get(Calendar.YEAR));
 159.            calendar.set(Calendar.MONTH, calendar2.get(Calendar.MONTH));
 160.            calendar.set(Calendar.DATE, calendar2.get(Calendar.DATE));
 161.            calendar.set(Calendar.HOUR, 0);
 162.            calendar.set(Calendar.MINUTE, 0);
 163.            calendar.set(Calendar.SECOND, 0);
 164.            fromDate = calendar.getTime();
 165.    
 166.            calendar.clear();
 167.            calendar2.setTime(toDate);
 168.            calendar.set(Calendar.YEAR, calendar2.get(Calendar.YEAR));
 169.            calendar.set(Calendar.MONTH, calendar2.get(Calendar.MONTH));
 170.            calendar.set(Calendar.DATE, calendar2.get(Calendar.DATE));
 171.            calendar.set(Calendar.HOUR, 23);
 172.            calendar.set(Calendar.MINUTE, 59);
 173.            calendar.set(Calendar.SECOND, 59);
 174.            toDate = calendar.getTime();
 175.    
 176.            return true;
 177.        }
