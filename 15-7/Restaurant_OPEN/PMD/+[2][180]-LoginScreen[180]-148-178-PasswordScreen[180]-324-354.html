 293.        public void doLogin() {
 294.            try {
 295.                Application application = Application.getInstance();
 296.    
 297.                int userId = captureUserId();
 298.    
 299.                String newPass = capturePassword();
 300.    
 301.                UserDAO dao = new UserDAO();
 302.                User user = dao.findUser(userId);
 303.                if (!user.getPassword().equalsIgnoreCase(
 304.                        PasswordHasher.hashPassword(newPass))) {
 305.                    throw new RuntimeException(POSConstants.WRONG_PASSWORD);
 306.                }
 307.    
 308.                Shift currentShift = ShiftUtil.getCurrentShift();
 309.                if (currentShift == null) {
 310.                    throw new RuntimeException(POSConstants.NO_SHIFT_CONFIGURED);
 311.                }
 312.    
 313.                adjustUserShift(user, currentShift);
 314.    
 315.                application.setCurrentUser(user);
 316.                application.setCurrentShift(currentShift);
 317.    
 318.                tfUserId.setText("");
 319.                tfPassword.setText("");
 320.                application.getRootView().showView(SwitchboardView.VIEW_NAME);
 321.            } catch (Exception e1) {
 322.                MessageDialog.showError(e1.getMessage());
 323.            }
 324.        }

 326.        private void adjustUserShift(User user, Shift currentShift) {
 327.            Application application = Application.getInstance();
 328.            Calendar currentTime = Calendar.getInstance();
 329.    
 330.            if (user.isClockedIn() != null && user.isClockedIn().booleanValue()) {
 331.                Shift userShift = user.getCurrentShift();
 332.                Date userLastClockInTime = user.getLastClockInTime();
 333.                long elaspedTimeSinceLastLogin = Math.abs(currentTime
 334.                        .getTimeInMillis() - userLastClockInTime.getTime());
 335.    
 336.                if (userShift != null) {
 337.                    if (!userShift.equals(currentShift)) {
 338.                        reClockInUser(currentTime, user, currentShift);
 339.                    } else if (userShift.getShiftLength() != null
 340.                            && (elaspedTimeSinceLastLogin >= userShift
 341.                                    .getShiftLength())) {
 342.                        reClockInUser(currentTime, user, currentShift);
 343.                    }
 344.                } else {
 345.                    user.doClockIn(application.getTerminal(), currentShift,
 346.                            currentTime);
 347.                }
 348.            } else {
 349.                user.doClockIn(application.getTerminal(), currentShift, currentTime);
 350.            }
 351.        }

 367.        private void reClockInUser(Calendar currentTime, User user,
 368.                Shift currentShift) {
 369.            POSMessageDialog
 370.                    .showMessage("You will be clocked out from previous Shift");
 371.    
 372.            Application application = Application.getInstance();
 373.            AttendenceHistoryDAO attendenceHistoryDAO = new AttendenceHistoryDAO();
 374.    
 375.            AttendenceHistory attendenceHistory = attendenceHistoryDAO
 376.                    .findHistoryByClockedInTime(user);
 377.            if (attendenceHistory == null) {
 378.                attendenceHistory = new AttendenceHistory();
 379.                Date lastClockInTime = user.getLastClockInTime();
 380.                Calendar c = Calendar.getInstance();
 381.                c.setTime(lastClockInTime);
 382.                attendenceHistory.setClockInTime(lastClockInTime);
 383.                attendenceHistory.setClockInHour(Short.valueOf((short) c
 384.                        .get(Calendar.HOUR)));
 385.                attendenceHistory.setUser(user);
 386.                attendenceHistory.setTerminal(application.getTerminal());
 387.                attendenceHistory.setShift(user.getCurrentShift());
 388.            }
 389.    
 390.            user.doClockOut(attendenceHistory, currentShift, currentTime);
 391.    
 392.            user.doClockIn(application.getTerminal(), currentShift, currentTime);
 393.        }
