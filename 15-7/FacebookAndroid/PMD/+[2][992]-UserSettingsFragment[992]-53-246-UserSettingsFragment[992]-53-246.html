 171.                                connectedStateLabel.setCompoundDrawables(
 172.                                        null, userProfilePic, null, null);

 175.                                ImageDownloader.downloadAsync(request);
 176.                            }
 177.                        }
 178.                    }
 179.                    connectedStateLabel.setText(user.optString("name"));
 180.                } else {

 182.                            R.string.usersettings_fragment_logged_in));

 184.                            R.drawable.profile_default_icon);

 186.                            getResources().getDimensionPixelSize(
 187.                                    R.dimen.usersettings_fragment_profile_picture_width),
 188.                            getResources().getDimensionPixelSize(
 189.                                    R.dimen.usersettings_fragment_profile_picture_height));
 190.                    connectedStateLabel.setCompoundDrawables(null, noProfilePic, null, null);
 191.                }
 192.            } else {
 193.                int textColor = getResources().getColor(
 194.                        R.color.usersettings_fragment_not_connected_text_color);

 196.                connectedStateLabel.setShadowLayer(0f, 0f, 0f, textColor);
