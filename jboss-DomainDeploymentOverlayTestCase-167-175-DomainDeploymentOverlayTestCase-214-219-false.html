 355.        private String performHttpCall(String host, String server, String deployment) throws Exception {
 356.            ModelNode op = new ModelNode();
 357.            op.get(OP).set(READ_RESOURCE_OPERATION);
 358.            op.get(OP_ADDR).add(HOST, host).add(SERVER, server).add(SOCKET_BINDING_GROUP, SOCKET_BINDING_GROUP_NAME).add(SOCKET_BINDING, "http");
 359.            op.get(INCLUDE_RUNTIME).set(true);
 360.            ModelNode socketBinding = validateResponse(client.execute(op));
 361.    
 362.            URL url = new URL("http",
 363.                    org.jboss.as.arquillian.container.NetworkUtils.formatPossibleIpv6Address(socketBinding.get("bound-address").asString()),
 364.                    socketBinding.get("bound-port").asInt(),
 365.                    "/" + deployment + "/SimpleServlet?env-entry=overlay-test");
 366.            return HttpRequest.get(url.toExternalForm(), 10, TimeUnit.SECONDS).trim();
 367.        }

 371.            StringBuilder content = new StringBuilder();
 372.            int c;
 373.            while (-1 != (c = reader.read())) {
 374.                content.append((char) c);
 375.            }

 355.        private String performHttpCall(String host, String server, String deployment) throws Exception {
 356.            ModelNode op = new ModelNode();
 357.            op.get(OP).set(READ_RESOURCE_OPERATION);
 358.            op.get(OP_ADDR).add(HOST, host).add(SERVER, server).add(SOCKET_BINDING_GROUP, SOCKET_BINDING_GROUP_NAME).add(SOCKET_BINDING, "http");
 359.            op.get(INCLUDE_RUNTIME).set(true);
 360.            ModelNode socketBinding = validateResponse(client.execute(op));
 361.    
 362.            URL url = new URL("http",
 363.                    org.jboss.as.arquillian.container.NetworkUtils.formatPossibleIpv6Address(socketBinding.get("bound-address").asString()),
 364.                    socketBinding.get("bound-port").asInt(),
 365.                    "/" + deployment + "/SimpleServlet?env-entry=overlay-test");
 366.            return HttpRequest.get(url.toExternalForm(), 10, TimeUnit.SECONDS).trim();
 367.        }

 369.        public static String getContent(HttpResponse response) throws IOException {
 370.            InputStreamReader reader = new InputStreamReader(response.getEntity().getContent());
 371.            StringBuilder content = new StringBuilder();
 372.            int c;
 373.            while (-1 != (c = reader.read())) {
 374.                content.append((char) c);
 375.            }
 376.            reader.close();
 377.            return content.toString();
 378.        }
