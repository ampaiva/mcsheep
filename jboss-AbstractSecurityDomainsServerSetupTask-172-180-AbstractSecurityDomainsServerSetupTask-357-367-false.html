 315.                    .append(SUBSYSTEM, SUBSYSTEM_SECURITY)
 316.                    .append(SECURITY_DOMAIN, domainName)
 317.                    .append(securityComponent, CLASSIC);
 318.            operations.add(Util.createAddOperation(address));
 319.    
 320.            for (final SecurityModule config : securityModules) {
 321.                final String code = config.getName();
 322.                final ModelNode securityModuleNode = Util.createAddOperation(address.append(subnodeName, code));
 323.    
 324.                final String flag = StringUtils.defaultIfEmpty(config.getFlag(), flagDefaultValue);
 325.                securityModuleNode.get(ModelDescriptionConstants.CODE).set(code);
 326.                securityModuleNode.get(flagAttributeName).set(flag);
 327.                Map configOptions = config.getOptions();
 328.                if (configOptions == null) {
 329.                    LOGGER.info("No module options provided.");
 330.                    configOptions = Collections.emptyMap();
 331.                }
 332.                if (LOGGER.isInfoEnabled()) {
 333.                    LOGGER.info("Adding " + securityComponent + " module [code=" + code + ", " + flagAttributeName + "=" + flag
 334.                            + ", options = " + configOptions + "]");
 335.                }
 336.                final ModelNode moduleOptionsNode = securityModuleNode.get(MODULE_OPTIONS);
 337.                for (final Map.Entry entry : configOptions.entrySet()) {
 338.                    final String optionName = entry.getKey();
 339.                    final String optionValue = entry.getValue();
 340.                    moduleOptionsNode.add(optionName, optionValue);
 341.                }
 342.                securityModuleNode.get(OPERATION_HEADERS).get(ALLOW_RESOURCE_SERVICE_RESTART).set(true);
 343.                operations.add(securityModuleNode);
 344.            }
 345.            return true;
 346.        }

 348.        /**
 349.         * Creates a {@link ModelNode} with configuration of the JSSE part of security domain.
 350.         *
 351.         * @param jsse
 352.         * @param domainName
 353.         * @return
 354.         */
 355.        private ModelNode createJSSENode(final JSSE jsse, String domainName) {
 356.            if (jsse == null) {
 357.                if (LOGGER.isInfoEnabled()) {
 358.                    LOGGER.info("No security configuration for JSSE module.");
 359.                }
 360.                return null;
 361.            }
 362.            final ModelNode securityComponentNode = new ModelNode();
 363.            securityComponentNode.get(OP).set(ADD);
 364.            securityComponentNode.get(OP_ADDR).add(SUBSYSTEM, SUBSYSTEM_SECURITY);
 365.            securityComponentNode.get(OP_ADDR).add(SECURITY_DOMAIN, domainName);
 366.    
 367.            securityComponentNode.get(OP_ADDR).add(JSSE, CLASSIC);
 368.            addSecureStore(jsse.getTrustStore(), TRUSTSTORE, securityComponentNode);
 369.            addSecureStore(jsse.getKeyStore(), KEYSTORE, securityComponentNode);
 370.            securityComponentNode.get(OPERATION_HEADERS).get(ALLOW_RESOURCE_SERVICE_RESTART).set(true);
 371.            return securityComponentNode;
 372.        }

 375.         * Adds given secureStore to a JSSE configuration represented by given ModelNode.
 376.         *
 377.         * @param secureStore
 378.         * @param storeName
 379.         * @param jsseNode
 380.         */
 381.        private void addSecureStore(SecureStore secureStore, String storeName, ModelNode jsseNode) {
 382.            if (secureStore == null) {
 383.                return;
 384.            }
 385.            if (secureStore.getUrl() != null) {
 386.                jsseNode.get(storeName, URL).set(secureStore.getUrl().toExternalForm());
 387.            }
 388.            if (secureStore.getPassword() != null) {
 389.                jsseNode.get(storeName, PASSWORD).set(secureStore.getPassword());
 390.            }
 391.            if (secureStore.getType() != null) {
 392.                jsseNode.get(storeName, TYPE).set(secureStore.getType());
 393.            }
 394.        }
