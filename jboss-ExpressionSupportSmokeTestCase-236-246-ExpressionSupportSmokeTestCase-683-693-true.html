 351.        private boolean isAttributeExcluded(PathAddress address, String attrName, ModelNode attrDesc, ModelNode resourceNoDefaults) {
 352.            if (!attrDesc.get(ACCESS_TYPE).isDefined()
 353.                    || !attrDesc.get(ACCESS_TYPE).asString().equalsIgnoreCase("read-write")) {
 354.                return true;
 355.            }
 356.            if (attrDesc.get(STORAGE).isDefined()
 357.                    && !attrDesc.get(STORAGE).asString().equalsIgnoreCase("configuration")) {
 358.                return true;
 359.            }

 361.                return true;

 370.                            && "virtual-server".equals(address.getLastElement().getKey())) {

 373.                        return true;
 374.                    }
 375.                }

 378.                    PathElement subPe = address.getElement(address.size() - 3);
 379.                    if ("subsystem".equals(subPe.getKey()) && "security".equals(subPe.getValue())
 380.                            && "security-domain".equals(address.getElement(address.size() - 2).getKey())) {
 381.                        // This is a kind of alias that shows child resources as a list. Validating it
 382.                        // after reload breaks because the real child resources get changed. It's deprecated, so
 383.                        // we could exclude all deprecated attributes, but for now I'd rather be specific
 384.                        return true;
 385.                    }
 386.                }

 393.                            && "cache-container".equals(containerPe.getKey())
 394.                            && "distributed-cache".equals(distPe.getKey())) {
 395.                        // This is a distributed cache attribute in Infinispan which has been deprecated
 396.                        return true;
 397.                    }
 398.                }
 399.            } else if (address.size() > 0 && "transactions".equals(address.getLastElement().getValue())
 400.                    &&  "subsystem".equals(address.getLastElement().getKey())) {
 401.                if (attrName.contains("jdbc")) {
 402.                    // Ignore jdbc store attributes unless jdbc store is enabled
 403.                    return !resourceNoDefaults.hasDefined("use-jdbc-store") || !resourceNoDefaults.get("use-jdbc-store").asBoolean();

 410.            return false;

 425.        private void handleSimpleCollection(PathAddress address, String attrName, ModelNode attrValue, ModelType valueType,
 426.                                            Map expressionAttrs, Map otherAttrs,
 427.                                            Map expectedAttrs) {
 428.            if (COMPLEX_TYPES.contains(valueType)) {
 429.                // Too complicated
 430.                noSimpleCollection++;
 431.                logNoExpressions(address, attrName);
 432.                otherAttrs.put(attrName, attrValue);
 433.            } else {
 434.                boolean hasExpression = false;
 435.                ModelNode updated = new ModelNode();
 436.                ModelNode expected = new ModelNode();
 437.                for (ModelNode item : attrValue.asList()) {
 438.                    ModelType itemType = item.getType();
 439.                    if (itemType == ModelType.PROPERTY) {
 440.                        Property prop = item.asProperty();
 441.                        ModelNode propVal = prop.getValue();
 442.                        ModelType propValType = propVal.getType();
 443.                        if (propVal.isDefined() && propValType != ModelType.EXPRESSION) {
 444.                            // Convert property value to expression
 445.                            if (propValType == ModelType.STRING) {
 446.                                checkForUnconvertedExpression(address, attrName, propVal);
 447.                            }
 448.                            String expression = "${exp.test:" + propVal.asString() + "}";
 449.                            updated.get(prop.getName()).set(expression);
 450.                            expected.get(prop.getName()).set(new ModelNode().set(new ValueExpression(expression)));
 451.                            hasExpression = true;
 452.    
 453.                        } else {
 454.                            updated.get(prop.getName()).set(propVal);
 455.                            expected.get(prop.getName()).set(propVal);
 456.                        }
 457.                    } else if (item.isDefined() && itemType != ModelType.EXPRESSION) {
 458.                        // Convert item to expression
 459.                        if (itemType == ModelType.STRING) {
 460.                            checkForUnconvertedExpression(address, attrName, item);
 461.                        }
 462.                        String expression = "${exp.test:" + item.asString() + "}";
 463.                        updated.add(expression);
 464.                        expected.add(new ModelNode().set(new ValueExpression(expression)));
 465.                        hasExpression = true;
 466.                    } else {
 467.                        updated.add(item);
 468.                        expected.add(item);
 469.                    }
 470.                }
 471.    
 472.                if (hasExpression) {
 473.                    simpleCollection++;
 474.                    logHandling("Added expression to SIMPLE " + attrValue.getType() + " attribute " + attrName + " at " + address.toModelNode().asString());
 475.                    expressionAttrs.put(attrName, updated);
 476.                    expectedAttrs.put(attrName, expected);
 477.                } else {
 478.                    // We didn't change anything
 479.                    noSimpleCollection++;
 480.                    logNoExpressions(address, attrName);
 481.                    otherAttrs.put(attrName, attrValue);
 482.                    expectedAttrs.put(attrName, attrValue);
 483.                }
 484.            }
 485.        }

 534.            }

 577.        private void handleComplexItem(PathAddress address, String attrName, ModelNode item, ModelNode valueTypeDesc,
 578.                                       ModelNode updatedItem, ModelNode itemToExpect) {
 579.            // Hack to deal with time unit processing
 580.            Set keys =  valueTypeDesc.keys();
 581.            boolean timeAttr = keys.size() == 2 && keys.contains("time") && keys.contains("unit");
 582.            boolean changed = false;
 583.            for (Property fieldProp : valueTypeDesc.asPropertyList()) {
 584.                String fieldName = fieldProp.getName();
 585.                if (!item.has(fieldName)) {
 586.                    continue;
 587.                }
 588.                boolean timeunit = timeAttr && "unit".equals(fieldName);
 589.                ModelNode fieldDesc = fieldProp.getValue();
 590.                ModelNode fieldValue = item.get(fieldName);
 591.                ModelType valueType = fieldValue.getType();
 592.                if (valueType == ModelType.UNDEFINED
 593.                        || valueType == ModelType.EXPRESSION
 594.                        || COMPLEX_TYPES.contains(valueType) // too complex
 595.                        || !fieldDesc.get(EXPRESSIONS_ALLOWED).asBoolean(false)) {
 596.                    updatedItem.get(fieldName).set(fieldValue);
 597.                    itemToExpect.get(fieldName).set(fieldValue);
 598.                } else {
 599.                    if (valueType == ModelType.STRING) {
 600.                        checkForUnconvertedExpression(address, attrName, item);
 601.                    }
 602.                    String valueString = timeunit ? fieldValue.asString().toLowerCase() : fieldValue.asString();
 603.                    String expression = "${exp.test:" + valueString + "}";
 604.                    updatedItem.get(fieldName).set(expression);
 605.                    itemToExpect.get(fieldName).set(new ModelNode().set(new ValueExpression(expression)));
 606.                    changed = true;
 607.                }
 608.            }
 609.    
 610.            if (!changed) {
 611.                // Use unchanged 'item'
 612.                updatedItem.set(item);
 613.                itemToExpect.set(item);
 614.            }
 615.        }

 667.        private void validateExpectedValues(PathAddress address, Map> expectedValues, String hostName) throws IOException, MgmtOperationException {
 668.    
 669.            Map expectedModel = expectedValues.get(address);
 670.            if (expectedModel != null && isValidatable(address)) {
 671.                ModelNode resource = readResource(address, true, true);
 672.                for (Map.Entry entry : expectedModel.entrySet()) {
 673.                    String attrName = entry.getKey();
 674.                    ModelNode expectedValue = entry.getValue();
 675.                    ModelNode modVal = resource.get(entry.getKey());
 676.                    validateAttributeValue(address, attrName, expectedValue, modVal);
 677.                }
 678.            }
 679.    
 680.            ModelNode description = readResourceDescription(address);
 681.    
 682.            // Recurse into children, being careful about what processes we are touching
 683.            boolean isHost = address.size() == 1 && HOST.equals(address.getLastElement().getKey());
 684.            for (Property descProp : description.get(CHILDREN).asPropertyList()) {
 685.                String childType = descProp.getName();
 686.                if (isHost && SERVER.equals(childType)) {
 687.                    continue;
 688.                }
 689.                boolean hostChild = address.size() == 0 && HOST.equals(childType);
 690.                List children = readChildrenNames(address, childType);
 691.                for (String child : children) {
 692.                    if (!hostChild || hostName.equals(child)) {
 693.                        validateExpectedValues(address.append(PathElement.pathElement(childType, child)), expectedValues, hostName);
 694.                    }
 695.                }
 696.            }
 697.        }

 202.        private void setExpressions(PathAddress address, String hostName, Map> expectedValues) throws IOException, MgmtOperationException {
 203.    
 204.            ModelNode description = readResourceDescription(address);
 205.            ModelNode resource = readResource(address, true, false);
 206.            ModelNode resourceNoDefaults = readResource(address, false, false);
 207.    
 208.            Map expressionAttrs = new HashMap();
 209.            Map otherAttrs = new HashMap();
 210.            Map expectedAttrs = new HashMap();
 211.            organizeAttributes(address, description, resource, resourceNoDefaults, expressionAttrs, otherAttrs, expectedAttrs);
 212.    
 213.    
 214.            for (Map.Entry entry : expressionAttrs.entrySet()) {
 215.                writeAttribute(address, entry.getKey(), entry.getValue());
 216.            }
 217.            // Set the other attrs as well just to exercise the write-attribute handlers
 218.            for (Map.Entry entry : otherAttrs.entrySet()) {
 219.                writeAttribute(address, entry.getKey(), entry.getValue());
 220.            }
 221.    
 222.            if (expectedAttrs.size() > 0 && immediateValidation) {
 223.                // Validate that our write-attribute calls resulted in the expected values in the model
 224.                ModelNode modifiedResource = readResource(address, true, true);
 225.                for (Map.Entry entry : expectedAttrs.entrySet()) {
 226.                    ModelNode expectedValue = entry.getValue();
 227.                    ModelNode modVal = modifiedResource.get(entry.getKey());
 228.                    validateAttributeValue(address, entry.getKey(), expectedValue, modVal);
 229.                }
 230.            }
 231.    
 232.            // Store the modified values for confirmation after HC reload
 233.            expectedValues.put(address, expectedAttrs);
 234.    
 235.            // Recurse into children, being careful about what processes we are touching
 236.            boolean isHost = address.size() == 1 && HOST.equals(address.getLastElement().getKey());
 237.            for (Property descProp : description.get(CHILDREN).asPropertyList()) {
 238.                String childType = descProp.getName();
 239.                if (isHost && SERVER.equals(childType)) {
 240.                    continue;
 241.                }
 242.                boolean hostChild = address.size() == 0 && HOST.equals(childType);
 243.                List children = readChildrenNames(address, childType);
 244.                for (String child : children) {
 245.                    if (!hostChild || hostName.equals(child)) {
 246.                        setExpressions(address.append(PathElement.pathElement(childType, child)), hostName, expectedValues);
 247.                    }
 248.                }
 249.            }
 250.        }

 252.        private void organizeAttributes(PathAddress address, ModelNode description, ModelNode resource, ModelNode resourceNoDefaults,
 253.                                        Map expressionAttrs, Map otherAttrs,
 254.                                        Map expectedAttrs) {
 255.            ModelNode attributeDescriptions = description.get(ATTRIBUTES);
 256.            for (Property descProp : attributeDescriptions.asPropertyList()) {
 257.                String attrName = descProp.getName();
 258.                ModelNode attrDesc = descProp.getValue();
 259.                if (isAttributeExcluded(address, attrName, attrDesc, resourceNoDefaults)) {
 260.                    continue;
 261.                }
 262.                ModelNode noDefaultValue = resourceNoDefaults.get(attrName);
 263.                if (!noDefaultValue.isDefined()) {
 264.                    // We need to see if it's legal to set this attribute, or whether it's undefined
 265.                    // because an alternative attribute is defined
 266.                    Set base = new HashSet();
 267.                    base.add(attrName);
 268.                    if (attrDesc.hasDefined(REQUIRES)) {
 269.                        for (ModelNode node : attrDesc.get(REQUIRES).asList()) {
 270.                            base.add(node.asString());
 271.                        }
 272.                    }
 273.                    boolean conflict = false;
 274.                    for (String baseAttr : base) {
 275.                        ModelNode baseAttrAlts = attributeDescriptions.get(baseAttr, ALTERNATIVES);
 276.                        if (baseAttrAlts.isDefined()) {
 277.                            for (ModelNode alt : baseAttrAlts.asList()) {
 278.                                String altName = alt.asString();
 279.                                if (resourceNoDefaults.hasDefined(alt.asString())
 280.                                        || expressionAttrs.containsKey(altName)
 281.                                        || otherAttrs.containsKey(altName)) {
 282.                                    conflict = true;
 283.                                    break;
 284.                                }
 285.                            }
 286.                        }
 287.                    }
 288.                    if (conflict) {
 289.                        conflicts++;
 290.                        logHandling("Skipping conflicted attribute " + attrName + " at " + address.toModelNode().asString());
 291.                        continue;
 292.                    }
 293.                }
 294.                ModelNode attrValue = resource.get(attrName);
 295.                ModelType attrType = attrValue.getType();
 296.                if (attrDesc.get(EXPRESSIONS_ALLOWED).asBoolean(false)) {
 297.    
 298.                    // If it's defined and not an expression, use the current value to create an expression
 299.                    if (attrType != ModelType.UNDEFINED && attrType != ModelType.EXPRESSION) {
 300.                        // Deal with complex types specially
 301.                        if (COMPLEX_TYPES.contains(attrType)) {
 302.                            ModelNode valueType = attrDesc.get(VALUE_TYPE);
 303.                            if (valueType.getType() == ModelType.TYPE) {
 304.                                // Simple collection whose elements support expressions
 305.                                handleSimpleCollection(address, attrName, attrValue, valueType.asType(), expressionAttrs,
 306.                                        otherAttrs, expectedAttrs);
 307.                            } else if (valueType.isDefined()) {
 308.                                handleComplexCollection(address, attrName, attrValue, attrType, valueType, expressionAttrs, otherAttrs,
 309.                                        expectedAttrs);
 310.                            } else {
 311.                                noSimple++;
 312.                                logNoExpressions(address, attrName);
 313.                                otherAttrs.put(attrName, attrValue);
 314.                                expectedAttrs.put(attrName, attrValue);
 315.                            }
 316.                        } else {
 317.                            if (attrType == ModelType.STRING) {
 318.                                checkForUnconvertedExpression(address, attrName, attrValue);
 319.                            }
 320.                            String expression = "${exp.test:" + attrValue.asString() + "}";
 321.                            expressionAttrs.put(attrName, new ModelNode(expression));
 322.                            expectedAttrs.put(attrName, new ModelNode().set(new ValueExpression(expression)));
 323.                            simple++;
 324.                            logHandling("Added expression to simple attribute " + attrName + " at " + address.toModelNode().asString());
 325.                        }
 326.                    } else {
 327.                        if (attrType != ModelType.EXPRESSION) {
 328.                            supportedUndefined++;
 329.                            logHandling("Expression supported but value undefined on simple attribute " + attrName + " at " + address.toModelNode().asString());
 330.                        } else {
 331.                            simple++;
 332.                            logHandling("Already found an expression on simple attribute " + attrName + " at " + address.toModelNode().asString());
 333.                        }
 334.                        otherAttrs.put(attrName, attrValue);
 335.                        expectedAttrs.put(attrName, attrValue);
 336.                    }
 337.                } else if (COMPLEX_TYPES.contains(attrType)
 338.                        && attrDesc.get(VALUE_TYPE).getType() != ModelType.TYPE
 339.                        && attrDesc.get(VALUE_TYPE).isDefined()) {
 340.                    handleComplexCollection(address, attrName, attrValue, attrType, attrDesc.get(VALUE_TYPE), expressionAttrs,
 341.                            otherAttrs, expectedAttrs);
 342.                } else /*if (!attrDesc.hasDefined(DEPRECATED))*/ {
 343.                    noSimple++;
 344.                    logNoExpressions(address, attrName);
 345.                    otherAttrs.put(attrName, attrValue);
 346.                    expectedAttrs.put(attrName, attrValue);
 347.                }
 348.            }
 349.        }

 351.        private boolean isAttributeExcluded(PathAddress address, String attrName, ModelNode attrDesc, ModelNode resourceNoDefaults) {
 352.            if (!attrDesc.get(ACCESS_TYPE).isDefined()
 353.                    || !attrDesc.get(ACCESS_TYPE).asString().equalsIgnoreCase("read-write")) {
 354.                return true;
 355.            }
 356.            if (attrDesc.get(STORAGE).isDefined()
 357.                    && !attrDesc.get(STORAGE).asString().equalsIgnoreCase("configuration")) {
 358.                return true;
 359.            }
 360.            if (attrDesc.get(ModelDescriptionConstants.DEPRECATED).isDefined()){
 361.                return true;
 362.            }
 363.    
 364.    
 365.            // Special cases
 366.            if ("default-web-module".equals(attrName)) {
 367.                if (address.size() > 1) {
 368.                    PathElement subPe = address.getElement(address.size() - 2);
 369.                    if ("subsystem".equals(subPe.getKey()) && "web".equals(subPe.getValue())
 370.                            && "virtual-server".equals(address.getLastElement().getKey())) {
 371.                        // This is not allowed if "enable-welcome-root" is "true", which is overly complex to validate
 372.                        // so skip it
 373.                        return true;
 374.                    }
 375.                }
 376.            } else if ("policy-modules".equals(attrName) || "login-modules".equals(attrName)) {

 380.                            && "security-domain".equals(address.getElement(address.size() - 2).getKey())) {

 382.                        // after reload breaks because the real child resources get changed. It's deprecated, so
 383.                        // we could exclude all deprecated attributes, but for now I'd rather be specific
 384.                        return true;
 385.                    }
 386.                }
 387.            } else if ("virtual-nodes".equals(attrName)) {

 391.                    PathElement distPe = address.getElement(address.size()-1);
 392.                    if ("subsystem".equals(subPe.getKey()) && "infinispan".equals(subPe.getValue())
 393.                            && "cache-container".equals(containerPe.getKey())
 394.                            && "distributed-cache".equals(distPe.getKey())) {

 397.                    }
 398.                }
 399.            } else if (address.size() > 0 && "transactions".equals(address.getLastElement().getValue())
 400.                    &&  "subsystem".equals(address.getLastElement().getKey())) {
 401.                if (attrName.contains("jdbc")) {
 402.                    // Ignore jdbc store attributes unless jdbc store is enabled
 403.                    return !resourceNoDefaults.hasDefined("use-jdbc-store") || !resourceNoDefaults.get("use-jdbc-store").asBoolean();

 405.                    // Ignore hornetq store attributes unless hornetq store is enabled

 407.                }
 408.            }
 409.    
 410.            return false;
 411.        }

 425.        private void handleSimpleCollection(PathAddress address, String attrName, ModelNode attrValue, ModelType valueType,
 426.                                            Map expressionAttrs, Map otherAttrs,
 427.                                            Map expectedAttrs) {
 428.            if (COMPLEX_TYPES.contains(valueType)) {
 429.                // Too complicated
 430.                noSimpleCollection++;
 431.                logNoExpressions(address, attrName);
 432.                otherAttrs.put(attrName, attrValue);
 433.            } else {
 434.                boolean hasExpression = false;
 435.                ModelNode updated = new ModelNode();
 436.                ModelNode expected = new ModelNode();
 437.                for (ModelNode item : attrValue.asList()) {
 438.                    ModelType itemType = item.getType();
 439.                    if (itemType == ModelType.PROPERTY) {
 440.                        Property prop = item.asProperty();
 441.                        ModelNode propVal = prop.getValue();
 442.                        ModelType propValType = propVal.getType();
 443.                        if (propVal.isDefined() && propValType != ModelType.EXPRESSION) {
 444.                            // Convert property value to expression
 445.                            if (propValType == ModelType.STRING) {
 446.                                checkForUnconvertedExpression(address, attrName, propVal);
 447.                            }
 448.                            String expression = "${exp.test:" + propVal.asString() + "}";
 449.                            updated.get(prop.getName()).set(expression);
 450.                            expected.get(prop.getName()).set(new ModelNode().set(new ValueExpression(expression)));
 451.                            hasExpression = true;
 452.    
 453.                        } else {
 454.                            updated.get(prop.getName()).set(propVal);
 455.                            expected.get(prop.getName()).set(propVal);
 456.                        }
 457.                    } else if (item.isDefined() && itemType != ModelType.EXPRESSION) {
 458.                        // Convert item to expression
 459.                        if (itemType == ModelType.STRING) {
 460.                            checkForUnconvertedExpression(address, attrName, item);
 461.                        }
 462.                        String expression = "${exp.test:" + item.asString() + "}";
 463.                        updated.add(expression);
 464.                        expected.add(new ModelNode().set(new ValueExpression(expression)));
 465.                        hasExpression = true;
 466.                    } else {
 467.                        updated.add(item);
 468.                        expected.add(item);
 469.                    }
 470.                }
 471.    
 472.                if (hasExpression) {
 473.                    simpleCollection++;
 474.                    logHandling("Added expression to SIMPLE " + attrValue.getType() + " attribute " + attrName + " at " + address.toModelNode().asString());
 475.                    expressionAttrs.put(attrName, updated);
 476.                    expectedAttrs.put(attrName, expected);
 477.                } else {
 478.                    // We didn't change anything
 479.                    noSimpleCollection++;
 480.                    logNoExpressions(address, attrName);
 481.                    otherAttrs.put(attrName, attrValue);
 482.                    expectedAttrs.put(attrName, attrValue);
 483.                }
 484.            }
 485.        }

 488.                                             ModelType attrType, ModelNode valueTypeDesc,

 490.                                             Map expectedAttrs) {

 509.        private void handleComplexList(PathAddress address, String attrName, ModelNode attrValue, ModelNode valueTypeDesc,
 510.                                             Map expressionAttrs, Map otherAttrs,
 511.                                             Map expectedAttrs) {
 512.            ModelNode updatedList = new ModelNode().setEmptyList();
 513.            ModelNode expectedList = new ModelNode().setEmptyList();
 514.            boolean changed = false;
 515.    
 516.            for (ModelNode item : attrValue.asList()) {
 517.                ModelNode updated = new ModelNode();
 518.                ModelNode toExpect = new ModelNode();
 519.                handleComplexItem(address, attrName, item, valueTypeDesc, updated, toExpect);
 520.                changed |= !updated.equals(item);
 521.                updatedList.add(updated);
 522.                expectedList.add(toExpect);
 523.            }
 524.    
 525.            if (changed) {
 526.                complexList++;
 527.                logHandling("Added expression to COMPLEX LIST attribute " + attrName + " at " + address.toModelNode().asString());
 528.                expressionAttrs.put(attrName, updatedList);
 529.                expectedAttrs.put(attrName, expectedList);
 530.            } else {
 531.                noComplexList++;
 532.                logNoExpressions(address, attrName);
 533.                otherAttrs.put(attrName, attrValue);
 534.            }
 535.        }

 556.        private void handleComplexProperty(PathAddress address, String attrName, ModelNode attrValue, ModelNode valueTypeDesc,
 557.                                             Map expressionAttrs, Map otherAttrs,
 558.                                             Map expectedAttrs) {
 559.            Property prop = attrValue.asProperty();
 560.            ModelNode propVal = prop.getValue();
 561.            ModelNode updatedPropVal = new ModelNode();
 562.            ModelNode propValToExpect = new ModelNode();
 563.            handleComplexItem(address, attrName, propVal, valueTypeDesc, updatedPropVal, propValToExpect);
 564.            if (!updatedPropVal.equals(propVal)) {
 565.                complexProperty++;
 566.                ModelNode updatedProp = new ModelNode().set(prop.getName(), updatedPropVal);
 567.                logHandling("Added expression to COMPLEX PROPERTY attribute " + attrName + " at " + address.toModelNode().asString());
 568.                expressionAttrs.put(attrName, updatedProp);
 569.                expectedAttrs.put(attrName, new ModelNode().set(prop.getName(), propValToExpect));
 570.            } else {
 571.                noComplexProperty++;
 572.                logNoExpressions(address, attrName);
 573.                otherAttrs.put(attrName, attrValue);
 574.            }
 575.        }

 577.        private void handleComplexItem(PathAddress address, String attrName, ModelNode item, ModelNode valueTypeDesc,
 578.                                       ModelNode updatedItem, ModelNode itemToExpect) {
 579.            // Hack to deal with time unit processing
 580.            Set keys =  valueTypeDesc.keys();
 581.            boolean timeAttr = keys.size() == 2 && keys.contains("time") && keys.contains("unit");
 582.            boolean changed = false;
 583.            for (Property fieldProp : valueTypeDesc.asPropertyList()) {
 584.                String fieldName = fieldProp.getName();
 585.                if (!item.has(fieldName)) {
 586.                    continue;
 587.                }
 588.                boolean timeunit = timeAttr && "unit".equals(fieldName);
 589.                ModelNode fieldDesc = fieldProp.getValue();
 590.                ModelNode fieldValue = item.get(fieldName);
 591.                ModelType valueType = fieldValue.getType();
 592.                if (valueType == ModelType.UNDEFINED
 593.                        || valueType == ModelType.EXPRESSION
 594.                        || COMPLEX_TYPES.contains(valueType) // too complex
 595.                        || !fieldDesc.get(EXPRESSIONS_ALLOWED).asBoolean(false)) {
 596.                    updatedItem.get(fieldName).set(fieldValue);
 597.                    itemToExpect.get(fieldName).set(fieldValue);
 598.                } else {
 599.                    if (valueType == ModelType.STRING) {
 600.                        checkForUnconvertedExpression(address, attrName, item);
 601.                    }
 602.                    String valueString = timeunit ? fieldValue.asString().toLowerCase() : fieldValue.asString();
 603.                    String expression = "${exp.test:" + valueString + "}";
 604.                    updatedItem.get(fieldName).set(expression);
 605.                    itemToExpect.get(fieldName).set(new ModelNode().set(new ValueExpression(expression)));
 606.                    changed = true;
 607.                }
 608.            }
 609.    
 610.            if (!changed) {
 611.                // Use unchanged 'item'
 612.                updatedItem.set(item);
 613.                itemToExpect.set(item);
 614.            }
 615.        }

 667.        private void validateExpectedValues(PathAddress address, Map> expectedValues, String hostName) throws IOException, MgmtOperationException {
 668.    
 669.            Map expectedModel = expectedValues.get(address);
 670.            if (expectedModel != null && isValidatable(address)) {
 671.                ModelNode resource = readResource(address, true, true);
 672.                for (Map.Entry entry : expectedModel.entrySet()) {
 673.                    String attrName = entry.getKey();
 674.                    ModelNode expectedValue = entry.getValue();
 675.                    ModelNode modVal = resource.get(entry.getKey());
 676.                    validateAttributeValue(address, attrName, expectedValue, modVal);
 677.                }
 678.            }
 679.    
 680.            ModelNode description = readResourceDescription(address);
 681.    
 682.            // Recurse into children, being careful about what processes we are touching
 683.            boolean isHost = address.size() == 1 && HOST.equals(address.getLastElement().getKey());
 684.            for (Property descProp : description.get(CHILDREN).asPropertyList()) {
 685.                String childType = descProp.getName();
 686.                if (isHost && SERVER.equals(childType)) {
 687.                    continue;
 688.                }
 689.                boolean hostChild = address.size() == 0 && HOST.equals(childType);
 690.                List children = readChildrenNames(address, childType);
 691.                for (String child : children) {
 692.                    if (!hostChild || hostName.equals(child)) {
 693.                        validateExpectedValues(address.append(PathElement.pathElement(childType, child)), expectedValues, hostName);
 694.                    }
 695.                }
 696.            }
 697.        }

 699.        private void validateAttributeValue(PathAddress address, String attrName, ModelNode expectedValue, ModelNode modelValue) {
 700.            switch (expectedValue.getType()) {
 701.                case EXPRESSION: {
 702.                    Assert.assertEquals(address + " attribute " + attrName + " value " + modelValue + " is an unconverted expression",
 703.                            expectedValue, modelValue);
 704.                    break;
 705.                }
 706.                case INT:
 707.                case LONG:
 708.                    Assert.assertTrue(address + " attribute " + attrName + " is a valid type", modelValue.getType() == ModelType.INT || modelValue.getType() == ModelType.LONG);
 709.                    Assert.assertEquals(address + " -- " + attrName, expectedValue.asLong(), modelValue.asLong());
 710.                    break;
 711.                default: {
 712.                    Assert.assertEquals(address + " -- " + attrName, expectedValue, modelValue);
 713.                }
 714.            }
 715.        }
