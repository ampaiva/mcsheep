 156.        /**
 157.         * Check response on given url
 158.         *
 159.         * @param url
 160.         * @param responseSubstring - if null we are checking response code only
 161.         * @return
 162.         * @throws Exception
 163.         */
 164.        private boolean checkGetURL(String url, String responseSubstring, String user, String pass) throws Exception {
 165.    
 166.            log.info("Checking URL=" + url);
 167.    
 168.            HttpClient httpClient;
 169.            if (url.startsWith("https")) {
 170.                httpClient = HttpClientUtils.wrapHttpsClient(new DefaultHttpClient());
 171.            } else {
 172.                httpClient = new DefaultHttpClient();
 173.            }
 174.    
 175.            ((DefaultHttpClient) httpClient).getCredentialsProvider().setCredentials(new AuthScope(AuthScope.ANY),
 176.                    new UsernamePasswordCredentials(user, pass));
 177.    
 178.            HttpGet get = new HttpGet(url);
 179.            HttpResponse hr;
 180.            try {
 181.                hr = httpClient.execute(get);
 182.            } catch (Exception e) {
 183.                if (responseSubstring == null)
 184.                    return false;
 185.                else // in case substring is defined, rethrow exception so, we can easier analyze the cause
 186.                    throw new Exception(e);
 187.            }
 188.    
 189.            int statusCode = hr.getStatusLine().getStatusCode();
 190.            if (statusCode != 200) {
 191.                log.info("statusCode not expected. statusCode=" + statusCode + ", URL=" + url);
 192.                return false;
 193.            }
 194.    
 195.            if (responseSubstring == null) {
 196.                // this indicates that negative test had problems
 197.                log.info("statusCode==200 on URL=" + url);
 198.                return true;
 199.            }
 200.    
 201.            String response = EntityUtils.toString(hr.getEntity());
 202.            if (response.indexOf(responseSubstring) != -1) {
 203.                return true;
 204.            } else {
 205.                log.info("Response doesn't contain expected substring (" + responseSubstring + ")");
 206.                return false;
 207.            }

 156.        /**
 157.         * Check response on given url
 158.         *
 159.         * @param url
 160.         * @param responseSubstring - if null we are checking response code only
 161.         * @return
 162.         * @throws Exception
 163.         */
 164.        private boolean checkGetURL(String url, String responseSubstring, String user, String pass) throws Exception {
 165.    
 166.            log.info("Checking URL=" + url);
 167.    
 168.            HttpClient httpClient;
 169.            if (url.startsWith("https")) {
 170.                httpClient = HttpClientUtils.wrapHttpsClient(new DefaultHttpClient());
 171.            } else {
 172.                httpClient = new DefaultHttpClient();
 173.            }
 174.    
 175.            ((DefaultHttpClient) httpClient).getCredentialsProvider().setCredentials(new AuthScope(AuthScope.ANY),
 176.                    new UsernamePasswordCredentials(user, pass));
 177.    
 178.            HttpGet get = new HttpGet(url);
 179.            HttpResponse hr;
 180.            try {
 181.                hr = httpClient.execute(get);
 182.            } catch (Exception e) {
 183.                if (responseSubstring == null)
 184.                    return false;
 185.                else // in case substring is defined, rethrow exception so, we can easier analyze the cause
 186.                    throw new Exception(e);
 187.            }
 188.    
 189.            int statusCode = hr.getStatusLine().getStatusCode();
 190.            if (statusCode != 200) {
 191.                log.info("statusCode not expected. statusCode=" + statusCode + ", URL=" + url);
 192.                return false;
 193.            }
 194.    
 195.            if (responseSubstring == null) {
 196.                // this indicates that negative test had problems
 197.                log.info("statusCode==200 on URL=" + url);
 198.                return true;
 199.            }
 200.    
 201.            String response = EntityUtils.toString(hr.getEntity());
 202.            if (response.indexOf(responseSubstring) != -1) {
 203.                return true;
 204.            } else {
 205.                log.info("Response doesn't contain expected substring (" + responseSubstring + ")");
 206.                return false;
 207.            }
 208.    
 209.        }
