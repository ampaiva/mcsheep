  96.        private void convert(SortedSet attributes) {
  97.            SortedSet newAttributes = new TreeSet();
  98.            for (String attribute : attributes) {
  99.                String[] split = attribute.split("-");
 100.                StringBuilder newString = new StringBuilder();
 101.                for (int i = 0, splitLength = split.length; i < splitLength; i++) {
 102.                    String s = split[i];
 103.                    if(i > 0) {
 104.                        String camelVersion = Character.toUpperCase(s.charAt(0)) + s.substring(1);
 105.                        newString.append(camelVersion);
 106.                    }
 107.                    else {
 108.                        newString.append(s);
 109.                    }
 110.                }
 111.                String name = newString.toString();
 112.                if(DODGY_NAME.containsKey(name)) {
 113.                    name = DODGY_NAME.get(name);
 114.                }
 115.                newAttributes.add(name);
 116.            }
 117.            attributes.clear();
 118.            attributes.addAll(newAttributes);
 119.        }
