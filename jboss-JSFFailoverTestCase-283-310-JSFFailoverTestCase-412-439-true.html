 316.        /**
 317.         * Test simple undeploy failover:
 318.         * 


 319.         * 1/ Start 2 containers and deploy  webapp.
 320.         * 2/ Query first container creating a web session.
 321.         * 3/ Undeploy application from the first container.
 322.         * 4/ Query second container verifying sessions got replicated.
 323.         * 5/ Redeploy application to the first container.
 324.         * 6/ Query first container verifying that updated sessions replicated back.
 325.         *
 326.         * @throws java.io.IOException
 327.         * @throws InterruptedException
 328.         * @throws URISyntaxException
 329.         */
 330.        @Test
 331.        public void testGracefulUndeployFailover(
 332.                @ArquillianResource() @OperateOnDeployment(DEPLOYMENT_1) URL baseURL1,
 333.                @ArquillianResource() @OperateOnDeployment(DEPLOYMENT_2) URL baseURL2)
 334.                throws IOException, InterruptedException, URISyntaxException {
 335.    
 336.            HttpClient client = HttpClients.createDefault();
 337.    
 338.            String url1 = baseURL1.toString() + "home.jsf";
 339.            String url2 = baseURL2.toString() + "home.jsf";
 340.    
 341.            try {
 342.                HttpResponse response;
 343.                NumberGuessState state;
 344.    
 345.                // First non-JSF request to the home page
 346.                response = client.execute(buildGetRequest(url1, null));
 347.                try {
 348.                    Assert.assertEquals(HttpServletResponse.SC_OK, response.getStatusLine().getStatusCode());
 349.                    state = parseState(response, null);
 350.                } finally {
 351.                    HttpClientUtils.closeQuietly(response);
 352.                }
 353.    
 354.                // We get a cookie!
 355.                String sessionId = state.sessionId;
 356.    
 357.                Assert.assertNotNull(sessionId);
 358.                Assert.assertEquals("0", state.smallest);
 359.                Assert.assertEquals("100", state.biggest);
 360.                Assert.assertEquals("10", state.remainingGuesses);
 361.    
 362.                // We do a JSF POST request, guessing "1"
 363.                response = client.execute(buildPostRequest(url1, state.sessionId, state.jsfViewState, "1"));
 364.                try {
 365.                    Assert.assertEquals(HttpServletResponse.SC_OK, response.getStatusLine().getStatusCode());
 366.                    state = parseState(response, sessionId);
 367.                } finally {
 368.                    HttpClientUtils.closeQuietly(response);
 369.                }
 370.    
 371.                Assert.assertEquals("2", state.smallest);
 372.                Assert.assertEquals("100", state.biggest);
 373.                Assert.assertEquals("9", state.remainingGuesses);
 374.    
 375.                // Gracefully undeploy from the 1st container.
 376.                undeploy(DEPLOYMENT_1);
 377.    
 378.                // Now we do a JSF POST request with a cookie on to the second node, guessing 100, expecting to find a replicated state.
 379.                response = client.execute(buildPostRequest(url2, state.sessionId, state.jsfViewState, "100"));
 380.                try {
 381.                    Assert.assertEquals(HttpServletResponse.SC_OK, response.getStatusLine().getStatusCode());
 382.                    state = parseState(response, sessionId);
 383.                } finally {
 384.                    HttpClientUtils.closeQuietly(response);
 385.                }
 386.    
 387.                // If the state would not be replicated, we would have 9 remaining guesses.
 388.                Assert.assertEquals("Session failed to replicate after container 1 was shutdown.", "8", state.remainingGuesses);
 389.    
 390.                // The server should accept our cookie and not try to set a different one
 391.                Assert.assertEquals(sessionId, state.sessionId);
 392.                Assert.assertEquals("2", state.smallest);
 393.                Assert.assertEquals("99", state.biggest);
 394.    
 395.                // Now we do a JSF POST request on the second node again, guessing "99"
 396.                response = client.execute(buildPostRequest(url2, sessionId, state.jsfViewState, "99"));
 397.                try {
 398.                    Assert.assertEquals(HttpServletResponse.SC_OK, response.getStatusLine().getStatusCode());
 399.                    state = parseState(response, sessionId);
 400.                } finally {
 401.                    HttpClientUtils.closeQuietly(response);
 402.                }
 403.    
 404.                Assert.assertEquals("7", state.remainingGuesses);
 405.                Assert.assertEquals("2", state.smallest);
 406.                Assert.assertEquals("98", state.biggest);
 407.    
 408.                // Redeploy
 409.                deploy(DEPLOYMENT_1);
 410.    
 411.                // And now we go back to the first node, guessing 2
 412.                response = client.execute(buildPostRequest(url1, state.sessionId, state.jsfViewState, "2"));
 413.                try {
 414.                    Assert.assertEquals(HttpServletResponse.SC_OK, response.getStatusLine().getStatusCode());
 415.                    state = parseState(response, sessionId);
 416.                } finally {
 417.                    HttpClientUtils.closeQuietly(response);
 418.                }
 419.    
 420.                Assert.assertEquals("Session failed to replicate after container 1 was brought up.", "6", state.remainingGuesses);
 421.                Assert.assertEquals(sessionId, state.sessionId);
 422.                Assert.assertEquals("3", state.smallest);
 423.                Assert.assertEquals("98", state.biggest);
 424.    
 425.                // One final guess on the first node, guess 50
 426.                response = client.execute(buildPostRequest(url1, state.sessionId, state.jsfViewState, "50"));
 427.                try {
 428.                    Assert.assertEquals(HttpServletResponse.SC_OK, response.getStatusLine().getStatusCode());
 429.                    state = parseState(response, sessionId);
 430.                } finally {
 431.                    HttpClientUtils.closeQuietly(response);
 432.                }
 433.    
 434.                Assert.assertEquals(sessionId, state.sessionId);
 435.                Assert.assertEquals("5", state.remainingGuesses);
 436.                Assert.assertEquals("3", state.smallest);
 437.                Assert.assertEquals("49", state.biggest);
 438.            } finally {
 439.                HttpClientUtils.closeQuietly(client);
 440.            }
 441.    
 442.            // Assert.fail("Show me the logs please!");
 443.        }
