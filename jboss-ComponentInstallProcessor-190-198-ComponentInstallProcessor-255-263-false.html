  84.        public void deploy(final DeploymentPhaseContext phaseContext) throws DeploymentUnitProcessingException {
  85.            final DeploymentUnit deploymentUnit = phaseContext.getDeploymentUnit();
  86.            final Module module = deploymentUnit.getAttachment(MODULE);
  87.            final EEModuleConfiguration moduleConfiguration = deploymentUnit.getAttachment(EE_MODULE_CONFIGURATION);
  88.            if (module == null || moduleConfiguration == null) {
  89.                return;
  90.            }
  91.            ComponentRegistry componentRegistry = deploymentUnit.getAttachment(COMPONENT_REGISTRY);
  92.    
  93.            final List dependencies = deploymentUnit.getAttachmentList(org.jboss.as.server.deployment.Attachments.JNDI_DEPENDENCIES);
  94.    
  95.            final ServiceName bindingDependencyService = JndiNamingDependencyProcessor.serviceName(deploymentUnit.getServiceName());
  96.    
  97.            // Iterate through each component, installing it into the container
  98.            for (final ComponentConfiguration configuration : moduleConfiguration.getComponentConfigurations()) {
  99.                try {
 100.                    ROOT_LOGGER.tracef("Installing component %s", configuration.getComponentClass().getName());
 101.                    deployComponent(phaseContext, configuration, dependencies, bindingDependencyService);
 102.                    componentRegistry.addComponent(configuration);
 103.    
 104.                    //we need to make sure that the web deployment has a dependency on all components it the app, so web components are started
 105.                    //when the web subsystem is starting
 106.                    //we only add a dependency on components in the same sub deployment, otherwise we get circular dependencies when initialize-in-order is used
 107.                    deploymentUnit.addToAttachmentList(org.jboss.as.server.deployment.Attachments.WEB_DEPENDENCIES, configuration.getComponentDescription().getStartServiceName());
 108.                } catch (Exception e) {
 109.                    throw EeLogger.ROOT_LOGGER.failedToInstallComponent(e, configuration.getComponentName());
 110.                }
 111.            }

 114.        @SuppressWarnings({ "unchecked", "rawtypes" })
 115.        protected void deployComponent(final DeploymentPhaseContext phaseContext, final ComponentConfiguration configuration, final List jndiDependencies, final ServiceName bindingDependencyService) throws DeploymentUnitProcessingException {
 116.    
 117.            final DeploymentUnit deploymentUnit = phaseContext.getDeploymentUnit();
 118.            final ServiceTarget serviceTarget = phaseContext.getServiceTarget();
 119.    
 120.            final String applicationName = configuration.getApplicationName();
 121.            final String moduleName = configuration.getModuleName();
 122.            final String componentName = configuration.getComponentName();
 123.            final EEApplicationClasses applicationClasses = deploymentUnit.getAttachment(Attachments.EE_APPLICATION_CLASSES_DESCRIPTION);
 124.            final Module module = deploymentUnit.getAttachment(org.jboss.as.server.deployment.Attachments.MODULE);
 125.    
 126.            //create additional injectors
 127.    
 128.            final ServiceName createServiceName = configuration.getComponentDescription().getCreateServiceName();
 129.            final ServiceName startServiceName = configuration.getComponentDescription().getStartServiceName();
 130.            final BasicComponentCreateService createService = configuration.getComponentCreateServiceFactory().constructService(configuration);
 131.            final ServiceBuilder createBuilder = serviceTarget.addService(createServiceName, createService);
 132.            // inject the DU
 133.            createBuilder.addDependency(deploymentUnit.getServiceName(), DeploymentUnit.class, createService.getDeploymentUnitInjector());
 134.    
 135.            final ComponentStartService startService = new ComponentStartService();
 136.            final ServiceBuilder startBuilder = serviceTarget.addService(startServiceName, startService);
 137.    
 138.            deploymentUnit.addToAttachmentList(org.jboss.as.server.deployment.Attachments.DEPLOYMENT_COMPLETE_SERVICES, startServiceName);
 139.    
 140.            //WFLY-1402 we don't add the bindings to the jndi dependencies list directly, instead
 141.            //the bindings depend on the this artificial service
 142.            ServiceName jndiDepServiceName = configuration.getComponentDescription().getServiceName().append(JNDI_BINDINGS_SERVICE);
 143.            final ServiceBuilder jndiDepServiceBuilder = serviceTarget.addService(jndiDepServiceName, Service.NULL);
 144.            jndiDependencies.add(jndiDepServiceName);
 145.    
 146.            // Add all service dependencies
 147.            for (DependencyConfigurator configurator : configuration.getCreateDependencies()) {
 148.                configurator.configureDependency(createBuilder, createService);
 149.            }
 150.            for (DependencyConfigurator configurator : configuration.getStartDependencies()) {
 151.                configurator.configureDependency(startBuilder, startService);
 152.            }
 153.    
 154.            // START depends on CREATE
 155.            startBuilder.addDependency(createServiceName, BasicComponent.class, startService.getComponentInjector());
 156.            Services.addServerExecutorDependency(startBuilder, startService.getExecutorInjector(), false);
 157.    
 158.            //don't start components until all bindings are up
 159.            startBuilder.addDependency(bindingDependencyService);
 160.            final ServiceName contextServiceName;
 161.            //set up the naming context if necessary
 162.            if (configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.CREATE) {
 163.                final NamingStoreService contextService = new NamingStoreService(true);
 164.                serviceTarget.addService(configuration.getComponentDescription().getContextServiceName(), contextService).install();
 165.            }
 166.    
 167.            final InjectionSource.ResolutionContext resolutionContext = new InjectionSource.ResolutionContext(
 168.                    configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.USE_MODULE,
 169.                    configuration.getComponentName(),
 170.                    configuration.getModuleName(),
 171.                    configuration.getApplicationName()
 172.            );
 173.    
 174.            // Iterate through each view, creating the services for each
 175.            for (ViewConfiguration viewConfiguration : configuration.getViews()) {
 176.                final ServiceName serviceName = viewConfiguration.getViewServiceName();
 177.                final ViewService viewService = new ViewService(viewConfiguration);
 178.                final ServiceBuilder componentViewServiceBuilder = serviceTarget.addService(serviceName, viewService);
 179.                componentViewServiceBuilder
 180.                        .addDependency(createServiceName, Component.class, viewService.getComponentInjector());
 181.                for(final DependencyConfigurator depConfig : viewConfiguration.getDependencies()) {
 182.                    depConfig.configureDependency(componentViewServiceBuilder, viewService);
 183.                }
 184.                componentViewServiceBuilder.install();
 185.                startBuilder.addDependency(serviceName);
 186.                // The bindings for the view
 187.                for (BindingConfiguration bindingConfiguration : viewConfiguration.getBindingConfigurations()) {
 188.                    final String bindingName = bindingConfiguration.getName();
 189.                    final ContextNames.BindInfo bindInfo = ContextNames.bindInfoFor(applicationName, moduleName, componentName, bindingName);
 190.                    final BinderService service = new BinderService(bindInfo.getBindName(), bindingConfiguration.getSource());
 191.    
 192.                    //these bindings should never be merged, if a view binding is duplicated it is an error
 193.                    jndiDepServiceBuilder.addDependency(bindInfo.getBinderServiceName());
 194.    
 195.                    ServiceBuilder serviceBuilder = serviceTarget.addService(bindInfo.getBinderServiceName(), service);
 196.                    bindingConfiguration.getSource().getResourceValue(resolutionContext, serviceBuilder, phaseContext, service.getManagedObjectInjector());
 197.                    serviceBuilder.addDependency(bindInfo.getParentContextServiceName(), ServiceBasedNamingStore.class, service.getNamingStoreInjector());
 198.                    serviceBuilder.install();
 199.                }
 200.            }
 201.    
 202.            if (configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.CREATE) {
 203.                // The bindings for the component
 204.                final Set bound = new HashSet();
 205.                processBindings(phaseContext, configuration, serviceTarget, resolutionContext, configuration.getComponentDescription().getBindingConfigurations(), jndiDepServiceBuilder, bound);
 206.    
 207.                //class level bindings should be ignored if the deployment is metadata complete
 208.                if (!MetadataCompleteMarker.isMetadataComplete(phaseContext.getDeploymentUnit())) {
 209.    
 210.                    // The bindings for the component class
 211.                    new ClassDescriptionTraversal(configuration.getComponentClass(), applicationClasses) {
 212.                        @Override
 213.                        protected void handle(final Class clazz, final EEModuleClassDescription classDescription) throws DeploymentUnitProcessingException {
 214.                            if (classDescription != null) {
 215.                                processBindings(phaseContext, configuration, serviceTarget, resolutionContext, classDescription.getBindingConfigurations(), jndiDepServiceBuilder, bound);
 216.                            }
 217.                        }
 218.                    }.run();
 219.    
 220.    
 221.                    for (InterceptorDescription interceptor : configuration.getComponentDescription().getAllInterceptors()) {
 222.                        final Class interceptorClass;
 223.                        try {
 224.                            interceptorClass = module.getClassLoader().loadClass(interceptor.getInterceptorClassName());
 225.                        } catch (ClassNotFoundException e) {
 226.                            throw EeLogger.ROOT_LOGGER.cannotLoadInterceptor(e, interceptor.getInterceptorClassName(), configuration.getComponentClass());
 227.                        }
 228.                        if (interceptorClass != null) {
 229.                            new ClassDescriptionTraversal(interceptorClass, applicationClasses) {
 230.                                @Override
 231.                                protected void handle(final Class clazz, final EEModuleClassDescription classDescription) throws DeploymentUnitProcessingException {
 232.                                    if (classDescription != null) {
 233.                                        processBindings(phaseContext, configuration, serviceTarget, resolutionContext, classDescription.getBindingConfigurations(), jndiDepServiceBuilder, bound);
 234.                                    }
 235.                                }
 236.                            }.run();
 237.                        }
 238.                    }
 239.                }
 240.            }
 241.    
 242.            createBuilder.install();
 243.            startBuilder.install();
 244.            jndiDepServiceBuilder.install();
 245.        }

 247.        @SuppressWarnings("unchecked")
 248.        private void processBindings(DeploymentPhaseContext phaseContext, ComponentConfiguration configuration, ServiceTarget serviceTarget, InjectionSource.ResolutionContext resolutionContext, List bindings, final ServiceBuilder jndiDepServiceBuilder, final Set bound) throws DeploymentUnitProcessingException {
 249.    
 250.            //we only handle java:comp bindings for components that have their own namespace here, the rest are processed by ModuleJndiBindingProcessor
 251.            for (BindingConfiguration bindingConfiguration : bindings) {
 252.                if (bindingConfiguration.getName().startsWith("java:comp") || !bindingConfiguration.getName().startsWith("java:")) {
 253.                    final String bindingName = bindingConfiguration.getName().startsWith("java:comp") ? bindingConfiguration.getName() : "java:comp/env/" + bindingConfiguration.getName();
 254.                    final ContextNames.BindInfo bindInfo = ContextNames.bindInfoForEnvEntry(configuration.getApplicationName(), configuration.getModuleName(), configuration.getComponentName(), configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.CREATE, bindingName);
 255.                    if (bound.contains(bindInfo.getBinderServiceName())) {
 256.                        continue;
 257.                    }
 258.                    bound.add(bindInfo.getBinderServiceName());
 259.                    try {
 260.                        final BinderService service = new BinderService(bindInfo.getBindName(), bindingConfiguration.getSource());
 261.                        jndiDepServiceBuilder.addDependency(bindInfo.getBinderServiceName());
 262.                        ServiceBuilder serviceBuilder = serviceTarget.addService(bindInfo.getBinderServiceName(), service);
 263.                        bindingConfiguration.getSource().getResourceValue(resolutionContext, serviceBuilder, phaseContext, service.getManagedObjectInjector());
 264.                        serviceBuilder.addDependency(bindInfo.getParentContextServiceName(), ServiceBasedNamingStore.class, service.getNamingStoreInjector());
 265.                        serviceBuilder.install();
 266.                    } catch (DuplicateServiceException e) {
 267.                        ServiceController registered = (ServiceController) CurrentServiceContainer.getServiceContainer().getService(bindInfo.getBinderServiceName());
 268.                        if (registered == null)
 269.                            throw e;
 270.    
 271.                        BinderService service = (BinderService) registered.getService();
 272.                        if (!service.getSource().equals(bindingConfiguration.getSource()))
 273.                            throw EeLogger.ROOT_LOGGER.conflictingBinding(bindingName, bindingConfiguration.getSource());
 274.                    } catch (CircularDependencyException e) {
 275.                        throw EeLogger.ROOT_LOGGER.circularDependency(bindingName);
 276.                    }
 277.                }
 278.            }

  80.    public final class ComponentInstallProcessor implements DeploymentUnitProcessor {
  81.    
  82.        private static final ServiceName JNDI_BINDINGS_SERVICE = ServiceName.of("JndiBindingsService");
  83.    
  84.        public void deploy(final DeploymentPhaseContext phaseContext) throws DeploymentUnitProcessingException {
  85.            final DeploymentUnit deploymentUnit = phaseContext.getDeploymentUnit();
  86.            final Module module = deploymentUnit.getAttachment(MODULE);
  87.            final EEModuleConfiguration moduleConfiguration = deploymentUnit.getAttachment(EE_MODULE_CONFIGURATION);
  88.            if (module == null || moduleConfiguration == null) {
  89.                return;
  90.            }
  91.            ComponentRegistry componentRegistry = deploymentUnit.getAttachment(COMPONENT_REGISTRY);
  92.    
  93.            final List dependencies = deploymentUnit.getAttachmentList(org.jboss.as.server.deployment.Attachments.JNDI_DEPENDENCIES);
  94.    
  95.            final ServiceName bindingDependencyService = JndiNamingDependencyProcessor.serviceName(deploymentUnit.getServiceName());
  96.    
  97.            // Iterate through each component, installing it into the container
  98.            for (final ComponentConfiguration configuration : moduleConfiguration.getComponentConfigurations()) {
  99.                try {
 100.                    ROOT_LOGGER.tracef("Installing component %s", configuration.getComponentClass().getName());
 101.                    deployComponent(phaseContext, configuration, dependencies, bindingDependencyService);
 102.                    componentRegistry.addComponent(configuration);
 103.    
 104.                    //we need to make sure that the web deployment has a dependency on all components it the app, so web components are started
 105.                    //when the web subsystem is starting
 106.                    //we only add a dependency on components in the same sub deployment, otherwise we get circular dependencies when initialize-in-order is used
 107.                    deploymentUnit.addToAttachmentList(org.jboss.as.server.deployment.Attachments.WEB_DEPENDENCIES, configuration.getComponentDescription().getStartServiceName());
 108.                } catch (Exception e) {
 109.                    throw EeLogger.ROOT_LOGGER.failedToInstallComponent(e, configuration.getComponentName());
 110.                }
 111.            }
 112.        }
 113.    
 114.        @SuppressWarnings({ "unchecked", "rawtypes" })
 115.        protected void deployComponent(final DeploymentPhaseContext phaseContext, final ComponentConfiguration configuration, final List jndiDependencies, final ServiceName bindingDependencyService) throws DeploymentUnitProcessingException {
 116.    
 117.            final DeploymentUnit deploymentUnit = phaseContext.getDeploymentUnit();
 118.            final ServiceTarget serviceTarget = phaseContext.getServiceTarget();
 119.    
 120.            final String applicationName = configuration.getApplicationName();
 121.            final String moduleName = configuration.getModuleName();
 122.            final String componentName = configuration.getComponentName();
 123.            final EEApplicationClasses applicationClasses = deploymentUnit.getAttachment(Attachments.EE_APPLICATION_CLASSES_DESCRIPTION);
 124.            final Module module = deploymentUnit.getAttachment(org.jboss.as.server.deployment.Attachments.MODULE);
 125.    
 126.            //create additional injectors
 127.    
 128.            final ServiceName createServiceName = configuration.getComponentDescription().getCreateServiceName();
 129.            final ServiceName startServiceName = configuration.getComponentDescription().getStartServiceName();
 130.            final BasicComponentCreateService createService = configuration.getComponentCreateServiceFactory().constructService(configuration);
 131.            final ServiceBuilder createBuilder = serviceTarget.addService(createServiceName, createService);
 132.            // inject the DU
 133.            createBuilder.addDependency(deploymentUnit.getServiceName(), DeploymentUnit.class, createService.getDeploymentUnitInjector());
 134.    
 135.            final ComponentStartService startService = new ComponentStartService();
 136.            final ServiceBuilder startBuilder = serviceTarget.addService(startServiceName, startService);
 137.    
 138.            deploymentUnit.addToAttachmentList(org.jboss.as.server.deployment.Attachments.DEPLOYMENT_COMPLETE_SERVICES, startServiceName);
 139.    
 140.            //WFLY-1402 we don't add the bindings to the jndi dependencies list directly, instead
 141.            //the bindings depend on the this artificial service
 142.            ServiceName jndiDepServiceName = configuration.getComponentDescription().getServiceName().append(JNDI_BINDINGS_SERVICE);
 143.            final ServiceBuilder jndiDepServiceBuilder = serviceTarget.addService(jndiDepServiceName, Service.NULL);
 144.            jndiDependencies.add(jndiDepServiceName);
 145.    
 146.            // Add all service dependencies
 147.            for (DependencyConfigurator configurator : configuration.getCreateDependencies()) {
 148.                configurator.configureDependency(createBuilder, createService);
 149.            }
 150.            for (DependencyConfigurator configurator : configuration.getStartDependencies()) {
 151.                configurator.configureDependency(startBuilder, startService);
 152.            }
 153.    
 154.            // START depends on CREATE
 155.            startBuilder.addDependency(createServiceName, BasicComponent.class, startService.getComponentInjector());
 156.            Services.addServerExecutorDependency(startBuilder, startService.getExecutorInjector(), false);
 157.    
 158.            //don't start components until all bindings are up
 159.            startBuilder.addDependency(bindingDependencyService);
 160.            final ServiceName contextServiceName;
 161.            //set up the naming context if necessary
 162.            if (configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.CREATE) {
 163.                final NamingStoreService contextService = new NamingStoreService(true);
 164.                serviceTarget.addService(configuration.getComponentDescription().getContextServiceName(), contextService).install();
 165.            }
 166.    
 167.            final InjectionSource.ResolutionContext resolutionContext = new InjectionSource.ResolutionContext(
 168.                    configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.USE_MODULE,
 169.                    configuration.getComponentName(),
 170.                    configuration.getModuleName(),
 171.                    configuration.getApplicationName()
 172.            );
 173.    
 174.            // Iterate through each view, creating the services for each
 175.            for (ViewConfiguration viewConfiguration : configuration.getViews()) {
 176.                final ServiceName serviceName = viewConfiguration.getViewServiceName();
 177.                final ViewService viewService = new ViewService(viewConfiguration);
 178.                final ServiceBuilder componentViewServiceBuilder = serviceTarget.addService(serviceName, viewService);
 179.                componentViewServiceBuilder
 180.                        .addDependency(createServiceName, Component.class, viewService.getComponentInjector());
 181.                for(final DependencyConfigurator depConfig : viewConfiguration.getDependencies()) {
 182.                    depConfig.configureDependency(componentViewServiceBuilder, viewService);
 183.                }
 184.                componentViewServiceBuilder.install();
 185.                startBuilder.addDependency(serviceName);
 186.                // The bindings for the view
 187.                for (BindingConfiguration bindingConfiguration : viewConfiguration.getBindingConfigurations()) {
 188.                    final String bindingName = bindingConfiguration.getName();
 189.                    final ContextNames.BindInfo bindInfo = ContextNames.bindInfoFor(applicationName, moduleName, componentName, bindingName);
 190.                    final BinderService service = new BinderService(bindInfo.getBindName(), bindingConfiguration.getSource());
 191.    
 192.                    //these bindings should never be merged, if a view binding is duplicated it is an error
 193.                    jndiDepServiceBuilder.addDependency(bindInfo.getBinderServiceName());
 194.    
 195.                    ServiceBuilder serviceBuilder = serviceTarget.addService(bindInfo.getBinderServiceName(), service);
 196.                    bindingConfiguration.getSource().getResourceValue(resolutionContext, serviceBuilder, phaseContext, service.getManagedObjectInjector());
 197.                    serviceBuilder.addDependency(bindInfo.getParentContextServiceName(), ServiceBasedNamingStore.class, service.getNamingStoreInjector());
 198.                    serviceBuilder.install();
 199.                }
 200.            }
 201.    
 202.            if (configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.CREATE) {
 203.                // The bindings for the component
 204.                final Set bound = new HashSet();
 205.                processBindings(phaseContext, configuration, serviceTarget, resolutionContext, configuration.getComponentDescription().getBindingConfigurations(), jndiDepServiceBuilder, bound);
 206.    
 207.                //class level bindings should be ignored if the deployment is metadata complete
 208.                if (!MetadataCompleteMarker.isMetadataComplete(phaseContext.getDeploymentUnit())) {
 209.    
 210.                    // The bindings for the component class
 211.                    new ClassDescriptionTraversal(configuration.getComponentClass(), applicationClasses) {
 212.                        @Override
 213.                        protected void handle(final Class clazz, final EEModuleClassDescription classDescription) throws DeploymentUnitProcessingException {
 214.                            if (classDescription != null) {
 215.                                processBindings(phaseContext, configuration, serviceTarget, resolutionContext, classDescription.getBindingConfigurations(), jndiDepServiceBuilder, bound);
 216.                            }
 217.                        }
 218.                    }.run();
 219.    
 220.    
 221.                    for (InterceptorDescription interceptor : configuration.getComponentDescription().getAllInterceptors()) {
 222.                        final Class interceptorClass;
 223.                        try {
 224.                            interceptorClass = module.getClassLoader().loadClass(interceptor.getInterceptorClassName());
 225.                        } catch (ClassNotFoundException e) {
 226.                            throw EeLogger.ROOT_LOGGER.cannotLoadInterceptor(e, interceptor.getInterceptorClassName(), configuration.getComponentClass());
 227.                        }
 228.                        if (interceptorClass != null) {
 229.                            new ClassDescriptionTraversal(interceptorClass, applicationClasses) {
 230.                                @Override
 231.                                protected void handle(final Class clazz, final EEModuleClassDescription classDescription) throws DeploymentUnitProcessingException {
 232.                                    if (classDescription != null) {
 233.                                        processBindings(phaseContext, configuration, serviceTarget, resolutionContext, classDescription.getBindingConfigurations(), jndiDepServiceBuilder, bound);
 234.                                    }
 235.                                }
 236.                            }.run();
 237.                        }
 238.                    }
 239.                }
 240.            }
 241.    
 242.            createBuilder.install();
 243.            startBuilder.install();
 244.            jndiDepServiceBuilder.install();
 245.        }
 246.    
 247.        @SuppressWarnings("unchecked")
 248.        private void processBindings(DeploymentPhaseContext phaseContext, ComponentConfiguration configuration, ServiceTarget serviceTarget, InjectionSource.ResolutionContext resolutionContext, List bindings, final ServiceBuilder jndiDepServiceBuilder, final Set bound) throws DeploymentUnitProcessingException {
 249.    
 250.            //we only handle java:comp bindings for components that have their own namespace here, the rest are processed by ModuleJndiBindingProcessor
 251.            for (BindingConfiguration bindingConfiguration : bindings) {
 252.                if (bindingConfiguration.getName().startsWith("java:comp") || !bindingConfiguration.getName().startsWith("java:")) {
 253.                    final String bindingName = bindingConfiguration.getName().startsWith("java:comp") ? bindingConfiguration.getName() : "java:comp/env/" + bindingConfiguration.getName();
 254.                    final ContextNames.BindInfo bindInfo = ContextNames.bindInfoForEnvEntry(configuration.getApplicationName(), configuration.getModuleName(), configuration.getComponentName(), configuration.getComponentDescription().getNamingMode() == ComponentNamingMode.CREATE, bindingName);
 255.                    if (bound.contains(bindInfo.getBinderServiceName())) {
 256.                        continue;
 257.                    }
 258.                    bound.add(bindInfo.getBinderServiceName());
 259.                    try {
 260.                        final BinderService service = new BinderService(bindInfo.getBindName(), bindingConfiguration.getSource());
 261.                        jndiDepServiceBuilder.addDependency(bindInfo.getBinderServiceName());
 262.                        ServiceBuilder serviceBuilder = serviceTarget.addService(bindInfo.getBinderServiceName(), service);
 263.                        bindingConfiguration.getSource().getResourceValue(resolutionContext, serviceBuilder, phaseContext, service.getManagedObjectInjector());
 264.                        serviceBuilder.addDependency(bindInfo.getParentContextServiceName(), ServiceBasedNamingStore.class, service.getNamingStoreInjector());
 265.                        serviceBuilder.install();
 266.                    } catch (DuplicateServiceException e) {
 267.                        ServiceController registered = (ServiceController) CurrentServiceContainer.getServiceContainer().getService(bindInfo.getBinderServiceName());
 268.                        if (registered == null)
 269.                            throw e;
 270.    
 271.                        BinderService service = (BinderService) registered.getService();
 272.                        if (!service.getSource().equals(bindingConfiguration.getSource()))
 273.                            throw EeLogger.ROOT_LOGGER.conflictingBinding(bindingName, bindingConfiguration.getSource());
 274.                    } catch (CircularDependencyException e) {
 275.                        throw EeLogger.ROOT_LOGGER.circularDependency(bindingName);
 276.                    }
 277.                }
 278.            }
 279.        }
