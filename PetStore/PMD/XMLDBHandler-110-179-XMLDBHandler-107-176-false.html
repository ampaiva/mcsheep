  82.      public void startElement(String uri, String localName, String qualifiedName, Attributes attributes) throws SAXException {
  83.        if (state == READY) {
  84.          if (qualifiedName.equals(elementTag)) {
  85.            state = PARSING;
  86.            values.clear();
  87.            for (int i = 0; i < attributes.getLength(); i++) {
  88.              setValue(qualifiedName + "/@" + attributes.getQName(i), attributes.getValue(i), state == PARSING ? values : context);
  89.            }
  90.            try {
  91.              if (lazyInstantiation) {
  92.                create();
  93.              }
  94.            } catch (PopulateException exception) {
  95.              throw new SAXException(exception.getMessage(), exception.getRootCause());
  96.            }
  97.            return;
  98.          }
  99.        } else if (state == OFF) {
 100.          if (qualifiedName.equals(rootElementTag)) {
 101.            state = READY;
 102.            context.clear();
 103.          }
 104.        }
 105.        if (state == READY) {
 106.          for (int i = 0; i < attributes.getLength(); i++) {
 107.            setValue(qualifiedName + "/@" + attributes.getQName(i), attributes.getValue(i), state == PARSING ? values : context);
 108.          }
 109.        }
 110.        if (state == READY || state == OFF) {
 111.          super.startElement(uri, localName, qualifiedName, attributes);
 112.        }
 113.        return;
 114.      }

 128.      public void endElement(String uri, String localName, String qualifiedName) throws SAXException {
 129.        if (state == READY) {
 130.          if (qualifiedName.equals(rootElementTag)) {
 131.            state = OFF;
 132.          } else {
 133.            setValue(qualifiedName, buffer.toString(), context);
 134.            buffer.setLength(0);
 135.          }
 136.        }
 137.        if (state == PARSING) {
 138.          if (qualifiedName.equals(elementTag)) {
 139.            try {
 140.              if (!lazyInstantiation) {
 141.                create();
 142.              } else {
 143.                update();
 144.              }
 145.            } catch (PopulateException exception) {
 146.              throw new SAXException(exception.getMessage(), exception.getRootCause());
 147.            }
 148.            state = READY;
 149.          } else {
 150.            setValue(qualifiedName, buffer.toString(), values);
 151.          }
 152.          buffer.setLength(0);
 153.        } else {
 154.          super.endElement(uri, localName, qualifiedName);
 155.        }
 156.        return;
 157.      }

 174.      private void setValue(String name, String value, Map map) {
 175.        String key = name;
 176.        if (values.get(key) == null) {
 177.          map.put(key, value);
 178.          key = name + "[" + 0 + "]";
 179.          map.put(key, value);
 180.        } else {
 181.          for (int i = 1;; i++) {
 182.              key = name + "[" + i + "]";
 183.              if (map.get(key) == null) {
 184.                map.put(key, value);
 185.                break;
 186.              }
 187.          }
 188.        }
 189.        return;
 190.      }
