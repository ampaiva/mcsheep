 127.        private HashMap getProtectedResources(Element root) {
 128.            HashMap resources = new HashMap();
 129.            // get the signon page and signon error page
 130.            signOnLoginPage = getTagValue(root, SIGNON_FORM_LOGIN_PAGE ).trim();
 131.            signOnErrorPage = getTagValue(root, SIGNON_FORM_ERROR_PAGE ).trim();
 132.    
 133.            // get protected pages //
 134.            NodeList outterList = root.getElementsByTagName(SECURITY_CONSTRAINT);
 135.            for (int outterLoop = 0; outterLoop < outterList.getLength(); outterLoop++) {
 136.                Element element = (Element)outterList.item(outterLoop);
 137.                // get  roles that can see this page
 138.                ArrayList roles = new ArrayList();
 139.                NodeList roleList = element.getElementsByTagName(AUTH_CONSTRAINT);
 140.                for (int roleLoop = 0; roleLoop < roleList.getLength(); roleLoop++) {
 141.                            Node  roleNode = roleList.item(roleLoop);
 142.                            String roleName = getSubTagValue(roleNode, ROLE_NAME);
 143.                            if ((roleName != null) && !roleName.equals("")) roles.add(roleName);
 144.                }
 145.    
 146.                NodeList list = element.getElementsByTagName(WEB_RESOURCE_COLLECTION);
 147.                for (int loop = 0; loop < list.getLength(); loop++) {
 148.                    Node node = list.item(loop);
 149.                    if (node != null) {
 150.                        String resourceName = getSubTagValue(node, WEB_RESOURCE_NAME);
 151.                        String urlPattern = getSubTagValue(node, URL_PATTERN);
 152.                        ProtectedResource resource = new ProtectedResource(resourceName, urlPattern, roles);
 153.                        if (!resources.containsKey(resourceName)) {
 154.                             resources.put(resourceName, resource);
 155.                        } else {
 156.                            System.err.println("*** Non Fatal errror: Protected Resource " + resourceName +
 157.                                           " defined more than once in screen definitions file");
 158.                        }
 159.                    }
 160.                }
 161.            }
 162.            return resources;
 163.        }
 164.        private String getSubTagAttribute(Element root, String tagName, String subTagName, String attribute) {
 165.            String returnString = "";
 166.            NodeList list = root.getElementsByTagName(tagName);
 167.            for (int loop = 0; loop < list.getLength(); loop++) {
 168.                Node node = list.item(loop);
 169.                if (node != null) {
 170.                    NodeList  children = node.getChildNodes();
 171.                    for (int innerLoop =0; innerLoop < children.getLength(); innerLoop++) {
 172.                        Node  child = children.item(innerLoop);
 173.                        if ((child != null) && (child.getNodeName() != null) && child.getNodeName().equals(subTagName) ) {
 174.                            if (child instanceof Element) {
 175.                                return ((Element)child).getAttribute(attribute);
 176.                            }
 177.                        }
 178.                    } // end inner loop
 179.                }
 180.            }
 181.            return returnString;
 182.        }
