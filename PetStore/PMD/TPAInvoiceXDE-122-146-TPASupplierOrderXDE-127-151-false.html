 127.        XMLDocumentUtils.serialize(XMLDocumentUtils.createTransformer(), supplierOrderDocument,
 128.                                   (isSupportingXSD() ? XSD_PUBLIC_ID : DTD_PUBLIC_ID),
 129.                                   systemId, isSupportingXSD(), XMLDocumentUtils.DEFAULT_ENCODING, result);
 130.        return;
 131.      }
