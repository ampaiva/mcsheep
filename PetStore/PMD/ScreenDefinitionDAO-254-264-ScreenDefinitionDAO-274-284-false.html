  94.        public static Element loadDocument(URL url) {
  95.            Document doc = null;
  96.            try {
  97.                InputSource xmlInp = new InputSource(url.openStream());
  98.    
  99.                DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
 100.                DocumentBuilder parser = docBuilderFactory.newDocumentBuilder();
 101.                doc = parser.parse(xmlInp);
 102.                Element root = doc.getDocumentElement();
 103.                root.normalize();
 104.                return root;
 105.            } catch (SAXParseException err) {
 106.                System.err.println ("ScreenFlowXmlDAO ** Parsing error" + ", line " +
 107.                            err.getLineNumber () + ", uri " + err.getSystemId ());
 108.                System.err.println("ScreenFlowXmlDAO error: " + err.getMessage ());
 109.            } catch (SAXException e) {
 110.                System.err.println("ScreenFlowXmlDAO error: " + e);
 111.            } catch (java.net.MalformedURLException mfx) {
 112.                System.err.println("ScreenFlowXmlDAO error: " + mfx);
 113.            } catch (java.io.IOException e) {
 114.                System.err.println("ScreenFlowXmlDAO error: " + e);
 115.            } catch (Exception pce) {
 116.                System.err.println("ScreenFlowXmlDAO error: " + pce);
 117.            }
 118.            return null;
 119.        }

 122.            Element root = loadDocument(location);

 130.        public static HashMap getScreenDefinitions(Element root) {
 131.            HashMap screensDefs = new HashMap();
 132.            NodeList list = root.getElementsByTagName(SCREEN_DEFINITION);
 133.            for (int loop = 0; loop < list.getLength(); loop++) {
 134.                Node node = list.item(loop);
 135.                if (node != null) {
 136.                    String language = null;
 137.                    String url = null;
 138.                    if (node instanceof Element) {
 139.                        language = ((Element)node).getAttribute(LANGUAGE);
 140.                        url = ((Element)node).getAttribute(URL);
 141.                    }
 142.                    if ((language != null) && (url != null) && !screensDefs.containsKey(language)) {
 143.                        screensDefs.put(language, url);
 144.                    } else {
 145.                        System.err.println("*** Non Fatal errror: ScreenDefinitions for language " + language +
 146.                                           " defined more than once in screen definitions file");
 147.                    }
 148.                }
 149.            }
 150.            return screensDefs;
 151.        }

 153.        public static Screens getScreens(Element root) {
 154.            // get the template
 155.            String defaultTemplate = getTagValue(root, DEFAULT_TEMPLATE);
 156.            if (defaultTemplate == null) {
 157.                System.err.println("*** ScreenDefinitionDAO error: " +
 158.                                   " Default Template not Defined.");
 159.                return null;
 160.            }
 161.    
 162.            Screens screens = new Screens(defaultTemplate);
 163.            getTemplates(root, screens);
 164.            // get screens
 165.            NodeList list = root.getElementsByTagName(SCREEN);
 166.            for (int loop = 0; loop < list.getLength(); loop++) {
 167.                Node node = list.item(loop);
 168.                if ((node != null) && node instanceof Element) {
 169.                    String templateName = ((Element)node).getAttribute(TEMPLATE);
 170.                    String screenName = ((Element)node).getAttribute(NAME);
 171.                    HashMap parameters = getParameters(node);
 172.                    Screen screen = new Screen(screenName, templateName, parameters);
 173.                    if (!screens.containsScreen(screenName)) {
 174.                        screens.addScreen(screenName, screen);
 175.                    } else {
 176.                        System.err.println("*** Non Fatal errror: Screen " + screenName +
 177.                                           " defined more than once in screen definitions file");
 178.                    }
 179.                }
 180.            }
 181.            return screens;
 182.        }

 188.        public static void getTemplates(Element root, Screens screens) {
 189.            NodeList list = root.getElementsByTagName(TEMPLATE);
 190.            for (int loop = 0; loop < list.getLength(); loop++) {
 191.                Node node = list.item(loop);
 192.                String templateName = null;
 193.                if (node != null) {
 194.                    if (node instanceof Element) {
 195.                        templateName = ((Element)node).getAttribute(NAME);
 196.                    }
 197.                    String templateURL = getSubTagValue(node, URL);
 198.    
 199.                    if (!screens.containsTemplate(templateName)) {
 200.                        screens.addTemplate(templateName, templateURL);
 201.                    } else {
 202.                        System.err.println("*** Non Fatal errror: Template " + templateName +
 203.                                           " defined more than once in screen definitions file");
 204.                    }
 205.                }
 206.            };
 207.        }

 209.        private static HashMap getParameters(Node node) {
 210.            HashMap params = new HashMap();
 211.            if (node != null) {
 212.                NodeList  children = node.getChildNodes();
 213.                for (int innerLoop =0; innerLoop < children.getLength(); innerLoop++) {
 214.                    Node  child = children.item(innerLoop);
 215.                    if ((child != null) && (child.getNodeName() != null) && child.getNodeName().equals(PARAMETER) ) {
 216.                        if (child instanceof Element) {
 217.                            Element childElement = ((Element)child);
 218.                            String key = childElement.getAttribute(KEY);
 219.                            String value = childElement.getAttribute(VALUE);
 220.                            String directString = childElement.getAttribute(DIRECT);
 221.                            boolean direct = false;
 222.                            if ((directString != null) && directString.equals("true")) {
 223.                                direct = true;
 224.                            }
 225.                            if (!params.containsKey(key)) {
 226.                                params.put(key, new Parameter(key, value, direct));
 227.                            } else {
 228.                                System.err.println("*** Non Fatal errror: " +
 229.                                                   "Parameter " + key + " is defined more than once");
 230.    
 231.                            }
 232.                        }
 233.                    }
 234.                } // end inner loop
 235.            }
 236.            return params;
 237.        }

 239.        public static String getSubTagValue(Node node, String subTagName) {
 240.            String returnString = "";
 241.            if (node != null) {
 242.                NodeList  children = node.getChildNodes();
 243.                for (int innerLoop =0; innerLoop < children.getLength(); innerLoop++) {
 244.                    Node  child = children.item(innerLoop);
 245.                    if ((child != null) && (child.getNodeName() != null) && child.getNodeName().equals(subTagName) ) {
 246.                        Node grandChild = child.getFirstChild();
 247.                        if (grandChild.getNodeValue() != null) return grandChild.getNodeValue();
 248.                    }
 249.                } // end inner loop
 250.            }
 251.            return returnString;
 252.        }
