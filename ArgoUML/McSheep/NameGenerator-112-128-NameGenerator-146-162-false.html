 102.        /**
 103.         * Generates the PHP package name for a given namespace, ignoring the root
 104.         * namespace (which is the model).
 105.         *
 106.         * @param modelElement The model element to generate the package name for.
 107.         *
 108.         * @return The PHP package name for the given model element
 109.         */
 110.        public static final String generatePackageName(Object modelElement) {
 111.            if (modelElement == null
 112.                    || !Model.getFacade().isANamespace(modelElement)
 113.                    || Model.getFacade().getNamespace(modelElement) == null) {
 114.                return null;
 115.            }
 116.    
 117.            if (!Model.getFacade().isAPackage(modelElement)) {
 118.                modelElement = Model.getFacade().getNamespace(modelElement);
 119.                if (Model.getFacade().getNamespace(modelElement) == null) {
 120.                    return null;
 121.                }
 122.            }
 123.    
 124.            String sPackageName = Model.getFacade().getName(modelElement);
 125.            while ((modelElement = Model.getFacade().getNamespace(modelElement))
 126.                    != null) {
 127.                if (Model.getFacade().getNamespace(modelElement) != null) {
 128.                    sPackageName = Model.getFacade().getName(modelElement) + '_'
 129.                            + sPackageName;
 130.                }
 131.            }
 132.    
 133.            return sPackageName;
 134.        }

 259.        /**
 260.         * Generates filename for a classifier
 261.         *
 262.         * @param modelElement  The model element to generate the filename for.
 263.         * @param sPath         The path to prefix the filename with.
 264.         * @param iMajorVersion PHP major version to generate name for.
 265.         *
 266.         * @return The generated filename.
 267.         * 
 268.         * TODO: Make this configurable to allow additional filenaming patterns,
 269.         * including MyClass.php, MyClass.class.php, and my_class.php. 
 270.         */
 271.        protected static final String generateFilename(Object modelElement,
 272.            String sPath, int iMajorVersion) {
 273.            if (!Model.getFacade().isAClassifier(modelElement)) {
 274.                throw new ClassCastException(modelElement.getClass()
 275.                        + " has wrong object type, Classifier required");
 276.            }
 277.    
 278.            if (sPath != null && sPath.trim().length() > 0) {
 279.                if (!sPath.endsWith(FILE_SEPARATOR)) {
 280.                    sPath += FILE_SEPARATOR;
 281.                }
 282.            } else {
 283.                sPath = "";
 284.            }
 285.    
 286.            String sPackagePath = NameGenerator.generatePackagePath(modelElement);
 287.            sPath += sPackagePath != null ? sPackagePath + FILE_SEPARATOR : "";
 288.    
 289.            String sFilename = sPath;
 290.            if (iMajorVersion > 4) {
 291.                if (Model.getFacade().isAInterface(modelElement)) {
 292.                    sFilename += "interface.";
 293.                } else if (Model.getFacade().isAClass(modelElement)) {
 294.                    sFilename += "class.";
 295.                } else {
 296.                    sFilename += "unknown.";
 297.                }
 298.            }
 299.    
 300.            String sClassName = Model.getFacade().getName(modelElement);
 301.            sFilename += sClassName + ".php";
 302.    
 303.            return sFilename;
 304.        }
