 339.         */
 340.        @Override
 341.        protected void setStandardBounds(int x, int y, int w, int h) {
 342.    
 343.            // In the rather unlikely case that we have no name, we give up.
 344.    
 345.            if (getNameFig() == null) {
 346.                return;
 347.            }
 348.    
 349.            // Remember where we are at present, so we can tell GEF later. Then
 350.            // check we are as big as the minimum size
 351.    
 352.            Rectangle oldBounds = getBounds();
 353.            Dimension minSize   = getMinimumSize();
 354.    
 355.            int newW = (minSize.width  > w) ? minSize.width  : w;
 356.            int newH = (minSize.height > h) ? minSize.height : h;
 357.    
 358.            Dimension stereoMin = getStereotypeFig().getMinimumSize();
 359.            Dimension nameMin   = getNameFig().getMinimumSize();
 360.    
 361.            // Work out the padding each side, depending on whether the stereotype
 362.            // is displayed and set bounds accordingly
 363.    
 364.            int extraEach = (newH - nameMin.height - stereoMin.height) / 2;
 365.            if (!(stereoMin.height == 0 && stereoMin.width == 0)) {
 366.                /* At least one stereotype is visible */
 367.                getStereotypeFig().setBounds(x, y + extraEach, newW, 
 368.                        getStereotypeFig().getHeight());
 369.            }
 370.            getNameFig().setBounds(x, y + stereoMin.height + extraEach, newW,
 371.                    nameMin.height);
 372.    
 373.            // Set the bounds of the bigPort and cover
 374.    
 375.            getBigPort().setBounds(x, y, newW, newH);
 376.            cover.setBounds(x, y, newW, newH);
 377.    
 378.            // Record the changes in the instance variables of our parent, tell GEF
 379.            // and trigger the edges to reconsider themselves.
 380.    
 381.            _x = x;
 382.            _y = y;
 383.            _w = newW;
 384.            _h = newH;
 385.    
 386.            firePropChange("bounds", oldBounds, getBounds());
 387.            updateEdges();
 388.        }
 389.

 391.        protected void updateLayout(UmlChangeEvent event) {
 392.            super.updateLayout(event);
 393.            if (event instanceof AddAssociationEvent
 394.                    || event instanceof AttributeChangeEvent) {
 395.                // TODO: We need to be more specific here about what to build
 396.                renderingChanged();
 397.                // TODO: Is this really needed?
 398.                damage();
 399.            }
 400.        }

 404.         */

 406.        public Selection makeSelection() {
 407.            return new SelectionClassifierRole(this);
 408.        }
