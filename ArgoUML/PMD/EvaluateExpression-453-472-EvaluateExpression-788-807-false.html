 498.        @Override
 499.        public void caseAFeatureCallParameters(AFeatureCallParameters node) {
 500.            inAFeatureCallParameters(node);
 501.            if (node.getLPar() != null) {
 502.                node.getLPar().apply(this);
 503.            }
 504.    
 505.            boolean hasDeclarator = false;
 506.            if (node.getDeclarator() != null) {
 507.                node.getDeclarator().apply(this);
 508.                hasDeclarator = true;
 509.            }
 510.            if (node.getActualParameterList() != null) {
 511.                List vars = null;
 512.                if (hasDeclarator) {
 513.                    List ret = new ArrayList();
 514.                    vars = (List) val;
 515.                    final PExpression exp = ((AActualParameterList) node
 516.                            .getActualParameterList()).getExpression();
 517.    
 518.                    /*
 519.                     * For a iterator call we should provide: (a) the variables (b)
 520.                     * the expression to be evaluated on each step (c) the
 521.                     * lambda-evaluator to evaluate it
 522.                     */
 523.    
 524.                    ret.add(vars);
 525.                    ret.add(exp);
 526.                    ret.add(new LambdaEvaluator() {
 527.    
 528.                        /**
 529.                         * @see org.argouml.profile.internal.ocl.LambdaEvaluator#evaluate(java.util.Map,
 530.                         *      java.lang.Object)
 531.                         */
 532.                        public Object evaluate(Map vti,
 533.                                Object expi) {
 534.                            
 535.                            Object state = EvaluateExpression.this.saveState();
 536.    
 537.                            EvaluateExpression.this.vt = vti;
 538.                            EvaluateExpression.this.val = null;
 539.                            EvaluateExpression.this.fwd = null;
 540.                            
 541.                            ((PExpression) expi).apply(EvaluateExpression.this);
 542.    
 543.                            Object reti = EvaluateExpression.this.val;
 544.                            EvaluateExpression.this.loadState(state);
 545.                            return reti;
 546.                        }
 547.    
 548.                    });
 549.    
 550.                    val = ret;
 551.                } else {
 552.                    node.getActualParameterList().apply(this);
 553.                }
 554.    
 555.            }
 556.            if (node.getRPar() != null) {
 557.                node.getRPar().apply(this);
 558.            }
 559.            outAFeatureCallParameters(node);
 560.        }
