 373.                    || theNewTarget instanceof Diagram) {
 374.                if (Model.getFacade().isAUMLElement(listTarget)) {
 375.                    Model.getPump().removeModelEventListener(this, listTarget,
 376.                            eventName);
 377.                    // Allow listening to other elements:
 378.                    removeOtherModelEventListeners(listTarget);
 379.                }
 380.    
 381.                if (Model.getFacade().isAUMLElement(theNewTarget)) {
 382.                    listTarget = theNewTarget;
 383.                    Model.getPump().addModelEventListener(this, listTarget,
 384.                            eventName);
 385.                    // Allow listening to other elements:
 386.                    addOtherModelEventListeners(listTarget);
 387.    
 388.                    rebuildModelList();
 389.    
 390.                } else {
 391.                    listTarget = null;
 392.                    removeAllElements();
 393.                }
 394.            }
 395.        }

 417.        /**
 418.         * Returns true if the given element is valid, i.e. it may be added to the
 419.         * list of elements.
 420.         *
 421.         * @param element the element to be tested
 422.         * @return true if valid
 423.         */
 424.        protected abstract boolean isValidElement(Object element);
 425.    
 426.        /**
 427.         * Returns true if some event is valid. An event is valid if the
 428.         * element changed in the event is valid. This is determined via a
 429.         * call to isValidElement.  This method can be overriden by
 430.         * subclasses if they cannot determine if it is a valid event just
 431.         * by checking the changed element.
 432.         *
 433.         * @param e the event
 434.         * @return boolean true if valid
 435.         */
 436.        protected boolean isValidEvent(PropertyChangeEvent e) {
 437.            boolean valid = false;
 438.            if (!(getChangedElement(e) instanceof Collection)) {
 439.                // TODO: Considering all delete events to be valid like below
 440.                // is going to cause lots of unecessary work and some problems
 441.                if ((e.getNewValue() == null && e.getOldValue() != null)
 442.                        // Don't test changed element if it was deleted
 443.                        || isValidElement(getChangedElement(e))) {
 444.                    valid = true; // we tried to remove a value
 445.                }
 446.            } else {
 447.                Collection col = (Collection) getChangedElement(e);
 448.                Iterator it = col.iterator();
 449.                if (!col.isEmpty()) {
 450.                    valid = true;
 451.                    while (it.hasNext()) {
 452.                        Object o = it.next();
 453.                        if (!isValidElement(o)) {
 454.                            valid = false;
 455.                            break;
 456.                        }
 457.                    }
 458.                } else {
 459.                    if (e.getOldValue() instanceof Collection
 460.                        && !((Collection) e.getOldValue()).isEmpty()) {
 461.                        valid = true;
 462.                    }
 463.                }
 464.            }
 465.            return valid;
 466.        }
