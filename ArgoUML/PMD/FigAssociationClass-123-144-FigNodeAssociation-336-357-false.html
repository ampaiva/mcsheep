     137.        /**
     138.         * Called when a model event is received from model subsystem.
     139.         * handles when a n-ary association becomes a binary association.
     140.         *
     141.         * @param mee the event
     142.         */
     143.        @Override
     144.        protected void updateLayout(UmlChangeEvent mee) {
     145.            super.updateLayout(mee);
     146.            if (mee.getSource() == getOwner()
     147.                    && mee instanceof RemoveAssociationEvent
     148.                    && "connection".equals(mee.getPropertyName())
     149.                    && Model.getFacade().getConnections(getOwner()).size() == 2) {
     150.                reduceToBinary();
     151.            }
     152.        }
     153.    
     154.        /**
     155.         * Called when deletion of an association end reduces the number of ends
     156.         * of an association down to only two. This Fig which represent the diamond
     157.         * node of a n-ary association needs to be replaced by a FigAssociation
     158.         * representing the binary relationship.
     159.         */
     160.        private void reduceToBinary() {
     161.            final Object association = getOwner();
     162.            assert (Model.getFacade().getConnections(association).size() == 2);
     163.            
     164.            // Detach any non-associationend edges (such as comment edges) already
     165.            // attached before this association node is removed.
     166.            // They'll later be re-attached to the new FigAssociation
     167.            final Collection existingEdges = getFigEdges();
     168.            for (Iterator it = existingEdges.iterator(); it.hasNext(); ) {
     169.                FigEdge edge = it.next();
     170.                if (edge instanceof FigAssociationEnd) {
     171.                    it.remove();
     172.                } else {
     173.                    removeFigEdge(edge);
     174.                }
     175.            }
     176.            
     177.            // Now we can remove ourself (which will also remove the
     178.            // attached association ends edges)
     179.            final LayerPerspective lay = (LayerPerspective) getLayer();
     180.            final MutableGraphModel gm = (MutableGraphModel) lay.getGraphModel();
     181.            gm.removeNode(association);
     182.            removeFromDiagram();
     183.            
     184.            // Create the new FigAssociation edge to replace the node
     185.            final GraphEdgeRenderer renderer =
     186.                lay.getGraphEdgeRenderer();
     187.            final FigEdgeModelElement figEdge =
     188.                (FigEdgeModelElement) renderer.getFigEdgeFor(
     189.                    gm, lay, association, null);
     190.            lay.add(figEdge);
     191.            gm.addEdge(association);
     192.            
     193.            // Add the non-associationend edges (such as comment edges) that were
     194.            // originally attached to this and attach them to the new
     195.            // FigAssociation and make sure they are positioned correctly.
     196.            for (FigEdge edge : existingEdges) {
     197.                figEdge.makeEdgePort();
     198.                if (edge.getDestFigNode() == this) {
     199.                    edge.setDestFigNode(figEdge.getEdgePort());
     200.                    edge.setDestPortFig(figEdge.getEdgePort());
     201.                }
     202.                if (edge.getSourceFigNode() == this) {
     203.                    edge.setSourceFigNode(figEdge.getEdgePort());
     204.                    edge.setSourcePortFig(figEdge.getEdgePort());
     205.                }
     206.            }
     207.            figEdge.computeRoute();
     208.        }
     209.        
     210.        /*
     211.         * Makes sure that the edges stick to the outline of the fig.
     212.         * @see org.tigris.gef.presentation.Fig#getGravityPoints()
     213.         */

     326.        /**
     327.         * Remove entire composite Fig from Diagram. Discover the attached
     328.         * FigEdgeAssociationClass and the FigClassAssociationClass attached to
     329.         * that. Remove them from the diagram before removing this.
     330.         */
     331.        @Override
     332.        protected void removeFromDiagramImpl() {
     333.            FigEdgeAssociationClass figEdgeLink = null;
     334.            final List edges = getFigEdges();
     335.    
     336.            if (edges != null) {
     337.                for (Iterator it = edges.iterator(); it.hasNext()
     338.                        && figEdgeLink == null;) {
     339.                    Object o = it.next();
     340.                    if (o instanceof FigEdgeAssociationClass) {
     341.                        figEdgeLink = (FigEdgeAssociationClass) o;
     342.                    }
     343.                }
     344.            }
     345.    
     346.            if (figEdgeLink != null) {
     347.                FigNode figClassBox = figEdgeLink.getDestFigNode();
     348.                if (!(figClassBox instanceof FigClassAssociationClass)) {
     349.                    figClassBox = figEdgeLink.getSourceFigNode();
     350.                }
     351.                figEdgeLink.removeFromDiagramImpl();
     352.                ((FigClassAssociationClass) figClassBox).removeFromDiagramImpl();
     353.            }
     354.    
     355.            super.removeFromDiagramImpl();
     356.        }
