 146.        private static void dumpTreeEntry(NamingEnumeration list, String s) throws NamingException {
 147.            System.out.println("\ndump " + s);
 148.            while (list.hasMore()) {
 149.                NameClassPair ncp = list.next();
 150.                System.out.println(ncp.toString());
 151.                if (s.length() == 0) {
 152.                    dumpJndi(ncp.getName());
 153.                } else {
 154.                    dumpJndi(s + "/" + ncp.getName());
 155.                }
 156.            }
