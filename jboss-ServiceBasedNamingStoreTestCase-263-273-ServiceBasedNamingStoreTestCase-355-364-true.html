 389.        private void assertContains(final List list, String name, Class type) {
 390.            for (NameClassPair value : list) {
 391.                if (value instanceof Binding) {
 392.                    assertNotNull(Binding.class.cast(value).getObject());
 393.                }
 394.                if (value.getName().equals(name) && value.getClassName().equals(type.getName())) {
 395.                    return;
 396.                }
 397.            }
 398.            fail("Child [" + name + "] not found in [" + list + "]");
 399.        }

 407.                }
