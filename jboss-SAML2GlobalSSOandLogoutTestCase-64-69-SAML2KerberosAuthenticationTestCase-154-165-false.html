 290.        /**
 291.         * Returns response body for the given URL request as a String. It also checks if the returned HTTP status code is the
 292.         * expected one. If the server returns {@link HttpServletResponse#SC_UNAUTHORIZED} and an username is provided, then the
 293.         * given user is authenticated against Kerberos and a new request is executed under the new subject.
 294.         *
 295.         * @param uri URI to which the request should be made
 296.         * @param user Username
 297.         * @param pass Password
 298.         * @return HTTP response body
 299.         * @throws IOException
 300.         * @throws URISyntaxException
 301.         * @throws PrivilegedActionException
 302.         * @throws LoginException
 303.         */
 304.        public static String makeCallWithKerberosAuthn(URI uri, URI idpUri, final String user, final String pass)
 305.                throws IOException, URISyntaxException,
 306.                PrivilegedActionException, LoginException {
 307.            
 308.            final String canonicalHost = Utils.getDefaultHost(true);
 309.            uri = Utils.replaceHost(uri, canonicalHost);
 310.            idpUri = Utils.replaceHost(idpUri, canonicalHost);
 311.    
 312.            LOGGER.info("Making call to: " + uri);
 313.            LOGGER.info("Expected IDP: " + idpUri);
 314.    
 315.            // Use our custom configuration to avoid reliance on external config
 316.            Configuration.setConfiguration(new Krb5LoginConfiguration());
 317.    
 318.            // 1. Authenticate to Kerberos.
 319.            final LoginContext lc = new LoginContext(Utils.class.getName(), new UsernamePasswordHandler(user, pass));
 320.            lc.login();
 321.    
 322.            // 2. Perform the work as authenticated Subject.
 323.            final String responseBody = Subject.doAs(lc.getSubject(), new HttpGetInKerberos(uri, idpUri));
 324.            lc.logout();
 325.            return responseBody;
 326.        }

 404.        /**
 405.         * Class which is intended to be run in context of a Kerberos-authenticated user, to test the http authentication via IdP.
 406.         */
 407.        private static class HttpGetInKerberos implements PrivilegedExceptionAction {
 408.    
 409.            private final URI uri;
 410.            private final URI idpUri;
 411.    
 412.            /**
 413.             * Initializes the instance.
 414.             * 
 415.             * @param uri URI of the web application
 416.             * @param idpUri URI of the respective identity provider
 417.             */
 418.            public HttpGetInKerberos(URI uri, URI idpUri) {
 419.                this.uri = uri;
 420.                this.idpUri = idpUri;
 421.            }
 422.    
 423.            /**
 424.             * Performs authentication via IdP and retrieves the document body from the {@link #uri}.
 425.             * 
 426.             * @return Body of the response retrieved from {@link #uri}
 427.             * @throws Exception
 428.             */
 429.            @Override
 430.            public String run() throws Exception {
 431.                final DefaultHttpClient httpClient = new DefaultHttpClient();
 432.                httpClient.getAuthSchemes().register(AuthPolicy.SPNEGO, new JBossNegotiateSchemeFactory(true));
 433.                httpClient.getCredentialsProvider().setCredentials(new AuthScope(null, -1, null), new NullHCCredentials());
 434.    
 435.                final HttpParams doNotRedirect = new BasicHttpParams();
 436.                doNotRedirect.setParameter(ClientPNames.HANDLE_REDIRECTS, false);
 437.                doNotRedirect.setParameter(ClientPNames.HANDLE_AUTHENTICATION, true);
 438.    
 439.                final HttpParams doRedirect = new BasicHttpParams();
 440.                doRedirect.setParameter(ClientPNames.HANDLE_AUTHENTICATION, true);
 441.                doRedirect.setParameter(ClientPNames.HANDLE_REDIRECTS, true);
 442.    
 443.                try {
 444.                    // 1. Login to IdP
 445.                    HttpGet initialIdpHttpGet = new HttpGet(this.idpUri); // GET /idp-test-DEP1
 446.                    initialIdpHttpGet.setParams(doRedirect);
 447.                    HttpResponse response = httpClient.execute(initialIdpHttpGet);
 448.                    assertThat("Unexpected status code when expecting successfull kerberos authentication", response
 449.                            .getStatusLine().getStatusCode(), equalTo(HttpStatus.SC_OK));
 450.                    consumeResponse(response);
 451.    
 452.                    // 2. Do the work, manually do the redirect
 453.                    HttpGet initialHttpGet = new HttpGet(this.uri); // GET /test-DEP1/printRoles?role=TheDuke2&role=...
 454.                    initialHttpGet.setParams(doNotRedirect);
 455.                    response = httpClient.execute(initialHttpGet);
 456.                    assertThat("Unexpected status code when expecting redirect to IdP", response.getStatusLine().getStatusCode(),
 457.                            equalTo(HttpStatus.SC_MOVED_TEMPORARILY));
 458.                    String initialHttpGetRedirect = response.getFirstHeader(HttpHeaders.LOCATION).getValue();
 459.                    consumeResponse(response);
 460.    
 461.                    HttpGet idpHttpGet = new HttpGet(initialHttpGetRedirect); // GET /idp-test-DEP1/?SAMLRequest=jZLfT4MwEMf.....
 462.                    idpHttpGet.setParams(doNotRedirect);
 463.                    response = httpClient.execute(idpHttpGet);
 464.                    assertThat("Unexpected status code when expecting redirect from SP with SAML request", response.getStatusLine()
 465.                            .getStatusCode(), equalTo(HttpStatus.SC_MOVED_TEMPORARILY));
 466.                    String idpHttpGetRedirect = response.getFirstHeader(HttpHeaders.LOCATION).getValue();
 467.                    consumeResponse(response);
 468.    
 469.                    HttpGet idpHttpGetRedirectForAuth = new HttpGet(idpHttpGetRedirect); // GET
 470.                                                                                         // /idp-test-DEP1/?SAMLRequest=jZLfT4MwEMf.....,
 471.                                                                                         // Authorization: Negotiate
 472.                    idpHttpGetRedirectForAuth.setParams(doNotRedirect);
 473.                    response = httpClient.execute(idpHttpGetRedirectForAuth);
 474.                    assertThat("Unexpected status code when expecting redirect from IdP with SAML response", response
 475.                            .getStatusLine().getStatusCode(), equalTo(HttpStatus.SC_MOVED_TEMPORARILY));
 476.                    String idpHttpGetRedirectAuth = response.getFirstHeader(HttpHeaders.LOCATION).getValue();
 477.                    consumeResponse(response);
 478.    
 479.                    HttpGet spHttpGet = new HttpGet(idpHttpGetRedirectAuth); // GET /test-DEP1/?SAMLResponse=...
 480.                    spHttpGet.setParams(doNotRedirect);
 481.                    response = httpClient.execute(spHttpGet);
 482.                    assertThat("Unexpected status code when expecting succesfull authentication to the SP", response
 483.                            .getStatusLine().getStatusCode(), equalTo(HttpStatus.SC_OK));
 484.                    return EntityUtils.toString(response.getEntity());
 485.                } finally {
 486.                    // When HttpClient instance is no longer needed,
 487.                    // shut down the connection manager to ensure
 488.                    // immediate deallocation of all system resources
 489.                    httpClient.getConnectionManager().shutdown();
 490.                }
 491.            }
 492.        }
