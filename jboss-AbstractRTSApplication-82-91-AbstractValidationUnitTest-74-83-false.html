 216.            try {
 217.                if (file != null) {
 218.                    url = file.toURI().toURL();
 219.                }
 220.            } catch (IOException e) {
 221.                url = null;
 222.            }
 223.            String fileName = xsdName;
 224.            int index = fileName.lastIndexOf("/");
 225.            if(index == -1) {
 226.                index = fileName.lastIndexOf("\\");
 227.            }
 228.            if(index != -1) {
 229.                fileName = fileName.substring(index + 1);
 230.            }
 231.            // Search
 232.            final ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
 233.            if (url == null)
 234.                url = classLoader.getResource("docs/schema/" + fileName);
 235.            if (url == null)
 236.                url = classLoader.getResource("docs/" + fileName);
 237.            if (url == null)
 238.                url = classLoader.getResource("schema/" + fileName);
 239.            if (url == null)
 240.                url = classLoader.getResource(fileName);
 241.            assertNotNull(xsdName + " not found", url);
 242.            return url;
 243.        }

 245.        /**
 246.         * Simple JBoss XSD filter
 247.         */
 248.        private static class SchemaFilter implements FilenameFilter {
 249.            private static final Pattern PATTERN = Pattern.compile("(jboss|wildfly)-.*\\.xsd$");
 250.            private final String[] exclusions;
 251.    
 252.            SchemaFilter() {
 253.                this.exclusions = new String[0];
 254.            }
 255.    
 256.            SchemaFilter(final String[] exclusions) {
 257.                this.exclusions = exclusions;
 258.            }
 259.    
 260.            @Override
 261.            public boolean accept(final File dir, final String name) {
 262.                final boolean accepted = PATTERN.matcher(name).find();
 263.                if (accepted) {
 264.                    // check for explicit excluded files
 265.                    for (final String excludedFile : exclusions) {
 266.                        if (excludedFile.equals(name)) {
 267.                            // file is in exclude list, so we don't accept this file
 268.                            return false;
 269.                        }
 270.                    }
 271.                }
 272.                return accepted;
 273.            }
 274.        }

 292.            private String formatMessage(SAXParseException e) {
 293.                StringBuffer sb = new StringBuffer();
 294.                sb.append(e.getLineNumber()).append(':').append(e.getColumnNumber());
 295.                if (e.getPublicId() != null)
 296.                    sb.append(" publicId='").append(e.getPublicId()).append('\'');
 297.                if (e.getSystemId() != null)
 298.                    sb.append(" systemId='").append(e.getSystemId()).append('\'');
 299.                sb.append(' ').append(e.getLocalizedMessage());
 300.                sb.append(" a possible cause may be that a subsystem is not using the most up to date schema.");
 301.                return sb.toString();
 302.            }
