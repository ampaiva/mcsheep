  91.        /**
  92.         * {@inheritDoc}
  93.         */
  94.        public void execute(OperationContext context, ModelNode operation) throws OperationFailedException {
  95.            if (context.isNormalServer()) {
  96.                context.addStep(new OperationStepHandler() {
  97.                    public void execute(OperationContext context, ModelNode operation) throws OperationFailedException {
  98.                        final ServiceRegistry registry = context.getServiceRegistry(false);
  99.                        if (registry != null) {
 100.                            try {
 101.                                context.getResult().set(getEndpointMetricsFragment(operation, registry));
 102.                            } catch (Exception e) {
 103.                                throw new OperationFailedException(getFallbackMessage() + ": " + e.getMessage());
 104.                            }
 105.                        } else {
 106.                            context.getResult().set(getFallbackMessage());
 107.                        }
 108.                        context.stepCompleted();
 109.                    }
 110.                }, OperationContext.Stage.RUNTIME);
 111.            } else {
 112.                context.getResult().set(getFallbackMessage());
 113.            }
 114.            context.stepCompleted();
 115.        }
 116.    
 117.        @SuppressWarnings("unchecked")
 118.        private ModelNode getEndpointMetricsFragment(final ModelNode operation, final ServiceRegistry registry) throws OperationFailedException {
 119.            final PathAddress address = PathAddress.pathAddress(operation.require(OP_ADDR));
 120.            String endpointId;
 121.            try {
 122.                endpointId = URLDecoder.decode(address.getLastElement().getValue(), "UTF-8");
 123.            } catch (final UnsupportedEncodingException e) {
 124.                throw new RuntimeException(e);
 125.            }
 126.            final String metricName = operation.require(NAME).asString();
 127.            final String webContext = endpointId.substring(0, endpointId.indexOf(":"));
 128.            final String endpointName = endpointId.substring(endpointId.indexOf(":") + 1);
 129.            ServiceName endpointServiceName = WSServices.ENDPOINT_SERVICE.append("context="+webContext).append(endpointName);
 130.            ServiceController service = (ServiceController) currentServiceContainer().getService(endpointServiceName);
 131.            ServerConfigService serverConfigService = (ServerConfigService)currentServiceContainer().getService(WSServices.CONFIG_SERVICE).getService();
 132.            Endpoint endpoint= service.getValue();
 133.            final ModelNode result = new ModelNode();
 134.            if (endpoint != null && serverConfigService.getValue().isStatisticsEnabled()) {
 135.                final EndpointMetrics endpointMetrics = endpoint.getEndpointMetrics();
 136.                if (MIN_PROCESSING_TIME.getName().equals(metricName)) {
 137.                    result.set(String.valueOf(endpointMetrics.getMinProcessingTime()));
 138.                } else if (MAX_PROCESSING_TIME.getName().equals(metricName)) {
 139.                    result.set(String.valueOf(endpointMetrics.getMaxProcessingTime()));
 140.                } else if (AVERAGE_PROCESSING_TIME.getName().equals(metricName)) {
 141.                    result.set(String.valueOf(endpointMetrics.getAverageProcessingTime()));
 142.                } else if (TOTAL_PROCESSING_TIME.getName().equals(metricName)) {
 143.                    result.set(String.valueOf(endpointMetrics.getTotalProcessingTime()));
 144.                } else if (REQUEST_COUNT.getName().equals(metricName)) {
 145.                    result.set(String.valueOf(endpointMetrics.getRequestCount()));
 146.                } else if (RESPONSE_COUNT.getName().equals(metricName)) {
 147.                    result.set(String.valueOf(endpointMetrics.getResponseCount()));
 148.                } else if (FAULT_COUNT.getName().equals(metricName)) {
 149.                    result.set(String.valueOf(endpointMetrics.getFaultCount()));
 150.                }
 151.            } else {
 152.                result.set(getFallbackMessage());
 153.            }
 154.            return result;
 155.        }

 157.        private static String getFallbackMessage() {
 158.            return WSLogger.ROOT_LOGGER.noMetricsAvailable();
 159.        }

 165.            return AccessController.doPrivileged(CurrentServiceContainer.GET_ACTION);
 166.        }
