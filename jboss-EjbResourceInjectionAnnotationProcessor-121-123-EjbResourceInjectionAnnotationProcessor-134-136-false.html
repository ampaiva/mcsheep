 149.        private void process(final DeploymentUnit deploymentUnit, final String beanInterface, final String beanName, final String lookup, final ClassInfo classInfo, final InjectionTarget targetDescription, final String localContextName, final EEModuleDescription eeModuleDescription) {
 150.    
 151.            if (!isEmpty(lookup) && !isEmpty(beanName)) {
 152.                if (logger.isDebugEnabled()) {
 153.                    logger.debug("Both beanName = " + beanName + " and lookup = " + lookup
 154.                            + " have been specified in @EJB annotation." + " lookup will be given preference. Class: "
 155.                            + classInfo.name());
 156.                }
 157.            }
 158.    
 159.            final EEModuleClassDescription classDescription = eeModuleDescription.addOrGetLocalClassDescription(classInfo.name().toString());
 160.    
 161.            final InjectionSource valueSource;
 162.            EjbInjectionSource ejbInjectionSource = null;
 163.            //give preference to lookup
 164.            if (!isEmpty(lookup)) {
 165.                if (!lookup.startsWith("java:")) {
 166.                    valueSource = new EjbLookupInjectionSource(lookup, targetDescription.getDeclaredValueClassName());
 167.                } else {
 168.                    valueSource = createLookup(lookup, appclient);
 169.                }
 170.            } else if (!isEmpty(beanName)) {
 171.                valueSource = ejbInjectionSource = new EjbInjectionSource(beanName, beanInterface, localContextName, deploymentUnit, appclient);
 172.            } else {
 173.                valueSource = ejbInjectionSource = new EjbInjectionSource(beanInterface, localContextName, deploymentUnit, appclient);
 174.            }
 175.            if (ejbInjectionSource != null) {
 176.                deploymentUnit.addToAttachmentList(EjbDeploymentAttachmentKeys.EJB_INJECTIONS, ejbInjectionSource);
 177.            }
 178.            // our injection comes from the local lookup, no matter what.
 179.            final ResourceInjectionConfiguration injectionConfiguration = targetDescription != null ?
 180.                    new ResourceInjectionConfiguration(targetDescription, createLookup(localContextName, appclient)) : null;
 181.    
 182.            // Create the binding from whence our injection comes.
 183.            final BindingConfiguration bindingConfiguration = new BindingConfiguration(localContextName, valueSource);
 184.    
 185.            classDescription.getBindingConfigurations().add(bindingConfiguration);
 186.            if (injectionConfiguration != null) {
 187.                classDescription.addResourceInjection(injectionConfiguration);
 188.            }
 189.        }

  87.            for (AnnotationInstance annotation : resourceAnnotations) {
  88.                final AnnotationTarget annotationTarget = annotation.target();
  89.                final EJBResourceWrapper annotationWrapper = new EJBResourceWrapper(annotation, propertyReplacer);
  90.                if (annotationTarget instanceof FieldInfo) {
  91.                    processField(deploymentUnit, annotationWrapper, (FieldInfo) annotationTarget, moduleDescription);
  92.                } else if (annotationTarget instanceof MethodInfo) {
  93.                    processMethod(deploymentUnit, annotationWrapper, (MethodInfo) annotationTarget, moduleDescription);
  94.                } else if (annotationTarget instanceof ClassInfo) {
  95.                    processClass(deploymentUnit, annotationWrapper, (ClassInfo) annotationTarget, moduleDescription);
  96.                }
  97.            }
  98.            final List ejbsAnnotations = index.getAnnotations(EJBS_ANNOTATION_NAME);
  99.            for (AnnotationInstance annotation : ejbsAnnotations) {
 100.                final AnnotationTarget annotationTarget = annotation.target();
 101.                if (annotationTarget instanceof ClassInfo) {
 102.                    final AnnotationValue annotationValue = annotation.value();
 103.                    final AnnotationInstance[] ejbAnnotations = annotationValue.asNestedArray();
 104.                    for (AnnotationInstance ejbAnnotation : ejbAnnotations) {
 105.                        final EJBResourceWrapper annotationWrapper = new EJBResourceWrapper(ejbAnnotation, propertyReplacer);
 106.                        processClass(deploymentUnit, annotationWrapper, (ClassInfo) annotationTarget, moduleDescription);
 107.                    }
 108.                } else {
 109.                    throw EjbLogger.ROOT_LOGGER.annotationOnlyAllowedOnClass(EJBs.class.getName(), annotation.target());
 110.                }
 111.            }
 112.        }

 114.        public void undeploy(DeploymentUnit context) {
 115.        }

 117.        private void processField(final DeploymentUnit deploymentUnit, final EJBResourceWrapper annotation, final FieldInfo fieldInfo, final EEModuleDescription eeModuleDescription) {
 118.            final String fieldName = fieldInfo.name();
 119.            final String fieldType = fieldInfo.type().name().toString();
 120.            final InjectionTarget targetDescription = new FieldInjectionTarget(fieldInfo.declaringClass().name().toString(), fieldName, fieldType);
 121.            final String localContextName = isEmpty(annotation.name()) ? fieldInfo.declaringClass().name().toString() + "/" + fieldInfo.name() : annotation.name();
 122.            final String beanInterfaceType = isEmpty(annotation.beanInterface()) || annotation.beanInterface().equals(Object.class.getName()) ? fieldType : annotation.beanInterface();
 123.            process(deploymentUnit, beanInterfaceType, annotation.beanName(), annotation.lookup(), fieldInfo.declaringClass(), targetDescription, localContextName, eeModuleDescription);
 124.        }
 125.    
 126.        private void processMethod(final DeploymentUnit deploymentUnit, final EJBResourceWrapper annotation, final MethodInfo methodInfo, final EEModuleDescription eeModuleDescription) {
 127.            final String methodName = methodInfo.name();
 128.            if (!methodName.startsWith("set") || methodInfo.args().length != 1) {
 129.                throw EjbLogger.ROOT_LOGGER.onlySetterMethodsAllowedToHaveEJBAnnotation(methodInfo);
 130.            }
 131.            final String methodParamType = methodInfo.args()[0].name().toString();
 132.            final InjectionTarget targetDescription = new MethodInjectionTarget(methodInfo.declaringClass().name().toString(), methodName, methodParamType);
 133.    
 134.            final String localContextName = isEmpty(annotation.name()) ? methodInfo.declaringClass().name().toString() + "/" + methodName.substring(3, 4).toLowerCase(Locale.ENGLISH) + methodName.substring(4) : annotation.name();
 135.            final String beanInterfaceType = isEmpty(annotation.beanInterface()) || annotation.beanInterface().equals(Object.class.getName()) ? methodParamType : annotation.beanInterface();
 136.            process(deploymentUnit, beanInterfaceType, annotation.beanName(), annotation.lookup(), methodInfo.declaringClass(), targetDescription, localContextName, eeModuleDescription);
 137.        }
 138.    
 139.        private void processClass(final DeploymentUnit deploymentUnit, final EJBResourceWrapper annotation, final ClassInfo classInfo, final EEModuleDescription eeModuleDescription) throws DeploymentUnitProcessingException {
 140.            if (isEmpty(annotation.name())) {
 141.                throw EjbLogger.ROOT_LOGGER.nameAttributeRequiredForEJBAnnotationOnClass(classInfo.toString());
 142.            }
 143.            if (isEmpty(annotation.beanInterface())) {
 144.                throw EjbLogger.ROOT_LOGGER.beanInterfaceAttributeRequiredForEJBAnnotationOnClass(classInfo.toString());
 145.            }
 146.            process(deploymentUnit, annotation.beanInterface(), annotation.beanName(), annotation.lookup(), classInfo, null, annotation.name(), eeModuleDescription);
 147.        }
 148.    
 149.        private void process(final DeploymentUnit deploymentUnit, final String beanInterface, final String beanName, final String lookup, final ClassInfo classInfo, final InjectionTarget targetDescription, final String localContextName, final EEModuleDescription eeModuleDescription) {
 150.    
 151.            if (!isEmpty(lookup) && !isEmpty(beanName)) {
 152.                if (logger.isDebugEnabled()) {
 153.                    logger.debug("Both beanName = " + beanName + " and lookup = " + lookup
 154.                            + " have been specified in @EJB annotation." + " lookup will be given preference. Class: "
 155.                            + classInfo.name());
 156.                }
 157.            }
 158.    
 159.            final EEModuleClassDescription classDescription = eeModuleDescription.addOrGetLocalClassDescription(classInfo.name().toString());
 160.    
 161.            final InjectionSource valueSource;
 162.            EjbInjectionSource ejbInjectionSource = null;
 163.            //give preference to lookup
 164.            if (!isEmpty(lookup)) {
 165.                if (!lookup.startsWith("java:")) {
 166.                    valueSource = new EjbLookupInjectionSource(lookup, targetDescription.getDeclaredValueClassName());
 167.                } else {
 168.                    valueSource = createLookup(lookup, appclient);
 169.                }
 170.            } else if (!isEmpty(beanName)) {
 171.                valueSource = ejbInjectionSource = new EjbInjectionSource(beanName, beanInterface, localContextName, deploymentUnit, appclient);
 172.            } else {
 173.                valueSource = ejbInjectionSource = new EjbInjectionSource(beanInterface, localContextName, deploymentUnit, appclient);
 174.            }
 175.            if (ejbInjectionSource != null) {
 176.                deploymentUnit.addToAttachmentList(EjbDeploymentAttachmentKeys.EJB_INJECTIONS, ejbInjectionSource);
 177.            }
 178.            // our injection comes from the local lookup, no matter what.
 179.            final ResourceInjectionConfiguration injectionConfiguration = targetDescription != null ?
 180.                    new ResourceInjectionConfiguration(targetDescription, createLookup(localContextName, appclient)) : null;
 181.    
 182.            // Create the binding from whence our injection comes.
 183.            final BindingConfiguration bindingConfiguration = new BindingConfiguration(localContextName, valueSource);
 184.    
 185.            classDescription.getBindingConfigurations().add(bindingConfiguration);
 186.            if (injectionConfiguration != null) {
 187.                classDescription.addResourceInjection(injectionConfiguration);
 188.            }
 189.        }
 190.    
 191.        private InjectionSource createLookup(final String localContextName, final boolean appclient) {
 192.            //appclient lookups are always optional
 193.            //as they could reference local interfaces that are not present
 194.            return new LookupInjectionSource(localContextName, appclient);
 195.        }
 196.    
 197.        private boolean isEmpty(final String string) {
 198.            return string == null || string.isEmpty();
 199.        }

 201.        private class EJBResourceWrapper {
 202.            private final String name;
 203.            private final String beanInterface;
 204.            private final String beanName;
 205.            private final String lookup;
 206.            private final PropertyReplacer propertyReplacer;
 207.    
 208.            public EJBResourceWrapper(AnnotationInstance annotation, PropertyReplacer propertyReplacer) {
 209.                this.propertyReplacer = propertyReplacer;
 210.                name = stringValueOrNull(annotation, "name");
 211.                beanInterface = classValueOrNull(annotation, "beanInterface");
 212.                beanName = stringValueOrNull(annotation, "beanName");
 213.                String lookupValue = stringValueOrNull(annotation, "lookup");
 214.                // if "lookup" isn't specified, then fallback on "mappedName". We treat "mappedName" the same as "lookup"
 215.                if (isEmpty(lookupValue)) {
 216.                    lookupValue = stringValueOrNull(annotation, "mappedName");
 217.                }
 218.                this.lookup = lookupValue;
 219.            }
 220.    
 221.            private String name() {
 222.                return name;
 223.            }
 224.    
 225.            private String beanInterface() {
 226.                return beanInterface;
 227.            }
 228.    
 229.            private String beanName() {
 230.                return beanName;
 231.            }
 232.    
 233.            private String lookup() {
 234.                return lookup;
 235.            }
 236.    
 237.            private String stringValueOrNull(final AnnotationInstance annotation, final String attribute) {
 238.                final AnnotationValue value = annotation.value(attribute);
 239.                return (value != null) ? propertyReplacer.replaceProperties(value.asString()) : null;
 240.            }
 241.    
 242.            private String classValueOrNull(final AnnotationInstance annotation, final String attribute) {
 243.                final AnnotationValue value = annotation.value(attribute);
 244.                return (value != null) ? value.asClass().name().toString() : null;
 245.            }
 246.        }
