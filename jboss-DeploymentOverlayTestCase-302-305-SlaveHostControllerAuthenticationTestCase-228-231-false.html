 176.        private static void reloadSlave() throws Exception {
 177.            ModelNode op = new ModelNode();
 178.            op.get(OP).set("reload");
 179.            op.get(OP_ADDR).add(HOST, "slave");
 180.            op.get(ADMIN_ONLY).set(false);
 181.            try {
 182.                domainSlaveClient.execute(new OperationBuilder(op).build());
 183.            } catch(IOException e) {
 184.                final Throwable cause = e.getCause();
 185.                if (!(cause instanceof ExecutionException) && !(cause instanceof CancellationException)) {
 186.                    throw e;
 187.                } // else ignore, this might happen if the channel gets closed before we got the response
 188.            }
 189.    
 190.            // Wait until host is reloaded
 191.            readHostControllerStatus(domainSlaveClient, TIMEOUT);
 192.        }

 194.        private static void readHostControllerStatus(ModelControllerClient client, long timeout) throws Exception {
 195.            final long time = System.currentTimeMillis() + timeout;
 196.            do {
 197.                Thread.sleep(250);
 198.                if (lookupHostInModel(client)) {
 199.                    return;
 200.                }
 201.            } while (System.currentTimeMillis() < time);
 202.    
 203.            Assert.fail("Cannot validate host 'slave' is running");
 204.        }

 206.        private static boolean lookupHostInModel(ModelControllerClient client) throws Exception {
 207.            final ModelNode operation = new ModelNode();
 208.            operation.get(OP).set(READ_ATTRIBUTE_OPERATION);
 209.            operation.get(OP_ADDR).add(HOST, "slave");
 210.            operation.get(NAME).set(HOST_STATE);
 211.    
 212.            try {
 213.                final ModelNode result = client.execute(operation);
 214.                if (result.get(OUTCOME).asString().equals(SUCCESS)){
 215.                    final ModelNode model = result.require(RESULT);
 216.                    if (model.asString().equalsIgnoreCase("running")) {
 217.                        return true;
 218.                    }
 219.                }
 220.            } catch (IOException e) {
 221.                //
 222.            }
 223.            return false;
 224.        }
