 259.        private ResourceTransformationDescriptionBuilder registerModuleTransformer(final ResourceTransformationDescriptionBuilder parent, final PathElement childPath,
 260.                                                                                   final ModulesToAttributeTransformer transformer, final ChildResourceTransformersRegistrar childRegistrar) {
 261.            final OperationTransformer addOrWriteTransformer = new OperationTransformer() {
 262.                public TransformedOperation transformOperation(TransformationContext context, PathAddress address, ModelNode operation)
 263.                        throws OperationFailedException {
 264.                    return transformer.transformParentOperation(context, address, operation);
 265.                }
 266.            };
 267.    
 268.            ResourceTransformationDescriptionBuilder child = parent.addChildResource(childPath)
 269.                    .setCustomResourceTransformer(transformer)
 270.                    .addOperationTransformationOverride(ADD).setCustomOperationTransformer(addOrWriteTransformer).inheritResourceAttributeDefinitions().end()
 271.                    .addOperationTransformationOverride(WRITE_ATTRIBUTE_OPERATION).setCustomOperationTransformer(addOrWriteTransformer).inheritResourceAttributeDefinitions().end()
 272.                    .addOperationTransformationOverride(UNDEFINE_ATTRIBUTE_OPERATION).setCustomOperationTransformer(addOrWriteTransformer).inheritResourceAttributeDefinitions().end()
 273.                    .addOperationTransformationOverride(REMOVE)
 274.                        .setCustomOperationTransformer(new OperationTransformer() {
 275.                            public TransformedOperation transformOperation(TransformationContext context, PathAddress address,
 276.                                    ModelNode operation) throws OperationFailedException {
 277.                                //Record that we removed the resource and return the original operation
 278.                                return new TransformedOperation(operation, TransformedOperation.ORIGINAL_RESULT);
 279.                            }
 280.                        }).end();
 281.            ResourceTransformationDescriptionBuilder childRedirectionBuilder = child.addChildRedirection(
 282.                        PathElement.pathElement(transformer.getResourceName()), CURRENT_PATH_TRANSFORMER)
 283.                    .setCustomResourceTransformer(ResourceTransformer.DISCARD)
 284.                    .addOperationTransformationOverride(ADD)
 285.                    .setCustomOperationTransformer(transformer)
 286.                    .inheritResourceAttributeDefinitions().end()
 287.                    .addOperationTransformationOverride(WRITE_ATTRIBUTE_OPERATION)
 288.                    .setCustomOperationTransformer(transformer)
 289.                    .inheritResourceAttributeDefinitions().end()
 290.                    .addOperationTransformationOverride(REMOVE)
 291.                    .setCustomOperationTransformer(transformer)
 292.                    .inheritResourceAttributeDefinitions().end();

 333.                ModelNode model = new ModelNode();
 334.                transformModulesToAttributes(address, resourceName, attributeName, context, model);

 347.            void transformModulesToAttributes(final PathAddress address, final String newName, final String oldName, final TransformationContext context, final ModelNode model) {

 351.                    for (Resource.ResourceEntry entry : children) {
 352.                        Resource moduleResource = context.readResourceFromRoot(address.append(entry.getPathElement()));
 353.                        modules.add(moduleResource.getModel());

 362.                for (ModelNode step : steps) {

 369.                //If the add is the form which has the attribute set, let it through - otherwise just discard
 370.                String op = operation.get(OP).asString();
 371.                if (ADD.equals(op)) {
 372.                    if (operation.hasDefined(attributeName)) {

 374.                    }

 384.            private JaspiModulesToAttributeTransformer(String resourceName, String oldName) {
 385.                super(resourceName, oldName);

 389.            public TransformedOperation transformOperation(TransformationContext context, PathAddress address, ModelNode operation) throws OperationFailedException {
 390.                //Within the authentication=jaspi resource the structure is:
 391.                // authentication=jaspi
 392.                //    auth-module=*  (should get pulled up into authentication=jaspi)
 393.                //    login-module-stack=*
 394.                //       login-module=*  (should get pulled up into login-module-stack
 395.                //
 396.                // If there are no auth-module children, and we are the login-module-stack, we cannot be added
 397.                PathAddress jaspiAddress = getJaspiAuthenticationAddress(address);
 398.    
 399.                ModelNode authModulesModel = new ModelNode();
 400.                transformModulesToAttributes(jaspiAddress, Constants.AUTH_MODULE, Constants.AUTH_MODULES, context, authModulesModel);
 401.    
 402.                ModelNode authModules = authModulesModel.get(Constants.AUTH_MODULES);
 403.                if (authModules.asList().size() == 0) {
 404.                    return new TransformedOperation(Util.createRemoveOperation(jaspiAddress), OperationResultTransformer.ORIGINAL_RESULT);
 405.                } else {
 406.                    List list = new ArrayList();
 407.    
 408.                    ModelNode addJaspi = Util.createAddOperation(jaspiAddress);
 409.                    addJaspi.get(Constants.AUTH_MODULES).set(authModules);
 410.                    list.add(addJaspi);
 411.    
 412.                    Resource jaspiResource = context.readResourceFromRoot(jaspiAddress);
 413.                    for (ResourceEntry entry : jaspiResource.getChildren(Constants.LOGIN_MODULE_STACK)) {
 414.                        PathAddress stackAddress = jaspiAddress.append(PathElement.pathElement(Constants.LOGIN_MODULE_STACK, entry.getName()));
 415.    
 416.                        ModelNode loginModulesModel = new ModelNode();
 417.                        transformModulesToAttributes(stackAddress, Constants.LOGIN_MODULE, Constants.LOGIN_MODULES, context, loginModulesModel);
 418.    
 419.                        ModelNode loginModules = loginModulesModel.get(Constants.LOGIN_MODULES);
 420.                        if (loginModules.asList().size() > 0) {
 421.                            ModelNode addStack = Util.createAddOperation(stackAddress);
 422.                            addStack.get(Constants.LOGIN_MODULES).set(loginModules);
 423.                            list.add(addStack);
 424.                        }
 425.                    }
 426.    
 427.                    return new TransformedOperation(createComposite(jaspiAddress, list.toArray(new ModelNode[list.size()])), OperationResultTransformer.ORIGINAL_RESULT);
 428.                }

 454.                final ModelNode transformedOp;

 468.                    ModelNode writeOp = Util.createEmptyOperation(WRITE_ATTRIBUTE_OPERATION, pa);
 469.                    writeOp.get(NAME).set(attributeName);
 470.                    writeOp.get(VALUE).set(operation.get(attributeName));
 471.                    steps.add(writeOp);
 472.                }
 473.    
 474.                return new TransformedOperation(transformedOp, OperationResultTransformer.ORIGINAL_RESULT);

 390.        private static class StandardWebExtensionAliasEntry extends AliasEntry {
 391.            public StandardWebExtensionAliasEntry(ManagementResourceRegistration target) {
 392.                super(target);
 393.            }
 394.    
 395.            @Override
 396.            public PathAddress convertToTargetAddress(PathAddress addr) {
 397.                final PathAddress targetAddress = getTargetAddress();
 398.                List list = new ArrayList();
 399.                int i = 0;
 400.                for (PathElement element : addr) {
 401.                    String key = element.getKey();
 402.                    try {
 403.                        if (i < targetAddress.size() && (key.equals(Constants.SSL) || key.equals(Constants.SSO) || key.equals(Constants.ACCESS_LOG) || key.equals(Constants.DIRECTORY))) {
 404.                            list.add(targetAddress.getElement(i));
 405.                        } else {
 406.                            list.add(element);
 407.                        }
 408.                        i++;
 409.                    } catch (Exception e) {
 410.                        throw new RuntimeException("Bad " + addr + " " + targetAddress);
 411.                    }
 412.                }
 413.                return PathAddress.pathAddress(list);
 414.            }
 415.        }
